388 /*********************************************************************
389  * @fn CyclingSensor_ProcessEvent
390  *
391  * @brief Cycling Application Task event processor. This function
392  * is called to process all events for the task. Events
393  * include timers, messages and any other user defined events.
394  *
395  * @param task_id - The OSAL assigned task ID.
396  * @param events - events to process. This is a bit map and can
397  * contain more than one event.
398  *
399  * @return events not processed
400  */
401 uint16 CyclingSensor_ProcessEvent( uint8 task_id, uint16 events )
402 {
403 
404  VOID task_id; // OSAL required parameter that isn't used in this function
405 
406  if ( events & SYS_EVENT_MSG )
407  {
408  uint8 *pMsg;
409 
410  if ( (pMsg = osal_msg_receive( sensor_TaskID )) != NULL )
411  {
412  sensor_ProcessOSALMsg( (osal_event_hdr_t *)pMsg );
413 
414  // Release the OSAL message
415  VOID osal_msg_deallocate( pMsg );
416  }
417 
418  // return unprocessed events
419  return (events ^ SYS_EVENT_MSG);
420  }
421 
422  if ( events & START_DEVICE_EVT )
423  {
424  // Start the Device
425  VOID GAPRole_StartDevice( &cyclingPeripheralCB );
426 
427  // Register with bond manager after starting device
428  GAPBondMgr_Register( (gapBondCBs_t *) &cyclingBondCB );
429 
430  return ( events ^ START_DEVICE_EVT );
431  }
432 
433  if ( events & CSC_PERIODIC_EVT )
434  {
435  // Perform sensor's periodic task
436  sensorPeriodicTask();
437 
438  return (events ^ CSC_PERIODIC_EVT);
439  }
440 
441  if ( events & CSC_CONN_PARAM_UPDATE_EVT )
442  {
443  // Send param update. If it fails, retry until successful.
444  GAPRole_SendUpdateParam( DEFAULT_DESIRED_MIN_CONN_INTERVAL, DEFAULT_DESIRED_MAX_CONN_INTERVAL,
445  DEFAULT_DESIRED_SLAVE_LATENCY, DEFAULT_DESIRED_CONN_TIMEOUT,
446  GAPROLE_RESEND_PARAM_UPDATE );
447 
448  // Assuming service discovery complete, start neglect timer
449  if ( USING_NEGLECT_TIMEOUT )
450  {
451  osal_start_timerEx( sensor_TaskID, CSC_NEGLECT_TIMEOUT_EVT, NEGLECT_TIMEOUT_DELAY );
452  }
453 
454  return (events ^ CSC_CONN_PARAM_UPDATE_EVT);
455  }
456 
457  if ( events & CSC_NEGLECT_TIMEOUT_EVT )
458  {
459  // No user input, terminate connection.
460  GAPRole_TerminateConnection();
461 
462  return ( events ^ CSC_NEGLECT_TIMEOUT_EVT );
463  }
464 
465  if ( events & CSC_RESET_EVT )
466  {
467  // Soft reset in action
468  if ( gapProfileState == GAPROLE_CONNECTED )
469  {
470  // Exit the connection
471  GAPRole_TerminateConnection();
472 
473  // There is no callback for manual termination of the link. change state variable here.
474  gapProfileState = GAPROLE_WAITING;
475 
476  // Set timer to give the end advertising event time to finish
477  osal_start_timerEx( sensor_TaskID, CSC_RESET_EVT, 500 );
478  }
479  else if ( gapProfileState == GAPROLE_ADVERTISING )
480  {
481  uint8 value = FALSE;
482 
483  // Turn off advertising
484  GAPRole_SetParameter( GAPROLE_ADVERT_ENABLED, sizeof( uint8 ), &value );
485 
486  // Set timer to give the end advertising event time to finish
487  osal_start_timerEx( sensor_TaskID, CSC_RESET_EVT, 500 );
488  }
489  else if ( USING_WHITE_LIST == TRUE )
490  {
491  //temporary variable
492  uint8 value = GAP_FILTER_POLICY_ALL;
493 
494  // Turn off white list filter policy
495  GAPRole_SetParameter(GAPROLE_ADV_FILTER_POLICY, sizeof( uint8 ), &value);
496 
497  sensorUsingWhiteList = FALSE;
498 
499  // Clear the white list
500  HCI_LE_ClearWhiteListCmd();
501 
502  // Set timer to give the end advertising event time to finish
503  osal_start_timerEx( sensor_TaskID, CSC_RESET_EVT, 500 );
504  }
505  else if ( (gapProfileState == GAPROLE_STARTED) ||
506  (gapProfileState == GAPROLE_WAITING) ||
507  (gapProfileState == GAPROLE_WAITING_AFTER_TIMEOUT) )
508  {
509  uint8 eraseBonds = TRUE;
510 
511  // Erase all bonds
512  GAPBondMgr_SetParameter( GAPBOND_ERASE_ALLBONDS, 0, &eraseBonds );
513 
514  // Turn on GREEN LED for set time
515  HalLedSet( HAL_LED_1, HAL_LED_MODE_BLINK );
516  }
517 
518  return (events ^ CSC_RESET_EVT);
519  }
520 
521  // Discard unknown events
522  return 0;
523 }

544 /*********************************************************************
545  * @fn sensor_HandleKeys
546  *
547  * @brief Handles all key events for this device.
548  *
549  * @param shift - true if in shift/alt.
550  * @param keys - bit field for key events. Valid entries:
551  * HAL_KEY_SW_2
552  * HAL_KEY_SW_1
553  *
554  * @return none
555  */
556 static void sensor_HandleKeys( uint8 shift, uint8 keys )
557 {
558 
559  if ( keys == ( HAL_KEY_SW_1 | HAL_KEY_SW_2 ) )
560  {
561  // Reset in progress has started
562  resetInProgress = TRUE;
563 
564  // Set OSAL timer for reset
565  osal_start_timerEx( sensor_TaskID, CSC_RESET_EVT, CSC_RESET_DELAY );
566  }
567  else if ( keys & HAL_KEY_SW_1 )
568  {
569  if ( resetInProgress == TRUE )
570  {
571  // Cancel the reset
572  resetInProgress = FALSE;
573  osal_stop_timerEx ( sensor_TaskID, CSC_RESET_EVT );
574  }
575 
576  // set simulated measurement flag index
577  if (++sensorFlagsIdx == FLAGS_IDX_MAX)
578  {
579  sensorFlagsIdx = 0;
580  }
581  }
582  else if ( keys & HAL_KEY_SW_2 )
583  {
584  if ( resetInProgress == TRUE )
585  {
586  // Cancel the reset
587  resetInProgress = FALSE;
588  osal_stop_timerEx ( sensor_TaskID, CSC_RESET_EVT );
589  }
590 
591  // if not in a connection, toggle advertising on and off
592  if ( gapProfileState != GAPROLE_CONNECTED )
593  {
594  uint8 status;
595 
596  // Set fast advertising interval for user-initiated connections
597  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MIN, DEFAULT_FAST_ADV_INTERVAL );
598  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MAX, DEFAULT_FAST_ADV_INTERVAL );
599  GAP_SetParamValue( TGAP_GEN_DISC_ADV_MIN, DEFAULT_WHITE_LIST_ADV_DURATION );
600 
601  // toggle GAP advertisement status
602  GAPRole_GetParameter( GAPROLE_ADVERT_ENABLED, &status );
603  status = !status;
604 
605  // If not already using white list, begin to do so.
606  // Only do so if about to begin advertising
607  if ( USING_WHITE_LIST && status == TRUE )
608  {
609  uint8 bondCount = 0;
610 
611  GAPBondMgr_GetParameter( GAPBOND_BOND_COUNT, &bondCount );
612 
613  if ((sensorUsingWhiteList == FALSE) && (bondCount > 0) )
614  {
615  uint8 value = GAP_FILTER_POLICY_WHITE;
616 
617  GAPRole_SetParameter(GAPROLE_ADV_FILTER_POLICY, sizeof( uint8 ), &value);
618 
619  sensorUsingWhiteList = TRUE;
620  }
621  }
622 
623  GAPRole_SetParameter( GAPROLE_ADVERT_ENABLED, sizeof( uint8 ), &status );
624 
625  // Set state variable
626  if (status == FALSE)
627  {
628  sensorAdvCancelled = TRUE;
629  }
630  }
631  }
632  // End of key press
633  else if ( keys == 0x00 )
634  {
635  if ( resetInProgress == TRUE )
636  {
637  resetInProgress = FALSE;
638 
639  osal_stop_timerEx( sensor_TaskID, CSC_RESET_EVT );
640  }
641  }
642 }

644 /*********************************************************************
645  * @fn sensorMeasNotify
646  *
647  * @brief Prepare and send a CSC measurement notification
648  *
649  * @return none
650  */
651 static void sensorMeasNotify( void )
652 {
653  uint8 *p = sensorMeas.value;
654  uint8 flags = sensorFlags[sensorFlagsIdx];
655 
656  // Build CSC measurement structure from simulated values
657  // Flags simulate the isPresent bits.
658  *p++ = flags;
659 
660  // If present, add Speed data into measurement
661  if (flags & CSC_FLAGS_SPEED)
662  {
663  *p++ = BREAK_UINT32(cummWheelRevs, 0);
664  *p++ = BREAK_UINT32(cummWheelRevs, 1);
665  *p++ = BREAK_UINT32(cummWheelRevs, 2);
666  *p++ = BREAK_UINT32(cummWheelRevs, 3);
667 
668  *p++ = LO_UINT16(lastWheelEvtTime);
669  *p++ = HI_UINT16(lastWheelEvtTime);
670 
671  // Update simulated values (simulate in the reverse direction)
672  if (cummWheelRevs < WHEEL_REV_INCREMENT) //don't allow revolutions to roll over
673  {
674  cummWheelRevs = 0;
675  }
676  else
677  {
678  cummWheelRevs -= WHEEL_REV_INCREMENT;
679  }
680 
681  lastWheelEvtTime += WHEEL_EVT_INCREMENT;
682  }
683 
684  // If present, add Cadence data into measurement
685  if (flags & CSC_FLAGS_CADENCE)
686  {
687  *p++ = LO_UINT16(cummCrankRevs);
688  *p++ = HI_UINT16(cummCrankRevs);
689 
690  *p++ = LO_UINT16(lastCrankEvtTime);
691  *p++ = HI_UINT16(lastCrankEvtTime);
692 
693  // Update Simualted Values
694  cummCrankRevs += CRANK_REV_INCREMENT;
695  lastCrankEvtTime += CRANK_EVT_INCREMENT;
696  }
697 
698  // Get length
699  sensorMeas.len = (uint8) (p - sensorMeas.value);
700 
701  // Send to service to send the notification
702  Cycling_MeasNotify( gapConnHandle, &sensorMeas );
703 }

705 /*********************************************************************
706  * @fn SensorGapStateCB
707  *
708  * @brief Notification from the profile of a state change.
709  *
710  * @param newState - new state
711  *
712  * @return none
713  */
714 static void SensorGapStateCB( gaprole_States_t newState )
715 {
716  // If connected
717  if (newState == GAPROLE_CONNECTED)
718  {
719  // Get connection handle
720  GAPRole_GetParameter(GAPROLE_CONNHANDLE, &gapConnHandle);
721 
722  // Set timer to update connection parameters
723  // 5 seconds should allow enough time for Service Discovery by the collector to finish
724  osal_start_timerEx( sensor_TaskID, CSC_CONN_PARAM_UPDATE_EVT, SVC_DISC_DELAY);
725  }
726  // If disconnected
727  else if (gapProfileState == GAPROLE_CONNECTED &&
728  newState != GAPROLE_CONNECTED)
729  {
730  uint8 advState = TRUE;
731  uint8 bondCount = 0;
732 
733  // Stop periodic measurement
734  osal_stop_timerEx( sensor_TaskID, CSC_PERIODIC_EVT );
735 
736  // Reset client characteristic configuration descriptors
737  Cycling_HandleConnStatusCB( gapConnHandle, LINKDB_STATUS_UPDATE_REMOVED );
738 
739  // If not already using white list, begin to do so.
740  GAPBondMgr_GetParameter( GAPBOND_BOND_COUNT, &bondCount );
741 
742  if( USING_WHITE_LIST && sensorUsingWhiteList == FALSE && bondCount > 0 )
743  {
744  uint8 value = GAP_FILTER_POLICY_WHITE;
745 
746  GAPRole_SetParameter(GAPROLE_ADV_FILTER_POLICY, sizeof( uint8 ), &value);
747 
748  sensorUsingWhiteList = TRUE;
749  }
750 
751  if ( newState == GAPROLE_WAITING_AFTER_TIMEOUT )
752  {
753  // link loss timeout-- use fast advertising
754  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MIN, DEFAULT_FAST_ADV_INTERVAL );
755  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MAX, DEFAULT_FAST_ADV_INTERVAL );
756  GAP_SetParamValue( TGAP_GEN_DISC_ADV_MIN, DEFAULT_WHITE_LIST_ADV_DURATION );
757  }
758  else
759  {
760  // Else use slow advertising
761  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MIN, DEFAULT_SLOW_ADV_INTERVAL );
762  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MAX, DEFAULT_SLOW_ADV_INTERVAL );
763  GAP_SetParamValue( TGAP_GEN_DISC_ADV_MIN, DEFAULT_WHITE_LIST_ADV_DURATION );
764  }
765 
766  // Enable advertising
767  GAPRole_SetParameter( GAPROLE_ADVERT_ENABLED, sizeof( uint8 ), &advState );
768  }
769  // if advertising stopped
770  else if ( gapProfileState == GAPROLE_ADVERTISING &&
771  newState == GAPROLE_WAITING )
772  {
773  uint8 whiteListUsed = FALSE;
774 
775  // if white list is in use, disable to allow general access
776  if( sensorUsingWhiteList == TRUE )
777  {
778  uint8 value = GAP_FILTER_POLICY_ALL;
779 
780  GAPRole_SetParameter(GAPROLE_ADV_FILTER_POLICY, sizeof( uint8), &value);
781 
782  whiteListUsed = TRUE;
783 
784  sensorUsingWhiteList = FALSE;
785  }
786 
787  // if advertising stopped by user
788  if ( sensorAdvCancelled )
789  {
790  sensorAdvCancelled = FALSE;
791  }
792  // if fast advertising interrupted to cancel white list
793  else if ( ( (!USING_WHITE_LIST) || whiteListUsed) &&
794  (GAP_GetParamValue( TGAP_GEN_DISC_ADV_INT_MIN ) == DEFAULT_FAST_ADV_INTERVAL ) )
795  {
796  uint8 advState = TRUE;
797 
798  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MIN, DEFAULT_FAST_ADV_INTERVAL );
799  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MAX, DEFAULT_FAST_ADV_INTERVAL );
800  GAP_SetParamValue( TGAP_GEN_DISC_ADV_MIN, DEFAULT_FAST_ADV_DURATION );
801  GAPRole_SetParameter( GAPROLE_ADVERT_ENABLED, sizeof( uint8 ), &advState );
802  }
803  // if fast advertising switch to slow or if was already slow but using white list.
804  else if ( ((!USING_WHITE_LIST) || whiteListUsed) ||
805  (GAP_GetParamValue( TGAP_GEN_DISC_ADV_INT_MIN ) == DEFAULT_FAST_ADV_INTERVAL) )
806  {
807  uint8 advState = TRUE;
808 
809  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MIN, DEFAULT_SLOW_ADV_INTERVAL );
810  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MAX, DEFAULT_SLOW_ADV_INTERVAL );
811  GAP_SetParamValue( TGAP_GEN_DISC_ADV_MIN, DEFAULT_SLOW_ADV_DURATION );
812  GAPRole_SetParameter( GAPROLE_ADVERT_ENABLED, sizeof( uint8 ), &advState );
813  }
814  }
815  // if started
816  else if (newState == GAPROLE_STARTED)
817  {
818  // Set the system ID from the bd addr
819  uint8 systemId[DEVINFO_SYSTEM_ID_LEN];
820  GAPRole_GetParameter(GAPROLE_BD_ADDR, systemId);
821 
822  // shift three bytes up
823  systemId[7] = systemId[5];
824  systemId[6] = systemId[4];
825  systemId[5] = systemId[3];
826 
827  // set middle bytes to zero
828  systemId[4] = 0;
829  systemId[3] = 0;
830 
831  DevInfo_SetParameter(DEVINFO_SYSTEM_ID, DEVINFO_SYSTEM_ID_LEN, systemId);
832  }
833 
834  gapProfileState = newState;
835 }

837 /*********************************************************************
838  * @fn SensorCB
839  *
840  * @brief Callback function for CSC service.
841  *
842  * @param event - service event
843  * @param pNewCummVal - pointer to new wheel revolution data
844  * if specified by event. NULL otherwise.
845  * @return none
846  */
847 static void SensorCB( uint8 event, uint32 *pNewCummVal )

866

876

879  {
880  osal_start_timerEx( sensor_TaskID, CSC_NEGLECT_TIMEOUT_EVT, NEGLECT_TIMEOUT_DELAY );
881  }
882  break;
883 
884  case CSC_MEAS_NOTI_ENABLED:
885  // Cancel neglect timer
886  if ( USING_NEGLECT_TIMEOUT )
887  {
888  osal_stop_timerEx( sensor_TaskID, CSC_NEGLECT_TIMEOUT_EVT );
889  }
890

899  // Stop periodic measurement
900  osal_stop_timerEx( sensor_TaskID, CSC_PERIODIC_EVT );
901 
902  // Start neglect timer

904  {
905  osal_start_timerEx( sensor_TaskID, CSC_NEGLECT_TIMEOUT_EVT, NEGLECT_TIMEOUT_DELAY );

914  osal_stop_timerEx( sensor_TaskID, CSC_NEGLECT_TIMEOUT_EVT );

916  // Start neglect timer

918  }
919  break;
920 
921  default:
922  break;
923  }
924 }
