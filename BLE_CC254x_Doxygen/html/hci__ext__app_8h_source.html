54 /*** HCI Extension Commands ***/
55 
56 // The 10-bit OCF (Opcode Command Field) of the HCI Opcode is further
57 // devided into two subsections:
58 // - Subgroup (3 bits):
59 // - 0 (LL)
60 // - 1 (L2CAP)
61 // - 2 (ATT)
62 // - 3 (GATT)
63 // - 4 (GAP)
64 // - 5 (UTIL)
65 // - 6 (Reserved)
66 // - 7 (User Profile)
67 // - Command (7 bits) or Profile (7 bits) if Subgroup value is set to
68 // User Profile (i.e., all ones) in which case, another octet is
69 // required to represent user profile commands.
70 //
71 #define HCI_EXT_LL_SUBGRP 0x00
72 #define HCI_EXT_L2CAP_SUBGRP 0x01
73 #define HCI_EXT_ATT_SUBGRP 0x02
74 #define HCI_EXT_GATT_SUBGRP 0x03
75 #define HCI_EXT_GAP_SUBGRP 0x04
76 #define HCI_EXT_UTIL_SUBGRP 0x05
77 #define HCI_EXT_PROFILE_SUBGRP 0x07

134 /*** HCI Extension Events ***/
135 
136 // HCI extension events must start from 0x0400. The upper 6 bits of all
137 // zeros is reserved for the HCI extension embedded commands. The rest of
138 // the event bits should follow the HCI extension command format (i.e.,
139 // 3-bit Subgroup + 7-bit Event).
140 //
141 #define HCI_EXT_BASE_EVENT ( 0x0001 << 10 ) // 0x0400
142 
143 #define HCI_EXT_LL_EVENT ( HCI_EXT_BASE_EVENT | (HCI_EXT_LL_SUBGRP << 7) ) // 0x0400
144 #define HCI_EXT_L2CAP_EVENT ( HCI_EXT_BASE_EVENT | (HCI_EXT_L2CAP_SUBGRP << 7) ) // 0x0480
145 #define HCI_EXT_ATT_EVENT ( HCI_EXT_BASE_EVENT | (HCI_EXT_ATT_SUBGRP << 7) ) // 0x0500
146 #define HCI_EXT_GATT_EVENT ( HCI_EXT_BASE_EVENT | (HCI_EXT_GATT_SUBGRP << 7) ) // 0x0580
147 #define HCI_EXT_GAP_EVENT ( HCI_EXT_BASE_EVENT | (HCI_EXT_GAP_SUBGRP << 7) ) // 0x0600
