154 /**************************************************************************************************
155  * @fn HalBarReadMeasurement
156  *
157  * @brief Get temperature and pressure data (alternate reads)
158  *
159  * @param pBuf - buffer for temperature and pressure (4 bytes)
160  *
161  * @return TRUE if valid data
162  */
163 bool HalBarReadMeasurement(uint8 *pBuf)
164 {
165  bool success;
166  uint8 dOffset = 0;
167 
168  if (!fCmdOk)
169  {
170  return FALSE;
171  }
172 
173  HalBarSelect();
174 
175  if (barCmd==C953_PRESS_READ_COMMAND)
176  {
177  dOffset = 2;
178  }
179 
180  success = HalSensorReadReg( C953_REG_ADDR_PRESS_LSB, &barData[dOffset], C953_DATA_LEN );
181  if (success)
182  {
183  pBuf[0] = barData[0];
184  pBuf[1] = barData[1];
185  pBuf[2] = barData[2];
186  pBuf[3] = barData[3];
187 
188  // Alternate
189  if (barCmd==C953_PRESS_READ_COMMAND)
190  {
191  barCmd = C953_TEMP_READ_COMMAND;
192  } else
193  {
194  barCmd = C953_PRESS_READ_COMMAND;
195  }
196  }
197 
198  return success;
199 }

235

240  ST_ASSERT(val==C953_TEST_COMMAND);
241 
242  // Restore values
243  val = C953_OFF_COMMAND;
244  ST_ASSERT(HalSensorWriteReg(C953_REG_ADDR_COMMAND, &val, sizeof(val)));
245 
246  return TRUE;
247 }

267 /* Conversion algorithm for barometer temperature
268  * ==============================================
269  * Formula from application note, rev_X:
270  * Ta = ((c1 * Tr) / 2^24) + (c2 / 2^10)
271  *
272  * c1 - c8: calibration coefficients the can be read from the sensor
273  * c1 - c4: unsigned 16-bit integers
274  * c5 - c8: signed 16-bit integers
275 
276 double calcBarTmp(uint16 rawT)
277 {
278  uint16 c1, c2;
279 
280  c1 = m_barCalib.c1;
281  c2 = m_barCalib.c2;
282  m_raw_temp = rawT;
283 
284  int64 temp, val;
285  val = ((int64)(c1 * m_raw_temp) * 100);
286  temp = (val >> 24);
287  val = ((int64)c2 * 100);
288  temp += (val >> 10);
289 
290  return ((double)temp) / 100;
291 }
292 
293  * Conversion algorithm for barometer pressure (hPa)
294  * ==============================================
295  * Formula from application note, rev_X:
296  * Sensitivity = (c3 + ((c4 * Tr) / 2^17) + ((c5 * Tr^2) / 2^34))
297  * Offset = (c6 * 2^14) + ((c7 * Tr) / 2^3) + ((c8 * Tr^2) / 2^19)
298  * Pa = (Sensitivity * Pr + Offset) / 2^14
299 
300 double TcalcBarPress(uint16 rawT)
301 {
302  int64 s, o, pres, val;
303  uint16 c3, c4;
304  int16 c5, c6, c7, c8;
305  uint16 Pr;
306  int16 Tr;
307 
308  Pr = rawT;
309  Tr = m_raw_temp;
310  c3 = m_barCalib.c3;
311  c4 = m_barCalib.c4;
312  c5 = m_barCalib.c5;
313  c6 = m_barCalib.c6;
314  c7 = m_barCalib.c7;
315  c8 = m_barCalib.c8;
316 
317  // Sensitivity
318  s = (int64)c3;
319  val = (int64)c4 * Tr;
320  s += (val >> 17);
321  val = (int64)c5 * Tr * Tr;
322  s += (val >> 34);
323 
324  // Offset
325  o = (int64)c6 << 14;
326  val = (int64)c7 * Tr;
327  o += (val >> 3);
328  val = (int64)c8 * Tr * Tr;
329  o += (val >> 19);
330 
331  // Pressure (Pa)
332  pres = ((int64)(s * Pr) + o) >> 14;
333 
334  return (double)pres/100;
335 }
