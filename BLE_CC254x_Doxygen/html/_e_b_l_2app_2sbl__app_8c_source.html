78 /**************************************************************************************************
79  * @fn appForceBoot
80  *
81  * @brief Force the boot loader to run.
82  *

85  * None.

91  * @return None.

93 void appForceBoot(void)
94 {
95  uint16 crc[2];
96  SBL_NVM_GET(SBL_ADDR_CRC, crc, sizeof(crc));
97 
98  // A simple check if the image is built for a boot loader is a valid CRC & shadow where expected.
99  if ((crc[0] == 0xFFFF) || (crc[0] == 0x0000) ||
100  (crc[1] == 0xFFFF) || (crc[1] == 0x0000) ||
101  (crc[1] != crc[0]))
102  {
103  return;
104  }
105 
106  HAL_DISABLE_INTERRUPTS();
107 
108  crc[0] ^= 0xFFFF; // Only write to zero bits that are not already zero.
109  crc[1] = 0xFFFF; // No need to write any bits to zero.
110  SBL_NVM_SET(SBL_ADDR_CRC, crc, sizeof(crc));
111 
112  HAL_SYSTEM_RESET();
113 }
