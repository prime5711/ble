181 /*********************************************************************
182  * @fn SimpleBLEBroadcaster_Init
183  *
184  * @brief Initialization function for the Simple BLE Broadcaster App
185  * Task. This is called during initialization and should contain
186  * any application specific initialization (ie. hardware
187  * initialization/setup, table initialization, power up
188  * notificaiton ... ).
189  *
190  * @param task_id - the ID assigned by OSAL. This ID should be
191  * used to send messages and set timers.
192  *
193  * @return none
194  */
195 void SimpleBLEBroadcaster_Init( uint8 task_id )
196 {
197  simpleBLEBroadcaster_TaskID = task_id;
198 
199  // Setup the GAP Broadcaster Role Profile
200  {
201  #if defined( CC2540_MINIDK )
202  // For the CC2540DK-MINI keyfob, device doesn't start advertising until button is pressed
203  uint8 initial_advertising_enable = FALSE;
204  #else
205  // For other hardware platforms, device starts advertising upon initialization
206  uint8 initial_advertising_enable = TRUE;
207  #endif
208 
209  // By setting this to zero, the device will go into the waiting state after
210  // being discoverable for 30.72 second, and will not being advertising again
211  // until the enabler is set back to TRUE
212  uint16 gapRole_AdvertOffTime = 0;
213 
214  //uint8 advType = GAP_ADTYPE_ADV_NONCONN_IND; // use non-connectable advertisements
215  uint8 advType = GAP_ADTYPE_ADV_SCAN_IND; // use scannable unidirected advertisements
216 
217  // Set the GAP Role Parameters
218  GAPRole_SetParameter( GAPROLE_ADVERT_ENABLED, sizeof( uint8 ), &initial_advertising_enable );
219  GAPRole_SetParameter( GAPROLE_ADVERT_OFF_TIME, sizeof( uint16 ), &gapRole_AdvertOffTime );
220 
221  GAPRole_SetParameter( GAPROLE_SCAN_RSP_DATA, sizeof ( scanRspData ), scanRspData );
222  GAPRole_SetParameter( GAPROLE_ADVERT_DATA, sizeof( advertData ), advertData );
223 
224  GAPRole_SetParameter( GAPROLE_ADV_EVENT_TYPE, sizeof( uint8 ), &advType );
225  }
226 
227  // Set advertising interval
228  {
229  uint16 advInt = DEFAULT_ADVERTISING_INTERVAL;
230 
231  GAP_SetParamValue( TGAP_LIM_DISC_ADV_INT_MIN, advInt );
232  GAP_SetParamValue( TGAP_LIM_DISC_ADV_INT_MAX, advInt );
233  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MIN, advInt );
234  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MAX, advInt );
235  }
236 
237 #if defined( CC2540_MINIDK )
238 
239  // Register for all key events - This app will handle all key events
240  RegisterForKeys( simpleBLEBroadcaster_TaskID );
241 
242  // makes sure LEDs are off
243  HalLedSet( (HAL_LED_1 | HAL_LED_2), HAL_LED_MODE_OFF );
244 
245  // For keyfob board set GPIO pins into a power-optimized state
246  // Note that there is still some leakage current from the buzzer,
247  // accelerometer, LEDs, and buttons on the PCB.
248 
249  P0SEL = 0; // Configure Port 0 as GPIO
250  P1SEL = 0; // Configure Port 1 as GPIO
251  P2SEL = 0; // Configure Port 2 as GPIO
252 
253  P0DIR = 0xFC; // Port 0 pins P0.0 and P0.1 as input (buttons),
254  // all others (P0.2-P0.7) as output
255  P1DIR = 0xFF; // All port 1 pins (P1.0-P1.7) as output
256  P2DIR = 0x1F; // All port 1 pins (P2.0-P2.4) as output
257 
258  P0 = 0x03; // All pins on port 0 to low except for P0.0 and P0.1 (buttons)
259  P1 = 0; // All pins on port 1 to low
260  P2 = 0; // All pins on port 2 to low
261 
262 #endif // #if defined( CC2540_MINIDK )
263 
264 #if (defined HAL_LCD) && (HAL_LCD == TRUE)
265 
266  HalLcdWriteString( "BLE Broadcaster", HAL_LCD_LINE_1 );
267 
268 #endif // (defined HAL_LCD) && (HAL_LCD == TRUE)
269 
270  // Setup a delayed profile startup
271  osal_set_event( simpleBLEBroadcaster_TaskID, SBP_START_DEVICE_EVT );
272 }

274 /*********************************************************************
275  * @fn SimpleBLEBroadcaster_ProcessEvent
276  *
277  * @brief Simple BLE Broadcaster Application Task event processor. This
278  * function is called to process all events for the task. Events
279  * include timers, messages and any other user defined events.
280  *
281  * @param task_id - The OSAL assigned task ID.
282  * @param events - events to process. This is a bit map and can
283  * contain more than one event.
284  *
285  * @return events not processed
286  */
287 uint16 SimpleBLEBroadcaster_ProcessEvent( uint8 task_id, uint16 events )
288 {
289 
290  VOID task_id; // OSAL required parameter that isn't used in this function
291 
292  if ( events & SYS_EVENT_MSG )
293  {
294  uint8 *pMsg;
295 
296  if ( (pMsg = osal_msg_receive( simpleBLEBroadcaster_TaskID )) != NULL )
297  {
298  simpleBLEBroadcaster_ProcessOSALMsg( (osal_event_hdr_t *)pMsg );
299 
300  // Release the OSAL message
301  VOID osal_msg_deallocate( pMsg );
302  }
303 
304  // return unprocessed events
305  return (events ^ SYS_EVENT_MSG);
306  }
307 
308  if ( events & SBP_START_DEVICE_EVT )
309  {
310  // Start the Device
311  VOID GAPRole_StartDevice( &simpleBLEBroadcaster_BroadcasterCBs );
312 
313  return ( events ^ SBP_START_DEVICE_EVT );
314  }
315 
316  // Discard unknown events
317  return 0;
318 }

320 /*********************************************************************
321  * @fn simpleBLEBroadcaster_ProcessOSALMsg
322  *
323  * @brief Process an incoming task message.

325  * @param pMsg - message to process

329 static void simpleBLEBroadcaster_ProcessOSALMsg( osal_event_hdr_t *pMsg )

338 
339  default:
340  // do nothing
341  break;
342  }
343 }

345 #if defined( CC2540_MINIDK )
346 /*********************************************************************
347  * @fn simpleBLEBroadcaster_HandleKeys
348  *
349  * @brief Handles all key events for this device.
350  *
351  * @param shift - true if in shift/alt.
352  * @param keys - bit field for key events. Valid entries:
353  * HAL_KEY_SW_2
354  * HAL_KEY_SW_1
355  *
356  * @return none
357  */
358 static void simpleBLEBroadcaster_HandleKeys( uint8 shift, uint8 keys )
359 {
360  VOID shift; // Intentionally unreferenced parameter
361 
362  if ( keys & HAL_KEY_SW_2 )
363  {
364  // ressing the right key should toggle advertising on and off
365  uint8 current_adv_enabled_status;
366  uint8 new_adv_enabled_status;
367 
368  //Find the current GAP advertisement status
369  GAPRole_GetParameter( GAPROLE_ADVERT_ENABLED, &current_adv_enabled_status );
370 
371  if( current_adv_enabled_status == FALSE )
372  {
373  new_adv_enabled_status = TRUE;
374  }
375  else
376  {
377  new_adv_enabled_status = FALSE;
378  }
379 
380  //change the GAP advertisement status to opposite of current status
381  GAPRole_SetParameter( GAPROLE_ADVERT_ENABLED, sizeof( uint8 ), &new_adv_enabled_status );
382  }
383 }
384 #endif // CC2540_MINIDK

386 /*********************************************************************
387  * @fn peripheralStateNotificationCB
388  *
389  * @brief Notification from the profile of a state change.
390  *
391  * @param newState - new state
392  *
393  * @return none
394  */
395 static void peripheralStateNotificationCB( gaprole_States_t newState )
396 {
397  switch ( newState )
398  {
399  case GAPROLE_STARTED:
400  {
401  uint8 ownAddress[B_ADDR_LEN];
402 
403  GAPRole_GetParameter(GAPROLE_BD_ADDR, ownAddress);
404 
405  #if (defined HAL_LCD) && (HAL_LCD == TRUE)
406  // Display device address
407  HalLcdWriteString( bdAddr2Str( ownAddress ), HAL_LCD_LINE_2 );
408  HalLcdWriteString( "Initialized", HAL_LCD_LINE_3 );
409  #endif // (defined HAL_LCD) && (HAL_LCD == TRUE)
410  }
411  break;
412 
413  case GAPROLE_ADVERTISING:
414  {
415  #if (defined HAL_LCD) && (HAL_LCD == TRUE)
416  HalLcdWriteString( "Advertising", HAL_LCD_LINE_3 );
417  #endif // (defined HAL_LCD) && (HAL_LCD == TRUE)
418  }
419  break;
420 
421  case GAPROLE_WAITING:
422  {
423  #if (defined HAL_LCD) && (HAL_LCD == TRUE)
424  HalLcdWriteString( "Waiting", HAL_LCD_LINE_3 );
425  #endif // (defined HAL_LCD) && (HAL_LCD == TRUE)
426  }
427  break;
428 
429  case GAPROLE_ERROR:
430  {
431  #if (defined HAL_LCD) && (HAL_LCD == TRUE)
432  HalLcdWriteString( "Error", HAL_LCD_LINE_3 );
433  #endif // (defined HAL_LCD) && (HAL_LCD == TRUE)
434  }
435  break;
436 
437  default:
438  {
439  #if (defined HAL_LCD) && (HAL_LCD == TRUE)
440  HalLcdWriteString( "", HAL_LCD_LINE_3 );
441  #endif // (defined HAL_LCD) && (HAL_LCD == TRUE)
442  }
443  break;
444  }
445 }

448 /*********************************************************************
449  * @fn bdAddr2Str
450  *
451  * @brief Convert Bluetooth address to string. Only needed when
452  * LCD display is used.
453  *
454  * @return none
455  */
456 char *bdAddr2Str( uint8 *pAddr )
457 {
458  uint8 i;
459  char hex[] = "0123456789ABCDEF";
460  static char str[B_ADDR_STR_LEN];
461  char *pStr = str;
462 
463  *pStr++ = '0';
464  *pStr++ = 'x';
465 
466  // Start from end of addr
467  pAddr += B_ADDR_LEN;
468 
469  for ( i = B_ADDR_LEN; i > 0; i-- )
470  {
471  *pStr++ = hex[*--pAddr >> 4];
472  *pStr++ = hex[*pAddr & 0x0F];
473  }
474 
475  *pStr = 0;
476 
477  return str;
478 }
479 #endif // (defined HAL_LCD) && (HAL_LCD == TRUE)
