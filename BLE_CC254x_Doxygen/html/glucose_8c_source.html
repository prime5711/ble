873 /*********************************************************************
874  * @fn glucoseProcessCtlPntMsg
875  *
876  * @brief Process Control Point messages
877  *
878  * @return none
879  */
880 static void glucoseProcessCtlPntMsg( glucoseCtlPntMsg_t* pMsg)
881 {
882  uint8 opcode = pMsg->data[0];
883  uint8 oper = pMsg->data[1];
884  UTCTimeStruct time1, time2;
885  bool opcodeValid = true;
886  uint16 seqNum1, seqNum2;
887 
888  switch(opcode)
889  {
890  case CTL_PNT_OP_REQ:
891  case CTL_PNT_OP_CLR:
892  case CTL_PNT_OP_GET_NUM:
893  if(oper == CTL_PNT_OPER_NULL)
894  {
895  glucoseCtlPntResponse(CTL_PNT_RSP_OPER_INVALID, opcode);
896  opcodeValid = false;
897  }
898  break;
899 
900  case CTL_PNT_OP_ABORT:
901  if(oper != CTL_PNT_OPER_NULL)
902  {
903  glucoseCtlPntResponse(CTL_PNT_RSP_OPER_INVALID, opcode);
904  opcodeValid = false;
905  }
906  break;
907 
908  default:
909  glucoseCtlPntResponse(CTL_PNT_RSP_OPCODE_NOT_SUPPORTED, opcode);
910  opcodeValid = false;
911  break;
912  }
913 
914  if(opcodeValid)
915  {
916  switch(oper)
917  {
918  case CTL_PNT_OPER_NULL:
919  case CTL_PNT_OPER_ALL:
920  case CTL_PNT_OPER_FIRST:
921  case CTL_PNT_OPER_LAST:
922  if(pMsg->len == 2)
923  {
924  glucoseCtlPntHandleOpcode(opcode, oper, 0, NULL, NULL);
925  }
926  else
927  {
928  // No operand should exist, but msg is longer than 2 bytes
929  glucoseCtlPntResponse(CTL_PNT_RSP_OPERAND_INVALID, opcode);
930  }
931  break;
932 
933 
934  case CTL_PNT_OPER_RANGE:
935  // check filter type
936  if (pMsg->data[2] == CTL_PNT_FILTER_SEQNUM && pMsg->len == 7)
937  {
938  seqNum1 = BUILD_UINT16(pMsg->data[3], pMsg->data[4]);
939  seqNum2 = BUILD_UINT16(pMsg->data[5], pMsg->data[6]);
940 
941  if ( seqNum1 <= seqNum2 )
942  {
943  glucoseCtlPntHandleOpcode(opcode, oper, pMsg->data[2], &seqNum1, &seqNum2);
944  }
945  else
946  {
947  glucoseCtlPntResponse(CTL_PNT_RSP_OPERAND_INVALID, opcode);
948  }
949  }
950  else if (pMsg->data[2] == CTL_PNT_FILTER_TIME && pMsg->len == 17)
951  {
952  time1.year = BUILD_UINT16(pMsg->data[3], pMsg->data[4]);
953  time1.month = pMsg->data[5];
954  time1.day = pMsg->data[6];
955  time1.hour = pMsg->data[7];
956  time1.minutes = pMsg->data[8];
957  time1.seconds = pMsg->data[9];
958 
959  time2.year = BUILD_UINT16(pMsg->data[10], pMsg->data[11]);
960  time2.month = pMsg->data[12];
961  time2.day = pMsg->data[13];
962  time2.hour = pMsg->data[14];
963  time2.minutes = pMsg->data[15];
964  time2.seconds = pMsg->data[16];
965 
966  glucoseCtlPntHandleOpcode(opcode, oper, pMsg->data[2], &time1, &time2);
967  }
968  else
969  {
970  glucoseCtlPntResponse(CTL_PNT_RSP_OPERAND_INVALID, opcode);
971  }
972  break;
973 
974  case CTL_PNT_OPER_LESS_EQUAL:
975  case CTL_PNT_OPER_GREATER_EQUAL:
976  // check filter type
977  if (pMsg->data[2] == CTL_PNT_FILTER_SEQNUM && pMsg->len == 5)
978  {
979  seqNum1 = BUILD_UINT16(pMsg->data[3], pMsg->data[4]);
980 
981  glucoseCtlPntHandleOpcode(opcode, oper, pMsg->data[2], &seqNum1, NULL);
982  }
983  else if (pMsg->data[2] == CTL_PNT_FILTER_TIME && pMsg->len == 10)
984  {
985  time1.year = BUILD_UINT16(pMsg->data[3], pMsg->data[4]);
986  time1.month = pMsg->data[5];
987  time1.day = pMsg->data[6];
988  time1.hour = pMsg->data[7];
989  time1.minutes = pMsg->data[8];
990  time1.seconds = pMsg->data[9];
991 
992  glucoseCtlPntHandleOpcode(opcode, oper, pMsg->data[2], &time1, NULL);
993  }
994  else
995  {
996  glucoseCtlPntResponse(CTL_PNT_RSP_FILTER_NOT_SUPPORTED, opcode);
997  }
998  break;
999 
1000  default:
1001  glucoseCtlPntResponse(CTL_PNT_RSP_OPER_NOT_SUPPORTED, opcode);
1002  break;
1003  }
1004  }
1005 }
