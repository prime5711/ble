67 /**************************************************************************************************
68  * @fn HalSensorReadReg
69  *
70  * @brief This function implements the I2C protocol to read from a sensor. The sensor must
71  * be selected before this routine is called.
72  *
73  * @param addr - which register to read
74  * @param pBuf - pointer to buffer to place data
75  * @param nBytes - numbver of bytes to read
76  *
77  * @return TRUE if the required number of bytes are reveived
78  **************************************************************************************************/
79 bool HalSensorReadReg(uint8 addr, uint8 *pBuf, uint8 nBytes)
80 {
81  uint8 i = 0;
82 
83  /* Send address we're reading from */
84  if (HalI2CWrite(1,&addr) == 1)
85  {
86  /* Now read data */
87  i = HalI2CRead(nBytes,pBuf);
88  }
89 
90  return i == nBytes;
91 }

93 /**************************************************************************************************
94 * @fn HalSensorWriteReg
95 * @brief This function implements the I2C protocol to write to a sensor. he sensor must
96 * be selected before this routine is called.
97 *
98 * @param addr - which register to write
99 * @param pBuf - pointer to buffer containing data to be written
100 * @param nBytes - number of bytes to write
101 *
102 * @return TRUE if successful write
103 */
104 bool HalSensorWriteReg(uint8 addr, uint8 *pBuf, uint8 nBytes)
105 {
106  uint8 i;
107  uint8 *p = buffer;
108 
109  /* Copy address and data to local buffer for burst write */
110  *p++ = addr;
111  for (i = 0; i < nBytes; i++)
112  {
113  *p++ = *pBuf++;
114  }
115  nBytes++;
116 
117  /* Send address and data */
118  i = HalI2CWrite(nBytes, buffer);
119  if ( i!= nBytes)
120  HAL_TOGGLE_LED2();
121 
122  return (i == nBytes);
123 }

177 /*********************************************************************
178  * @fn HalDcDcControl
179  *
180  * @brief Control the DCDC converter based on what sensors are enabled.
181  *
182  * @descr This applies to IR Temp, Humidity and Gyroscope. If any of those
183  * are used, the DCDC-converter must be bypassed.
184  *
185  * @param sensorID - bit indicating one sensor
186  *
187  * @param powerOn - indicate whether the sensor is turned on or off
188  *
189  * @return none
190  */
191 void HalDcDcControl(uint8 sensorID, bool powerOn)
192 {
193  if (powerOn)
194  {
195  halSensorEnableMap |= sensorID;
196  }
197  else
198  {
199  halSensorEnableMap &= ~sensorID;
200  }
201 
202  if (halSensorEnableMap & HIGH_SUPPLY_SENSOR_MAP)
203  {
204  /* Bypass DCDC */
205  DCDC_SBIT = 0;
206  }
207  else
208  {
209  /* Enable DCDC */
210  DCDC_SBIT = 1;
211  }
212 }
