128 /**************************************************************************************************
129  * @fn sblRun
130  *
131  * @brief Serial Boot run code.
132  *
133  * input parameters
134  *
135  * None.
136  *
137  * output parameters
138  *
139  * None.
140  *
141  * @return None.
142  */
143 static void sblRun(void)
144 {
145  uint8 resetF = 0;
146  sblHalInit();
147 
148  while (1)
149  {
150  if (txLen == 0)
151  {
152  if ((resetF != 0) && ((UxCSR & CSR_ACTIVE) == 0)) // If the last Tx byte has flushed out.
153  {
154  break;
155  }
156  }
157  else if (UTXxIF != 0)
158  {
159  if (--txLen != 0)
160  {
161  UTXxIF = 0;
162  }
163 
164  UxDBUF = sbBuf[txIdx++];
165  }
166 
167  if (URXxIF)
168  {
169  uint8 ch = UxDBUF;
170  URXxIF = 0;
171  resetF |= sblParse(ch);
172  }
173  }
174 }

176 /**************************************************************************************************
177  * @fn sblParse
178  *
179  * @brief Serial Boot parser according to the RPC UART transport.
180  *
181  * input parameters
182  *
183  * @param ch - The Rx character to parse.
184  *
185  * output parameters
186  *
187  * None.
188  *
189  * @return TRUE if the downloaded code has been enabled; FALSE otherwise.
190  */
191 static uint8 sblParse(uint8 ch)
192 {
193  sbBuf[sbSte + sbIdx] = ch;
194 
195  switch (sbSte)
196  {
197  case SBL_SOF_STATE:
198  if (RPC_UART_SOF == ch)
199  {
200  sbSte = SBL_LEN_STATE;
201  }
202  break;
203 
204  case SBL_LEN_STATE:
205  sbFcs = sbIdx = 0;
206  sbSte = SBL_CMD1_STATE;
207  sbSte = ((sbLen = ch) >= SBL_MAX_SIZE) ? SBL_SOF_STATE : SBL_CMD1_STATE;
208  break;
209 
210  case SBL_CMD1_STATE:
211  sbSte = SBL_CMD2_STATE;
212  break;
213 
214  case SBL_CMD2_STATE:
215  sbSte = (sbLen == 0) ? SBL_FCS_STATE : SBL_DATA_STATE;
216  break;
217 
218  case SBL_DATA_STATE:
219  if (++sbIdx == sbLen)
220  {
221  sbSte = SBL_FCS_STATE;
222  }
223  break;
224 
225  case SBL_FCS_STATE:
226  sbSte = SBL_SOF_STATE;
227 
228  if ((sbFcs == ch) && ((sbBuf[SBL_CMD1_STATE] & RPC_SUBSYSTEM_MASK) == RPC_SYS_BOOT))
229  {
230  sblExec(sbBuf + SBL_LEN_STATE);
231  return sblResp(); // Send the SB response setup in the sbBuf passed to sblExec().
232  }
233  break;
234 
235  default:
236  break;
237  }
238 
239  sbFcs ^= ch;
240  return FALSE;
241 }
242 
243 /**************************************************************************************************
244  * @fn sblResp
245  *
246  * @brief Make the SB response.
247  *
248  * input parameters
249  *
250  * None.
251  *
252  * output parameters
253  *
254  * None.
255  *
256  * @return TRUE if the downloaded code has been enabled; FALSE otherwise.
257  */
258 static uint8 sblResp(void)
259 {
260  uint8 idx, fcs = 0, len = sbBuf[SBL_LEN_STATE] + SBL_FCS_STATE - 1;
261 
262  for (idx = SBL_LEN_STATE; idx < len; idx++)
263  {
264  fcs ^= sbBuf[idx];
265  }
266  sbBuf[idx++] = fcs;
267 
268  txLen = idx;
269  txIdx = 0;
270 
271  if ((sbBuf[SBL_CMD2_STATE] == (SBL_ENABLE_CMD | SBL_RSP_MASK)) &&
272  (sbBuf[SBL_DATA_STATE] == SBL_SUCCESS))
273  {
274  txLen++; // Send an extra garbage byte to flush the last good one before resetting.
275  return TRUE;
276  }
277  else
278  {
279  return FALSE;
280  }
281 }

283 /**************************************************************************************************
284  * @fn sblExec
285  *
286  * @brief Act on the SB command and received buffer.
287  *
288  * @param pBuf - A pointer to the RPC command buffer received.
289  *
290  * None.
291  *
292  * output parameters
293  *
294  * None.
295  *
296  * @return None.
297  **************************************************************************************************
298  */
299 static void sblExec(uint8 *pBuf)
300 {
301  uint16 t16 = BUILD_UINT16(pBuf[SBL_REQ_ADDR_LSB],pBuf[SBL_REQ_ADDR_MSB]) + OAD_DONGLE_SBL_IMG_BEG;
302  uint8 len = 1;
303  uint8 rsp = SBL_SUCCESS;
304 
305  switch (pBuf[RPC_POS_CMD1])
306  {
307  case SBL_WRITE_CMD:
308  if ((t16 >= OAD_DONGLE_SBL_IMG_END) || (t16 < OAD_DONGLE_SBL_IMG_BEG))
309  {
310  rsp = SBL_FAILURE;
311  break;
312  }
313  if ((t16 % SBL_PAGE_SIZE) == 0)
314  {
315  HalFlashErase(t16 / SBL_PAGE_SIZE);
316  }
317  HalFlashWrite(t16, (pBuf + SBL_REQ_DAT0), (SBL_RW_BUF_LEN / HAL_FLASH_WORD_SIZE));
318  break;
319 
320  case SBL_READ_CMD:
321  len = SBL_RW_BUF_LEN + SBL_READ_HDR_LEN;
322  pBuf[SBL_RSP_ADDR_MSB] = pBuf[SBL_REQ_ADDR_MSB];
323  pBuf[SBL_RSP_ADDR_LSB] = pBuf[SBL_REQ_ADDR_LSB];
324 
325  HalFlashRead(t16 / SBL_PAGE_SIZE,
326  (t16 % SBL_PAGE_SIZE) << 2, (pBuf + SBL_RSP_DAT0), SBL_RW_BUF_LEN);
327  break;
328 
329  case SBL_ENABLE_CMD:
330  // Bootload master must verify download by read back - no room for CRC checking code in dongle.
331  break;
332 
333  case SBL_HANDSHAKE_CMD:
334  break;
335 
336  default:
337  rsp = SBL_FAILURE;
338  break;
339  }
340 
341  pBuf[RPC_POS_LEN] = len;
342  pBuf[RPC_POS_CMD1] |= SBL_RSP_MASK;
343  pBuf[RPC_POS_DAT0] = rsp;
344 }
