226 /*********************************************************************
227  * @fn HeartRate_AddService
228  *
229  * @brief Initializes the Heart Rate service by registering
230  * GATT attributes with the GATT server.
231  *
232  * @param services - services to add. This is a bit map and can
233  * contain more than one service.
234  *
235  * @return Success or Failure
236  */
237 bStatus_t HeartRate_AddService( uint32 services )
238 {
239  uint8 status = SUCCESS;
240 
241  // Initialize Client Characteristic Configuration attributes
242  GATTServApp_InitCharCfg( INVALID_CONNHANDLE, heartRateMeasClientCharCfg );
243 
244  if ( services & HEARTRATE_SERVICE )
245  {
246  // Register GATT attribute list and CBs with GATT Server App
247  status = GATTServApp_RegisterService( heartRateAttrTbl,
248  GATT_NUM_ATTRS( heartRateAttrTbl ),
249  &heartRateCBs );
250  }
251 
252  return ( status );
253 }

372 /*********************************************************************
373  * @fn heartRate_ReadAttrCB
374  *
375  * @brief Read an attribute.
376  *
377  * @param connHandle - connection message was received on
378  * @param pAttr - pointer to attribute
379  * @param pValue - pointer to data to be read
380  * @param pLen - length of data to be read
381  * @param offset - offset of the first octet to be read
382  * @param maxLen - maximum length of data to be read
383  *
384  * @return Success or Failure
385  */
386 static uint8 heartRate_ReadAttrCB( uint16 connHandle, gattAttribute_t *pAttr,
387  uint8 *pValue, uint8 *pLen, uint16 offset, uint8 maxLen )
388 {
389  bStatus_t status = SUCCESS;
390 
391  // Make sure it's not a blob operation (no attributes in the profile are long)
392  if ( offset > 0 )
393  {
394  return ( ATT_ERR_ATTR_NOT_LONG );
395  }
396 
397  uint16 uuid = BUILD_UINT16( pAttr->type.uuid[0], pAttr->type.uuid[1]);
398 
399  if (uuid == BODY_SENSOR_LOC_UUID)
400  {
401  *pLen = 1;
402  pValue[0] = *pAttr->pValue;
403  }
404  else
405  {
406  status = ATT_ERR_ATTR_NOT_FOUND;
407  }
408 
409  return ( status );
410 }

412 /*********************************************************************
413  * @fn heartRate_WriteAttrCB
414  *
415  * @brief Validate attribute data prior to a write operation
416  *
417  * @param connHandle - connection message was received on
418  * @param pAttr - pointer to attribute
419  * @param pValue - pointer to data to be written
420  * @param len - length of data
421  * @param offset - offset of the first octet to be written
422  *
423  * @return Success or Failure
424  */
425 static bStatus_t heartRate_WriteAttrCB( uint16 connHandle, gattAttribute_t *pAttr,
426  uint8 *pValue, uint8 len, uint16 offset )
427 {
428  bStatus_t status = SUCCESS;
429 
430  uint16 uuid = BUILD_UINT16( pAttr->type.uuid[0], pAttr->type.uuid[1]);
431  switch ( uuid )
432  {
433  case HEARTRATE_CTRL_PT_UUID:
434  if ( offset > 0 )
435  {
436  status = ATT_ERR_ATTR_NOT_LONG;
437  }
438  else if (len != 1)
439  {
440  status = ATT_ERR_INVALID_VALUE_SIZE;
441  }
442  else if (*pValue != HEARTRATE_COMMAND_ENERGY_EXP)
443  {
444  status = HEARTRATE_ERR_NOT_SUP;
445  }
446  else
447  {
448  *(pAttr->pValue) = pValue[0];
449 
450  (*heartRateServiceCB)(HEARTRATE_COMMAND_SET);
451 
452  }
453  break;
454 
455  case GATT_CLIENT_CHAR_CFG_UUID:
456  status = GATTServApp_ProcessCCCWriteReq( connHandle, pAttr, pValue, len,
457  offset, GATT_CLIENT_CFG_NOTIFY );
458  if ( status == SUCCESS )
459  {
460  uint16 charCfg = BUILD_UINT16( pValue[0], pValue[1] );
461 
462  (*heartRateServiceCB)( (charCfg == GATT_CFG_NO_OPERATION) ?
463  HEARTRATE_MEAS_NOTI_DISABLED :
464  HEARTRATE_MEAS_NOTI_ENABLED );
465  }
466  break;
467 
468  default:
469  status = ATT_ERR_ATTR_NOT_FOUND;
470  break;
471  }
472 
473  return ( status );
474 }

476 /*********************************************************************
477  * @fn HeartRate_HandleConnStatusCB
478  *
479  * @brief Heart Rate Service link status change handler function.
480  *
481  * @param connHandle - connection handle
482  * @param changeType - type of change
483  *
484  * @return none
485  */
486 void HeartRate_HandleConnStatusCB( uint16 connHandle, uint8 changeType )
487 {
488  // Make sure this is not loopback connection
489  if ( connHandle != LOOPBACK_CONNHANDLE )
490  {
491  // Reset Client Char Config if connection has dropped
492  if ( ( changeType == LINKDB_STATUS_UPDATE_REMOVED ) ||
493  ( ( changeType == LINKDB_STATUS_UPDATE_STATEFLAGS ) &&
494  ( !linkDB_Up( connHandle ) ) ) )
495  {
496  GATTServApp_InitCharCfg( connHandle, heartRateMeasClientCharCfg );
497  }
498  }
499 }
