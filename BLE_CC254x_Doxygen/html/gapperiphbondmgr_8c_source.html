451 /*********************************************************************
452  * @brief Notify the Bond Manager that a connection has been made.
453  *
454  * Public function defined in gapperiphbondmgr.h.
455  */
456 bStatus_t GAPBondMgr_LinkEst( uint8 addrType, uint8 *pDevAddr, uint16 connHandle )
457 {
458  uint8 idx; // NV Index
459  uint8 publicAddr[B_ADDR_LEN] // Place to put the public address
460  = {0, 0, 0, 0, 0, 0};
461 
462  idx = GAPBondMgr_ResolveAddr( addrType, pDevAddr, publicAddr );
463  if ( idx < GAP_BONDINGS_MAX )
464  {
465  // Bonding found
466  uint8 stateFlags; // Bond state flags
467  smSecurityInfo_t localLTK; // LTK information
468  smSigningInfo_t signingInfo; // Signature information
469 
470  // Initialize the NV structures
471  osal_memset( &localLTK, 0, sizeof ( smSecurityInfo_t ) );
472  osal_memset( &signingInfo, 0, sizeof ( smSigningInfo_t ) );
473 
474  // Check if key is valid, then load the key
475  stateFlags = gapBondMgrGetStateFlags( idx );
476  if ( osal_snv_read( localLTKNvID(idx), sizeof ( smSecurityInfo_t ), &localLTK ) == SUCCESS )
477  {
478  if ( (localLTK.keySize >= MIN_ENC_KEYSIZE) && (localLTK.keySize <= MAX_ENC_KEYSIZE) )
479  {
480  // Load the key information for the bonding
481  VOID GAP_Bond( connHandle,
482  ((stateFlags & GAP_BONDED_STATE_AUTHENTICATED) ? TRUE : FALSE),
483  &localLTK );
484  }
485  }
486 
487  // Load the Signing Key
488  if ( osal_snv_read( devCSRKNvID(idx), KEYLEN, signingInfo.srk ) == SUCCESS )
489  {
490  if ( osal_isbufset( signingInfo.srk, 0xFF, KEYLEN ) == FALSE )
491  {
492  // Load the signing information for this connection
493  VOID osal_snv_read( devSignCounterNvID(idx), sizeof ( uint32 ), &(signingInfo.signCounter) );
494  VOID GAP_Signable( connHandle,
495  ((stateFlags & GAP_BONDED_STATE_AUTHENTICATED) ? TRUE : FALSE),
496  &signingInfo );
497  }
498  }
499 
500  // Has there been a service change?
501  if ( stateFlags & GAP_BONDED_STATE_SERVICE_CHANGED )
502  {
503  VOID GATT_ServiceChangedInd( connHandle, gapBondMgr_TaskID );
504  }
505  }
506  else
507  {
508  // Not bound
509  if ( gapBond_PairingMode == GAPBOND_PAIRING_MODE_INITIATE )
510  {
511  // Could add code to setup a timeout to initiate security
512  // with a Slave Security Request.
513  }
514  }
515 
516  return ( SUCCESS );
517 }

525 {

527 
528  switch ( addrType )
529  {
530  case ADDRTYPE_PUBLIC:
531  case ADDRTYPE_STATIC:
532  idx = gapBondMgrFindAddr( pDevAddr );
533  if ( (idx < GAP_BONDINGS_MAX) && (pResolvedAddr) )
534  {
535  VOID osal_memcpy( pResolvedAddr, pDevAddr, B_ADDR_LEN );
536  }
537  break;
538

540  // This could be a reconnection address

544  VOID gapBondMgrGetPublicAddr( idx, pResolvedAddr );
545  }
546  break;
547 
548  case ADDRTYPE_PRIVATE_RESOLVE:
549  // Master's don't use Private Resolvable addresses
550  break;
551 
552  default:
553  break;
554  }
555 
556  return ( idx );

561  *
562  * Public function defined in gapperiphbondmgr.h.
563  */
564 bStatus_t GAPBondMgr_ServiceChangeInd( uint16 connectionHandle, uint8 setParam )
565 {
566  bStatus_t ret = bleNoResources; // return value
567 
568  if ( connectionHandle == 0xFFFF )
569  {
570  uint8 idx; // loop counter
571 
572  // Run through the bond database and update the Service Change indication
573  for ( idx = 0; idx < GAP_BONDINGS_MAX; idx++ )
574  {
575  if ( gapBondMgrChangeState( idx, GAP_BONDED_STATE_SERVICE_CHANGED, setParam ) )
576  {
577  ret = SUCCESS;
578  }
579  }
580 
581  // If the service change indication is TRUE, tell the connected devices
582  if ( setParam )
583  {
584  // Run connected database
585  linkDB_PerformFunc( gapBondMgrSendServiceChange );
586  }
587  }
588  else
589  {
590  // Find connection information
591  linkDBItem_t *pLinkItem = linkDB_Find( connectionHandle );
592  if ( pLinkItem )
593  {
594  uint8 idx; // loop counter
595  idx = GAPBondMgr_ResolveAddr( pLinkItem->addrType, pLinkItem->addr, NULL );
596  if ( idx < GAP_BONDINGS_MAX )
597  {
598  // Bond found, update it.
599  VOID gapBondMgrChangeState( idx, GAP_BONDED_STATE_SERVICE_CHANGED, setParam );
600  ret = SUCCESS;
601  }
602 
603  // If the service change indication is TRUE, tell the connected device
604  if ( setParam )
605  {
606  gapBondMgrSendServiceChange( pLinkItem );
607  }
608  }
609  else
610  {
611  ret = bleNotConnected;
612  }
613  }
614 
615  return ( ret );
616 }

673  *
674  * @brief Change a state flag in the stateFlags field of the bond record.
675  *
676  * @param idx - Bond NV index
677  * @param state - state flage to set or clear
678  * @param set - TRUE to set the flag, FALSE to clear the flag
679  *
680  * @return TRUE if NV Record exists, FALSE if NV Record is empty
681  */
682 static uint8 gapBondMgrChangeState( uint8 idx, uint16 state, uint8 set )
683 {
684  gapBondRec_t bondRec; // Space to read a Bond record from NV
685 
686  // Look for public address that is used (not all 0xFF's)
687  if ( (osal_snv_read( mainRecordNvID(idx), sizeof ( gapBondRec_t ), &bondRec ) == SUCCESS)
688  && (osal_isbufset( bondRec.publicAddr, 0xFF, B_ADDR_LEN ) == FALSE) )
689  {
690  // Update the state of the bonded device.
691  uint8 stateFlags = bondRec.stateFlags;
692  if ( set )
693  {
694  stateFlags |= state;
695  }
696  else
697  {
698  stateFlags &= ~(state);
699  }
700 
701  if ( stateFlags != bondRec.stateFlags )
702  {
703  bondRec.stateFlags = stateFlags;
704  VOID osal_snv_write( mainRecordNvID(idx), sizeof ( gapBondRec_t ), &bondRec );
705  }
706  return ( TRUE );
707  }
708  return ( FALSE );
709 }

1116  }
1117  }
1118  break;
1119 
1120  case GAP_AUTHENTICATION_COMPLETE_EVENT:
1121  {
1122  gapAuthCompleteEvent_t *pPkt = (gapAuthCompleteEvent_t *)pMsg;
1123 
1124  // Should we save bonding information
1125  if ( (pPkt->hdr.status == SUCCESS) && (pPkt->authState & SM_AUTH_STATE_BONDING) )
1126  {
1127  gapBondRec_t bondRec;
1128 
1129  VOID osal_memset( &bondRec, 0, sizeof ( gapBondRec_t ) ) ;
1130 
1131  // Do we have a public address in the data?
1132  if ( pPkt->pIdentityInfo )

1135  }
1136  else

1138  linkDBItem_t *pLinkItem = linkDB_Find( pPkt->connectionHandle );

1140  {
1141  VOID osal_memcpy( bondRec.publicAddr, pLinkItem->addr, B_ADDR_LEN );
1142  }
1143  else
1144  {
1145  // We don't have an address, so ignore the message.
1146  break;
1147  }
1148  }
1149 
1150  // Save off of the authentication state
1151  bondRec.stateFlags |= (pPkt->authState & SM_AUTH_STATE_AUTHENTICATED) ? GAP_BONDED_STATE_AUTHENTICATED : 0;
1152 
1153  VOID gapBondMgrAddBond( &bondRec,
1154  (gapBondLTK_t *)pPkt->pSecurityInfo,
1155  (gapBondLTK_t *)pPkt->pDevSecInfo,

1193 
1194  case GAP_PAIRING_REQ_EVENT:

1197
