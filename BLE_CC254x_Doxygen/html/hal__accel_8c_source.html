218 /**************************************************************************************************
219  * @fn halProcessMotionDetectInterrupt
220  *
221  * @brief Processes motion detection interrupt by informing the entity that
222  * enabled motion detection that it has occurred. Also disables further
223  * motion detection interrupts.
224  *
225  * @param
226  *
227  * @return
228  **************************************************************************************************/
229 static void halAccelProcessMotionDetectInterrupt( void )
230 {
231  /* Disable port interrupt */
232  P1IEN &= (uint8) ~HAL_ACCEL_P1_INTERRUPT_PINS;
233 
234  /* Disable P1 interrupts */
235  IEN2 &= ~(BV( 4 ));
236 
237  /* Set event to process motion detection not in interrupt context */
238  osal_set_event( Hal_TaskID, HAL_MOTION_DETECTED_EVENT );
239 }
240 
241 /**************************************************************************************************
242  * @fn halAccelProcessMotionDetectEvent

246  * @param
247  *
248  * @return
249  **************************************************************************************************/
250 void halAccelProcessMotionDetectEvent( void )
251 {
252  /* Inform application that motion was detected */
253  if (pHalAccelProcessFunction != NULL)

255  pHalAccelProcessFunction();
256  }

259 /**************************************************************************************************
260  * @fn HalAccelInit
261  *
262  * @brief This function initializes the HAL Accelerometer abstraction layer.
263  *
264  * input parameters
265  *
266  * None.
267  *
268  * output parameters
269  *
270  * None.
271  *
272  * @return None.
273  */
274 void HalAccelInit( void )
275 {
276  /* Configure POWER PIN as output */
277  HAL_ACCEL_POWER_PxDIR |= HAL_ACCEL_POWER_PIN_BV;
278 
279  /* Set POWER output high */
280  HAL_ACCEL_POWER_PIN = 1;
281 
282  /* Configure pin function as GPIO for pins related to accelerometer */
283  HAL_ACCEL_POWER_PxSEL &= ~HAL_ACCEL_POWER_PIN_BV;
284  P1SEL &= (uint8) ~HAL_ACCEL_P1_GPIO_PINS;
285 
286  /* Configure direction of pins related to accelerometer */
287  P1DIR &= (uint8) ~HAL_ACCEL_P1_INPUT_PINS;
288 
289  /* Configure port 1 inputs (i.e. accelerometer interrupt pin) as pulldown
290  * to save current.
291  */
292  P2INP |= 0x40;
293 
294  /* Indicate no callback has been registered */
295  pHalAccelProcessFunction = NULL;
296 }

298 /**************************************************************************************************
299  * @fn HalAccelEnable
300  *
301  * @brief This function configures the accelerometer for operation.
302  *
303  * input parameters
304  *
305  * None.
306  *
307  * output parameters
308  *
309  * None.
310  *
311  * @return None.
312  */
313 void HalAccelEnable( void )
314 {
315  uint8 i, dummy;
316 
317  /* Ensure accelerometer is in standby mode */
318  HalMotionI2cWrite( HAL_MOTION_DEVICE_ACCELEROMETER,
319  ACC_REG_ADDR_CTRL_REG1,
320  &HalAccelEnableConfigTable[0].data,
321  1 );
322 
323  /* Read INT_REL register to clear any prior accelerometer interrupts */
324  HalMotionI2cRead( HAL_MOTION_DEVICE_ACCELEROMETER,
325  ACC_REG_ADDR_INT_REL,
326  1,
327  &dummy );
328 
329  /* Configure accelerometer for measurement mode */
330  for (i = 0; i < HAL_ACCEL_ENABLE_CONFIG_TABLE_SIZE; i++)
331  {
332  HalMotionI2cWrite( HAL_MOTION_DEVICE_ACCELEROMETER,
333  HalAccelEnableConfigTable[i].addr,
334  &HalAccelEnableConfigTable[i].data,
335  1 );
336  }
337 }

368 /**************************************************************************************************
369  * @fn HalAccelMotionDetect
370  *
371  * @brief Places the Accelerometer in motion detection mode.
372  *
373  * input parameters
374  *
375  * @param cback - callback function to invoke if motion is detected
376  *
377  * output parameters
378  *
379  * None.
380  *
381  * @return None.
382  */
383 void HalAccelMotionDetect( halAccelCBack_t cback )
384 {
385  uint8 i;
386 
387  /* Register the callback fucntion */
388  pHalAccelProcessFunction = cback;
389 
390  /* Configure accelerometer for motion detection mode */
391  for (i = 0; i < HAL_ACCEL_MOTION_DETECT_CONFIG_TABLE_SIZE; i++)
392  {
393  HalMotionI2cWrite( HAL_MOTION_DEVICE_ACCELEROMETER,
394  HalAccelMotionDetectConfigTable[i].addr,
395  &HalAccelMotionDetectConfigTable[i].data,
396  1 );
397  }
398 
399  /* Clear any prior P1 interrupts */
400  P1IFG = (uint8) (~HAL_ACCEL_P1_INTERRUPT_PINS);
401  P1IF = 0;
402 
403  /* Enable interrupt when motion is detected */
404  P1IEN |= HAL_ACCEL_P1_INTERRUPT_PINS;
405 
406  /* Enable P1 interrupts */
407  IEN2 |= BV( 4 );
408 }

414 /**************************************************************************************************

418  *
419  * @param

421  * @return
422  **************************************************************************************************/
423 HAL_ISR_FUNCTION( halAccelPort1Isr, P1INT_VECTOR )

426

428  if ((P1IFG & HAL_ACCEL_P1_INTERRUPT_PINS) && (P1IEN & HAL_ACCEL_P1_INTERRUPT_PINS))

430  halAccelProcessMotionDetectInterrupt();
431  }
432 
433  P1IFG = 0;
434  P1IF = 0;
435 
436  HAL_EXIT_ISR();
437 }
