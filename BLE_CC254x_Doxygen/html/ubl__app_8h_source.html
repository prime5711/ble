61 // The SB page boundary since all SB addresses are "actual address / flash word size".
62 // Note for MSP - flash word size is 1, but 4 must be used for inter-compatibility w/ SoC.
63 #define UBL_PAGE_SIZE (HAL_FLASH_PAGE_SIZE / HAL_FLASH_WORD_SIZE)
64 
65 #define UBL_ADDR_OSET (uint16)(UBL_PAGE_FIRST * HAL_FLASH_PAGE_SIZE / HAL_FLASH_WORD_SIZE)
66 // AES Control Block occupies the first 32 bytes of the RC Image.
67 #define UBL_ADDR_CTRL (UBL_ADDR_OSET + 0x00) // 0x800
68 #define UBL_ADDR_CRC (UBL_ADDR_OSET + 0x10) // 0x810
69 #define UBL_ADDR_IMG_LEN (UBL_ADDR_OSET + 0x14) // 0x814
70 
71 // The UBL is in Bank0, page 0, and Bank7, pages 122-126.
72 // The Application image area first & last pages need to be carefully synchronized with the UBL and
73 // Application linker files as well as with the ASM IVED relays.
74 #define UBL_PAGE_FIRST 1
75 // Not including pages reserved for SNV for ease-of-use with encrypting-signing boot loader build.
76 #define UBL_PAGE_LAST 119
77 //efine UBL_PAGE_LAST 121

84 // The last address allowed for reading is determined by the 'imageLen' field of the Control Block
85 // in order to prevent access to the SNV pages after they contain a secret or security key.
