385 /*********************************************************************
386  * @fn Hidapp_ProcessEvent
387  *
388  * @brief This function processes the OSAL events and messages for the
389  * Sample Application.
390  *
391  * input parameters
392  *
393  * @param taskId - Task ID assigned to this app by OSAL at system initialization.
394  * @param events - Bit mask of the pending event(s).
395  *
396  * output parameters
397  *
398  * None.
399  *
400  * @return The events bit map received via parameter with the bits cleared
401  * which correspond to the event(s) that were processed on this invocation.
402  */
403 uint16 Hidapp_ProcessEvent(uint8 taskId, uint16 events)
404 {
405  (void) taskId; // Unused argument
406 
407  if ( events & SYS_EVENT_MSG )
408  {
409  osal_event_hdr_t *pMsg;
410 
411  while ((pMsg = (osal_event_hdr_t *) osal_msg_receive(hidappTaskId)) != NULL)
412  {
413  switch (pMsg->event)
414  {
415  case GATT_MSG_EVENT:
416  {
417  hidappProcessGATTMsg( (gattMsgEvent_t *)pMsg );
418  }
419  break;
420  case KEY_CHANGE:
421  {
422  hidappHandleKeys( ((keyChange_t *)pMsg)->keys, ((keyChange_t *)pMsg)->state );
423  }
424  break;
425 
426  default:
427  break;
428  }
429 
430  VOID osal_msg_deallocate((uint8 *) pMsg);
431  }
432 
433  return (events ^ SYS_EVENT_MSG);
434  }
435 
436  if ( events & HIDAPP_EVT_START )
437  {
438  hidappStart();
439 
440  return (events ^ HIDAPP_EVT_START);
441  }
442 
443  if ( events & HIDAPP_EVT_REPORT_RETRY )
444  {
445  // Report retries event
446  if ( ( hidappSendInReport( &lastInReport ) == FALSE ) &&
447  ( reportRetries < USB_HID_INPUT_RETRY_COUNT ) )
448  {
449  reportRetries++;
450  osal_start_timerEx( hidappTaskId, HIDAPP_EVT_REPORT_RETRY,
451  (reportRetries*HIDAPP_INPUT_RETRY_TIMEOUT) );
452  }
453  else
454  {
455  // Done retrying
456  reportRetries = 0;
457  }
458 
459  return (events ^ HIDAPP_EVT_REPORT_RETRY);
460  }
461 
462  if ( events & HIDAPP_EVT_START_DISCOVERY )
463  {
464  // Start initial discovery
465  hidappStartDiscovery();
466 
467  return ( events ^ HIDAPP_EVT_START_DISCOVERY );
468  }
469 
470  if ( events & HIDAPP_EVT_INIT_CONNECT )
471  {
472  // Still trying to re-connect
473  HalLedSet( HAL_LED_1, HAL_LED_MODE_BLINK ); // green led
474 
475  return ( events ^ HIDAPP_EVT_INIT_CONNECT );
476  }
477 
478  return ( 0 ); /* Discard unknown events. */
479 }

602 /*********************************************************************
603  * @fn hidappProcessGATTMsg
604  *
605  * @brief Process incoming GATT messages.
606  *
607  * @param pPkt - pointer to message.
608  *
609  * @return none
610  */
611 
612 static void hidappProcessGATTMsg( gattMsgEvent_t *pPkt )
613 {
614  // Build the message first
615  switch ( pPkt->method )
616  {
617  case ATT_HANDLE_VALUE_NOTI:
618  // First try to send out pending HID report
619  if ( reportRetries > 0 )
620  {
621  hidappSendInReport( &lastInReport );
622 
623  reportRetries = 0;
624  osal_stop_timerEx( hidappTaskId, HIDAPP_EVT_REPORT_RETRY );
625  }
626 
627  // Send incoming HID report
628  if ( hidappSendInReport( &(pPkt->msg.handleValueNoti) ) == FALSE )
629  {
630  // Save report for retries later
631  osal_memcpy( &lastInReport, &(pPkt->msg.handleValueNoti), sizeof( attHandleValueNoti_t ) );
632 
633  reportRetries = 1;
634  osal_start_timerEx( hidappTaskId, HIDAPP_EVT_REPORT_RETRY, HIDAPP_INPUT_RETRY_TIMEOUT );
635  }
636  break;
637 
638  case ATT_FIND_BY_TYPE_VALUE_RSP:
639  // Response from GATT_DiscPrimaryServiceByUUID
640  // Service found, store handles
641  if ( pPkt->msg.findByTypeValueRsp.numInfo > 0 )
642  {
643  serviceStartHandle = pPkt->msg.findByTypeValueRsp.handlesInfo[0].handle;
644  serviceEndHandle = pPkt->msg.findByTypeValueRsp.handlesInfo[0].grpEndHandle;
645  }
646  // If procedure complete
647  else if ( pPkt->hdr.status == bleProcedureComplete )
648  {
649  if ( serviceStartHandle != 0 )
650  {
651  if ( serviceToDiscover == GATT_SERVICE_UUID )
652  {
653  // Begin the search for characteristic handle of the service
654  readReq.startHandle = serviceStartHandle;
655  readReq.endHandle = serviceEndHandle;
656  readReqType.len = 2;
657  readReqType.uuid[0] = LO_UINT16(SERVICE_CHANGED_UUID);
658  readReqType.uuid[1] = HI_UINT16(SERVICE_CHANGED_UUID);
659  readReq.type = readReqType;
660 
661  GATT_DiscCharsByUUID(connHandle, &readReq, hidappTaskId );
662  }
663  else if ( serviceToDiscover == HID_SERV_UUID )
664  {
665  // Discover all characteristics
666  GATT_DiscAllChars( connHandle, serviceStartHandle, serviceEndHandle, hidappTaskId );
667  }
668  }
669  }
670  break;
671 
672  case ATT_READ_BY_TYPE_RSP:
673  // Response from Discover all Characteristics.
674  // Success indicates packet with characteristic discoveries.
675  if ( pPkt->hdr.status == SUCCESS )
676  {
677  attReadByTypeRsp_t rsp = pPkt->msg.readByTypeRsp;
678  uint8 idx = 0;
679 
680  if ( serviceToDiscover == GATT_SERVICE_UUID )
681  {
682  // We have discovered the GATT Service Characteristic handle
683  uint8 low = LO_UINT16(pPkt->msg.readByTypeRsp.dataList[3]);
684  uint8 high = HI_UINT16(pPkt->msg.readByTypeRsp.dataList[4]);
685  serviceChangeHandle = BUILD_UINT16(low, high);
686  // Break to skip next part, it doesn't apply here
687  break;
688  }
689 
690  // Search characteristics for those with notification permissions.
691  while( idx < ((rsp.numPairs * rsp.len ) - 1))
692  {
693  // Check permissions of characteristic for notification permission
694  if ( (rsp.dataList[idx+2] & GATT_PROP_NOTIFY ) )
695  {
696  uint16* pHandle = (mouseCharHandle == GATT_INVALID_HANDLE) ? &mouseCharHandle : &keyCharHandle ;
697
