122 /*********************************************************************
123  * @fn glucoseIndGattMsg
124  *
125  * @brief Handle indications and notifications.
126  *
127  * @param pMsg - GATT message.
128  *
129  * @return none
130  */
131 void glucoseIndGattMsg( gattMsgEvent_t *pMsg )
132 {
133  uint8 i;
134 
135  // Look up the handle in the handle cache
136  for ( i = 0; i < HDL_CACHE_LEN; i++ )
137  {
138  if ( pMsg->msg.handleValueInd.handle == glucoseHdlCache[i] )
139  {
140  break;
141  }
142  }
143 
144  // Perform processing for this handle
145  switch ( i )
146  {
147  case HDL_GLUCOSE_START:
148  {
149  uint8* p = pMsg->msg.handleValueNoti.value;
150 
151  // restart procedure timer
152  if (glucCollWritePending == true)
153  {
154  osal_start_timerEx( glucCollTaskId, PROCEDURE_TIMEOUT_EVT, GLUCOSE_PROCEDURE_TIMEOUT );
155  }
156 
157  memset(&glucoseMeas, 0, sizeof(glucoseMeas));
158 
159  // Flags
160  glucoseMeas.flags = *p++;
161 
162  // Sequence number
163  glucoseMeas.seqNum = BUILD_UINT16(p[0], p[1]);
164  LCD_WRITE_STRING_VALUE( "SeqNum:", glucoseMeas.seqNum, 10, HAL_LCD_LINE_1 );
165  p += 2;
166 
167  // Base time
168  memcpy(glucoseMeas.baseTime, p, 7);
169  p += 7;
170 
171  // Time offset;
172  if (glucoseMeas.flags & GLUCOSE_MEAS_FLAG_TIME_OFFSET)
173  {
174  glucoseMeas.timeOffset = BUILD_UINT16(p[0], p[1]);
175  p += 2;
176  }
177 
178  // Glucose concentration
179  if(glucoseMeas.flags & GLUCOSE_MEAS_FLAG_CONCENTRATION)
180  {
181  glucoseMeas.concentration = BUILD_UINT16(p[0], p[1]);
182  if(glucoseMeas.flags & GLUCOSE_MEAS_FLAG_UNITS)
183  LCD_WRITE_STRING_VALUE( STR_MMOL_PER_L, glucoseMeas.concentration, 10, HAL_LCD_LINE_2 );
184  else
185  LCD_WRITE_STRING_VALUE( STR_MG_PER_DL, glucoseMeas.concentration, 10, HAL_LCD_LINE_2 );
186  p += 2;
187 
188  // Type sample location
189  glucoseMeas.typeSampleLocation = *p++;
190  }
191 
192  // Sensor status annunciation
193  if (glucoseMeas.flags & GLUCOSE_MEAS_FLAG_STATUS_ANNUNCIATION)
194  {
195  glucoseMeas.sensorStatus = BUILD_UINT16(p[0], p[1]);
196  p += 2;
197  }
198  }
199  break;
200 
201  case HDL_GLUCOSE_CONTEXT_START:
202  {
203  uint8* p = pMsg->msg.handleValueNoti.value;
204 
205  // restart procedure timer
206  if (glucCollWritePending == true)
207  {
208  osal_start_timerEx( glucCollTaskId, PROCEDURE_TIMEOUT_EVT, GLUCOSE_PROCEDURE_TIMEOUT );
209  }
210 
211  memset(&glucoseContext, 0, sizeof(glucoseContext));
212 
213  // Flags
214  glucoseContext.flags = *p++;
215 
216  // Sequence number
217  glucoseContext.seqNum = BUILD_UINT16(p[0], p[1]);
218  p += 2;
219 
220  // Extended flags
221  if(glucoseContext.flags & GLUCOSE_CONTEXT_FLAG_EXTENDED)
222  {
223  glucoseContext.extendedFlags = *p++;
224  }
225 
226  // Carbohydrate
227  if(glucoseContext.flags & GLUCOSE_CONTEXT_FLAG_CARBO)
228  {
229  // carbohydrate ID
230  glucoseContext.carboId = *p++;
231 
232  // Carbohydrate
233  glucoseContext.carboVal = BUILD_UINT16(p[0], p[1]);
234  p += 2;
235  }
236 
237  // Meal
238  if(glucoseContext.flags & GLUCOSE_CONTEXT_FLAG_MEAL)
239  {
240  glucoseContext.mealVal = *p++;
241  }
242 
243  // Tester health
244  if(glucoseContext.flags & GLUCOSE_CONTEXT_FLAG_TESTER_HEALTH)
245  {
246  glucoseContext.TesterHealthVal = *p++;
247  }
248 
249  // Exercise
250  if(glucoseContext.flags & GLUCOSE_CONTEXT_FLAG_EXERCISE)
251  {
252  // Duration
253  glucoseContext.exerciseDuration = BUILD_UINT16(p[0], p[1]);
254  p += 2;
255 
256  // Intensity
257  glucoseContext.exerciseIntensity = *p++;
258  }
259 
260  // Medication
261  if(glucoseContext.flags & GLUCOSE_CONTEXT_FLAG_MEDICATION)
262  {
263  // Medication ID
264  glucoseContext.medId = *p++;
265 
266  // Medication
267  glucoseContext.medVal = BUILD_UINT16(p[0], p[1]);
268  p += 2;
269  }
270 
271  // HbA1c
272  if(glucoseContext.flags & GLUCOSE_CONTEXT_FLAG_HbA1c)
273  {
274  glucoseContext.HbA1cVal = BUILD_UINT16(p[0], p[1]);
275  LCD_WRITE_STRING_VALUE( "HbA1c:", glucoseContext.HbA1cVal, 10, HAL_LCD_LINE_3 );
276  p += 2;
277  }
278 
279  }
280  break;
281 
282  case HDL_GLUCOSE_CTL_PNT_START:
283  {
284  uint8* pValue = pMsg->msg.handleValueInd.value;
285 
286  // stop procedure timer
287  osal_stop_timerEx( glucCollTaskId, PROCEDURE_TIMEOUT_EVT );
288 
289  if(pValue[0] == CTL_PNT_OP_NUM_RSP)
290  {
291  if(pMsg->msg.handleValueInd.len >= 3)
292  {
293  LCD_WRITE_STRING("Matching ", HAL_LCD_LINE_1);
294  LCD_WRITE_STRING( "Records:", HAL_LCD_LINE_2 );
295  LCD_WRITE_STRING_VALUE("", BUILD_UINT16(pValue[2], pValue[3]), 10, HAL_LCD_LINE_3 );
296  }
297  }
298  else if(pValue[0] == CTL_PNT_OP_REQ_RSP && glucCollClearPending)
299  {
300  glucCollClearPending = false;
301 
302  if(pMsg->msg.handleValueInd.len >= 3)
303  {
304  switch(pValue[3])
305  {
306  case CTL_PNT_RSP_SUCCESS:
307  LCD_WRITE_STRING("Records", HAL_LCD_LINE_1);
308  LCD_WRITE_STRING("Cleared", HAL_LCD_LINE_2 );
309  LCD_WRITE_STRING("", HAL_LCD_LINE_3 );
310  break;
311  case CTL_PNT_RSP_NO_RECORDS:
312  LCD_WRITE_STRING("No Matching", HAL_LCD_LINE_1);
313  LCD_WRITE_STRING("Records", HAL_LCD_LINE_2 );
314  LCD_WRITE_STRING("to Delete", HAL_LCD_LINE_3 );
315  break;
316  default:
317  LCD_WRITE_STRING("Error:", HAL_LCD_LINE_1);
318  LCD_WRITE_STRING_VALUE("", pValue[3], 10, HAL_LCD_LINE_2 );
319  LCD_WRITE_STRING( "", HAL_LCD_LINE_3 );
320  break;
321  }
322  }
323  }
324  else if(pValue[0] == CTL_PNT_OP_REQ_RSP)
325  {
326  if(pMsg->msg.handleValueInd.len >= 3)
327  {
328  switch(pValue[3])
329  {
330  case CTL_PNT_RSP_SUCCESS:
331  break;
332  case CTL_PNT_RSP_NO_RECORDS:
333  LCD_WRITE_STRING("No Matching", HAL_LCD_LINE_1);
334  LCD_WRITE_STRING("Records", HAL_LCD_LINE_2 );
335  LCD_WRITE_STRING("Found", HAL_LCD_LINE_3 );
336  break;
337  default:
338  LCD_WRITE_STRING("Error:", HAL_LCD_LINE_1);
339  LCD_WRITE_STRING_VALUE("", pValue[3], 10, HAL_LCD_LINE_2 );
340  LCD_WRITE_STRING( "", HAL_LCD_LINE_3 );
341  break;
342  }
343  }
344  }
345  }
346  break;
347 
348  default:
349  break;
350  }
351 
352  // Send confirm for indication
353  if ( pMsg->method == ATT_HANDLE_VALUE_IND )
354  {
355  ATT_HandleValueCfm( pMsg->connHandle );
356  }
357 
358 }
