39 /*********************************************************************
40  NOTE: If polling is used, the hal_driver task schedules the KeyRead()
41  to occur every 100ms. This should be long enough to naturally
42  debounce the keys. The KeyRead() function remembers the key
43  state of the previous poll and will only return a non-zero
44  value if the key state changes.
45 
46  NOTE: If interrupts are used, the KeyRead() function is scheduled
47  25ms after the interrupt occurs by the ISR. This delay is used
48  for key debouncing. The ISR disables any further Key interrupt
49  until KeyRead() is executed. KeyRead() will re-enable Key
50  interrupts after executing. Unlike polling, when interrupts
51  are enabled, the previous key state is not remembered. This
52  means that KeyRead() will return the current state of the keys
53  (not a change in state of the keys).
54 
55  NOTE: If interrupts are used, the KeyRead() fucntion is scheduled by
56  the ISR. Therefore, the joystick movements will only be detected
57  during a pushbutton interrupt caused by S1 or the center joystick
58  pushbutton.
59 
60  NOTE: When a switch like S1 is pushed, the S1 signal goes from a normally
61  high state to a low state. This transition is typically clean. The
62  duration of the low state is around 200ms. When the signal returns
63  to the high state, there is a high likelihood of signal bounce, which
64  causes a unwanted interrupts. Normally, we would set the interrupt
65  edge to falling edge to generate an interrupt when S1 is pushed, but
66  because of the signal bounce, it is better to set the edge to rising
67  edge to generate an interrupt when S1 is released. The debounce logic
68  can then filter out the signal bounce. The result is that we typically
69  get only 1 interrupt per button push. This mechanism is not totally
70  foolproof because occasionally, signal bound occurs during the falling
71  edge as well. A similar mechanism is used to handle the joystick
72  pushbutton on the DB. For the EB, we do not have independent control
73  of the interrupt edge for the S1 and center joystick pushbutton. As
74  a result, only one or the other pushbuttons work reasonably well with
75  interrupts. The default is the make the S1 switch on the EB work more
76  reliably.

175 /**************************************************************************************************
176  * @fn HalKeyInit
177  *
178  * @brief Initilize Key Service
179  *
180  * @param none
181  *
182  * @return None
183  **************************************************************************************************/
184 void HalKeyInit( void )
185 {
186 #if (HAL_KEY == TRUE)
187  /* Initialize previous key to 0 */
188  halKeySavedKeys = HAL_KEY_CODE_NOKEY;
189 
190  /* The advanced remote doesn't have the same 8X8 row/column matrix as in other
191  * products. Instead, a 3X16 row/column matrix is used, with the rows continuing
192  * to be utilized by GPIOs, but the columns are generated via a 16 bit
193  * shift register. Controls for the shift register are, however, utilized with
194  * GPIOs.
195  *
196  * Another difference is that the GPIOs utilized for the rows are split between
197  * P0 and P1.
198  */
199 
200  /* Configure pin function as GPIO for pins related to keypad */
201  P0SEL &= (uint8) ~HAL_KEY_P0_GPIO_PINS;
202  P1SEL &= (uint8) ~HAL_KEY_P1_GPIO_PINS;
203 
204  /* Configure direction of pins related to keypad */
205  P0DIR |= (uint8) HAL_KEY_P0_OUTPUT_PINS;
206  P0DIR &= (uint8) ~HAL_KEY_P0_INPUT_PINS;
207 
208  P1DIR |= (uint8) HAL_KEY_P1_OUTPUT_PINS;
209  P1DIR &= (uint8) ~HAL_KEY_P1_INPUT_PINS;
210 
211  P0INP = 0x00;
212 
213  /* Turn off LED pins */
214  P1_0 = P1_1 = P1_4 = 0x00;
215 
216  /* Set up initial value on output pins */
217  halPowerDownShiftRegister();
218 
219  /* Initialize callback function */
220  pHalKeyProcessFunction = NULL;
221 
222  /* Start with key is not configured */
223  HalKeyConfigured = FALSE;
224 
225  halKeyTimerRunning = FALSE;
226 #endif /* HAL_KEY */
227 }

229 /**************************************************************************************************
230  * @fn HalKeyConfig
231  *
232  * @brief Configure the Key serivce
233  *
234  * @param interruptEnable - TRUE/FALSE, enable/disable interrupt
235  * cback - pointer to the CallBack function
236  *
237  * @return None
238  **************************************************************************************************/
239 void HalKeyConfig (bool interruptEnable, halKeyCBack_t cback)
240 {
241 #if (HAL_KEY == TRUE)
242  /* Enable/Disable Interrupt */
243  Hal_KeyIntEnable = interruptEnable;
244 
245  /* Register the callback fucntion */
246  pHalKeyProcessFunction = cback;
247 
248  /* Determine if interrupt is enabled or not */
249  if (Hal_KeyIntEnable)
250  {
251  /* Configure interrupt for falling edge */
252  PICTL |= HAL_KEY_BIT0;
253 
254  /* Enable interrupts for individual port pins */
255  P0IEN |= HAL_KEY_P0_INTERRUPT_PINS;
256 
257  /* Enable P0 interrupts */
258  IEN1 |= HAL_KEY_BIT5;
259 
260  /* Do this only after the hal_key is configured - to work with sleep stuff */
261  if (HalKeyConfigured == TRUE)
262  {
263  osal_stop_timerEx( Hal_TaskID, HAL_KEY_EVENT); /* Cancel polling if active */
264  }
265  }
266  else /* Interrupts NOT enabled */
267  {
268  // disable interrupt
269  P0IEN &= ~(HAL_KEY_P0_INTERRUPT_PINS);
270  IEN1 &= ~(HAL_KEY_BIT5);
271 
272  osal_start_timerEx (Hal_TaskID, HAL_KEY_EVENT, HAL_KEY_POLLING_VALUE); /* Kick off polling */
273  }
274 
275  /* Key now is configured */
276  HalKeyConfigured = TRUE;
277 #endif /* HAL_KEY */
278 }

280 /**************************************************************************************************
281  * @fn HalKeyRead
282  *
283  * @brief Read the current value of a key
284  *
285  * @param None
286  *
287  * @return keys - current keys status
288  **************************************************************************************************/
289 uint8 HalKeyRead ( void )
290 {
291  uint8 colkeys = 0, rowkeys = 0;
292  uint8 keys, colcode = HAL_KEY_NUM_COLUMNS, rowcode = HAL_KEY_NUM_ROWS;
293 
294 #if (HAL_KEY == TRUE)
295 
296  // Disable interrupt, as interrupt can be triggered without key press during
297  // scanning process
298  P0IEN &= ~(HAL_KEY_P0_INTERRUPT_PINS);
299 
300  // Detect row first while all columns are still asserted.
301  keys = (P0 & HAL_KEY_P0_INPUT_PINS);
302  for (uint8 row = 0; row < HAL_KEY_NUM_ROWS; row++)
303  {
304  if (((1 << row) & keys) == 0)
305  {
306  if (++rowkeys == 1)
307  {
308  // remember first row code
309  rowcode = row;
310  }
311  else
312  {
313  // multiple key presses detected
314  break;
315  }
316  }
317  }
318 
319  /* Now check the columns. Since the pin supplying power to the shift register
320  * can't power all the KPa outputs, we will change the KPb pins to have pull down
321  * registers, which will result in all KPb pins reading 0. We will then shift 1
322  * 1 through the shift register, and the column that is pressed will show up as
323  * a 1 on the corresponding KPb pin.
324  */
325  P2INP |= HAL_KEY_BIT5;
326  halPowerUpShiftRegister();
327 
328  // Wait some time for things to settle
329  halSleepWait( 50 );
330 
331  /* Fix to address sending of false key release for power key and fav key */
332  halSetShiftRegisterData( 0 );
333  halClockShiftRegister();
334 
335  /* Check each column by shifting a 1 through the shift register */
336  halSetShiftRegisterData( 1 );
337  halClockShiftRegister();
338  halSetShiftRegisterData( 0 );
339 
340  for (uint8 col = 0; col < HAL_KEY_NUM_COLUMNS; col++)
341  {
342  // read all rows
343  keys = (P0 & HAL_KEY_P0_INPUT_PINS);
344 
345  // de-assert the column
346  halClockShiftRegister();
347 
348  // check whether the specific row was on.
349  if (keys & ((uint8)1 << rowcode))
350  {
351  if (++colkeys == 1)
352  {
353  // remember first col code
354  colcode = col;
355  }
356  else
357  {
358  // multiple key presses detected
359  break;
360  }
361  }
362  }
363 
364  // assert all columns for interrupt and for lower current consumption
365  halPowerDownShiftRegister();
366 
367  // change KPb pins back to pull up
368  P2INP &= ~HAL_KEY_BIT5;
369 
370  if (Hal_KeyIntEnable)
371  {
372  // clear interrupt flag. It is necessary since key scanning sets
373  // interrupt flag bits.
374  P0IFG = (uint8) (~HAL_KEY_P0_INTERRUPT_PINS);
375  P0IF = 0;
376 
377  // re-enable interrupts
378  P0IEN |= HAL_KEY_P0_INTERRUPT_PINS;
379  }
380 
381 #endif /* HAL_KEY */
382 
383  // multiple key presses not supported
384  if ((colcode == HAL_KEY_NUM_COLUMNS) ||
385  (rowcode == HAL_KEY_NUM_ROWS) ||
386  (rowkeys > 1) ||
387  (colkeys > 1))
388  {
389  keys = HAL_KEY_CODE_NOKEY; // no key pressed
390  }
391  else
392  {
393  keys = (rowcode << 4) | colcode;
394  }
395 
396  return keys;
397 }

400 /**************************************************************************************************
401  * @fn HalKeyPoll
402  *
403  * @brief Called by hal_driver to poll the keys
404  *
405  * @param None
406  *
407  * @return None
408  **************************************************************************************************/
409 void HalKeyPoll (void)
410 {
411 #if (HAL_KEY == TRUE)
412 
413  uint8 keys = 0;
414 
415  /*
416  * If interrupts are enabled, get the status of the interrupt-driven keys from 'halSaveIntKey'
417  * which is updated by the key ISR. If Polling, read these keys directly.
418  */
419  keys = HalKeyRead();
420 
421  /* Exit if polling and no keys have changed */
422  if (!Hal_KeyIntEnable)
423  {
424  if (keys == halKeySavedKeys)
425  {
426  return;
427  }
428  halKeySavedKeys = keys; /* Store the current keys for comparation next time */
429  }
430 
431  /* Invoke Callback if new keys were depressed */
432  if ((keys != HAL_KEY_CODE_NOKEY || Hal_KeyIntEnable) &&
433  (pHalKeyProcessFunction))
434  {
435  // When interrupt is enabled, send HAL_KEY_CODE_NOKEY as well so that
436  // application would know the previous key is no longer depressed.
437  (pHalKeyProcessFunction) (keys, HAL_KEY_STATE_NORMAL);
438  }
439 
440  if (Hal_KeyIntEnable)
441  {
442  if (keys != HAL_KEY_CODE_NOKEY)
443  {
444  // In order to trigger callback again as far as the key is depressed,
445  // timer is called here.
446  osal_start_timerEx(Hal_TaskID, HAL_KEY_EVENT, 50);
447  }
448  else
449  {
450  halKeyTimerRunning = FALSE;
451  }
452  }
453 #endif /* HAL_KEY */
454 
455 }

457 /**************************************************************************************************
458  * @fn halProcessKeyInterrupt
459  *
460  * @brief Checks to see if it's a valid key interrupt, saves interrupt driven key states for
461  * processing by HalKeyRead(), and debounces keys by scheduling HalKeyRead() 25ms later.
462  *
463  * @param
464  *
465  * @return
466  **************************************************************************************************/
467 void halProcessKeyInterrupt (void)
468 {
469 
470 #if (HAL_KEY == TRUE)
471 
472  if (P0IFG & HAL_KEY_P0_INTERRUPT_PINS)
473  {
474  // Disable interrupt
475  P0IEN &= (uint8) ~HAL_KEY_P0_INTERRUPT_PINS;
476 
477  // interrupt flag has been set
478  P0IFG = (uint8) (~HAL_KEY_P0_INTERRUPT_PINS); // clear interrupt flag
479  if (!halKeyTimerRunning)
480  {
481  halKeyTimerRunning = TRUE;
482  osalTimeUpdate();
483  osal_start_timerEx (Hal_TaskID, HAL_KEY_EVENT, HAL_KEY_DEBOUNCE_VALUE);
484  }
485  // Enable interrupt
486  P0IEN |= HAL_KEY_P0_INTERRUPT_PINS;
487  }
488 #endif /* HAL_KEY */
489 }
