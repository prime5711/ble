301 /*********************************************************************
302  * @fn BLE_Bridge_ProcessEvent
303  *
304  * @brief SBLE_Bridge Application Task event processor. This function
305  * is called to process all events for the task. Events
306  * include timers, messages and any other user defined events.
307  *
308  * @param task_id - The OSAL assigned task ID.
309  * @param events - events to process. This is a bit map and can
310  * contain more than one event.
311  *
312  * @return events not processed
313  */
314 uint16 BLE_Bridge_ProcessEvent( uint8 task_id, uint16 events )
315 {
316 
317  VOID task_id; // OSAL required parameter that isn't used in this function
318 
319  if ( events & SYS_EVENT_MSG )
320  {
321  uint8 *pMsg;
322 
323  if ( (pMsg = osal_msg_receive( BLE_Bridge_TaskID )) != NULL )
324  {
325  BLE_Bridge_ProcessOSALMsg( (osal_event_hdr_t *)pMsg );
326 
327  // Release the OSAL message
328  VOID osal_msg_deallocate( pMsg );
329  }
330 
331  // return unprocessed events
332  return (events ^ SYS_EVENT_MSG);
333  }
334 
335  if ( events & SBP_START_DEVICE_EVT )
336  {
337  // Start the Device
338  VOID GAPRole_StartDevice( &BLE_Bridge_PeripheralCBs );
339 
340  // Start timer for polling to see if there is data to send from SerialBuffer
341  osal_start_timerEx( BLE_Bridge_TaskID, SBP_SEND_EVT, SBP_SEND_EVT_PERIOD );
342 
343  return ( events ^ SBP_START_DEVICE_EVT );
344  }
345 
346  if ( events & SBP_SEND_EVT )
347  {
348  // Restart timer
349  if ( SBP_SEND_EVT_PERIOD )
350  {
351  osal_start_timerEx( BLE_Bridge_TaskID, SBP_SEND_EVT, SBP_SEND_EVT_PERIOD );
352  }
353 
354  // Send data if not caught up and connected
355  if ((buffer_tail != serialBufferOffset) && (connected_flag == TRUE))
356  {
357  //calculate how many bytes can be sent
358  uint16 diff = circular_diff(serialBufferOffset, buffer_tail);
359  //send data and update tail
360  uint8 bytes_sent = sendData(diff);
361  buffer_tail = circular_add(buffer_tail,bytes_sent);
362  //if we sent anything over-the-air, let host MCU know it can send more bytes
363  if (bytes_sent)
364  {
365  //keep trying to send ACK until it is a success. this may not be the desirable approach
366  while(sendAckMessage(bytes_sent));
367  }
368  }
369  return (events ^ SBP_SEND_EVT);
370  }
371 
372  // Discard unknown events
373  return 0;
374 }

455 /*********************************************************************
456  * @fn sendData
457  *
458  * @brief parse available serial data and send max possible amount over the air
459  *
460  * @param diff - how many bytes can be sent
461  *
462  * @return none
463  */
464 
465 static uint8 sendData(uint16 diff)
466 {
467  //can send max 4 packets per connection interval
468  uint8 packets_sent = 0;
469  //ensure queue of notification is successful
470  bool send_error = FALSE;
471  //return value to update tail and send ack to msp
472  uint8 bytes_sent = 0;
473 
474  attHandleValueNoti_t noti;
475  //dummy handle
476  noti.handle = 0x2E;
477 
478  //counter
479  uint8 i;
480 
481  while ((packets_sent < 4) && (diff >= 20) && (send_error == FALSE))
482  {
483  //send 20 bytes
484  noti.len = 20;
485  for (i = 0; i < 20; i++)
486  {
487  noti.value[i] = serialBuffer[circular_add(buffer_tail , bytes_sent+i)];
488  }
489  //connection handle currently hardcoded
490  if (!(GATT_Notification(0, &noti, FALSE))) //if sucessful
491  {
492  bytes_sent += 20;
493  diff -= 20;
494  packets_sent++;
495  }
496  else
497  {
498  send_error = TRUE;
499  }
500  }
501  //send remaining bytes
502  if ((packets_sent < 4) && (diff > 0) && (send_error == FALSE))
503  {
504  noti.len = diff;
505  for (i = 0; i < diff; i++)
506  {
507  noti.value[i] = serialBuffer[circular_add(buffer_tail, bytes_sent + i)];
508  }
509  if (!(GATT_Notification(0, &noti, FALSE))) //if sucessful
510  {
511  bytes_sent += i;
512  diff -= i;//amount of data sent
513  }
514  else
515  {
516  send_error = TRUE;
517  }
518  }
519  return bytes_sent;
520 }
