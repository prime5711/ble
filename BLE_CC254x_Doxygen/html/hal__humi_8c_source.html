98 /**************************************************************************************************
99 * @fn HalHumiInit
100 *
101 * @brief Initialise the humidity sensor driver
102 *
103 * @return none
104 **************************************************************************************************/
105 void HalHumiInit(void)
106 {
107  HalHumiSelect();
108 
109  // Set 11 bit resolution
110  HalSensorReadReg(SHT21_CMD_READ_U_R,&usr,1);
111  usr &= USR_REG_RES_MASK;
112  usr |= USR_REG_11BITRES;
113  HalSensorWriteReg(SHT21_CMD_WRITE_U_R,&usr,1);
114  success = FALSE;
115 }

118 /**************************************************************************************************
119 * @fn HalHumiExecMeasurementStep
120 *
121 * @brief Execute measurement step
122 *
123 * @return none
124 */
125 bool HalHumiExecMeasurementStep(uint8 state)
126 {
127  HalHumiSelect();
128 
129  switch (state)
130  {
131  case 0:
132  // Turn on DC-DC control
133  HalDcDcControl(ST_HUMID,true);
134 
135  // Start temperature read
136  success = HalHumiWriteCmd(SHT21_CMD_TEMP_T_NH);
137  break;
138 
139  case 1:
140  // Read and store temperature value
141  if (success)
142  {
143  success = HalHumiReadData(buf, DATA_LEN);
144 
145  // Start for humidity read
146  if (success)
147  {
148  success = HalHumiWriteCmd(SHT21_CMD_HUMI_T_NH);
149  }
150  }
151  break;
152 
153  case 2:
154  // Read and store humidity value
155  if (success)
156  {
157  success = HalHumiReadData(buf+DATA_LEN, DATA_LEN);
158  }
159 
160  // Turn of DC-DC control
161  HalDcDcControl(ST_HUMID,false);
162  break;
163  }
164 
165  return success;
166 }
