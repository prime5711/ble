374 /*********************************************************************
375  * @fn TimeApp_ProcessEvent
376  *
377  * @brief Time App Application Task event processor. This function
378  * is called to process all events for the task. Events
379  * include timers, messages and any other user defined events.
380  *
381  * @param task_id - The OSAL assigned task ID.
382  * @param events - events to process. This is a bit map and can
383  * contain more than one event.
384  *
385  * @return events not processed
386  */
387 uint16 TimeApp_ProcessEvent( uint8 task_id, uint16 events )
388 {
389 
390  VOID task_id; // OSAL required parameter that isn't used in this function
391 
392  if ( events & SYS_EVENT_MSG )
393  {
394  uint8 *pMsg;
395 
396  if ( (pMsg = osal_msg_receive( timeAppTaskId )) != NULL )
397  {
398  timeApp_ProcessOSALMsg( (osal_event_hdr_t *)pMsg );
399 
400  // Release the OSAL message
401  VOID osal_msg_deallocate( pMsg );
402  }
403 
404  // return unprocessed events
405  return ( events ^ SYS_EVENT_MSG );
406  }
407 
408  if ( events & START_DEVICE_EVT )
409  {
410  // Start the Device
411  VOID GAPRole_StartDevice( &timeAppPeripheralCB );
412 
413  // Register with bond manager after starting device
414  GAPBondMgr_Register( (gapBondCBs_t *) &timeAppBondCB );
415 
416  return ( events ^ START_DEVICE_EVT );
417  }
418 
419  if ( events & START_DISCOVERY_EVT )
420  {
421  if ( timeAppPairingStarted )
422  {
423  // Postpone discovery until pairing completes
424  timeAppDiscPostponed = TRUE;
425  }
426  else
427  {
428  timeAppDiscState = timeAppDiscStart();
429  }
430  return ( events ^ START_DISCOVERY_EVT );
431  }
432 
433  if ( events & CLOCK_UPDATE_EVT )
434  {
435  timeAppClockDisplay();
436 
437  // Restart clock tick timer
438  osal_start_timerEx( timeAppTaskId, CLOCK_UPDATE_EVT, DEFAULT_CLOCK_UPDATE_PERIOD );
439 
440  return ( events ^ CLOCK_UPDATE_EVT );
441  }
442 
443  // Discard unknown events
444  return 0;
445 }

473 /*********************************************************************
474  * @fn timeApp_HandleKeys
475  *
476  * @brief Handles all key events for this device.
477  *
478  * @param shift - true if in shift/alt.
479  * @param keys - bit field for key events. Valid entries:
480  * HAL_KEY_SW_2
481  * HAL_KEY_SW_1
482  *
483  * @return none
484  */
485 static void timeApp_HandleKeys( uint8 shift, uint8 keys )
486 {
487  if ( keys & HAL_KEY_UP )
488  {
489  // Start or stop advertising
490  if ( timeAppGapState != GAPROLE_CONNECTED )
491  {
492  uint8 advState;
493 
494  // Set fast advertising interval for user-initiated connections
495  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MIN, DEFAULT_FAST_ADV_INTERVAL );
496  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MAX, DEFAULT_FAST_ADV_INTERVAL );
497  GAP_SetParamValue( TGAP_GEN_DISC_ADV_MIN, DEFAULT_FAST_ADV_DURATION );
498 
499  // Toggle advertising state
500  GAPRole_GetParameter( GAPROLE_ADVERT_ENABLED, &advState );
501  advState = !advState;
502  GAPRole_SetParameter( GAPROLE_ADVERT_ENABLED, sizeof( uint8 ), &advState );
503 
504  // Set state variable
505  if (advState == FALSE)
506  {
507  timeAppAdvCancelled = TRUE;
508  }
509  }
510  }
511 
512  if ( keys & HAL_KEY_LEFT )
513  {
514  attWriteReq_t req;
515 
516  // Send command to alert notificaton control point
517  if ( ( timeAppGapState == GAPROLE_CONNECTED ) &&
518  ( timeAppHdlCache[HDL_ALERT_NTF_CTRL] != 0 ) )
519  {
520  // Send write request
521  req.len = 2;
522  req.value[0] = *pTimeAppAlertCmd;
523  req.value[1] = ALERT_NOTIF_CAT_ALL;
524  req.sig = 0;
525  req.cmd = 0;
526  req.handle = timeAppHdlCache[HDL_ALERT_NTF_CTRL];
527  GATT_WriteCharValue( timeAppConnHandle, &req, timeAppTaskId );
528  LCD_WRITE_STRING_VALUE( "Alert cmd:", *pTimeAppAlertCmd, 10, HAL_LCD_LINE_1);
529  }
530 
531  // Cycle through command test values
532  if ( *pTimeAppAlertCmd == ALERT_NOTIF_DISABLE_UNREAD )
533  {
534  pTimeAppAlertCmd = timeAppAlertCmd;
535  }
536  else
537  {
538  pTimeAppAlertCmd++;
539  }
540  }
541 
542  if ( keys & HAL_KEY_RIGHT )
543  {
544  attWriteReq_t req;
545 
546  // Do a reference time update
547  if ( ( timeAppGapState == GAPROLE_CONNECTED ) &&
548  ( timeAppHdlCache[HDL_REF_TIME_UPD_CTRL] != 0 ) )
549  {
550  // Send write command
551  req.len = 1;
552  req.value[0] = timeAppRefUpdateVal;
553  req.sig = 0;
554  req.cmd = 1;
555  req.handle = timeAppHdlCache[HDL_REF_TIME_UPD_CTRL];
556  GATT_WriteNoRsp( timeAppConnHandle, &req );
557 
558  LCD_WRITE_STRING_VALUE( "Time update:", timeAppRefUpdateVal, 10, HAL_LCD_LINE_1);
559 
560  // Toggle between two reference time update values
561  if ( timeAppRefUpdateVal == REF_TIME_UPDATE_GET )
562  {
563  timeAppRefUpdateVal = REF_TIME_UPDATE_CANCEL;
564  }
565  else
566  {
567  timeAppRefUpdateVal = REF_TIME_UPDATE_GET;
568  }
569  }
570  }
571 
572  if ( keys & HAL_KEY_CENTER )
573  {
574  // If connected, terminate connection
575  if ( timeAppGapState == GAPROLE_CONNECTED )
576  {
577  GAPRole_TerminateConnection();
578  }
579  }
580 
581  if ( keys & HAL_KEY_DOWN )
582  {
583  attWriteReq_t req;
584 
585  // Write ringer control point
586  if ( ( timeAppGapState == GAPROLE_CONNECTED ) &&
587  ( timeAppHdlCache[HDL_PAS_CTRL] != 0 ) )
588  {
589  // Send write command
590  req.len = 1;
591  req.value[0] = timeAppRingerCmd;
592  req.sig = 0;
593  req.cmd = 1;
594  req.handle = timeAppHdlCache[HDL_PAS_CTRL];
595  GATT_WriteNoRsp( timeAppConnHandle, &req );
596 
597  LCD_WRITE_STRING_VALUE( "Ringer ctrl:", timeAppRingerCmd, 10, HAL_LCD_LINE_1);
598 
599  // Toggle between values
600  if ( ++timeAppRingerCmd > RINGER_CANCEL_SILENT )
601  {
602  timeAppRingerCmd = RINGER_SILENT_MODE;
603  }
604  }
605  }
606 
607  if ( keys & HAL_KEY_SW_6 )
608  {
609 
610  }
611 
612 }

614 /*********************************************************************
615  * @fn timeAppProcessGattMsg
616  *
617  * @brief Process GATT messages
618  *
619  * @return none
620  */
621 static void timeAppProcessGattMsg( gattMsgEvent_t *pMsg )
622 {
623  if ( pMsg->method == ATT_HANDLE_VALUE_NOTI ||
624  pMsg->method == ATT_HANDLE_VALUE_IND )
625  {
626  timeAppIndGattMsg( pMsg );
627  }
628  else if ( ( pMsg->method == ATT_READ_RSP || pMsg->method == ATT_WRITE_RSP ) ||
629  ( pMsg->method == ATT_ERROR_RSP &&
630  ( pMsg->msg.errorRsp.reqOpcode == ATT_READ_REQ ||
631  pMsg->msg.errorRsp.reqOpcode == ATT_WRITE_REQ ) ) )
632  {
633  timeAppConfigState = timeAppConfigGattMsg ( timeAppConfigState, pMsg );
634  if ( timeAppConfigState == TIMEAPP_CONFIG_CMPL )
635  {
636  timeAppDiscoveryCmpl = TRUE;
637  }
638  }
639  else
640  {
641  timeAppDiscState = timeAppDiscGattMsg( timeAppDiscState, pMsg );
642  if ( timeAppDiscState == DISC_IDLE )
643  {
644  // Start characteristic configuration
645  timeAppConfigState = timeAppConfigNext( TIMEAPP_CONFIG_START );
646  }
647  }
648 }

665 /*********************************************************************

670  * @param newState - new state
671  *
672  * @return none
673  */
674 static void timeAppGapStateCB( gaprole_States_t newState )
675 {
676  // if connected
677  if ( newState == GAPROLE_CONNECTED )
678  {
679  linkDBItem_t *pItem;
680 
681  // Get connection handle
682  GAPRole_GetParameter( GAPROLE_CONNHANDLE, &timeAppConnHandle );
683 
684  // Get peer bd address
685  if ( (pItem = linkDB_Find( timeAppConnHandle )) != NULL)
686  {
687  // If connected to device without bond do service discovery
688  if ( !osal_memcmp( pItem->addr, timeAppBondedAddr, B_ADDR_LEN ) )
689  {
