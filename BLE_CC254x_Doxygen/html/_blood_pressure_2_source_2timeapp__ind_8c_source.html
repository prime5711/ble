94 /*********************************************************************
95  * @fn timeAppIndGattMsg
96  *
97  * @brief Handle indications and notifications.
98  *
99  * @param pMsg - GATT message.
100  *
101  * @return none
102  */
103 void timeAppIndGattMsg( gattMsgEvent_t *pMsg )
104 {
105  uint8 i;
106 
107  // Look up the handle in the handle cache
108  for ( i = 0; i < HDL_CACHE_LEN; i++ )
109  {
110  if ( pMsg->msg.handleValueNoti.handle == timeAppHdlCache[i] )
111  {
112  break;
113  }
114  }
115 
116  // Perform processing for this handle
117  switch ( i )
118  {
119  case HDL_CURR_TIME_CT_TIME_START:
120  // Set clock to time read from time server
121  timeAppClockSet( pMsg->msg.handleValueNoti.value );
122  break;
123 
124 
125  default:
126  break;
127  }
128 
129  // Send confirm for indication
130  if ( pMsg->method == ATT_HANDLE_VALUE_IND )
131  {
132  ATT_HandleValueCfm( pMsg->connHandle );
133  }
134 }
