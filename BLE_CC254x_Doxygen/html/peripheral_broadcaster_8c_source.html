178 /*********************************************************************
179  * @brief Set a GAP Role parameter.
180  *
181  * Public function defined in peripheral.h.
182  */
183 bStatus_t GAPRole_SetParameter( uint16 param, uint8 len, void *pValue )
184 {
185  bStatus_t ret = SUCCESS;
186  switch ( param )
187  {
188  case GAPROLE_IRK:
189  if ( len == KEYLEN )
190  {
191  VOID osal_memcpy( gapRole_IRK, pValue, KEYLEN ) ;
192  }
193  else
194  {
195  ret = bleInvalidRange;
196  }
197  break;
198 
199  case GAPROLE_SRK:
200  if ( len == KEYLEN )
201  {
202  VOID osal_memcpy( gapRole_SRK, pValue, KEYLEN ) ;
203  }
204  else
205  {
206  ret = bleInvalidRange;
207  }
208  break;
209 
210  case GAPROLE_SIGNCOUNTER:
211  if ( len == sizeof ( uint32 ) )
212  {
213  gapRole_signCounter = *((uint32*)pValue);
214  }
215  else
216  {
217  ret = bleInvalidRange;
218  }
219  break;
220 
221  case GAPROLE_ADVERT_ENABLED:
222  if ( len == sizeof( uint8 ) )
223  {
224  if ( (gapRole_state == GAPROLE_CONNECTED) || (gapRole_state == GAPROLE_CONNECTED_ADV) )
225  {
226  uint8 advEnabled = *((uint8*)pValue);
227 
228  if ( (gapRole_state == GAPROLE_CONNECTED) && (advEnabled == TRUE) )
229  {
230  // Turn on advertising
231  osal_set_event( gapRole_TaskID, START_ADVERTISING_EVT );
232  }
233  else if ( (gapRole_state == GAPROLE_CONNECTED_ADV) && (advEnabled == FALSE) )
234  {
235  // Turn off Advertising
236  GAP_EndDiscoverable( gapRole_TaskID );
237  }
238  }
239  else
240  {
241  uint8 oldAdvEnabled = gapRole_AdvEnabled;
242  gapRole_AdvEnabled = *((uint8*)pValue);
243 
244  if ( (oldAdvEnabled) && (gapRole_AdvEnabled == FALSE) )
245  {
246  // Turn off Advertising
247  VOID GAP_EndDiscoverable( gapRole_TaskID );
248  }
249  else if ( (oldAdvEnabled == FALSE) && (gapRole_AdvEnabled) )
250  {
251  // Turn on Advertising
252  if ( (gapRole_state == GAPROLE_STARTED)
253  || (gapRole_state == GAPROLE_WAITING )
254  || (gapRole_state == GAPROLE_WAITING_AFTER_TIMEOUT) )
255  {
256  VOID osal_set_event( gapRole_TaskID, START_ADVERTISING_EVT );
257  }
258  }
259  }
260  }
261  else
262  {
263  ret = bleInvalidRange;
264  }
265  break;
266 
267  case GAPROLE_ADVERT_OFF_TIME:
268  if ( len == sizeof ( uint16 ) )
269  {
270  gapRole_AdvertOffTime = *((uint16*)pValue);
271  }
272  else
273  {
274  ret = bleInvalidRange;
275  }
276  break;
277 
278  case GAPROLE_ADVERT_DATA:
279  if ( len <= B_MAX_ADV_LEN )
280  {
281  VOID osal_memset( gapRole_AdvertData, 0, B_MAX_ADV_LEN );
282  VOID osal_memcpy( gapRole_AdvertData, pValue, len );
283  gapRole_AdvertDataLen = len;
284  }
285  else
286  {
287  ret = bleInvalidRange;
288  }
289  break;
290 
291  case GAPROLE_SCAN_RSP_DATA:
292  if ( len <= B_MAX_ADV_LEN )
293  {
294  VOID osal_memset( gapRole_ScanRspData, 0, B_MAX_ADV_LEN );
295  VOID osal_memcpy( gapRole_ScanRspData, pValue, len );
296  gapRole_ScanRspDataLen = len;
297  }
298  else
299  {
300  ret = bleInvalidRange;
301  }
302  break;
303 
304  case GAPROLE_ADV_EVENT_TYPE:
305  if ( (len == sizeof ( uint8 )) && (*((uint8*)pValue) <= GAP_ADTYPE_ADV_LDC_DIRECT_IND) )
306  {
307  gapRole_AdvEventType = *((uint8*)pValue);
308  }
309  else
310  {
311  ret = bleInvalidRange;
312  }
313  break;
314 
315  case GAPROLE_ADV_DIRECT_TYPE:
316  if ( (len == sizeof ( uint8 )) && (*((uint8*)pValue) <= ADDRTYPE_PRIVATE_RESOLVE) )
317  {
318  gapRole_AdvDirectType = *((uint8*)pValue);
319  }
320  else
321  {
322  ret = bleInvalidRange;
323  }
324  break;
325 
326  case GAPROLE_ADV_DIRECT_ADDR:
327  if ( len == B_ADDR_LEN )
328  {
329  VOID osal_memcpy( gapRole_AdvDirectAddr, pValue, B_ADDR_LEN ) ;
330  }
331  else
332  {
333  ret = bleInvalidRange;
334  }
335  break;
336 
337  case GAPROLE_ADV_CHANNEL_MAP:
338  if ( (len == sizeof ( uint8 )) && (*((uint8*)pValue) <= 0x07) )
339  {
340  gapRole_AdvChanMap = *((uint8*)pValue);
341  }
342  else
343  {
344  ret = bleInvalidRange;
345  }
346  break;
347 
348  case GAPROLE_ADV_FILTER_POLICY:
349  if ( (len == sizeof ( uint8 )) && (*((uint8*)pValue) <= GAP_FILTER_POLICY_WHITE) )
350  {
351  gapRole_AdvFilterPolicy = *((uint8*)pValue);
352  }
353  else
354  {
355  ret = bleInvalidRange;
356  }
357  break;
358 
359  case GAPROLE_RSSI_READ_RATE:
360  if ( len == sizeof ( uint16 ) )
361  {
362  gapRole_RSSIReadRate = *((uint16*)pValue);
363 
364  if ( (gapRole_RSSIReadRate) && (gapRole_state == GAPROLE_CONNECTED) )
365  {
366  // Start the RSSI Reads
367  VOID osal_start_timerEx( gapRole_TaskID, RSSI_READ_EVT, gapRole_RSSIReadRate );
368  }
369  }
370  else
371  {
372  ret = bleInvalidRange;
373  }
374  break;
375 
376  case GAPROLE_PARAM_UPDATE_ENABLE:
377  if ( (len == sizeof ( uint8 )) && (*((uint8*)pValue) <= TRUE) )
378  {
379  gapRole_ParamUpdateEnable = *((uint8*)pValue);
380  }
381  else
382  {
383  ret = bleInvalidRange;
384  }
385  break;
386 
387  case GAPROLE_MIN_CONN_INTERVAL:
388  {
389  uint16 newInterval = *((uint16*)pValue);
390  if ( len == sizeof ( uint16 ) &&
391  ( newInterval >= MIN_CONN_INTERVAL ) &&
392  ( newInterval <= MAX_CONN_INTERVAL ) )
393  {
394  gapRole_MinConnInterval = newInterval;
395  }
396  else
397  {
398  ret = bleInvalidRange;
399  }
400  }
401  break;
402 
403  case GAPROLE_MAX_CONN_INTERVAL:
404  {
405  uint16 newInterval = *((uint16*)pValue);
406  if ( len == sizeof ( uint16 ) &&
407  ( newInterval >= MIN_CONN_INTERVAL) &&
408  ( newInterval <= MAX_CONN_INTERVAL) )
409  {
410  gapRole_MaxConnInterval = newInterval;
411  }
412  else
413  {
414  ret = bleInvalidRange;
415  }
416  }
417  break;
418 
419  case GAPROLE_SLAVE_LATENCY:
420  {
421  uint16 latency = *((uint16*)pValue);
422  if ( len == sizeof ( uint16 ) && (latency < MAX_SLAVE_LATENCY) )
423  {
424  gapRole_SlaveLatency = latency;
425  }
426  else
427  {
428  ret = bleInvalidRange;
429  }
430  }
431  break;
432 
433  case GAPROLE_TIMEOUT_MULTIPLIER:
434  {
435  uint16 newTimeout = *((uint16*)pValue);
436  if ( len == sizeof ( uint16 )
437  && (newTimeout >= MIN_TIMEOUT_MULTIPLIER) && (newTimeout <= MAX_TIMEOUT_MULTIPLIER) )
438  {
439  gapRole_TimeoutMultiplier = newTimeout;
440  }
441  else
442  {
443  ret = bleInvalidRange;
444  }
445  }
446  break;
447 
448  default:
449  // The param value isn't part of this profile, try the GAP.
450  if ( (param < TGAP_PARAMID_MAX) && (len == sizeof ( uint16 )) )
451  {
452  ret = GAP_SetParamValue( param, *((uint16*)pValue) );
453  }
454  else
455  {
456  ret = INVALIDPARAMETER;
457  }
458  break;
459  }
460 
461  return ( ret );
462 }

657 /*********************************************************************
658  * @brief Task Event Processor function.
659  *
660  * Internal function defined in peripheral.h.
661  */
662 uint16 GAPRole_ProcessEvent( uint8 task_id, uint16 events )
663 {
664  VOID task_id; // OSAL required parameter that isn't used in this function
665 
666  if ( events & SYS_EVENT_MSG )
667  {
668  uint8 *pMsg;
669 
670  if ( (pMsg = osal_msg_receive( gapRole_TaskID )) != NULL )
671  {
672  gapRole_ProcessOSALMsg( (osal_event_hdr_t *)pMsg );
673 
674  // Release the OSAL message
675  VOID osal_msg_deallocate( pMsg );
676  }
677 
678  // return unprocessed events
679  return (events ^ SYS_EVENT_MSG);
680  }
681 
682  if ( events & GAP_EVENT_SIGN_COUNTER_CHANGED )
683  {
684  // Sign counter changed, save it to NV
685  VOID osal_snv_write( BLE_NVID_SIGNCOUNTER, sizeof( uint32 ), &gapRole_signCounter );
686 
687  return ( events ^ GAP_EVENT_SIGN_COUNTER_CHANGED );
688  }
689 
690  if ( events & START_ADVERTISING_EVT )
691  {
692  if ( gapRole_AdvEnabled )
693  {
694  gapAdvertisingParams_t params;
695 
696  // Setup advertisement parameters
697  if ( gapRole_state == GAPROLE_CONNECTED )
698  {
699  // While in a connection, we can only advertise non-connectable undirected.
700  params.eventType = GAP_ADTYPE_ADV_NONCONN_IND;
701  }
702  else
703  {
704  params.eventType = gapRole_AdvEventType;
705  params.initiatorAddrType = gapRole_AdvDirectType;
706  VOID osal_memcpy( params.initiatorAddr, gapRole_AdvDirectAddr, B_ADDR_LEN );
707  }
708  params.channelMap = gapRole_AdvChanMap;
709  params.filterPolicy = gapRole_AdvFilterPolicy;
710 
711  if ( GAP_MakeDiscoverable( gapRole_TaskID, &params ) != SUCCESS )
712  {
713  gapRole_state = GAPROLE_ERROR;
714  if ( pGapRoles_AppCGs && pGapRoles_AppCGs->pfnStateChange )
715  {
716  pGapRoles_AppCGs->pfnStateChange( gapRole_state );
717  }
718  }
719  }
720  return ( events ^ START_ADVERTISING_EVT );

823  * @fn gapRole_ProcessGAPMsg
824  *
825  * @brief Process an incoming task message.
826  *

832 {

834 
835  switch ( pMsg->opcode )
836  {
837  case GAP_DEVICE_INIT_DONE_EVENT:
838  {
839  gapDeviceInitDoneEvent_t *pPkt = (gapDeviceInitDoneEvent_t *)pMsg;
840  bStatus_t stat = pPkt->hdr.status;
841 
842  if ( stat == SUCCESS )

845  VOID osal_snv_write( BLE_NVID_IRK, KEYLEN, gapRole_IRK );
846  VOID osal_snv_write( BLE_NVID_CSRK, KEYLEN, gapRole_SRK );

849  VOID osal_memcpy( gapRole_bdAddr, pPkt->devAddr, B_ADDR_LEN );
850 
851  gapRole_state = GAPROLE_STARTED;
852 
853  // Update the advertising data
854  stat = GAP_UpdateAdvertisingData( gapRole_TaskID,
855  TRUE, gapRole_AdvertDataLen, gapRole_AdvertData );
856  }
857 
858  if ( stat != SUCCESS )
859  {
860  gapRole_state = GAPROLE_ERROR;
861  }
862 
863  notify = TRUE;
864  }
865  break;
866 
867  case GAP_ADV_DATA_UPDATE_DONE_EVENT:
868  {
869  gapAdvDataUpdateEvent_t *pPkt = (gapAdvDataUpdateEvent_t *)pMsg;
870 
871  if ( pPkt->hdr.status == SUCCESS )
872  {
873  if ( pPkt->adType )
874  {

876  pPkt->hdr.status = GAP_UpdateAdvertisingData( gapRole_TaskID,
877  FALSE, gapRole_ScanRspDataLen, gapRole_ScanRspData );
878  }
879  else
880  {
881  // Start advertising
882  VOID osal_set_event( gapRole_TaskID, START_ADVERTISING_EVT );
883  }
884  }
885 
886  if ( pPkt->hdr.status != SUCCESS )
887  {
888  // Set into Error state
889  gapRole_state = GAPROLE_ERROR;
890  notify = TRUE;
891  }
892  }
893  break;
894 
895  case GAP_MAKE_DISCOVERABLE_DONE_EVENT:
896  case GAP_END_DISCOVERABLE_DONE_EVENT:
897  {
898  gapMakeDiscoverableRspEvent_t *pPkt = (gapMakeDiscoverableRspEvent_t *)pMsg;
899 
900  if ( pPkt->hdr.status == SUCCESS )
901  {
902  if ( pMsg->opcode == GAP_MAKE_DISCOVERABLE_DONE_EVENT )
903  {
904  if ( gapRole_state == GAPROLE_CONNECTED )
905  {
906  gapRole_state = GAPROLE_CONNECTED_ADV;
907  }
908  else
909  {
910  gapRole_state = GAPROLE_ADVERTISING;
911  }
912  }
913  else // GAP_END_DISCOVERABLE_DONE_EVENT
914  {
915 
916  if ( gapRole_AdvertOffTime != 0 )
917  {
918  if ( ( gapRole_AdvEnabled ) )
919  {
920  VOID osal_start_timerEx( gapRole_TaskID, START_ADVERTISING_EVT, gapRole_AdvertOffTime );
921  }
922  }
923  else
924  {
925  // Since gapRole_AdvertOffTime is set to 0, the device should not
926  // automatically become discoverable again after a period of time.
927  // Set enabler to FALSE; device will become discoverable again when
928  // this value gets set to TRUE
929  gapRole_AdvEnabled = FALSE;
930  }
931 
932  // In the Advertising Off period
933  gapRole_state = GAPROLE_WAITING;
934 
935  }
936  }
937  else
938  {
939  gapRole_state = GAPROLE_ERROR;
940  }
941  notify = TRUE;
942  }
943  break;
944 
945  case GAP_LINK_ESTABLISHED_EVENT:
946  {
947  gapEstLinkReqEvent_t *pPkt = (gapEstLinkReqEvent_t *)pMsg;
948 
949  if ( pPkt->hdr.status == SUCCESS )
950  {
951  VOID osal_memcpy( gapRole_ConnectedDevAddr, pPkt->devAddr, B_ADDR_LEN );
952  gapRole_ConnectionHandle = pPkt->connectionHandle;
953  gapRole_state = GAPROLE_CONNECTED;
954 
955  if ( gapRole_RSSIReadRate )
956  {
957  // Start the RSSI Reads
958  VOID osal_start_timerEx( gapRole_TaskID, RSSI_READ_EVT, gapRole_RSSIReadRate );
959  }
960 
961  // Check whether update parameter request is enabled, and check the connection parameters
962  if ( ( gapRole_ParamUpdateEnable == TRUE ) &&
963  ( (pPkt->connInterval < gapRole_MinConnInterval) ||
964  (pPkt->connInterval > gapRole_MaxConnInterval) ||

969  }

973  }
974  else
975  {
976  gapRole_state = GAPROLE_ERROR;
977  }
978  notify = TRUE;
979  }
980  break;
981

1011  }

1033  default:

1035  }
1036 
1037  if ( notify == TRUE )

1041  {

1065 /*********************************************************************
1066  * @fn gapRole_SendUpdateParam
1067  *
1068  * @brief Send an Update Connection Parameters.
1069  *
1070  * @param connInterval - current connection interval
1071  * @param connLatency - current connection latency
1072  *
1073  * @return none
1074  */

1076 {

1078  uint32 timeout; // Calculated response timeout
1079 
1080  // Calculate the current interval
1081  uint16 effectiveOldInterval = (connInterval * (connLatency + 1));
1082 
1083  // Calculate the interval we want
1084  uint16 effectiveNewMaxInterval = (gapRole_MaxConnInterval * (gapRole_SlaveLatency + 1));
1085

1088  updateReq.intervalMax = gapRole_MaxConnInterval;
1089  updateReq.slaveLatency = gapRole_SlaveLatency;
1090  updateReq.timeoutMultiplier = gapRole_TimeoutMultiplier;
1091 
1092  VOID L2CAP_ConnParamUpdateReq( gapRole_ConnectionHandle, &updateReq, gapRole_TaskID );
1093 
1094  // Set up the timeout for expected response
1095  if( effectiveOldInterval > effectiveNewMaxInterval )
1096  {
1097  timeout = (uint32)(effectiveOldInterval) * CONN_INTERVAL_MULTIPLIER;
1098  }
1099  else
1100  {
1101  timeout = (uint32)(effectiveNewMaxInterval) * CONN_INTERVAL_MULTIPLIER;
1102  }
1103 
1104  if( timeout > MAX_TIMEOUT_VALUE )
1105  {
1106  timeout = MAX_TIMEOUT_VALUE;
1107  }
1108 
1109  VOID osal_start_timerEx( gapRole_TaskID, UPDATE_PARAMS_TIMEOUT_EVT, (uint16)(timeout) );
