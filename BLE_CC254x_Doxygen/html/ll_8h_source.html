62 // check if connection parameter ranges for CI (min/max), SL, and LSTO are valid
63 #define LL_INVALID_CONN_TIME_PARAM( ciMin, ciMax, sl, lsto ) \
64  (((ciMin) < LL_CONN_INTERVAL_MIN) || \
65  ((ciMin) > LL_CONN_INTERVAL_MAX) || \
66  ((ciMax) < LL_CONN_INTERVAL_MIN) || \
67  ((ciMax) > LL_CONN_INTERVAL_MAX) || \
68  ((ciMax) < (ciMin)) || \
69  ((sl) > LL_SLAVE_LATENCY_MAX) || \
70  ((lsto) < LL_CONN_TIMEOUT_MIN) || \
71  ((lsto) > LL_CONN_TIMEOUT_MAX))
72 
73 // check if the CI/SL/LSTO combination is valid
74 // based on: LSTO > (1 + Slave Latency) * (Connection Interval * 2)
75 // Note: The CI * 2 requirement based on ESR05 V1.0, Erratum 3904.
76 // Note: LSTO time is normalized to units of 1.25ms (i.e. 10ms = 8 * 1.25ms).
77 #define LL_INVALID_CONN_TIME_PARAM_COMBO( ci, sl, lsto ) \
78  ((uint32)((lsto)*8) <= ((uint32)(1+(sl)) * (uint32)((ci)*2)))
