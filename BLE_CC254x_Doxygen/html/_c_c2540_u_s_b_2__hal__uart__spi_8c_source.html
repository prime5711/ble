6  Description: Implementation of a specialized SPI Transport driver for emulating a UART API.
7 
8 
9  The SPI frame format is
10  <SOF> <LEN> <DATA Byte 1> < DATA Byte2 >...<DATA Byte N> <FCS>
11 
12  5 GPIO lines are required: MOSI, MISO, SPI Clock, SPI CSn/MRDY, and SRDY.
13  MRDY/CS = SPI Master Ready to send and serves as a SPI Slave wakeup from sleep.
14  SRDY = SPI Slave Ready to send and serves as a SPI Master wakeup from sleep.
15 
16  The SRDY output from SPI Slave to SPI Master will transition from high to low
17  if the SPI slave receives an MRDY falling edge ISR from the SPI Master. But, if
18  the SPI slave has nothing to send, the SRDY line will be pulled high again.
19 
20  The SPI Slave does not re-enter sleep for the following conditions:
21  - Slave is asserting SRDY (i.e. it is waiting to send a SPI packet).
22  - Master is asserting MRDY.
23  - Slave has parsed the SOF byte of a SPI packet, but has not received the entire packet yet.
24  Note how important this case is to support a master that has to interrupt a SPI packet
25  transmission in order to service a higher priority SPI slave (i.e. the master has executed the
26  protocol to ensure that the SPI slave is awake, but cannot hold MRDY low for the entire,
27  contiguous packet for having to service some other SPI slave mid-packet).
28 
29  The incoming bytes may not be just garbage bytes clocked out by the synchronous transfer, but
30  instead be meaningful bytes of a Tx-packet from the other side.

385  * @fn HalUARTInitSPI

387  * @brief Initialize the SPI UART Transport.
388  *

399 static void HalUARTInitSPI(void)
400 {
401 #if (HAL_UART_SPI == 1)
402  PERCFG &= ~HAL_UART_PERCFG_BIT; /* Set UART0 I/O to Alt. 1 location on P0 */
403 #else
404  PERCFG |= HAL_UART_PERCFG_BIT; /* Set UART1 I/O to Alt. 2 location on P1 */
405 #endif
406 #if defined HAL_SPI_MASTER
407  PxSEL |= HAL_UART_Px_SEL_M; /* SPI-Master peripheral select */
408  UxCSR = 0; /* Mode is SPI-Master Mode */
409  UxGCR = 15; /* Cfg for the max Rx/Tx baud of 2-MHz */
410  UxBAUD = 255;
411 #elif !defined HAL_SPI_MASTER
412  PxSEL |= HAL_UART_Px_SEL_S; /* SPI-Slave peripheral select */
413  UxCSR = CSR_SLAVE; /* Mode is SPI-Slave Mode */
414 #endif
415  UxUCR = UCR_FLUSH; /* Flush it */
416  UxGCR |= BV(5); /* Set bit order to MSB */
417 
418  P2DIR &= ~P2DIR_PRIPO;
419  P2DIR |= HAL_UART_PRIPO;
420 
421  /* Setup GPIO for interrupts by falling edge on SPI_RDY_IN */
422  PxIEN |= SPI_RDYIn_BIT;
423  PICTL |= PICTL_BIT;
424 
425  SPI_CLR_RDY_OUT();
426  PxDIR |= SPI_RDYOut_BIT;
427 
428  /* Setup Tx by DMA */
429  halDMADesc_t *ch = HAL_DMA_GET_DESC1234( HAL_SPI_CH_TX );
430 
431  /* Abort any pending DMA operations (in case of a soft reset) */
432  HAL_DMA_ABORT_CH( HAL_SPI_CH_TX );
433 
434  /* The start address of the destination */
435  HAL_DMA_SET_DEST( ch, DMA_UxDBUF );
436 
437  /* Using the length field to determine how many bytes to transfer */
438  HAL_DMA_SET_VLEN( ch, HAL_DMA_VLEN_USE_LEN );
439 
440  /* One byte is transferred each time */
441  HAL_DMA_SET_WORD_SIZE( ch, HAL_DMA_WORDSIZE_BYTE );
442 
443  /* The bytes are transferred 1-by-1 on Tx Complete trigger */
444  HAL_DMA_SET_TRIG_MODE( ch, HAL_DMA_TMODE_SINGLE );
445  HAL_DMA_SET_TRIG_SRC( ch, DMATRIG_TX );
446 
447  /* The source address is incremented by 1 byte after each transfer */
448  HAL_DMA_SET_SRC_INC( ch, HAL_DMA_SRCINC_1 );
449  HAL_DMA_SET_SOURCE( ch, spiTxPkt );
450 
451  /* The destination address is constant - the Tx Data Buffer */
452  HAL_DMA_SET_DST_INC( ch, HAL_DMA_DSTINC_0 );
453 
454  /* The DMA Tx done is serviced by ISR */
455  HAL_DMA_SET_IRQ( ch, HAL_DMA_IRQMASK_ENABLE );
456 
457  /* Xfer all 8 bits of a byte xfer */
458  HAL_DMA_SET_M8( ch, HAL_DMA_M8_USE_8_BITS );
459 
460  /* DMA has highest priority for memory access */
461  HAL_DMA_SET_PRIORITY( ch, HAL_DMA_PRI_HIGH );
462 
463  /* Setup Rx by DMA */
464  ch = HAL_DMA_GET_DESC1234( HAL_SPI_CH_RX );
465 
466  /* Abort any pending DMA operations (in case of a soft reset) */
467  HAL_DMA_ABORT_CH( HAL_SPI_CH_RX );
468 
469  /* The start address of the source */
470  HAL_DMA_SET_SOURCE( ch, DMA_UxDBUF );
471 
472  /* Using the length field to determine how many bytes to transfer */
473  HAL_DMA_SET_VLEN( ch, HAL_DMA_VLEN_USE_LEN );
474 
475  /* The trick is to cfg DMA to xfer 2 bytes for every 1 byte of Rx.
476  * The byte after the Rx Data Buffer is the Baud Cfg Register,
477  * which always has a known value. So init Rx buffer to inverse of that
478  * known value. DMA word xfer will flip the bytes, so every valid Rx byte
479  * in the Rx buffer will be preceded by a DMA_PAD char equal to the
480  * Baud Cfg Register value.
481  */
482  HAL_DMA_SET_WORD_SIZE( ch, HAL_DMA_WORDSIZE_WORD );
483 
484  /* The bytes are transferred 1-by-1 on Rx Complete trigger */
485  HAL_DMA_SET_TRIG_MODE( ch, HAL_DMA_TMODE_SINGLE_REPEATED );
486  HAL_DMA_SET_TRIG_SRC( ch, DMATRIG_RX );
487 
488  /* The source address is constant - the Rx Data Buffer */
489  HAL_DMA_SET_SRC_INC( ch, HAL_DMA_SRCINC_0 );
490 
491  /* The destination address is incremented by 1 word after each transfer */
492  HAL_DMA_SET_DST_INC( ch, HAL_DMA_DSTINC_1 );
493  HAL_DMA_SET_DEST( ch, spiRxBuf );
494  HAL_DMA_SET_LEN( ch, SPI_MAX_PKT_LEN );
495 
496  /* The DMA is to be polled and shall not issue an IRQ upon completion */
497  HAL_DMA_SET_IRQ( ch, HAL_DMA_IRQMASK_DISABLE );
498 
499  /* Xfer all 8 bits of a byte xfer */
500  HAL_DMA_SET_M8( ch, HAL_DMA_M8_USE_8_BITS );
501 
502  /* DMA has highest priority for memory access */
503  HAL_DMA_SET_PRIORITY( ch, HAL_DMA_PRI_HIGH );
504 
505  volatile uint8 dummy = *(volatile uint8 *)DMA_UxDBUF; /* Clear the DMA Rx trigger */
506  HAL_DMA_CLEAR_IRQ(HAL_SPI_CH_RX);
507  HAL_DMA_ARM_CH(HAL_SPI_CH_RX);
508  (void)memset(spiRxBuf, (DMA_PAD ^ 0xFF), SPI_MAX_PKT_LEN * sizeof(uint16));
509 }

567  }

822 /**************************************************************************************************
823  * @fn spiParseRx
824  *
825  * @brief Parse all available bytes from the spiRxBuf[]; parse Rx data into the spiRxDat[].
826  *
827  * input parameters
828  *
829  * None.
830  *
831  * output parameters
832  *
833  * None.
834  *
835  * @return None.
836  */
837 static void spiParseRx(void)
838 {
839  while (1)
840  {
841  if (!SPI_NEW_RX_BYTE(spiRxIdx))
842  {
843 #if defined HAL_SPI_MASTER
844  if (SPI_RDY_IN() && (spiTxLen == 0))
845  {
846  SPI_CLOCK_RX(1);
847  continue;
848  }
849 #endif
850  break;
851  }
852 
853  uint8 ch = SPI_GET_RX_BYTE(spiRxIdx);
854  SPI_CLR_RX_BYTE(spiRxIdx);
855  SPI_LEN_T_INCR(spiRxIdx);
856 
857  switch (spiRxPktState)
858  {
859  case SPIRX_STATE_SOF:
860  if (ch == SPI_SOF)
861  {
862  spiRxPktState = SPIRX_STATE_LEN;
863 
864  /* At this point, the master has effected the protocol for ensuring that the SPI slave is
865  * awake, so set the spiRxLen to non-zero to prevent the slave from re-entering sleep until
866  * the entire packet is received - even if the master interrupts the sending of the packet
867  * by de-asserting/re-asserting MRDY one or more times
868  */
869  spiRxLen = 1;
870  }
871  break;
872 
873  case SPIRX_STATE_LEN:
874  if ((ch == 0) || (ch > SPI_MAX_DAT_LEN))
875  {
876  spiRxPktState = SPIRX_STATE_SOF;
877  spiRxLen = 0;
878  }
879  else
880  {
881  spiRxFcs = spiRxLen = ch;
882  spiRxTemp = spiRxTail;
883  spiRxCnt = 0;
884  spiRxPktState = SPIRX_STATE_DATA;
885 #if defined HAL_SPI_MASTER
886  if (!SPI_NEW_RX_BYTE(spiRxIdx)) /* Fix for simultaneous TX/RX to avoid extra clock pulses to SPI Slave */
887  {
888  halIntState_t intState;
889  HAL_ENTER_CRITICAL_SECTION(intState);
890  SPI_CLOCK_RX(ch + 1); /* Clock out the SPI Frame Data bytes and FCS */
891  HAL_EXIT_CRITICAL_SECTION(intState);
892  }
893 #endif
894  }
895  break;
896 
897  case SPIRX_STATE_DATA:
898  spiRxFcs ^= ch;
899  spiRxDat[spiRxTemp] = ch;
900 
901  SPI_LEN_T_INCR(spiRxTemp);
902 
903  if (++spiRxCnt == spiRxLen)
904  {
905  spiRxPktState = SPIRX_STATE_FCS;
906  }
907  break;
908 
909  case SPIRX_STATE_FCS:
910  spiRxPktState = SPIRX_STATE_SOF;
911 #ifdef POWER_SAVING
912  pktFound = TRUE;
913 #endif
914  SPI_CLR_RDY_OUT(); /* SPI_RDYOut = 1 */
915 
916  if (ch == spiRxFcs)
917  {
918  spiRxTail = spiRxTemp;
919  }
920  else
921  {
922  dbgFcsByte = ch;
923 #ifdef RBA_UART_TO_SPI
924  badFcsPktCount++;
925 #endif
926  }
927  spiRxCnt = spiRxLen = 0;
928  break;
929 
930  default:
931  HAL_ASSERT(0);
932  break;
933  }
934  }
