357 /*********************************************************************
358  * @fn scanParamWriteAttrCB
359  *
360  * @brief Validate attribute data prior to a write operation
361  *
362  * @param connHandle - connection message was received on
363  * @param pAttr - pointer to attribute
364  * @param pValue - pointer to data to be written
365  * @param len - length of data
366  * @param offset - offset of the first octet to be written
367  *
368  * @return Success or Failure
369  */
370 static bStatus_t scanParamWriteAttrCB( uint16 connHandle, gattAttribute_t *pAttr,
371  uint8 *pValue, uint8 len, uint16 offset )
372 {
373  bStatus_t status = SUCCESS;
374 
375  // Make sure it's not a blob operation (no attributes in the profile are long)
376  if ( offset > 0 )
377  {
378  return ( ATT_ERR_ATTR_NOT_LONG );
379  }
380 
381  uint16 uuid = BUILD_UINT16( pAttr->type.uuid[0], pAttr->type.uuid[1]);
382 
383  // Only one writeable attribute
384  if ( uuid == SCAN_INTERVAL_WINDOW_UUID )
385  {
386  if ( len == SCAN_INTERVAL_WINDOW_CHAR_LEN )
387  {
388  uint16 interval = BUILD_UINT16( pValue[0], pValue[1] );
389  uint16 window = BUILD_UINT16( pValue[0], pValue[1] );
390 
391  // Validate values
392  if ( window <= interval )
393  {
394  osal_memcpy( pAttr->pValue, pValue, len );
395 
396  (*scanParamServiceCB)( SCAN_INTERVAL_WINDOW_SET );
397  }
398  else
399  {
400  status = ATT_ERR_INVALID_VALUE;
401  }
402  }
403  else
404  {
405  status = ATT_ERR_INVALID_VALUE_SIZE;
406  }
407  }
408  else if ( uuid == GATT_CLIENT_CHAR_CFG_UUID )
409  {
410  status = GATTServApp_ProcessCCCWriteReq( connHandle, pAttr, pValue, len,
411  offset, GATT_CLIENT_CFG_NOTIFY );
412  }
413  else
414  {
415  status = ATT_ERR_ATTR_NOT_FOUND;
416  }
417 
418  return ( status );
419 }

421 /*********************************************************************
422  * @fn ScanParam_HandleConnStatusCB
423  *
424  * @brief Service link status change handler function.
425  *
426  * @param connHandle - connection handle
427  * @param changeType - type of change
428  *
429  * @return none
430  */
431 void ScanParam_HandleConnStatusCB( uint16 connHandle, uint8 changeType )
432 {
433  // Make sure this is not loopback connection
434  if ( connHandle != LOOPBACK_CONNHANDLE )
435  {
436  // Reset Client Char Config if connection has dropped
437  if ( ( changeType == LINKDB_STATUS_UPDATE_REMOVED ) ||
438  ( ( changeType == LINKDB_STATUS_UPDATE_STATEFLAGS ) &&
439  ( !linkDB_Up( connHandle ) ) ) )
440  {
441  GATTServApp_InitCharCfg( connHandle, scanParamRefreshClientCharCfg );
442  }
