495 /*********************************************************************
496  * @fn SensorTag_ProcessEvent
497  *
498  * @brief Simple BLE Peripheral Application Task event processor. This function
499  * is called to process all events for the task. Events
500  * include timers, messages and any other user defined events.
501  *
502  * @param task_id - The OSAL assigned task ID.
503  * @param events - events to process. This is a bit map and can
504  * contain more than one event.
505  *
506  * @return events not processed
507  */
508 uint16 SensorTag_ProcessEvent( uint8 task_id, uint16 events )
509 {
510  VOID task_id; // OSAL required parameter that isn't used in this function
511 
512  if ( events & SYS_EVENT_MSG )
513  {
514  uint8 *pMsg;
515 
516  if ( (pMsg = osal_msg_receive( sensorTag_TaskID )) != NULL )
517  {
518  sensorTag_ProcessOSALMsg( (osal_event_hdr_t *)pMsg );
519 
520  // Release the OSAL message
521  VOID osal_msg_deallocate( pMsg );
522  }
523 
524  // return unprocessed events
525  return (events ^ SYS_EVENT_MSG);
526  }
527 
528  // Handle system reset (long press on side key)
529  if ( events & ST_SYS_RESET_EVT )
530  {
531  if (sysResetRequest)
532  {
533  HAL_SYSTEM_RESET();
534  }
535  return ( events ^ ST_SYS_RESET_EVT );
536  }
537 
538  if ( events & ST_START_DEVICE_EVT )
539  {
540  // Start the Device
541  VOID GAPRole_StartDevice( &sensorTag_PeripheralCBs );
542 
543  // Start Bond Manager
544  VOID GAPBondMgr_Register( &sensorTag_BondMgrCBs );
545 
546  return ( events ^ ST_START_DEVICE_EVT );
547  }
548 
550  // IR TEMPERATURE //
552  if ( events & ST_IRTEMPERATURE_READ_EVT )
553  {
554  if ( irTempEnabled )
555  {
556  if (HalIRTempStatus() == TMP006_DATA_READY)
557  {
558  readIrTempData();
559  osal_start_timerEx( sensorTag_TaskID, ST_IRTEMPERATURE_READ_EVT, sensorTmpPeriod-TEMP_MEAS_DELAY );
560  }
561  else if (HalIRTempStatus() == TMP006_OFF)
562  {
563  HalIRTempTurnOn();
564  osal_start_timerEx( sensorTag_TaskID, ST_IRTEMPERATURE_READ_EVT, TEMP_MEAS_DELAY );
565  }
566  }
567  else
568  {
569  //Turn off Temperatur sensor
570  VOID HalIRTempTurnOff();
571  VOID resetCharacteristicValue(IRTEMPERATURE_SERV_UUID,SENSOR_DATA,0,IRTEMPERATURE_DATA_LEN);
572  VOID resetCharacteristicValue(IRTEMPERATURE_SERV_UUID,SENSOR_CONF,ST_CFG_SENSOR_DISABLE,sizeof ( uint8 ));
573  }
574 
575  return (events ^ ST_IRTEMPERATURE_READ_EVT);
576  }
577 
579  // Accelerometer //
581  if ( events & ST_ACCELEROMETER_SENSOR_EVT )
582  {
583  if(accConfig != ST_CFG_SENSOR_DISABLE)
584  {
585  readAccData();
586  osal_start_timerEx( sensorTag_TaskID, ST_ACCELEROMETER_SENSOR_EVT, sensorAccPeriod );
587  }
588  else
589  {
590  VOID resetCharacteristicValue( ACCELEROMETER_SERV_UUID, SENSOR_DATA, 0, ACCELEROMETER_DATA_LEN );
591  VOID resetCharacteristicValue( ACCELEROMETER_SERV_UUID, SENSOR_CONF, ST_CFG_SENSOR_DISABLE, sizeof ( uint8 ));
592  }
593 
594  return (events ^ ST_ACCELEROMETER_SENSOR_EVT);
595  }
596 
598  // Humidity //
600  if ( events & ST_HUMIDITY_SENSOR_EVT )
601  {
602  if (humiEnabled)
603  {
604  HalHumiExecMeasurementStep(humiState);
605  if (humiState == 2)
606  {
607  readHumData();
608  humiState = 0;
609  osal_start_timerEx( sensorTag_TaskID, ST_HUMIDITY_SENSOR_EVT, sensorHumPeriod );
610  }
611  else
612  {
613  humiState++;
614  osal_start_timerEx( sensorTag_TaskID, ST_HUMIDITY_SENSOR_EVT, HUM_FSM_PERIOD );
615  }
616  }
617  else
618  {
619  resetCharacteristicValue( HUMIDITY_SERV_UUID, SENSOR_DATA, 0, HUMIDITY_DATA_LEN);
620  resetCharacteristicValue( HUMIDITY_SERV_UUID, SENSOR_CONF, ST_CFG_SENSOR_DISABLE, sizeof ( uint8 ));
621  }
622 
623  return (events ^ ST_HUMIDITY_SENSOR_EVT);
624  }
625 
627  // Magnetometer //
629  if ( events & ST_MAGNETOMETER_SENSOR_EVT )
630  {
631  if(magEnabled)
632  {
633  if (HalMagStatus() == MAG3110_DATA_READY)
634  {
635  readMagData();
636  }
637  else if (HalMagStatus() == MAG3110_OFF)
638  {
639  HalMagTurnOn();
640  }
641 
642  osal_start_timerEx( sensorTag_TaskID, ST_MAGNETOMETER_SENSOR_EVT, sensorMagPeriod );
643  }
644  else
645  {
646  HalMagTurnOff();
