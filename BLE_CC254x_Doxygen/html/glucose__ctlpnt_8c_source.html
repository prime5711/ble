109 /*********************************************************************
110  * @fn glucoseCtlPntWriteFilter
111  *
112  * @brief Write Control Point Filter Requests
113  *
114  * @param opcode - control point opcode
115  * @param oper - control point operator
116  * @param filterType - control point filter type
117  * @param param1 - first filter
118  * @param param2 - second filter (if applicable), otherwise NULL
119 
120  *
121  * @return status of write
122  */
123 
124 uint8 glucoseCtlPntWriteFilter(uint8 opcode, uint8 oper, uint8 filterType,
125  void* param1, void* param2)
126 {
127  attWriteReq_t writeReq;
128  UTCTimeStruct *time1, *time2;
129  uint16 *seqNum1, *seqNum2;
130 
131  uint8 *p = writeReq.value;
132 
133  *p++ = opcode;
134  *p++ = oper;
135 
136  // The operator will tells us whether to include the filters or not
137  // Note day and month are converted to date time struct values
138  switch(oper)
139  {
140  case CTL_PNT_OPER_LESS_EQUAL:
141  case CTL_PNT_OPER_GREATER_EQUAL:
142  *p++ = filterType;
143  if (filterType == CTL_PNT_FILTER_SEQNUM)
144  {
145  seqNum1 = param1;
146  *p++ = LO_UINT16(*seqNum1);
147  *p++ = HI_UINT16(*seqNum1);
148  }
149  else
150  {
151  time1 = param1;
152  *p++ = LO_UINT16(time1->year);
153  *p++ = HI_UINT16(time1->year);
154  *p++ = (time1->month + 1);
155  *p++ = (time1->day + 1);
156  *p++ = time1->hour;
157  *p++ = time1->minutes;
158  *p++ = time1->seconds;
159  }
160  break;
161  case CTL_PNT_OPER_RANGE:
162  *p++ = filterType;
163  if (filterType == CTL_PNT_FILTER_SEQNUM)
164  {
165  seqNum1 = param1;
166  seqNum2 = param2;
167  *p++ = LO_UINT16(*seqNum1);
168  *p++ = HI_UINT16(*seqNum1);
169  *p++ = LO_UINT16(*seqNum2);
170  *p++ = HI_UINT16(*seqNum2);
171  }
172  else
173  {
174  time1 = param1;
175  time2 = param2;
176  *p++ = LO_UINT16(time1->year);
177  *p++ = HI_UINT16(time1->year);
178  *p++ = (time1->month + 1);
179  *p++ = (time1->day + 1);
180  *p++ = time1->hour;
181  *p++ = time1->minutes;
182  *p++ = time1->seconds;
183 
184  *p++ = LO_UINT16(time2->year);
185  *p++ = HI_UINT16(time2->year);
186  *p++ = (time2->month + 1);
187  *p++ = (time2->day + 1);
188  *p++ = time2->hour;
189  *p++ = time2->minutes;
190  *p++ = time2->seconds;
191  }
192  break;
193  default:
194  break;
195  }
196 
197  writeReq.len = (p - writeReq.value);
198  writeReq.sig = 0;
199  writeReq.cmd = 0;
200 
201  writeReq.handle = glucoseHdlCache[HDL_GLUCOSE_CTL_PNT_START];
202  return GATT_WriteCharValue( glucCollConnHandle, &writeReq, glucCollTaskId );
203 }
