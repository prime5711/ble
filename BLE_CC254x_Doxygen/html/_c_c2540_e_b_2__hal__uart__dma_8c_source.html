361 /******************************************************************************
362  * @fn HalUARTInitDMA
363  *
364  * @brief Initialize the UART
365  *
366  * @param none
367  *
368  * @return none
369  *****************************************************************************/
370 static void HalUARTInitDMA(void)
371 {
372  halDMADesc_t *ch;
373 #if (HAL_UART_DMA == 1)
374  PERCFG &= ~HAL_UART_PERCFG_BIT; // Set UART0 I/O to Alt. 1 location on P0.
375 #else
376  PERCFG |= HAL_UART_PERCFG_BIT; // Set UART1 I/O to Alt. 2 location on P1.
377 #endif
378  PxSEL |= HAL_UART_Px_SEL; // Enable Peripheral control of Rx/Tx on Px.
379  UxCSR = CSR_MODE; // Mode is UART Mode.
380  UxUCR = UCR_FLUSH; // Flush it.
381 
382  P2DIR &= ~P2DIR_PRIPO;
383  P2DIR |= HAL_UART_PRIPO;
384 
385  if (DMA_PM)
386  {
387  // Setup GPIO for interrupts by falling edge on DMA_RDY_IN.
388  PxIEN |= DMA_RDYIn_BIT;
389  PICTL |= PICTL_BIT;
390 
391  HAL_UART_DMA_CLR_RDY_OUT();
392  PxDIR |= DMA_RDYOut_BIT;
393  }
394 
395 #if !HAL_UART_TX_BY_ISR
396  // Setup Tx by DMA.
397  ch = HAL_DMA_GET_DESC1234( HAL_DMA_CH_TX );
398 
399  // Abort any pending DMA operations (in case of a soft reset).
400  HAL_DMA_ABORT_CH( HAL_DMA_CH_TX );
401 
402  // The start address of the destination.
403  HAL_DMA_SET_DEST( ch, DMA_UxDBUF );
404 
405  // Using the length field to determine how many bytes to transfer.
406  HAL_DMA_SET_VLEN( ch, HAL_DMA_VLEN_USE_LEN );
407 
408  // One byte is transferred each time.
409  HAL_DMA_SET_WORD_SIZE( ch, HAL_DMA_WORDSIZE_BYTE );
410 
411  // The bytes are transferred 1-by-1 on Tx Complete trigger.
412  HAL_DMA_SET_TRIG_MODE( ch, HAL_DMA_TMODE_SINGLE );
413  HAL_DMA_SET_TRIG_SRC( ch, DMATRIG_TX );
414 
415  // The source address is incremented by 1 byte after each transfer.
416  HAL_DMA_SET_SRC_INC( ch, HAL_DMA_SRCINC_1 );
417 
418  // The destination address is constant - the Tx Data Buffer.
419  HAL_DMA_SET_DST_INC( ch, HAL_DMA_DSTINC_0 );
420 
421  // The DMA Tx done is serviced by ISR in order to maintain full thruput.
422  HAL_DMA_SET_IRQ( ch, HAL_DMA_IRQMASK_ENABLE );
423 
424  // Xfer all 8 bits of a byte xfer.
425  HAL_DMA_SET_M8( ch, HAL_DMA_M8_USE_8_BITS );
426 
427  // DMA has highest priority for memory access.
428  HAL_DMA_SET_PRIORITY( ch, HAL_DMA_PRI_HIGH);
429 #endif
430 
431  // Setup Rx by DMA.
432  ch = HAL_DMA_GET_DESC1234( HAL_DMA_CH_RX );
433 
434  // Abort any pending DMA operations (in case of a soft reset).
435  HAL_DMA_ABORT_CH( HAL_DMA_CH_RX );
436 
437  // The start address of the source.
438  HAL_DMA_SET_SOURCE( ch, DMA_UxDBUF );
439 
440  // Using the length field to determine how many bytes to transfer.
441  HAL_DMA_SET_VLEN( ch, HAL_DMA_VLEN_USE_LEN );
442 
443  /* The trick is to cfg DMA to xfer 2 bytes for every 1 byte of Rx.
444  * The byte after the Rx Data Buffer is the Baud Cfg Register,
445  * which always has a known value. So init Rx buffer to inverse of that
446  * known value. DMA word xfer will flip the bytes, so every valid Rx byte
447  * in the Rx buffer will be preceded by a DMA_PAD char equal to the
448  * Baud Cfg Register value.
449  */
450  HAL_DMA_SET_WORD_SIZE( ch, HAL_DMA_WORDSIZE_WORD );
451 
452  // The bytes are transferred 1-by-1 on Rx Complete trigger.
453  HAL_DMA_SET_TRIG_MODE( ch, HAL_DMA_TMODE_SINGLE_REPEATED );
454  HAL_DMA_SET_TRIG_SRC( ch, DMATRIG_RX );
455 
456  // The source address is constant - the Rx Data Buffer.
457  HAL_DMA_SET_SRC_INC( ch, HAL_DMA_SRCINC_0 );
458 
459  // The destination address is incremented by 1 word after each transfer.
460  HAL_DMA_SET_DST_INC( ch, HAL_DMA_DSTINC_1 );
461  HAL_DMA_SET_DEST( ch, dmaCfg.rxBuf );
462  HAL_DMA_SET_LEN( ch, HAL_UART_DMA_RX_MAX );
463 
464  // The DMA is to be polled and shall not issue an IRQ upon completion.
465  HAL_DMA_SET_IRQ( ch, HAL_DMA_IRQMASK_DISABLE );
466 
467  // Xfer all 8 bits of a byte xfer.
468  HAL_DMA_SET_M8( ch, HAL_DMA_M8_USE_8_BITS );
469 
470  // DMA has highest priority for memory access.
471  HAL_DMA_SET_PRIORITY( ch, HAL_DMA_PRI_HIGH);
472 
473  volatile uint8 dummy = *(volatile uint8 *)DMA_UxDBUF; // Clear the DMA Rx trigger.
474  HAL_DMA_CLEAR_IRQ(HAL_DMA_CH_RX);
475  HAL_DMA_ARM_CH(HAL_DMA_CH_RX);
476  (void)memset(dmaCfg.rxBuf, (DMA_PAD ^ 0xFF), HAL_UART_DMA_RX_MAX * sizeof(uint16));
477 }

479 /******************************************************************************
480  * @fn HalUARTOpenDMA
481  *
482  * @brief Open a port according tp the configuration specified by parameter.
483  *
484  * @param config - contains configuration information
485  *
486  * @return none
487  *****************************************************************************/
488 static void HalUARTOpenDMA(halUARTCfg_t *config)
489 {
490  dmaCfg.uartCB = config->callBackFunc;
491 
492  // Only supporting subset of baudrate for code size - other is possible.
493  HAL_ASSERT((config->baudRate == HAL_UART_BR_9600) ||
494  (config->baudRate == HAL_UART_BR_19200) ||
495  (config->baudRate == HAL_UART_BR_38400) ||
496  (config->baudRate == HAL_UART_BR_57600) ||
497  (config->baudRate == HAL_UART_BR_115200));
498 
499  if (config->baudRate == HAL_UART_BR_57600 ||
500  config->baudRate == HAL_UART_BR_115200)
501  {
502  UxBAUD = 216;
503  }
504  else
505  {
506  UxBAUD = 59;
507  }
508 
509  switch (config->baudRate)
510  {
511  case HAL_UART_BR_9600:
512  UxGCR = 8;
513  break;
514  case HAL_UART_BR_19200:
515  UxGCR = 9;
516  break;
517  case HAL_UART_BR_38400:
518  case HAL_UART_BR_57600:
519  UxGCR = 10;
520  break;
521  default:
522  // HAL_UART_BR_115200
523  UxGCR = 11;
524  break;
525  }
526 
527  if (DMA_PM || config->flowControl)
528  {
529  UxUCR = UCR_FLOW | UCR_STOP; // 8 bits/char; no parity; 1 stop bit; stop bit hi.
530  PxSEL |= HAL_UART_Px_CTS; // Enable Peripheral control of CTS flow control on Px.
531  }
532  else
533  {
534  UxUCR = UCR_STOP; // 8 bits/char; no parity; 1 stop bit; stop bit hi.
535  }
536 
537  UxCSR = (CSR_MODE | CSR_RE);
538 
539  if (DMA_PM)
540  {
541  PxIFG = 0;
542  PxIF = 0;
543  IENx |= IEN_BIT;
544  }
545  else if (UxUCR & UCR_FLOW)
546  {
547  // DMA Rx is always on (self-resetting). So flow must be controlled by the S/W polling the
548  // circular Rx queue depth. Start by allowing flow.
549  HAL_UART_DMA_SET_RDY_OUT();
550  PxDIR |= HAL_UART_Px_RTS;
551  }
552 
553 #if HAL_UART_TX_BY_ISR
554  UTXxIF = 1; // Prime the ISR pump.
555 #endif
556 }

558 /*****************************************************************************
559  * @fn HalUARTReadDMA
560  *
561  * @brief Read a buffer from the UART
562  *
563  * @param buf - valid data buffer at least 'len' bytes in size
564  * len - max length number of bytes to copy to 'buf'
565  *
566  * @return length of buffer that was read
567  *****************************************************************************/
568 static uint16 HalUARTReadDMA(uint8 *buf, uint16 len)
569 {
570  uint16 cnt;
571 
572  for (cnt = 0; cnt < len; cnt++)
573  {
574  if (!HAL_UART_DMA_NEW_RX_BYTE(dmaCfg.rxHead))
575  {
576  break;
577  }
578  *buf++ = HAL_UART_DMA_GET_RX_BYTE(dmaCfg.rxHead);
579  HAL_UART_DMA_CLR_RX_BYTE(dmaCfg.rxHead);
580  HAL_UART_RX_IDX_T_INCR(dmaCfg.rxHead);
581  }
582 
583  /* Update pointers after reading the bytes */
584  dmaCfg.rxTail = dmaCfg.rxHead;
585 
586  if (!DMA_PM && (UxUCR & UCR_FLOW))
587  {
588  HAL_UART_DMA_SET_RDY_OUT(); // Re-enable the flow asap (i.e. not wait until next uart poll).
589  }
590 
591  return cnt;
592 }

595  * @fn HalUARTWriteDMA
596  *
597  * @brief Write a buffer to the UART, enforcing an all or none policy if the requested length
598  * exceeds the space available.
599  *
600  * @param buf - pointer to the buffer that will be written, not freed
601  * len - length of
602  *
603  * @return length of the buffer that was sent
604  *****************************************************************************/
605 static uint16 HalUARTWriteDMA(uint8 *buf, uint16 len)
606 {
607 #if HAL_UART_TX_BY_ISR
608  // Enforce all or none.
609  if (HAL_UART_DMA_TX_AVAIL() < len)
610  {
611  return 0;
612  }
613 
614  for (uint16 cnt = 0; cnt < len; cnt++)
615  {
616  dmaCfg.txBuf[dmaCfg.txTail] = *buf++;
617  dmaCfg.txMT = 0;
618 
619  if (dmaCfg.txTail >= HAL_UART_DMA_TX_MAX-1)
620  {
621  dmaCfg.txTail = 0;
622  }
623  else
624  {
625  dmaCfg.txTail++;
626  }
627 
628  // Keep re-enabling ISR as it might be keeping up with this loop due to other ints.
629  IEN2 |= UTXxIE;
630  }
631 #else
632  txIdx_t txIdx;
633  uint8 txSel;
634  halIntState_t his;
635 
636  HAL_ENTER_CRITICAL_SECTION(his);
637  txSel = dmaCfg.txSel;
638  txIdx = dmaCfg.txIdx[txSel];
639  HAL_EXIT_CRITICAL_SECTION(his);
640 
641  // Enforce all or none.
642  if ((len + txIdx) > HAL_UART_DMA_TX_MAX)
643  {
644  return 0;
645  }
646 
647  (void)memcpy(&(dmaCfg.txBuf[txSel][txIdx]), buf, len);
648 
649  HAL_ENTER_CRITICAL_SECTION(his);
650  /* If an ongoing DMA Tx finished while this buffer was being *appended*, then another DMA Tx
651  * will have already been started on this buffer, but it did not include the bytes just appended.
652  * Therefore these bytes have to be re-copied to the start of the new working buffer.
653  */
654  if (txSel != dmaCfg.txSel)
655  {
656  HAL_EXIT_CRITICAL_SECTION(his);
657  txSel ^= 1;
658 
659  (void)memcpy(&(dmaCfg.txBuf[txSel][0]), buf, len);
660  HAL_ENTER_CRITICAL_SECTION(his);
661  dmaCfg.txIdx[txSel] = len;
662  }
663  else
664  {
665  dmaCfg.txIdx[txSel] = txIdx + len;
666  }
667 
668  // If there is no ongoing DMA Tx, then the channel must be armed here.
669  if (dmaCfg.txIdx[(txSel ^ 1)] == 0)
670  {
671  HAL_EXIT_CRITICAL_SECTION(his);
672  HalUARTArmTxDMA();
673  }
674  else
675  {
676  dmaCfg.txMT = FALSE;
677  HAL_EXIT_CRITICAL_SECTION(his);
678  }
679 #endif
680 
681  return len;
682 }

684 /******************************************************************************
685  * @fn HalUARTPollDMA
686  *
687  * @brief Poll a USART module implemented by DMA, including the hybrid solution in which the Rx
688  * is driven by DMA but the Tx is driven by ISR.
689  *
690  * @param none
691  *
692  * @return none
693  *****************************************************************************/

695 {

697  uint16 cnt;
698 
699 #if DMA_PM

701  {

703  {

705 #if HAL_UART_TX_BY_ISR

707 #else

709 #endif

714

716  {
717  (void)osal_set_event(Hal_TaskID, HAL_PWRMGR_HOLD_EVENT);
718  }
719 
720  if ((dmaRdyDly = ST0) == 0) // Reserve zero to signify that the delay expired.
721  {
722  dmaRdyDly = 0xFF;
723  }
724  HAL_UART_DMA_SET_RDY_OUT();
725  }
726  else if ((dmaRdyDly != 0) && (!DMA_PM_DLY || ((uint8)(ST0 - dmaRdyDly) > DMA_PM_DLY)))
727  {
728  dmaRdyDly = 0;
729  (void)osal_set_event(Hal_TaskID, HAL_PWRMGR_CONSERVE_EVENT);
730  }
731  }
732  PxIEN |= DMA_RDYIn_BIT;
733 #endif
734 
735 #if !HAL_UART_TX_BY_ISR
736  HalUARTPollTxTrigDMA();
737 #endif
738 
739 
740  cnt = HalUARTRxAvailDMA(); // Wait to call until after the above DMA Rx bug work-around.
741 
742 #if HAL_UART_DMA_IDLE
743  if (dmaCfg.rxTick)
744  {
745  // Use the LSB of the sleep timer (ST0 must be read first anyway) to measure the Rx timeout.
746  if ((ST0 - dmaCfg.rxTick) > HAL_UART_DMA_IDLE)
747  {
748  dmaCfg.rxTick = 0;
749  evt = HAL_UART_RX_TIMEOUT;
750  }
751  }
752  else if (cnt != 0)
753  {
754  if ((dmaCfg.rxTick = ST0) == 0) // Zero signifies that the Rx timeout is not running.
755  {
756  dmaCfg.rxTick = 0xFF;
757  }
758  }
759 #else
760  if (cnt != 0)
761  {
762  evt = HAL_UART_RX_TIMEOUT;
763  }
764 #endif
765 
766  if (cnt >= HAL_UART_DMA_FULL)
767  {
768  evt |= HAL_UART_RX_FULL;
769  }
770  else if (cnt >= HAL_UART_DMA_HIGH)
771  {
772  evt |= HAL_UART_RX_ABOUT_FULL;
773 
774  if (!DMA_PM && (UxUCR & UCR_FLOW))
775  {
776  HAL_UART_DMA_CLR_RDY_OUT(); // Disable Rx flow.
777  }
778  }
779 
780  if (dmaCfg.txMT)
781  {

783  evt |= HAL_UART_TX_EMPTY;
784  }
785 
786  if ((evt != 0) && (dmaCfg.uartCB != NULL))
787  {
788  dmaCfg.uartCB(HAL_UART_DMA-1, evt);
789  }
790 
791  if (DMA_PM && (dmaRdyDly == 0) && !HalUARTBusyDMA())
792  {
793  HAL_UART_DMA_CLR_RDY_OUT();
794  }
795 }

797 /**************************************************************************************************
798  * @fn HalUARTRxAvailDMA()
799  *
800  * @brief Calculate Rx Buffer length - the number of bytes in the buffer.
801  *
802  * @param none
803  *
804  * @return length of current Rx Buffer
805  **************************************************************************************************/
806 rxIdx_t sweepIdx = 0;
807 
808 static uint16 HalUARTRxAvailDMA(void)
809 {
810  uint16 cnt = 0;
811 #ifndef POWER_SAVING
812  bool detectOverflow = FALSE;
813 #endif
814  // First, synchronize the Rx tail marker with where the DMA Rx engine is working.
815  rxIdx_t tail = dmaCfg.rxTail;
816 
817 #ifndef POWER_SAVING
818  if (!DMA_PM && (UxUCR & UCR_FLOW))
819  {
820  HAL_UART_DMA_CLR_RDY_OUT(); // Stop the inflow for counting the bytes
821  }
822 #endif
823 
824  do
825  {
826  if (!HAL_UART_DMA_NEW_RX_BYTE(tail))
827  {
828  break;
829  }
830  else
831  {
832  cnt++;
833  }
834 
835  HAL_UART_RX_IDX_T_INCR(tail);
836  } while (cnt < HAL_UART_DMA_RX_MAX);
837 
838  if(!cnt)
839  {
840  while(sweepIdx < HAL_UART_DMA_RX_MAX)
841  {
842  if (HAL_UART_DMA_NEW_RX_BYTE(sweepIdx))
843  {
844  dmaCfg.rxTail = sweepIdx;
845  dmaCfg.rxHead = sweepIdx;
846  cnt = 0;
847 #ifndef POWER_SAVING
848  detectOverflow = TRUE;
849 #endif
850  break;
851  }
852 
853  sweepIdx++;
854  }
855 
856  if ( sweepIdx == HAL_UART_DMA_RX_MAX )
857  {
858  sweepIdx = 0;
859  }
860  }
861 
862 #ifndef POWER_SAVING
863  if ( (!DMA_PM && (UxUCR & UCR_FLOW)) && (!detectOverflow ) )
864  {
865  HAL_UART_DMA_SET_RDY_OUT(); // Re-enable the flow asap
866  }
867 #endif
868  return cnt;
869 }
