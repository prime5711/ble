67 void accInit(void)
68 {
69  //*** Enable Accelerometer power ***
70  P2DIR |= 0x01; // Set P2_0 as output high
71  P2SEL &= ~0x01;
72  P2 |= 0x01;
73 
74  //*** Setup USART 0 SPI at alternate location 2 ***
75  // USART 0 at alternate location 2
76  PERCFG |= 0x01;
77  // Peripheral function on SCK, MISO and MOSI (P1_3-5)
78  P1SEL |= 0x38;
79  // Configure CS (P1_2) as output
80  P1DIR |= 0x04;
81  CS = CS_DISABLED;
82 
83  //*** Setup the SPI interface ***
84  // SPI master mode
85  U0CSR = 0x00;
86  // Negative clock polarity, Phase: data out on CPOL -> CPOL-inv
87  // data in on CPOL-inv -> CPOL
88  // MSB first
89  U0GCR = 0x20;
90  // SCK frequency = 3MHz (MBA250 max=10MHz, CC254x max = 4MHz)
91  U0GCR |= 0x10;
92  U0BAUD = 0x80;
93 
94  // *** Configure accelerometer ***
95  // Wait 2ms for accelerometer to power up and settle
96  WAIT_MS(2);
97 
98  // Disable all interrupts
99  ACC_INT_DISABLE(ACC_INT_ENABLE0, 0xFF);
100  ACC_INT_DISABLE(ACC_INT_ENABLE1, 0xFF);
101 
102  // Set 2G range
103  accWriteReg(ACC_RANGE, ACC_RANGE_2G);
104 
105  // Filter detection bandwidth (Time between samples = 1/(2*filter_bw))
106  accWriteReg(ACC_BW, ACC_BW_250HZ);
107 
108 
109 #ifdef ENABLE_BMA250_INTERRUPT
110  // *** New data interrupt on INT1, active high ***
111  // INT1 as push-pull, active high
112  accWriteReg(ACC_INT_PIN_BEHAVIOR, ACC_INT1_LVL);
113 
114  // Map New data interrupt to INT1
115  accWriteReg(ACC_INT_MAPPING1, ACC_INT1_MAP_DATA);
116 
117  // Enable new data interrupts
118  ACC_INT_ENABLE(ACC_INT_ENABLE1, ACC_INT_DATA_EN);
119 #endif
120 
121  acc_initialized = TRUE;
122 } // accInit

190 void accReadAcc(int8 *pXVal, int8 *pYVal, int8 *pZVal)
191 {
192  int8 readout[6] = {0,0,0,0,0,0};
193 
194  // Read all data from accelerometer
195  CS = CS_ENABLED;
196  spiWriteByte(0x80|ACC_X_LSB); // Write start address
197  for(uint8 i = 0; i<6; i++)

200  }
201  CS = CS_DISABLED;

223 {
