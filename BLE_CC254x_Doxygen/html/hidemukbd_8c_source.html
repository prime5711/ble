350 /*********************************************************************
351  * @fn HidEmuKbd_ProcessEvent
352  *
353  * @brief HidEmuKbd Application Task event processor. This function
354  * is called to process all events for the task. Events
355  * include timers, messages and any other user defined events.
356  *
357  * @param task_id - The OSAL assigned task ID.
358  * @param events - events to process. This is a bit map and can
359  * contain more than one event.
360  *
361  * @return events not processed
362  */
363 uint16 HidEmuKbd_ProcessEvent( uint8 task_id, uint16 events )
364 {
365 
366  VOID task_id; // OSAL required parameter that isn't used in this function
367 
368  if ( events & SYS_EVENT_MSG )
369  {
370  uint8 *pMsg;
371 
372  if ( (pMsg = osal_msg_receive( hidEmuKbdTaskId )) != NULL )
373  {
374  hidEmuKbd_ProcessOSALMsg( (osal_event_hdr_t *)pMsg );
375 
376  // Release the OSAL message
377  VOID osal_msg_deallocate( pMsg );
378  }
379 
380  // return unprocessed events
381  return (events ^ SYS_EVENT_MSG);
382  }
383 
384  if ( events & START_DEVICE_EVT )
385  {
386  return ( events ^ START_DEVICE_EVT );
387  }
388 
389  return 0;
390 }

414 /*********************************************************************
415  * @fn hidEmuKbd_HandleKeys
416  *
417  * @brief Handles all key events for this device.
418  *
419  * @param shift - true if in shift/alt.
420  * @param keys - bit field for key events. Valid entries:
421  * HAL_KEY_SW_2
422  * HAL_KEY_SW_1
423  *
424  * @return none
425  */
426 static void hidEmuKbd_HandleKeys( uint8 shift, uint8 keys )
427 {
428  static uint8 prevKey1 = 0;
429  static uint8 prevKey2 = 0;
430 
431  (void)shift; // Intentionally unreferenced parameter
432 
433  if ( (keys & HAL_KEY_SW_1) && (prevKey1 == 0) )
434  {
435  // pressed
436  hidEmuKbdSendReport( KEY_LEFT_ARROW );
437  prevKey1 = 1;
438  }
439  else if ( !(keys & HAL_KEY_SW_1) && (prevKey1 == 1) )
440  {
441  // released
442  hidEmuKbdSendReport( KEY_NONE );
443  prevKey1 = 0;
444  }
445 
446  if ( (keys & HAL_KEY_SW_2) && (prevKey2 == 0) )
447  {
448  // pressed
449  if ( !hidBootMouseEnabled )
450  {
451  hidEmuKbdSendReport( KEY_RIGHT_ARROW );
452  }
453  else
454  {
455  hidEmuKbdSendMouseReport( MOUSE_BUTTON_1 );
456  }
457  prevKey2 = 1;
458  }
459  else if ( !(keys & HAL_KEY_SW_2) && (prevKey2 == 1) )
460  {
461  // released
462  if ( !hidBootMouseEnabled )
463  {
464  hidEmuKbdSendReport( KEY_NONE );
465  }
466  else
467  {
468  hidEmuKbdSendMouseReport ( MOUSE_BUTTON_NONE );
469  }
470  prevKey2 = 0;
471  }
472 }

550 /*********************************************************************
551  * @fn hidEmuKbdRptCB
552  *
553  * @brief HID Dev report callback.
554  *
555  * @param id - HID report ID.
556  * @param type - HID report type.
557  * @param uuid - attribute uuid.
558  * @param oper - operation: read, write, etc.
559  * @param len - Length of report.
560  * @param pData - Report data.
561  *
562  * @return GATT status code.
563  */
564 static uint8 hidEmuKbdRptCB( uint8 id, uint8 type, uint16 uuid,
565  uint8 oper, uint8 *pLen, uint8 *pData )
566 {
567  uint8 status = SUCCESS;
568 
569  // write
570  if ( oper == HID_DEV_OPER_WRITE )
571  {
572  if ( uuid == REPORT_UUID )
573  {
574  // process write to LED output report; ignore others
575  if ( type == HID_REPORT_TYPE_OUTPUT )
576  {
577  status = hidEmuKbdRcvReport( *pLen, pData );
578  }
579  }
580 
581  if ( status == SUCCESS )
582  {
583  status = HidKbd_SetParameter( id, type, uuid, *pLen, pData );
584  }
585  }
586  // read
587  else if ( oper == HID_DEV_OPER_READ )
588  {
589  status = HidKbd_GetParameter( id, type, uuid, pLen, pData );
590  }
591  // notifications enabled
592  else if ( oper == HID_DEV_OPER_ENABLE )
593  {
594  if ( id == HID_RPT_ID_MOUSE_IN && type == HID_REPORT_TYPE_INPUT )
595  {
596  hidBootMouseEnabled = TRUE;
597  }
598  }
599  // notifications disabled
600  else if ( oper == HID_DEV_OPER_DISABLE )
601  {
602  if ( id == HID_RPT_ID_MOUSE_IN && type == HID_REPORT_TYPE_INPUT )
603  {
604  hidBootMouseEnabled = FALSE;
605  }
606  }
607 
608  return status;
609 }
