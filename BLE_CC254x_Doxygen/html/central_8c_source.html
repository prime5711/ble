334 bStatus_t GAPCentralRole_StartRssi( uint16 connHandle, uint16 period )
335 {
336  gapCentralRoleRssi_t *pRssi;
337 
338  // Verify link is up
339  if (!linkDB_Up(connHandle))
340  {
341  return bleIncorrectMode;
342  }
343 
344  // If already allocated
345  if ((pRssi = gapCentralRole_RssiFind( connHandle )) != NULL)
346  {
347  // Stop timer
348  osal_CbTimerStop( pRssi->timerId );
349  }
350  // Allocate structure
351  else if ((pRssi = gapCentralRole_RssiAlloc( connHandle )) != NULL)
352  {
353  pRssi->period = period;
354  }
355  // Allocate failed
356  else
357  {
358  return bleNoResources;
359  }
360 
361  // Start timer
362  osal_CbTimerStart( gapCentralRole_timerCB, (uint8 *) pRssi,
363  period, &pRssi->timerId );
364 
365  return SUCCESS;
366 }

431 uint16 GAPCentralRole_ProcessEvent( uint8 taskId, uint16 events )
432 {
433  if ( events & SYS_EVENT_MSG )
434  {
435  uint8 *pMsg;
436 
437  if ( (pMsg = osal_msg_receive( gapCentralRoleTaskId )) != NULL )
438  {
439  gapCentralRole_ProcessOSALMsg( (osal_event_hdr_t *) pMsg );
440 
441  // Release the OSAL message
442  VOID osal_msg_deallocate( pMsg );
443  }
444 
445  // return unprocessed events
446  return (events ^ SYS_EVENT_MSG);
447  }
448 
449  if ( events & GAP_EVENT_SIGN_COUNTER_CHANGED )
450  {
451  // Sign counter changed, save it to NV
452  VOID osal_snv_write( BLE_NVID_SIGNCOUNTER, sizeof( uint32 ), &gapCentralRoleSignCounter );
453 
454  return ( events ^ GAP_EVENT_SIGN_COUNTER_CHANGED );
455  }
456 
457  // Discard unknown events
458  return 0;
459 }

461 /*********************************************************************
462  * @fn gapCentralRole_ProcessOSALMsg
463  *
464  * @brief Process an incoming task message.
465  *
466  * @param pMsg - message to process
467  *
468  * @return none
469  */
470 static void gapCentralRole_ProcessOSALMsg( osal_event_hdr_t *pMsg )
471 {
472  switch ( pMsg->event )
473  {
474  case HCI_GAP_EVENT_EVENT:
475  if ( pMsg->status == HCI_COMMAND_COMPLETE_EVENT_CODE )
476  {
477  hciEvt_CmdComplete_t *pPkt = (hciEvt_CmdComplete_t *) pMsg;
478 
479  if ( pPkt->cmdOpcode == HCI_READ_RSSI )
480  {
481  uint16 connHandle = BUILD_UINT16( pPkt->pReturnParam[1], pPkt->pReturnParam[2] );
482  int8 rssi = (int8) pPkt->pReturnParam[3];
483 
484  // Report RSSI to app
485  if ( pGapCentralRoleCB && pGapCentralRoleCB->rssiCB )
486  {
487  pGapCentralRoleCB->rssiCB( connHandle, rssi );
488  }
489  }
490  }
491  break;
492 
493  case GAP_MSG_EVENT:
494  gapCentralRole_ProcessGAPMsg( (gapEventHdr_t *) pMsg );
495  break;
496 
497  case GAPCENTRALROLE_RSSI_MSG_EVT:
498  {
499  gapCentralRoleRssi_t *pRssi = ((gapCentralRoleRssiEvent_t *) pMsg)->pRssi;
500 
501  // If link is up and RSSI reads active
502  if (pRssi->connHandle != GAP_CONNHANDLE_ALL &&
503  linkDB_Up(pRssi->connHandle))
504  {
505  // Restart timer
506  osal_CbTimerStart( gapCentralRole_timerCB, (uint8 *) pRssi,
507  pRssi->period, &pRssi->timerId );
508 
509  // Read RSSI
510  VOID HCI_ReadRssiCmd( pRssi->connHandle );
511  }
512  }
513  break;
514 
515  default:
516  break;
517  }
518 }

520 /*********************************************************************
521  * @fn gapCentralRole_ProcessGAPMsg
522  *
523  * @brief Process an incoming task message from GAP.
524  *
525  * @param pMsg - message to process
526  *
527  * @return none
528  */
529 static void gapCentralRole_ProcessGAPMsg( gapEventHdr_t *pMsg )
530 {
531  switch ( pMsg->opcode )
532  {
533  case GAP_DEVICE_INIT_DONE_EVENT:
534  {
535  gapDeviceInitDoneEvent_t *pPkt = (gapDeviceInitDoneEvent_t *) pMsg;
536 
537  if ( pPkt->hdr.status == SUCCESS )
538  {
539  // Save off the generated keys
540  VOID osal_snv_write( BLE_NVID_IRK, KEYLEN, gapCentralRoleIRK );
541  VOID osal_snv_write( BLE_NVID_CSRK, KEYLEN, gapCentralRoleSRK );
542 
543  // Save off the information
544  VOID osal_memcpy( gapCentralRoleBdAddr, pPkt->devAddr, B_ADDR_LEN );
545  }
546  }
547  break;
548 
549  case GAP_LINK_ESTABLISHED_EVENT:
550  {
551  gapEstLinkReqEvent_t *pPkt = (gapEstLinkReqEvent_t *) pMsg;
552 
553  if (pPkt->hdr.status == SUCCESS)
554  {
555  // Notify the Bond Manager of the connection
556  VOID GAPBondMgr_LinkEst( pPkt->devAddrType, pPkt->devAddr,
557  pPkt->connectionHandle, GAP_PROFILE_CENTRAL );
558  }
559  }
560  break;
561 
562  case GAP_LINK_TERMINATED_EVENT:
563  {
564  uint16 connHandle = ((gapTerminateLinkEvent_t *) pMsg)->connectionHandle;
565 
566  VOID GAPBondMgr_ProcessGAPMsg( (gapEventHdr_t *)pMsg );
567 
568  // Cancel RSSI reads
569  GAPCentralRole_CancelRssi( connHandle );
570  }
571  break;
572 
573  // temporary workaround
574  case GAP_SLAVE_REQUESTED_SECURITY_EVENT:
575  VOID GAPBondMgr_ProcessGAPMsg( pMsg );
576  break;
577 
578  default:
579  break;
580  }
581 
582  // Pass event to app
583  if ( pGapCentralRoleCB && pGapCentralRoleCB->eventCB )
584  {
585  pGapCentralRoleCB->eventCB( (gapCentralRoleEvent_t *) pMsg );
586  }
587 }
