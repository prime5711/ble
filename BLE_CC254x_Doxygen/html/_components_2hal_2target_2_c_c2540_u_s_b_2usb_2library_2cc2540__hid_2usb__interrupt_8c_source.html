47  uint8 usbcif;
48 
49  HAL_ENTER_ISR();
50  HAL_USB_ENABLE(); // Never access the USB controller before PLL is stable.
51  usbcif = USBCIF;
52 
53  if ((usbcif & (USBCIF_RSTIF | USBCIF_RESUMEIF)) || (P2IFG & P2IFG_DPIF)) {
54  if (usbcif & USBCIF_RSTIF) {
55  // All interrupts (except suspend) are by default enabled by hardware, so
56  // re-initialize the enable bits to avoid unwanted interrupts
57  USBCIE = usbirqData.irqMask;
58  USBIIE = usbirqData.irqMask >> 4;
59  USBOIE = (usbirqData.irqMask >> 9) & 0x3E;
60 
61  // Enable suspend mode when suspend signaling is detected on the bus
62  USBPOW |= USBPOW_SUSPEND_EN;
63  }
64 
65  usbcif |= USBCIF_RESUMEIF;
66  usbirqData.inSuspend = FALSE;
67  }
68  else if (usbcif & USBCIF_SUSPENDIF) {
69  HAL_USB_PLL_DISABLE(); // Disable USB clock (PLL).
70  usbirqData.inSuspend = TRUE;
71  }
72 
73  eventMask = usbcif;
74  eventMask |= (uint16)USBIIF << 4;
75  eventMask |= (uint16)USBOIF << 9;
76  usbirqData.eventMask |= eventMask; // Record events (keeping existing).
77 
78  HAL_USB_INT_CLEAR();
79  HAL_EXIT_ISR();
80 }
