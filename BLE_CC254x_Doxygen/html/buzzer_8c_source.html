49 void buzzerInit(void)
50 {
51 #if defined ( CC2540_MINIDK )
52  // Buzzer connected at P1_6
53  // We will use Timer 3 Channel 0 at alternate location 2
54  // Channel 0 will toggle on compare with 0 and counter will
55  // count in up/down mode to T3CC0.
56 
57  PERCFG |= 0x20; // Timer 3 Alternate location 2
58  P1DIR |= 0x40; // P1_6 = output
59  P1SEL |= 0x40; // Peripheral function on P1_6
60 
61  T3CTL &= ~0x10; // Stop timer 3 (if it was running)
62  T3CTL |= 0x04; // Clear timer 3
63  T3CTL &= ~0x08; // Disable Timer 3 overflow interrupts
64  T3CTL |= 0x03; // Timer 3 mode = 3 - Up/Down
65 
66  T3CCTL0 &= ~0x40; // Disable channel 0 interrupts
67  T3CCTL0 |= 0x04; // Ch0 mode = compare
68  T3CCTL0 |= 0x10; // Ch0 output compare mode = toggle on compare
69 #endif
70 }

80 uint8 buzzerStart(uint16 frequency)
81 {
82 #if defined ( CC2540_MINIDK )
83  buzzerInit();
84 
85  uint8 prescaler = 0;
86 
87  // Get current Timer tick divisor setting
88  uint8 tickSpdDiv = (CLKCONSTA & 0x38)>>3;
89 
90  // Check if frequency too low
91  if (frequency < (244 >> tickSpdDiv)){ // 244 Hz = 32MHz / 256 (8bit counter) / 4 (up/down counter and toggle on compare) / 128 (max timer prescaler)
92  buzzerStop(); // A lower tick speed will lower this number accordingly.
93  return 0;
94  }
95 
96  // Calculate nr of ticks required to achieve target frequency
97  uint32 ticks = (8000000/frequency) >> tickSpdDiv; // 8000000 = 32M / 4;
98 
99  // Fit this into an 8bit counter using the timer prescaler
100  while ((ticks & 0xFFFFFF00) != 0)
101  {
102  ticks >>= 1;
103  prescaler += 32;
104  }
105 
106  // Update registers
107  T3CTL &= ~0xE0;
108  T3CTL |= prescaler;
109  T3CC0 = (uint8)ticks;
110 
111  // Start timer
112  T3CTL |= 0x10;
113 #endif
114 
115  return 1;
116 }
