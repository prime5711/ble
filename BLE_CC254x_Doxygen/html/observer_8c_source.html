228 uint16 GAPObserverRole_ProcessEvent( uint8 taskId, uint16 events )
229 {
230  if ( events & SYS_EVENT_MSG )
231  {
232  uint8 *pMsg;
233 
234  if ( (pMsg = osal_msg_receive( gapObserverRoleTaskId )) != NULL )
235  {
236  gapObserverRole_ProcessOSALMsg( (osal_event_hdr_t *) pMsg );
237 
238  // Release the OSAL message
239  VOID osal_msg_deallocate( pMsg );
240  }
241 
242  // return unprocessed events
243  return (events ^ SYS_EVENT_MSG);
244  }
245 
246  // Discard unknown events
247  return 0;
248 }

253  * @brief Process an incoming task message.

255  * @param pMsg - message to process

257  * @return none
258  */
259 static void gapObserverRole_ProcessOSALMsg( osal_event_hdr_t *pMsg )
260 {
261  switch ( pMsg->event )
262  {
263  case HCI_GAP_EVENT_EVENT:
264  if ( pMsg->status == HCI_COMMAND_COMPLETE_EVENT_CODE )
265  {
266  //hciEvt_CmdComplete_t *pPkt = (hciEvt_CmdComplete_t *) pMsg;
267  }
268  break;
269 
270  case GAP_MSG_EVENT:
271  gapObserverRole_ProcessGAPMsg( (gapEventHdr_t *) pMsg );
272  break;
273 
274  default:
275  break;
276  }
277 }

279 /*********************************************************************
280  * @fn gapObserverRole_ProcessGAPMsg
281  *
282  * @brief Process an incoming task message from GAP.
283  *
284  * @param pMsg - message to process
285  *
286  * @return none
287  */
288 static void gapObserverRole_ProcessGAPMsg( gapEventHdr_t *pMsg )
289 {
290  switch ( pMsg->opcode )
291  {
292  case GAP_DEVICE_INIT_DONE_EVENT:
293  {
294  gapDeviceInitDoneEvent_t *pPkt = (gapDeviceInitDoneEvent_t *) pMsg;
295 
296  if ( pPkt->hdr.status == SUCCESS )
297  {
298  // Save off the information
299  VOID osal_memcpy( gapObserverRoleBdAddr, pPkt->devAddr, B_ADDR_LEN );
300  }
301  }
302  break;
303 
304  default:
305  break;
306  }
307 
308  // Pass event to app
309  if ( pGapObserverRoleCB && pGapObserverRoleCB->eventCB )
310  {
311  pGapObserverRoleCB->eventCB( (gapObserverRoleEvent_t *) pMsg );
312  }
313 }
