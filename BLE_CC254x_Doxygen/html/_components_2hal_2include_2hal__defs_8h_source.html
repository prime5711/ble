115 /*
116  * This macro is for use by other macros to form a fully valid C statement.
117  * Without this, the if/else conditionals could show unexpected behavior.
118  *
119  * For example, use...
120  * #define SET_REGS() st( ioreg1 = 0; ioreg2 = 0; )
121  * instead of ...
122  * #define SET_REGS() { ioreg1 = 0; ioreg2 = 0; }
123  * or
124  * #define SET_REGS() ioreg1 = 0; ioreg2 = 0;
125  * The last macro would not behave as expected in the if/else construct.
126  * The second to last macro will cause a compiler error in certain uses
127  * of if/else construct
128  *
129  * It is not necessary, or recommended, to use this macro where there is
130  * already a valid C statement. For example, the following is redundant...
131  * #define CALL_FUNC() st( func(); )
132  * This should simply be...
133  * #define CALL_FUNC() func()
134  *
135  * (The while condition below evaluates false without generating a
136  * constant-controlling-loop type of warning on most compilers.)
137  */
