35 void usbHidProcessEvents(void)
36 {
37  // Handle USB resume
38  if (USBIRQ_GET_EVENT_MASK() & USBIRQ_EVENT_RESUME) {
39  USBIRQ_CLEAR_EVENTS(USBIRQ_EVENT_RESUME); // Clear USB resume interrupt
40  if (pFnSuspendExitHook!=NULL)
41  {
42  pFnSuspendExitHook();
43  }
44  }
45 
46  // Handle USB reset
47  if (USBIRQ_GET_EVENT_MASK() & USBIRQ_EVENT_RESET) {
48  USBIRQ_CLEAR_EVENTS(0xFFFF);
49  usbfwResetHandler();
50  }
51 
52  // Handle USB suspend
53  if (USBIRQ_GET_EVENT_MASK() & USBIRQ_EVENT_SUSPEND) {
54  USBIRQ_CLEAR_EVENTS(USBIRQ_EVENT_SUSPEND); // Clear USB suspend interrupt
55  if (pFnSuspendEnterHook!=NULL)
56  {
57  pFnSuspendEnterHook();
58  }
59  }
60 
61  // Handle USB packets on EP0
62  if (USBIRQ_GET_EVENT_MASK() & USBIRQ_EVENT_SETUP) {
63  USBIRQ_CLEAR_EVENTS(USBIRQ_EVENT_SETUP);
64  usbfwSetupHandler();
65  }
66 }
