97 /*********************************************************************
98  * @fn timeAppConfigNext()
99  *
100  * @brief Perform the characteristic configuration read or
101  * write procedure.
102  *
103  * @param state - Configuration state.
104  *
105  * @return New configuration state.
106  */
107 uint8 timeAppConfigNext( uint8 state )
108 {
109  attReadReq_t readReq;
110  attWriteReq_t writeReq;
111  bool read;
112 
113  // Find next non-zero cached handle of interest
114  while ( state < TIMEAPP_CONFIG_MAX &&
115  timeAppHdlCache[timeAppConfigList[state]] == 0 )
116  {
117  state++;
118  }
119 
120  // Return if reached end of list
121  if ( state == TIMEAPP_CONFIG_MAX )
122  {
123  return TIMEAPP_CONFIG_CMPL;
124  }
125 
126  // Determine what to do with characteristic
127  switch ( timeAppConfigList[state] )
128  {
129  // Read these characteristics
130  case HDL_CURR_TIME_CT_TIME_START:
131  read = TRUE;
132  break;
133 
134  // Set notification for these characteristics
135  case HDL_CURR_TIME_CT_TIME_CCCD:
136 
137  read = FALSE;
138  writeReq.len = 2;
139  writeReq.value[0] = LO_UINT16(GATT_CLIENT_CFG_NOTIFY);
140  writeReq.value[1] = HI_UINT16(GATT_CLIENT_CFG_NOTIFY);
141  writeReq.sig = 0;
142  writeReq.cmd = 0;
143  break;
144 
145  default:
146  break;
147  }
148 
149  if(timeConfigDone==TRUE)
150  {
151  return state;
152  }
153 
154  // Do a GATT read or write
155  if ( read )
156  {
157  readReq.handle = timeAppHdlCache[timeAppConfigList[state]];
158  GATT_ReadCharValue( gapConnHandle, &readReq, thermometerTaskId );
159 
160  //Only reading time right now
161  timeConfigDone = TRUE;
162  }
163  else
164  {
165  writeReq.handle = timeAppHdlCache[timeAppConfigList[state]];
166  GATT_WriteCharValue( gapConnHandle, &writeReq, thermometerTaskId );
167  }
168 
169  return state;
170 }
171 
172 /*********************************************************************
173  * @fn timeAppConfigGattMsg()
174  *
175  * @brief Handle GATT messages for characteristic configuration.
176  *
177  * @param state - Discovery state.
178  * @param pMsg - GATT message.
179  *
180  * @return New configuration state.
181  */
182 uint8 timeAppConfigGattMsg( uint8 state, gattMsgEvent_t *pMsg )
183 {
184 
185  if ( (pMsg->method == ATT_READ_RSP || pMsg->method == ATT_WRITE_RSP) &&
186  (pMsg->hdr.status == SUCCESS) )
187  {
188  // Process response
189  switch ( timeAppConfigList[state] )
190  {
191  case HDL_CURR_TIME_CT_TIME_START:
192  // Set clock to time read from time server
193  timeAppClockSet( pMsg->msg.readRsp.value );
194  break;
195 
196  default:
197  break;
198  }
199  }
200  return timeAppConfigNext( state + 1 );
201 }
