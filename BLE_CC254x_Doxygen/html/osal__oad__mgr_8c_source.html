107 /*********************************************************************
108  * @fn osalInitTasks
109  *
110  * @brief This function invokes the initialization function for each task.
111  *
112  * @param void
113  *
114  * @return none
115  */
116 void osalInitTasks( void )
117 {
118  uint8 taskID = 0;
119 
120  tasksEvents = (uint16 *)osal_mem_alloc( sizeof( uint16 ) * tasksCnt);
121  osal_memset( tasksEvents, 0, (sizeof( uint16 ) * tasksCnt));
122 
123  /* LL Task */
124  LL_Init( taskID++ );
125 
126  /* Hal Task */
127  Hal_Init( taskID++ );
128 
129  /* HCI Task */
130  HCI_Init( taskID++ );
131 
132 #if defined ( OSAL_CBTIMER_NUM_TASKS )
133  /* Callback Timer Tasks */
134  osal_CbTimerInit( taskID );
135  taskID += OSAL_CBTIMER_NUM_TASKS;
136 #endif
137 
138  /* L2CAP Task */
139  L2CAP_Init( taskID++ );
140 
141  /* GAP Task */
142  GAP_Init( taskID++ );
143 
144  /* GATT Task */
145  GATT_Init( taskID++ );
146 
147  /* SM Task */
148  SM_Init( taskID++ );
149 
150  /* Profiles */
151  GAPCentralRole_Init( taskID++ );
152 
153  /* Application */
154  OADManager_Init( taskID );
155 }
