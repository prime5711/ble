213 // switch to the 16MHz HSOSC and wait until it is stable
214 #define SET_OSC_TO_HSOSC() \
215 { \
216  CLKCONCMD = (CLKCONCMD & 0x80) | CLKCONCMD_16MHZ; \
217  while ( (CLKCONSTA & ~0x80) != CLKCONCMD_16MHZ ); \
218 }
219 
220 // switch to the 32MHz XOSC and wait until it is stable
221 #define SET_OSC_TO_XOSC() \
222 { \
223  CLKCONCMD = (CLKCONCMD & 0x80) | CLKCONCMD_32MHZ; \
224  while ( (CLKCONSTA & ~0x80) != CLKCONCMD_32MHZ ); \
225 }
226 
227 // set 32kHz OSC and wait until it is stable
228 #define SET_32KHZ_OSC() \
229 { \
230  CLKCONCMD = (CLKCONCMD & ~0x80) | OSC_32KHZ; \
231  while ( (CLKCONSTA & 0x80) != OSC_32KHZ ); \
232 }
233 
234 #define START_HSOSC_XOSC() \
235 { \
236  SLEEPCMD &= ~OSC_PD; /* start 16MHz RCOSC & 32MHz XOSC */ \
237  while (!(SLEEPSTA & XOSC_STB)); /* wait for stable 32MHz XOSC */ \
238 }
239 
240 #define STOP_HSOSC() \
241 { \
242  SLEEPCMD |= OSC_PD; /* stop 16MHz RCOSC */ \
243 }

326 // The TI reference design uses UART1 Alt. 2 in SPI mode.
327 #define XNV_SPI_INIT() \
328 st( \
329  /* Mode select UART1 SPI Mode as master. */\
330  U1CSR = 0; \
331  \
332  /* Setup for 115200 baud. */\
333  U1GCR = 11; \
334  U1BAUD = 216; \
335  \
336  /* Set bit order to MSB */\
337  U1GCR |= BV(5); \
338  \
339  /* Set UART1 I/O to alternate 2 location on P1 pins. */\
340  PERCFG |= 0x02; /* U1CFG */\
341  \
342  /* Select peripheral function on I/O pins but SS is left as GPIO for separate control. */\
343  P1SEL |= 0xE0; /* SELP1_[7:4] */\
344  /* P1.1,2,3: reset, LCD CS, XNV CS. */\
345  P1SEL &= ~0x0E; \
346  P1 |= 0x0E; \
347  P1_1 = 0; \
348  P1DIR |= 0x0E; \
349  \
350  /* Give UART1 priority over Timer3. */\
351  P2SEL &= ~0x20; /* PRI2P1 */\
352  \
353  /* When SPI config is complete, enable it. */\
354  U1CSR |= 0x40; \
355  /* Release XNV reset. */\
356  P1_1 = 1; \
357 )
