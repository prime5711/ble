86 void accInit(void)
87 {
88  //*** Setup USART 0 SPI at alternate location 2 ***
89 
90  // USART 0 at alternate location 2
91  PERCFG |= 0x01;
92  // Peripheral function on SCK, MISO and MOSI (P1_3-5)
93  P1SEL |= 0x38;
94  // Configure CS (P1_7/P1_2) as output
95 #ifdef REV_1_0
96  P1DIR |= 0x04;
97 #elif (defined REV_0_6)
98  P1DIR |= 0x80;
99 #endif
100  CS = CS_DISABLED;
101 
102  //*** Setup the SPI interface ***
103  // SPI master mode
104  U0CSR = 0x00;
105  // Negative clock polarity, Phase: data out on CPOL -> CPOL-inv
106  // data in on CPOL-inv -> CPOL
107  // MSB first
108  U0GCR = 0x20;
109  // SCK frequency = 480.5kHz (max 500kHz)
110  U0GCR |= 0x0D;
111  U0BAUD = 0xEC;
112 
113  uint8 readValue;
114  accWriteReg(CTRL, RANGE_2G | MODE_100HZ_MEAS);
115  WAIT_1_3US(80);
116  do{
117  accReadReg(STATUS, &readValue);
118  WAIT_1_3US(80);
119  }while(readValue & 0x08);
120  acc_initialized = TRUE;
121 }
122 
126 void accStop(void)
127 {
128  if (acc_initialized) {
129  accWriteReg(CTRL, MODE_PD);
130  acc_initialized = FALSE;
131  }
132 }

211 {
212  U0CSR &= ~0x02; // Clear TX_BYTE
