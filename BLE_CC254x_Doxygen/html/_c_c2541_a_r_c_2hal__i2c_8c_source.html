194 /**************************************************************************************************
195  * @fn i2cMstStrt
196  *
197  * @brief Attempt to send an I2C bus START and Slave Address as an I2C bus Master.
198  *
199  * input parameters
200  *
201  * @param address - I2C address of the slave device we're attempting to talk to
202  * @param RD_WRn - The LSB of the Slave Address as Read/~Write.
203  *
204  * output parameters

206  * None.
207  *
208  * @return The I2C status of the START request or of the Slave Address Ack.
209  */
210 static uint8 i2cMstStrt(uint8 address, uint8 RD_WRn)
211 {
212  I2C_STRT();
213 
214  if (I2CSTAT == mstStarted)

216  address <<= 1;
217  I2C_WRITE(address | RD_WRn);
218  }
219 
220  return I2CSTAT;
221 }

246 /**************************************************************************************************
247  * @fn HalI2CRead
248  *
249  * @brief Read from the I2C bus as a Master.
250  *
251  * input parameters
252  *
253  * @param address - I2C address of the slave device we're attempting to read from
254  * @param len - Number of bytes to read.
255  * @param pBuf - Pointer to the data buffer to put read bytes.
256  *
257  * output parameters
258  *
259  * None.
260  *
261  * @return The number of bytes successfully read.
262  */
263 i2cLen_t HalI2CRead(uint8 address, i2cLen_t len, uint8 *pBuf)
264 {
265  uint8 cnt = 0;
266 
267  if (i2cMstStrt(address, I2C_MST_RD_BIT) != mstAddrAckR)
268  {
269  len = 0;
270  }
271 
272  // All bytes are ACK'd except for the last one which is NACK'd. If only
273  // 1 byte is being read, a single NACK will be sent. Thus, we only want
274  // to enable ACK if more than 1 byte is going to be read.
275  if (len > 1)
276  {
277  I2C_SET_ACK();
278  }
279 
280  while (len > 0)
281  {
282  // slave devices require NACK to be sent after reading last byte
283  if (len == 1)
284  {
285  I2C_SET_NACK();
286  }
287 
288  // read a byte from the I2C interface
289  I2C_READ(*pBuf++);
290  cnt++;
291  len--;
292 
293  if (I2CSTAT != mstDataAckR)
294  {
295  if (I2CSTAT != mstDataNackR)
296  {
297  // something went wrong, so don't count last byte
298  cnt--;
299  }
300  break;
301  }
302  }
303 
304  I2C_STOP();
305  return cnt;
306 }

308 /**************************************************************************************************
309  * @fn HalI2CWrite
310  *
311  * @brief Write to the I2C bus as a Master.
312  *
313  * input parameters
314  *
315  * @param address - I2C address of the slave device we're attempting to write to
316  * @param len - Number of bytes to write.
317  * @param pBuf - Pointer to the data buffer to write.
318  *
319  * output parameters
320  *
321  * None.
322  *
323  * @return The number of bytes successfully written.
324  */
325 i2cLen_t HalI2CWrite(uint8 address, i2cLen_t len, uint8 *pBuf)
326 {
327  if (i2cMstStrt(address, 0) != mstAddrAckW)
328  {
329  len = 0;
330  }
331 
332  for (i2cLen_t cnt = 0; cnt < len; cnt++)
333  {
334  I2C_WRITE(*pBuf++);
335 
336  if (I2CSTAT != mstDataAckW)
337  {
338  if (I2CSTAT == mstDataNackW)
339  {
340  len = cnt + 1;
341  }
342  else
343  {
344  len = cnt;
345  }
346  break;
347  }
348  }
349 
350  I2C_STOP();
351  return len;
352 }

356 /**************************************************************************************************
357  * @fn i2cSlvRx
358  *
359  * @brief Block, reading from the I2C bus as a Slave.
360  *
361  * input parameters
362  *
363  * None.
364  *
365  * output parameters
366  *
367  * None.
368  *
369  * @return None.
370  */
371 static i2cLen_t i2cSlvRx(void)
372 {
373  uint8 ch, idx = 0;
374 
375  do {
376  SLV_READ(ch);
377 
378  if ((I2CSTAT == slvDataAckR) || (I2CSTAT == slvDataNackR))
379  {
380  i2cRxBuf[idx++] = ch;
381  }
382 
383  if (idx == HAL_I2C_BUF_MAX)
384  {
385  I2C_SET_NACK();
386  }
387  } while (I2CSTAT == slvDataAckR);
388 
389  i2cRxLen = idx;
390  i2cRxIdx = 0;
391  I2C_CLR_NACK();
392 
393  return idx;
394 }

396 /**************************************************************************************************
397  * @fn i2cSlvTx
398  *
399  * @brief Block, writing to the I2C bus as a Slave.
400  *
401  * input parameters
402  *
403  * None.
404  *
405  * output parameters
406  *
407  * None.
408  *
409  * @return None.
410  */
411 static i2cLen_t i2cSlvTx(void)
412 {
413  uint8 idx = i2cTxIdx, len = i2cTxLen;
414 
415  while (1)
416  {
417  if (len == 1)
418  {
419  I2C_SET_NACK(); // Setup to write last byte.
420  }
421  SLV_WRITE(i2cTxBuf[idx]);
422 
423  if ((I2CSTAT == slvDataAckW) || (I2CSTAT == slvLastAckW) || (I2CSTAT == slvDataNackW))
424  {
425  idx++;
426 
427  if ((--len == 0) || (I2CSTAT == slvDataNackW))
428  {
429  break;
430  }
431  }
432  else
433  {
434  break;
435  }
436  }
437 
438  i2cTxIdx = idx;
439  i2cTxLen = len;
440  I2C_CLR_NACK();
441 
442  return len;
443 }

512 /**************************************************************************************************
513  * @fn HalI2CWrite
514  *
515  * @brief Write the data to write to the I2C bus as a Slave.
516  *
517  * input parameters
518  *
519  * @param len - Length of the data buffer to write.
520  * @param pBuf - Pointer to the data buffer to write.
521  *
522  * output parameters
523  *
524  * None.
525  *
526  * @return The number of bytes setup to write.
527  */
528 i2cLen_t HalI2CWrite(i2cLen_t len, uint8 *pBuf)
529 {
530 #if ((HAL_I2C_BUF_MAX != 255) && (HAL_I2C_BUF_MAX != 65535))
531  if (len > HAL_I2C_BUF_MAX)
532  {
533  len = HAL_I2C_BUF_MAX;
534  }
535 #endif
536 
537  if (i2cTxLen != 0) // Refuse to overwrite old msg with new data so no race with ISR.
538  {
539  return 0;
540  }
541 
542  (void)osal_memcpy(i2cTxBuf, pBuf, len);
543  i2cTxIdx = 0;
544  i2cTxLen = len;
545 
546 #if HAL_I2C_POLLED
547  if (I2CSTAT == slvAddrAckW)
548  {
549  i2cSlvTx();
550  }
551 #else
552  // When the I2C slave is actively addressed for write and there is no Tx ready,
553  // this driver gets a promise from the Application that a write will be made and effects
554  // clock stretching, waiting for the Application to compose and now here to write the bytes
555  // to be transmitted. Unfortunately, when the slave is clock stretching, the H/W continues to
556  // fire the ISR and the Application cannot run (no throughput), so the I2C ISR has to be disabled
557  // while purposely clock stretching.
558  if (!I2C_INT_ENABLED())
559  {
560  I2C_INT_ENABLE(); // Constant ISR's being held off will trigger & invoke i2cSlvTx().
561  }
562 #endif
563 
564  return len;
565 }

567 /**************************************************************************************************
568  * @fn HalI2CPoll
569  *
570  * @brief Poll the I2C module as a Slave when not running by ISR.
571  *
572  * input parameters
573  *
574  * None.
575  *
576  * output parameters
577  *
578  * None.
579  *
580  * @return None.
581  */
582 #if HAL_I2C_POLLED
583 void HalI2CPoll(void)
584 #else
585 HAL_ISR_FUNCTION(halI2CIsr, I2C_VECTOR)
586 #endif
587 {
588  HAL_ENTER_ISR();
589 
590  switch (I2CSTAT)
591  {
592  case slvAddrAckR:
593  if (i2cRxLen == 0)
594  {
595  i2cSlvRx(); // Block, receiving bytes from the Master.
596  }
597  else // Rx overrun.
598  {
599  volatile uint8 ch;
600  // Throw away these new bytes & leave old msg so no race with background HalI2CRead().
601  I2C_SET_NACK();
602  SLV_READ(ch);
603  I2C_CLR_NACK();
604  }
605 
606  (void)i2cCB(i2cRxLen); // Alert Application that a Master Tx is ready to read.
607  break;
608 
609  case slvAddrAckW:
610  if (i2cTxLen != 0)
611  {
612  i2cSlvTx(); // Block, transmitting bytes to the Master.
613  }
614  else if (i2cCB(0) == FALSE)
615  {
616  I2C_SET_NACK();
617  SLV_WRITE(0);
618  I2C_CLR_NACK();
619  }
620  else
621  {
622  /* Slave will be clock-stretching, blocking master, until Application makes a HalI2CWrite().
623  * While addressed as a slave, the I2C ISR keeps firing even though the I2CSTAT is not
624  * changing and the flags are cleared below. So this special ISR disable here to be
625  * re-enabled within the slave HalI2CWrite() function.
626  */
627  I2C_INT_DISABLE();
628  }
629  break;
630 
631  case i2cIdle: // Not expected, but not really an error, so no need to execute a STOP.
632  break;
633 
634  default:
635  I2C_STOP();
636  I2C_CLR_NACK(); // Setup to Ack the next time addressed.
637  break;
638  }
639 
640  // Clear the CPU interrupt flag for Port_2 PxIFG has to be cleared before PxIF.
641  I2C_PXIFG = 0;
642  I2C_IF = 0;
643 
644  HAL_EXIT_ISR();
645 }
646 #endif
