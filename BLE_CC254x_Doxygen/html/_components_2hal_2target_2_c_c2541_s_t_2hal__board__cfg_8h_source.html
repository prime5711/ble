219 /* Setting Clocks */
220 
221 // switch to the 16MHz HSOSC and wait until it is stable
222 #define SET_OSC_TO_HSOSC() \
223 { \
224  CLKCONCMD = (CLKCONCMD & 0x80) | CLKCONCMD_16MHZ; \
225  while ( (CLKCONSTA & ~0x80) != CLKCONCMD_16MHZ ); \
226 }
227 
228 // switch to the 32MHz XOSC and wait until it is stable
229 #define SET_OSC_TO_XOSC() \
230 { \
231  CLKCONCMD = (CLKCONCMD & 0x80) | CLKCONCMD_32MHZ; \
232  while ( (CLKCONSTA & ~0x80) != CLKCONCMD_32MHZ ); \
233 }
234 
235 // set 32kHz OSC and wait until it is stable
236 #define SET_32KHZ_OSC() \
237 { \
238  CLKCONCMD = (CLKCONCMD & ~0x80) | OSC_32KHZ; \
239  while ( (CLKCONSTA & 0x80) != OSC_32KHZ ); \
240 }
241 
242 #define START_HSOSC_XOSC() \
243 { \
244  SLEEPCMD &= ~OSC_PD; /* start 16MHz RCOSC & 32MHz XOSC */ \
245  while (!(SLEEPSTA & XOSC_STB)); /* wait for stable 32MHz XOSC */ \
246 }
247 
248 #define STOP_HSOSC() \
249 { \
250  SLEEPCMD |= OSC_PD; /* stop 16MHz RCOSC */ \
251 }
252 /* ----------- Board Initialization ---------- */
253 
254 #define HAL_BOARD_INIT() \
255 { \
256  /* Set to 16Mhz to set 32kHz OSC, then back to 32MHz */ \
257  START_HSOSC_XOSC(); \
258  SET_OSC_TO_HSOSC(); \
259  SET_32KHZ_OSC(); \
260  SET_OSC_TO_XOSC(); \
261  STOP_HSOSC(); \
262  \
263  /* Turn on cache prefetch mode */ \
264  PREFETCH_ENABLE(); \
265  \
266  /* set direction for GPIO outputs */ \
267  LED1_DDR |= LED1_BV; \
268  LED2_DDR |= LED2_BV; \
269  GYRO_VDD_DDR |= GYRO_VDD_BV; \
270  DCDC_DDR |= DCDC_BV; /* Set P0_7 as output */ \
271  GYRO_VDD_SBIT = 1; /* Gyro must be on for I2C to work */ \
272  P0DIR |= BV(5); /* Unused pin as output */ \
273  P2DIR |= BV(0); /* Unused pin as output */ \
274  P1DIR |= 0x3C; /* UART pins as output */ \
275  P0INP = 0x4E; /* Tri-state inputs */ \
276 }
