152 /**************************************************************************************************
153  * @fn uartInit
154  *
155  * @brief Initialize the UART for SBL polling use.
156  *
157  * input parameters
158  *
159  * None.
160  *
161  * output parameters
162  *
163  * None.
164  *
165  * @return None.
166  */
167 static void uartInit(void)
168 {
169 #if defined POWER_SAVING
170  /* Although the SBL does not implement PM, follow the protocol so that a Host App does not need
171  * to drive the UART differently for communication with the NP versus the BL.
172  */
173  HAL_UART_SBL_SET_RDY_OUT();
174  PxDIR |= DMA_RDYOut_BIT;
175 #endif
176 
177  UxCSR = CSR_MODE; // Mode is UART Mode.
178  UxUCR = UCR_FLUSH; // Flush it.
179 #if (HAL_UART_SBL == 1)
180  PERCFG &= ~HAL_UART_PERCFG_BIT; // Set UART0 I/O to Alt. 1 location on P0.
181 #else
182  PERCFG |= HAL_UART_PERCFG_BIT; // Set UART1 I/O to Alt. 2 location on P1.
183 #endif
184  PxSEL |= HAL_UART_Px_SEL; // Enable Peripheral control of Rx/Tx on Px.
185 
186  P2DIR &= ~P2DIR_PRIPO;
187  P2DIR |= HAL_UART_PRIPO;
188 
189  // HAL_UART_BR_115200
190  UxBAUD = 216;
191  UxGCR = 11;
192 
193  UxUCR = UCR_STOP;
194  UxCSR = (CSR_MODE | CSR_RE);
195 }

197 /**************************************************************************************************
198  * @fn sblRun
199  *
200  * @brief Serial Boot run code for the UART transport.
201  *
202  * input parameters
203  *
204  * None.
205  *
206  * output parameters
207  *
208  * None.
209  *
210  * @return None.
211  */
212 static void sblRun(void)
213 {
214  uint8 resetF = 0;
215  uartInit();
216 
217  while (1)
218  {
219  if (txLen == 0)
220  {
221  if ((resetF != 0) && ((UxCSR & CSR_ACTIVE) == 0)) // If the last Tx byte has flushed out.
222  {
223  HAL_SYSTEM_RESET();
224  }
225  }
226  else if (UxCSR & CSR_TX_BYTE)
227  {
228  UxCSR &= ~CSR_TX_BYTE;
229  UxDBUF = pTxBuf[txIdx];
230 
231  if (++txIdx == txLen)
232  {
233  txLen = 0;
234  }
235  }
236 #if defined POWER_SAVING
237  else if (!HAL_UART_SBL_RDY_IN())
238  {
239  // Master may have timed-out the SRDY asserted state & may need a new edge.
240  HAL_UART_SBL_CLR_RDY_OUT();
241  ASM_NOP; ASM_NOP; ASM_NOP; ASM_NOP;
242  HAL_UART_SBL_SET_RDY_OUT();
243  }
244 #endif
245 
246  if (UxCSR & CSR_RX_BYTE)
247  {
248  resetF |= sblPoll();
249  }
250  }
251 }

253 /**************************************************************************************************
254  * @fn sblWait
255  *
256  * @brief Serial Boot wait & poll for a Force SBL or Force Jump after a hard reset.
257  *
258  * input parameters
259  *
260  * None.
261  *
262  * output parameters
263  *
264  * None.
265  *
266  * @return TRUE or FALSE to Force Boot Loader mode or not.
267  */
268 static uint8 sblWait(void)
269 {
270  uartInit();
271 
272  for (uint32 delay = 0; delay < 0x00100000; delay++)
273  {
274  if (UxCSR & CSR_RX_BYTE)
275  {
276  uint8 ch = UxDBUF;
277 
278  if (ch == SBL_FORCE_BOOT)
279  {
280  return TRUE;
281  }
282  else if (ch == SBL_FORCE_RUN)
283  {
284  return FALSE;
285  }
286  }
287  }
288 
289  return FALSE;
290 }
