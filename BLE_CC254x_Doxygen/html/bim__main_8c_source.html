103 /**************************************************************************************************
104  * @fn halSleepExec
105  *
106  * @brief This function puts the CC254x to sleep by writing to the PCON register.
107  * The instruction after writing to PCON must not be 4-byte aligned or excessive
108  * power consumption may result. Since the write to PCON is 3 instructions, this
109  * function is forced to be even-byte aligned. Thus, this function must not have any
110  * automatic variables and the write to PCON must be the first C statement.
111  * See the linker file ".xcl" for actual placement of this function.
112  *
113  * input parameters
114  *
115  * @param None.
116  *
117  * output parameters
118  *
119  * None.
120  *
121  * @return None.
122  **************************************************************************************************
123  */
124 #pragma optimize=none
125 void halSleepExec(void)
126 {
127  PCON = 0x01;
128  ASM_NOP;
129 }

132  * @fn crcCalcDMA
133  *
134  * @brief Run the CRC16 Polynomial calculation over the image specified,
135  * using DMA to read the flash memory into the CRC register
136  *
137  * input parameters
138  *
139  * @param page - Flash page on which to beging the CRC calculation.
140  *
141  * output parameters
142  *
143  * None.
144  *
145  * @return The CRC16 calculated.
146  **************************************************************************************************
147  */
148 static uint16 crcCalcDMA(uint8 page)
149 {
150  uint16 crc;
151  uint8 pageBeg;
152  uint8 pageEnd;
153  const img_hdr_t *pImgHdr;
154 
155  HalFlashRead(page, 0, pgBuf, HAL_FLASH_PAGE_SIZE);
156 
157  pImgHdr = (const img_hdr_t *)(pgBuf + BIM_HDR_OSET);
158 
159  pageBeg = page;
160  pageEnd = pImgHdr->len / (HAL_FLASH_PAGE_SIZE / HAL_FLASH_WORD_SIZE);
161 
162  // One page is used for BIM, so we move this image's last page forward
163  pageEnd += pageBeg;
164 
165  // If image A, set last page to be ImgA size + ImgB size
166  if (pageBeg == BIM_IMG_A_PAGE)
167  {
168  pageEnd += BIM_IMG_B_AREA;
169  }
170 
171  ADCCON1 &= 0xF3; // CRC configuration of LRSR.
172 
173  // CRC seed of 0x0000.
174  RNDL = 0x00;
175  RNDL = 0x00;
176 
177  // Handle first page differently to skip CRC and CRC shadow when calculating
178  DMAExecCrc(pageBeg, 4, HAL_FLASH_PAGE_SIZE-4);
179 
180  // Do remaining pages
181  for (uint8 pg = pageBeg + 1; pg < pageEnd; pg++)
182  {
183  if (pg == BIM_IMG_B_PAGE)
184  {
185  pg += BIM_IMG_B_AREA;
186  }
187 
188  DMAExecCrc(pg, 0, HAL_FLASH_PAGE_SIZE);
189  }
190 
191  crc = RNDH;
192  crc = (crc << 8) | RNDL;
193 
194  return crc;
195 }

310 /**************************************************************************************************
311  * @fn main
312  *
313  * @brief C-code main function.
314  *
315  * input parameters

326 void main(void)
327 {
328  uint16 crc[2];
329 
330  // Prefer to run Image-B over Image-A so that Image-A does not have to invalidate itself.
331  HalFlashRead(BIM_IMG_B_PAGE, BIM_CRC_OSET, (uint8 *)crc, 4);
332 
333  if ((crc[0] != 0xFFFF) && (crc[0] != 0x0000))
334  {
335  if (crc[0] == crc[1])
336  {
337  JumpToImageAorB = 1;
338  // Simulate a reset for the Application code by an absolute jump to the expected INTVEC addr.
339  asm("LJMP 0x4030");
340  HAL_SYSTEM_RESET(); // Should not get here.
341  }
342  /* This check is disruptive when an OAD process to Image-A is interrupted - this check must
343  * complete before the still good Image-A is run.
344  else if (crc[1] == 0xFFFF) // If first run of an image that was physically downloaded.*/
345  {
346  crcCheck(BIM_IMG_B_PAGE, crc);
347  }
348 
349  }
350 
351  HalFlashRead(BIM_IMG_A_PAGE, BIM_CRC_OSET, (uint8 *)crc, 4);
352 
353  if ((crc[0] != 0xFFFF) && (crc[0] != 0x0000))
354  {
355  if (crc[0] == crc[1])
356  {
357  JumpToImageAorB = 0;
358  // Simulate a reset for the Application code by an absolute jump to the expected INTVEC addr.
359  asm("LJMP 0x0830");
360  HAL_SYSTEM_RESET(); // Should not get here.
361  }
362  else if (crc[1] == 0xFFFF) // If first run of an image that was physically downloaded.
363  {
364  crcCheck(BIM_IMG_A_PAGE, crc);
365  }
366  }
367 
368  SLEEPCMD |= 0x03; // PM3, All clock oscillators off, voltage regulator off.
369  halSleepExec();
370  HAL_SYSTEM_RESET(); // Should not get here.
371 }

373 /**************************************************************************************************
374  * @fn DMAExecCrc
375  *
376  * @brief This function assumes CRC has been initialized and sets up and
377  * starts a dma tranfer from a flash page to the CRC HW module.
378  *
379  * @note This function assumes DMA channel 0 is available for use.
380  *
381  * input parameters
382  *
383  * @param page - A valid flash page number.
384  * @param offset - A valid offset into the page.
385  * @param len - A valid number of bytes to calculate crc of.
386  *
387  * @return None.
388  **************************************************************************************************
389  */
390 void DMAExecCrc(uint8 page, uint16 offset, uint16 len) {
391 
392  uint8 memctr = MEMCTR; // Save to restore.
393 
394  // Calculate the offset into the containing flash bank as it gets mapped into XDATA.
395  uint16 address = (offset + HAL_FLASH_PAGE_MAP) +
396  ((page % HAL_FLASH_PAGE_PER_BANK) * HAL_FLASH_PAGE_SIZE);
397 
398  // Pointer to DMA config structure
399  halDMADesc_t *dmaCh0_p = &dmaCh0;
400 
401 #if !defined HAL_OAD_BOOT_CODE
402  halIntState_t is;
403 #endif
404 
405  page /= HAL_FLASH_PAGE_PER_BANK; // Calculate the flash bank from the flash page.
406 
407 #if !defined HAL_OAD_BOOT_CODE
408  HAL_ENTER_CRITICAL_SECTION(is);
409 #endif
410 
411  // Calculate and map the containing flash bank into XDATA.
412  MEMCTR = (MEMCTR & 0xF8) | page; // page is actually bank
413 
414  // Start address for CRC calculation in the XDATA mapped flash bank
415  HAL_DMA_SET_SOURCE(dmaCh0_p, address);
416 
417  // Destination for data transfer, RNDH mapped to XDATA
418  HAL_DMA_SET_DEST(dmaCh0_p, 0x70BD);
419 
420  // One whole page (or len) at a time
421  HAL_DMA_SET_LEN(dmaCh0_p, len);
422 
423  // 8-bit, block, no trigger
424  HAL_DMA_SET_WORD_SIZE(dmaCh0_p, HAL_DMA_WORDSIZE_BYTE);
425  HAL_DMA_SET_TRIG_MODE(dmaCh0_p, HAL_DMA_TMODE_BLOCK);
426  HAL_DMA_SET_TRIG_SRC(dmaCh0_p, HAL_DMA_TRIG_NONE);
427 
428  // SRC += 1, DST = constant, no IRQ, all 8 bits, high priority
429  HAL_DMA_SET_SRC_INC(dmaCh0_p, HAL_DMA_SRCINC_1);
430  HAL_DMA_SET_DST_INC(dmaCh0_p, HAL_DMA_DSTINC_0);
431  HAL_DMA_SET_IRQ(dmaCh0_p, HAL_DMA_IRQMASK_DISABLE);
432  HAL_DMA_SET_M8(dmaCh0_p, HAL_DMA_M8_USE_8_BITS);
433  HAL_DMA_SET_PRIORITY(dmaCh0_p, HAL_DMA_PRI_HIGH);
434 
435  // Tell DMA Controller where above configuration can be found
436  HAL_DMA_SET_ADDR_DESC0(&dmaCh0);
437 
438  // Arm the DMA channel (0)
439  HAL_DMA_ARM_CH(0);
440 
441  // 9 cycles wait
442  asm("nop"); asm("nop"); asm("nop"); asm("nop"); asm("nop"); asm("nop");
443  asm("nop"); asm("nop"); asm("nop"); asm("nop"); asm("nop");
444 
445  // Start DMA tranfer
446  HAL_DMA_MAN_TRIGGER(0);
447 
448  // Wait for dma to finish.
449  while(DMAREQ & 0x1);
450 
451  // Restore bank mapping
452  MEMCTR = memctr;
453 
454 #if !defined HAL_OAD_BOOT_CODE
455  HAL_EXIT_CRITICAL_SECTION(is);
456 #endif
457 }
