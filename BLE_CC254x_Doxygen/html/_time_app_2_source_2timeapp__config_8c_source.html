120 /*********************************************************************
121  * @fn timeAppConfigNext()
122  *
123  * @brief Perform the characteristic configuration read or
124  * write procedure.
125  *
126  * @param state - Configuration state.
127  *
128  * @return New configuration state.
129  */
130 uint8 timeAppConfigNext( uint8 state )
131 {
132  attReadReq_t readReq;
133  attWriteReq_t writeReq;
134  bool read;
135  static uint8 alertNtfCtrlCmd;
136 
137  // Find next non-zero cached handle of interest
138  while ( state < TIMEAPP_CONFIG_MAX &&
139  timeAppHdlCache[timeAppConfigList[state]] == 0 )
140  {
141  state++;
142  }
143 
144  // Return if reached end of list
145  if ( state == TIMEAPP_CONFIG_MAX )
146  {
147  return TIMEAPP_CONFIG_CMPL;
148  }
149 
150  // Determine what to do with characteristic
151  switch ( timeAppConfigList[state] )
152  {
153  // Read these characteristics
154  case HDL_CURR_TIME_LOC_INFO:
155  case HDL_CURR_TIME_REF_INFO:
156  case HDL_DST_CHG_TIME_DST:
157  case HDL_NWA_NWA_START:
158  case HDL_CURR_TIME_CT_TIME_START:
159  case HDL_ALERT_NTF_NEW_CAT:
160  case HDL_ALERT_NTF_UNREAD_CAT:
161  case HDL_PAS_ALERT_START:
162  case HDL_PAS_RINGER_START:
163  case HDL_REF_TIME_UPD_STATE:
164  read = TRUE;
165  break;
166 
167  // Set notification for these characteristics
168  case HDL_CURR_TIME_CT_TIME_CCCD:
169  case HDL_ALERT_NTF_UNREAD_CCCD:
170  case HDL_ALERT_NTF_NEW_CCCD:
171  case HDL_BATT_LEVEL_CCCD:
172  case HDL_PAS_ALERT_CCCD:
173  case HDL_PAS_RINGER_CCCD:
174  read = FALSE;
175  writeReq.len = 2;
176  writeReq.value[0] = LO_UINT16(GATT_CLIENT_CFG_NOTIFY);
177  writeReq.value[1] = HI_UINT16(GATT_CLIENT_CFG_NOTIFY);
178  writeReq.sig = 0;
179  writeReq.cmd = 0;
180  break;
181 
182  // Set indication for these characteristics
183  case HDL_NWA_NWA_CCCD:
184  read = FALSE;
185  writeReq.len = 2;
186  writeReq.value[0] = LO_UINT16(GATT_CLIENT_CFG_INDICATE);
187  writeReq.value[1] = HI_UINT16(GATT_CLIENT_CFG_INDICATE);
188  writeReq.sig = 0;
189  writeReq.cmd = 0;
190  break;
191 
192  // Alert notification control point
193  case HDL_ALERT_NTF_CTRL:
194  // initialize control point command
195  if (state == TIMEAPP_ALERT_NTF_CTRL_START)
196  {
197  alertNtfCtrlCmd = ALERT_NOTIF_ENABLE_NEW;
198  }
199 
200  read = FALSE;
201  writeReq.len = 2;
202  writeReq.value[0] = alertNtfCtrlCmd;
203  writeReq.value[1] = ALERT_NOTIF_CAT_ALL;
204  writeReq.sig = 0;
205  writeReq.cmd = 0;
206 
207  // set next command to send
208  if (alertNtfCtrlCmd == ALERT_NOTIF_ENABLE_NEW)
209  {
210  alertNtfCtrlCmd = ALERT_NOTIF_NOTIFY_NEW;
211  }
212  else if (alertNtfCtrlCmd == ALERT_NOTIF_NOTIFY_NEW)
213  {
214  alertNtfCtrlCmd = ALERT_NOTIF_ENABLE_UNREAD;
215  }
216  else if (alertNtfCtrlCmd == ALERT_NOTIF_ENABLE_UNREAD)
217  {
218  alertNtfCtrlCmd = ALERT_NOTIF_NOTIFY_UNREAD;
219  }
220  break;
221 
222  default:
223  break;
224  }
225 
226  // Do a GATT read or write
227  if ( read )
228  {
229  readReq.handle = timeAppHdlCache[timeAppConfigList[state]];
230  GATT_ReadCharValue( timeAppConnHandle, &readReq, timeAppTaskId );
231  }
232  else
233  {
234  writeReq.handle = timeAppHdlCache[timeAppConfigList[state]];
235  GATT_WriteCharValue( timeAppConnHandle, &writeReq, timeAppTaskId );
236  }
237 
238  return state;
239 }

241 /*********************************************************************
242  * @fn timeAppConfigGattMsg()
243  *
244  * @brief Handle GATT messages for characteristic configuration.
245  *
246  * @param state - Discovery state.
247  * @param pMsg - GATT message.
248  *
249  * @return New configuration state.
250  */
251 uint8 timeAppConfigGattMsg( uint8 state, gattMsgEvent_t *pMsg )
252 {
253  if ( state > TIMEAPP_CONFIG_MAX )
254  {
255  return TIMEAPP_CONFIG_CMPL;
256  }
257 
258  if ( (pMsg->method == ATT_READ_RSP || pMsg->method == ATT_WRITE_RSP) &&
259  (pMsg->hdr.status == SUCCESS) )
260  {
261  // Process response
262  switch ( timeAppConfigList[state] )
263  {
264  case HDL_CURR_TIME_CT_TIME_START:
265  // Set clock to time read from time server
266  timeAppClockSet( pMsg->msg.readRsp.value );
267  break;
268 
269  case HDL_CURR_TIME_LOC_INFO:
270  break;
271 
272  case HDL_CURR_TIME_REF_INFO:
273  break;
274 
275  case HDL_DST_CHG_TIME_DST:
276  break;
277 
278  case HDL_NWA_NWA_START:
279  // Display network availability state
280  if ( pMsg->msg.readRsp.value[0] == 1 )
281  {
282  LCD_WRITE_STRING( "Network: Yes", HAL_LCD_LINE_1 );
283  }
284  else
285  {
286  LCD_WRITE_STRING( "Network: None", HAL_LCD_LINE_1 );
287  }
288  break;
289 
290  case HDL_BATT_LEVEL_START:
291  // Display battery level
292  LCD_WRITE_STRING_VALUE( "Battery%", pMsg->msg.readRsp.value[0], 10, HAL_LCD_LINE_2 );
293  break;
294 
295  case HDL_CURR_TIME_CT_TIME_CCCD:
296  break;
297 
298  case HDL_ALERT_NTF_NEW_CAT:
299  break;
300 
301  case HDL_ALERT_NTF_UNREAD_CAT:
302  break;
303 
304  case HDL_ALERT_NTF_UNREAD_CCCD:
305  break;
306 
307  case HDL_ALERT_NTF_NEW_CCCD:
308  break;
309 
310  case HDL_NWA_NWA_CCCD:
311  break;
312 
313  case HDL_PAS_ALERT_START:
314  // Display phone alert status
315  LCD_WRITE_STRING_VALUE( "Phone Alert:", pMsg->msg.readRsp.value[0], 16, HAL_LCD_LINE_1 );
316  break;
317 
318  case HDL_PAS_RINGER_START:
319  // Display ringer state
320  if ( pMsg->msg.readRsp.value[0] == 0 )
321  {
322  LCD_WRITE_STRING( "Ringer Off", HAL_LCD_LINE_2 );
323  }
324  else
325  {
326  LCD_WRITE_STRING( "Ringer On", HAL_LCD_LINE_2 );
327  }
328  break;
329 
330  default:
331  break;
332  }
333  }
334 
335  return timeAppConfigNext( state + 1 );
336 }
