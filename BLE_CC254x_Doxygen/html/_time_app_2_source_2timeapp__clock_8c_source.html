119 /*********************************************************************
120  * @fn timeAppClockDisplay()
121  *
122  * @brief Write the clock time to the display.
123  *
124  * @return none
125  */
126 void timeAppClockDisplay( void )
127 {
128  char displayBuf[HAL_LCD_MAX_CHARS];
129  char *p = displayBuf;
130  UTCTimeStruct time;
131 
132  // Get time structure from OSAL
133  osal_ConvertUTCTime( &time, osal_getClock() );
134 
135  // Display is in the format:
136  // HH:MM MmmDD YYYY
137 
138  p = num2Str( p, time.hour );
139  *p++ = ':';
140  p = num2Str( p, time.minutes );
141  *p++ = ' ';
142 
143  *p++ = timeAppMonthStr[time.month][0];
144  *p++ = timeAppMonthStr[time.month][1];
145  *p++ = timeAppMonthStr[time.month][2];
146 
147  p = num2Str( p, time.day + 1 );
148  *p++ = ' ';
149 
150  _ltoa( time.year, (uint8 *) p, 10 );
151 
152  LCD_WRITE_STRING( displayBuf, HAL_LCD_LINE_3 );
153 }

168 /*********************************************************************
169  * @fn timeAppClockSet()
170  *
171  * @brief Set the clock.
172  *
173  * @param pData - Pointer to a Date Time characteristic structure
174  *
175  * @return none
176  */
177 void timeAppClockSet( uint8 *pData )
178 {
179  UTCTimeStruct time;
180 
181  // Parse time service structure to OSAL time structure
182  time.year = BUILD_UINT16(pData[0], pData[1]);
183  if ( time.year == 0 )
184  {
185  time.year = 2000;
186  }
187  pData += 2;
188  time.month = *pData++;
189  if ( time.month > 0 )
190  {
191  time.month--;
192  }
193  time.day = *pData++;
194  if ( time.day > 0 )
195  {
196  time.day--;
197  }
198  time.hour = *pData++;
199  time.minutes = *pData++;
200  time.seconds = *pData;
201 
202  // Update OSAL time
203  osal_setClock( osal_ConvertUTCSecs( &time ) );
204 
205  timeAppClockDisplay();
206 }
