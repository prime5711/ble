106 /*********************************************************************
107  * @fn timeAppIndGattMsg
108  *
109  * @brief Handle indications and notifications.
110  *
111  * @param pMsg - GATT message.
112  *
113  * @return none
114  */
115 void timeAppIndGattMsg( gattMsgEvent_t *pMsg )
116 {
117  uint8 i;
118 
119  // Look up the handle in the handle cache
120  for ( i = 0; i < HDL_CACHE_LEN; i++ )
121  {
122  if ( pMsg->msg.handleValueNoti.handle == timeAppHdlCache[i] )
123  {
124  break;
125  }
126  }
127 
128  // Perform processing for this handle
129  switch ( i )
130  {
131  case HDL_CURR_TIME_CT_TIME_START:
132  // Set clock to time read from time server
133  timeAppClockSet( pMsg->msg.handleValueNoti.value );
134  break;
135 
136  case HDL_NWA_NWA_START:
137  // Display network availability state
138  if ( pMsg->msg.handleValueInd.value[0] == 1 )
139  {
140  LCD_WRITE_STRING( "Network: Yes", HAL_LCD_LINE_1 );
141  }
142  else
143  {
144  LCD_WRITE_STRING( "Network: None", HAL_LCD_LINE_1 );
145  }
146  break;
147 
148  case HDL_ALERT_NTF_UNREAD_START:
149  // Display unread message alert
150  {
151  uint8 *p = pMsg->msg.handleValueNoti.value;
152  uint8 len = pMsg->msg.handleValueNoti.len;
153 
154  if ( p[0] <= ALERT_CAT_ID_MAX && len == 2 )
155  {
156  LCD_WRITE_STRING_VALUE( (char *) timeAppAlertCatStr[p[0]], p[1], 10, HAL_LCD_LINE_1 );
157  LCD_WRITE_STRING( "", HAL_LCD_LINE_2 );
158  }
159  }
160  break;
161 
162  case HDL_ALERT_NTF_NEW_START:
163  // Display incoming message
164  timeAppDisplayAlert( pMsg->msg.handleValueNoti.value,
165  pMsg->msg.handleValueNoti.len );
166  break;
167 
168  case HDL_BATT_LEVEL_START:
169  // Display battery level
170  LCD_WRITE_STRING_VALUE( "Battery%", pMsg->msg.handleValueNoti.value[0], 10, HAL_LCD_LINE_2 );
171  break;
172 
173  case HDL_PAS_ALERT_START:
174  // Display phone alert status
175  LCD_WRITE_STRING_VALUE( "Phone Alert:", pMsg->msg.handleValueNoti.value[0], 16, HAL_LCD_LINE_1 );
176  break;
177 
178  case HDL_PAS_RINGER_START:
179  // Display ringer state
180  if ( pMsg->msg.handleValueNoti.value[0] == 0 )
181  {
182  LCD_WRITE_STRING( "Ringer Off", HAL_LCD_LINE_2 );
183  }
184  else
185  {
186  LCD_WRITE_STRING( "Ringer On", HAL_LCD_LINE_2 );
187  }
188  break;
189 
190  default:
191  break;
192  }
193 
194  // Send confirm for indication
195  if ( pMsg->method == ATT_HANDLE_VALUE_IND )
196  {
197  ATT_HandleValueCfm( pMsg->connHandle );
198  }
199 }

201 /*********************************************************************
202  * @fn timeAppIndGattMsg
203  *
204  * @brief Handle indications and notifications.
205  *
206  * @param pValue - Pointer to buffer containing a new incoming alert
207  * characteristic.
208  * @param len - length of pValue.
209  *
210  * @return none
211  */
212 static void timeAppDisplayAlert( uint8 *pValue, uint8 len )
213 {
214  char buf[HAL_LCD_MAX_CHARS + 1];
215  uint8 buflen;
216 
217  // Verify alert category and length
218  if ( pValue[0] <= ALERT_CAT_ID_MAX && len >= 2 )
219  {
220  // Write category and number of unread to first line
221  LCD_WRITE_STRING_VALUE( (char *) timeAppAlertCatStr[pValue[0]], pValue[1], 10, HAL_LCD_LINE_1 );
222  pValue += 2;
223  len -= 2;
224 
225  // Write alert text to second line
226  buflen = MIN( HAL_LCD_MAX_CHARS, len );
227  memcpy( buf, pValue, buflen );
228  buf[buflen] = '\0';
229  LCD_WRITE_STRING( buf, HAL_LCD_LINE_2 );
230  }
231 }
