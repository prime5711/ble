57 // The SB page boundary since all SB addresses are "actual address / flash word size".
58 // Note for MSP - flash word size is 1, but 4 must be used for inter-compatibility w/ SoC.

66 // Commands to Bootloader
67 #define SBL_WRITE_CMD 0x01
68 #define SBL_READ_CMD 0x02
69 #define SBL_ENABLE_CMD 0x03
70 #define SBL_HANDSHAKE_CMD 0x04
71 
72 // If an Application is to increase the number of NV pages used, the SBL AP can dynamically change
73 // the NV area "skipped" by the BootLoader. The payload is the Address to begin skipping / 4.
74 #define SBL_SKIP_ADDR_BEG_CMD 0x08
75 // If SBL_READ_CMD is disabled for security and the BootLoader must calculated the CRC, the code
76 // address end for the calculation with differ with the flash size. Instead of requiring a
77 // different BootLoader image for each flash size, and this flexibility to dynamically increase
78 // this code address end. This flexibility will also allow custom use of the end of the flash range
79 // for a 2nd, proprietary NV or other storage area.
80 #define SBL_CODE_ADDR_END_CMD 0x09
81 
82 // Commands to Target Application
83 #define SBL_TGT_BOOTLOAD 0x10 // Erase the image valid signature & jump to bootloader.
84 
85 // Responses from Bootloader - for historical consistency, SBL has OR'ed the MSBit of all commands
86 // when responding - this is probably not necessary for smooth functionality.
87 #define SBL_RSP_MASK 0x80
