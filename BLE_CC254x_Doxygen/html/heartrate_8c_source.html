359 /*********************************************************************
360  * @fn HeartRate_ProcessEvent
361  *
362  * @brief Heart Rate Application Task event processor. This function
363  * is called to process all events for the task. Events
364  * include timers, messages and any other user defined events.
365  *
366  * @param task_id - The OSAL assigned task ID.
367  * @param events - events to process. This is a bit map and can
368  * contain more than one event.
369  *
370  * @return events not processed
371  */
372 uint16 HeartRate_ProcessEvent( uint8 task_id, uint16 events )
373 {
374 
375  VOID task_id; // OSAL required parameter that isn't used in this function
376 
377  if ( events & SYS_EVENT_MSG )
378  {
379  uint8 *pMsg;
380 
381  if ( (pMsg = osal_msg_receive( heartRate_TaskID )) != NULL )
382  {
383  heartRate_ProcessOSALMsg( (osal_event_hdr_t *)pMsg );
384 
385  // Release the OSAL message
386  VOID osal_msg_deallocate( pMsg );
387  }
388 
389  // return unprocessed events
390  return (events ^ SYS_EVENT_MSG);
391  }
392 
393  if ( events & START_DEVICE_EVT )
394  {
395  // Start the Device
396  VOID GAPRole_StartDevice( &heartRatePeripheralCB );
397 
398  // Register with bond manager after starting device
399  GAPBondMgr_Register( (gapBondCBs_t *) &heartRateBondCB );
400 
401  return ( events ^ START_DEVICE_EVT );
402  }
403 
404  if ( events & HEART_PERIODIC_EVT )
405  {
406  // Perform periodic heart rate task
407  heartRatePeriodicTask();
408 
409  return (events ^ HEART_PERIODIC_EVT);
410  }
411 
412  if ( events & BATT_PERIODIC_EVT )
413  {
414  // Perform periodic battery task
415  heartRateBattPeriodicTask();
416 
417  return (events ^ BATT_PERIODIC_EVT);
418  }
419 
420  // Discard unknown events
421  return 0;
422 }

443 /*********************************************************************
444  * @fn heartRate_HandleKeys
445  *
446  * @brief Handles all key events for this device.
447  *
448  * @param shift - true if in shift/alt.
449  * @param keys - bit field for key events. Valid entries:
450  * HAL_KEY_SW_2
451  * HAL_KEY_SW_1
452  *
453  * @return none
454  */
455 static void heartRate_HandleKeys( uint8 shift, uint8 keys )
456 {
457  if ( keys & HAL_KEY_SW_1 )
458  {
459  // set simulated measurement flag index
460  if (++heartRateFlagsIdx == FLAGS_IDX_MAX)
461  {
462  heartRateFlagsIdx = 0;
463  }
464  }
465 
466  if ( keys & HAL_KEY_SW_2 )
467  {
468  // if not in a connection, toggle advertising on and off
469  if( gapProfileState != GAPROLE_CONNECTED )
470  {
471  uint8 status;
472 
473  // Set fast advertising interval for user-initiated connections
474  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MIN, DEFAULT_FAST_ADV_INTERVAL );
475  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MAX, DEFAULT_FAST_ADV_INTERVAL );
476  GAP_SetParamValue( TGAP_GEN_DISC_ADV_MIN, DEFAULT_FAST_ADV_DURATION );
477 
478  // toggle GAP advertisement status
479  GAPRole_GetParameter( GAPROLE_ADVERT_ENABLED, &status );
480  status = !status;
481  GAPRole_SetParameter( GAPROLE_ADVERT_ENABLED, sizeof( uint8 ), &status );
482 
483  // Set state variable
484  if (status == FALSE)
485  {
486  heartRateAdvCancelled = TRUE;
487  }
488  }
489  }
490 }

492 /*********************************************************************
493  * @fn heartRateMeasNotify
494  *
495  * @brief Prepare and send a heart rate measurement notification
496  *
497  * @return none
498  */
499 static void heartRateMeasNotify(void)
500 {
501  uint8 *p = heartRateMeas.value;
502  uint8 flags = heartRateFlags[heartRateFlagsIdx];
503 
504  // build heart rate measurement structure from simulated values
505  *p++ = flags;
506  *p++ = heartRateBpm;
507  if (flags & HEARTRATE_FLAGS_FORMAT_UINT16)
508  {
509  // additional byte for 16 bit format
510  *p++ = 0;
511  }
512  if (flags & HEARTRATE_FLAGS_ENERGY_EXP)
513  {
514  *p++ = LO_UINT16(heartRateEnergy);
515  *p++ = HI_UINT16(heartRateEnergy);
516  }
517  if (flags & HEARTRATE_FLAGS_RR)
518  {
519  *p++ = LO_UINT16(heartRateRrInterval1);
520  *p++ = HI_UINT16(heartRateRrInterval1);
521  *p++ = LO_UINT16(heartRateRrInterval2);
522  *p++ = HI_UINT16(heartRateRrInterval2);
523  }
524  heartRateMeas.len = (uint8) (p - heartRateMeas.value);
525  HeartRate_MeasNotify( gapConnHandle, &heartRateMeas );
526 
527  // update simulated values
528  heartRateEnergy += ENERGY_INCREMENT;
529  if (++heartRateBpm == BPM_MAX)
530  {
531  heartRateBpm = BPM_DEFAULT;
532  }
533  heartRateRrInterval1 = heartRateRrInterval2 = HEARTRATE_BPM_TO_RR(heartRateBpm);
534 }

536 /*********************************************************************
537  * @fn HeartRateGapStateCB
538  *
539  * @brief Notification from the profile of a state change.
540  *
541  * @param newState - new state
542  *
543  * @return none
544  */
545 static void HeartRateGapStateCB( gaprole_States_t newState )
546 {
547  // if connected
548  if (newState == GAPROLE_CONNECTED)
549  {
550  // get connection handle
551  GAPRole_GetParameter(GAPROLE_CONNHANDLE, &gapConnHandle);
552  }
553  // if disconnected
554  else if (gapProfileState == GAPROLE_CONNECTED &&
555  newState != GAPROLE_CONNECTED)
556  {
557  uint8 advState = TRUE;
558 
559  // stop periodic measurement
560  osal_stop_timerEx( heartRate_TaskID, HEART_PERIODIC_EVT );
561 
562  // reset client characteristic configuration descriptors
563  HeartRate_HandleConnStatusCB( gapConnHandle, LINKDB_STATUS_UPDATE_REMOVED );
564  Batt_HandleConnStatusCB( gapConnHandle, LINKDB_STATUS_UPDATE_REMOVED );
565 
566  if ( newState == GAPROLE_WAITING_AFTER_TIMEOUT )
567  {
568  // link loss timeout-- use fast advertising
569  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MIN, DEFAULT_FAST_ADV_INTERVAL );
570  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MAX, DEFAULT_FAST_ADV_INTERVAL );
571  GAP_SetParamValue( TGAP_GEN_DISC_ADV_MIN, DEFAULT_FAST_ADV_DURATION );
572  }
573  else
574  {
575  // Else use slow advertising
576  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MIN, DEFAULT_SLOW_ADV_INTERVAL );
577  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MAX, DEFAULT_SLOW_ADV_INTERVAL );
578  GAP_SetParamValue( TGAP_GEN_DISC_ADV_MIN, DEFAULT_SLOW_ADV_DURATION );
579  }
580 
581  // Enable advertising
582  GAPRole_SetParameter( GAPROLE_ADVERT_ENABLED, sizeof( uint8 ), &advState );
583  }
584  // if advertising stopped
585  else if ( gapProfileState == GAPROLE_ADVERTISING &&
586  newState == GAPROLE_WAITING )
587  {
588  // if advertising stopped by user
589  if ( heartRateAdvCancelled )
590  {
591  heartRateAdvCancelled = FALSE;
592  }
593  // if fast advertising switch to slow
594  else if ( GAP_GetParamValue( TGAP_GEN_DISC_ADV_INT_MIN ) == DEFAULT_FAST_ADV_INTERVAL )
595  {
596  uint8 advState = TRUE;
597 
598  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MIN, DEFAULT_SLOW_ADV_INTERVAL );
599  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MAX, DEFAULT_SLOW_ADV_INTERVAL );
600  GAP_SetParamValue( TGAP_GEN_DISC_ADV_MIN, DEFAULT_SLOW_ADV_DURATION );
601  GAPRole_SetParameter( GAPROLE_ADVERT_ENABLED, sizeof( uint8 ), &advState );
602  }
603  }
604  // if started
605  else if (newState == GAPROLE_STARTED)
606  {
607  // Set the system ID from the bd addr
608  uint8 systemId[DEVINFO_SYSTEM_ID_LEN];
609  GAPRole_GetParameter(GAPROLE_BD_ADDR, systemId);
610 
611  // shift three bytes up
612  systemId[7] = systemId[5];
613  systemId[6] = systemId[4];
614  systemId[5] = systemId[3];
615 
616  // set middle bytes to zero
617  systemId[4] = 0;
618  systemId[3] = 0;
619 
620  DevInfo_SetParameter(DEVINFO_SYSTEM_ID, DEVINFO_SYSTEM_ID_LEN, systemId);
621  }
622 
623  gapProfileState = newState;
624 }

626 /*********************************************************************
627  * @fn heartRateCB
628  *
629  * @brief Callback function for heart rate service.
630  *
631  * @param event - service event
632  *
633  * @return none
634  */
635 static void heartRateCB(uint8 event)
636 {
637  if (event == HEARTRATE_MEAS_NOTI_ENABLED)
638  {
639  // if connected start periodic measurement
640  if (gapProfileState == GAPROLE_CONNECTED)
641  {
642  osal_start_timerEx( heartRate_TaskID, HEART_PERIODIC_EVT, DEFAULT_HEARTRATE_PERIOD );
643  }
644  }
645  else if (event == HEARTRATE_MEAS_NOTI_DISABLED)
