29 // **************** Process USB class requests with OUT data phase ************************
30 void usbcrHookProcessOut(void)
31 {
32  // Process USB class requests with OUT data phase, or stall endpoint 0 when unsupported
33  switch (usbSetupHeader.request) {
34  case SET_REPORT: usbcrSetReport(); break;
35  case SET_PROTOCOL: usbcrSetProtocol(); break;
36  case SET_IDLE: usbcrSetIdle(); break;
37  default: usbfwData.ep0Status = EP_STALL; break;
38  }
39 }

72 // ************************ USB standard request event processing *************************
73 void usbsrHookProcessEvent(uint8 event, uint8 index)
74 {
75  (void)index;
76 
77  // Process relevant events, one at a time.
78  switch (event) {
79  case USBSR_EVENT_CONFIGURATION_CHANGING : //(the device configuration is about to change)
80  break;
81  case USBSR_EVENT_CONFIGURATION_CHANGED :// (the device configuration has changed)
82  break;
83  case USBSR_EVENT_INTERFACE_CHANGING ://(the alternate setting of the given interface is about to change)
84  break;
85  case USBSR_EVENT_INTERFACE_CHANGED : //(the alternate setting of the given interface has changed)
86  break;
87  case USBSR_EVENT_REMOTE_WAKEUP_ENABLED ://(remote wakeup has been enabled by the host)
88  break;
89  case USBSR_EVENT_REMOTE_WAKEUP_DISABLED ://(remote wakeup has been disabled by the host)
90  break;
91  case USBSR_EVENT_EPIN_STALL_CLEARED ://(the given IN endpoint's stall condition has been cleared the host)
92  break;
93  case USBSR_EVENT_EPIN_STALL_SET ://(the given IN endpoint has been stalled by the host)
94  break;
95  case USBSR_EVENT_EPOUT_STALL_CLEARED ://(the given OUT endpoint's stall condition has been cleared the host)
96  break;
97  case USBSR_EVENT_EPOUT_STALL_SET ://(the given OUT endpoint has been stalled by the PC)
98  break;
99  }
