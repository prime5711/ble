188 /*********************************************************************
189  * @fn SK_AddService
190  *
191  * @brief Initializes the Simple Key service by registering
192  * GATT attributes with the GATT server.
193  *
194  * @param services - services to add. This is a bit map and can
195  * contain more than one service.
196  *
197  * @return Success or Failure
198  */
199 bStatus_t SK_AddService( uint32 services )
200 {
201  uint8 status = SUCCESS;
202 
203  // Initialize Client Characteristic Configuration attributes
204  GATTServApp_InitCharCfg( INVALID_CONNHANDLE, skConfig );
205 
206  // Register with Link DB to receive link status change callback
207  VOID linkDB_Register( sk_HandleConnStatusCB );
208 
209 
210  if ( services & SK_SERVICE )
211  {
212  // Register GATT attribute list and CBs with GATT Server App
213  status = GATTServApp_RegisterService( simplekeysAttrTbl,
214  GATT_NUM_ATTRS( simplekeysAttrTbl ),
215  &skCBs );
216  }
217 
218  return ( status );
219 }

221 /*********************************************************************
222  * @fn SK_SetParameter
223  *
224  * @brief Set a Simple Key Profile parameter.
225  *
226  * @param param - Profile parameter ID
227  * @param len - length of data to right
228  * @param pValue - pointer to data to write. This is dependent on
229  * the parameter ID and WILL be cast to the appropriate
230  * data type (example: data type of uint16 will be cast to
231  * uint16 pointer).
232  *
233  * @return bStatus_t
234  */
235 bStatus_t SK_SetParameter( uint8 param, uint8 len, void *pValue )
236 {
237  bStatus_t ret = SUCCESS;
238  switch ( param )
239  {
240  case SK_KEY_ATTR:
241  if ( len == sizeof ( uint8 ) )
242  {
243  skKeyPressed = *((uint8*)pValue);
244 
245  // See if Notification/Indication has been enabled
246  GATTServApp_ProcessCharCfg( skConfig, &skKeyPressed, FALSE,
247  simplekeysAttrTbl, GATT_NUM_ATTRS( simplekeysAttrTbl ),
248  INVALID_TASK_ID );
249  }
250  else
251  {
252  ret = bleInvalidRange;
253  }
254  break;
255 
256  default:
257  ret = INVALIDPARAMETER;
258  break;
259  }
260 
261  return ( ret );
262 }

294 /*********************************************************************
295  * @fn sk_ReadAttrCB
296  *
297  * @brief Read an attribute.
298  *
299  * @param connHandle - connection message was received on
300  * @param pAttr - pointer to attribute
301  * @param pValue - pointer to data to be read
302  * @param pLen - length of data to be read
303  * @param offset - offset of the first octet to be read
304  * @param maxLen - maximum length of data to be read
305  *
306  * @return Success or Failure
307  */
308 static uint8 sk_ReadAttrCB( uint16 connHandle, gattAttribute_t *pAttr,
309  uint8 *pValue, uint8 *pLen, uint16 offset, uint8 maxLen )
310 {
311  bStatus_t status = SUCCESS;
312 
313  // Make sure it's not a blob operation (no attributes in the profile are long
314  if ( offset > 0 )
315  {
316  return ( ATT_ERR_ATTR_NOT_LONG );
317  }
318 
319  if ( pAttr->type.len == ATT_BT_UUID_SIZE )
320  {
321  // 16-bit UUID
322  uint16 uuid = BUILD_UINT16( pAttr->type.uuid[0], pAttr->type.uuid[1]);
323  switch ( uuid )
324  {
325  // No need for "GATT_SERVICE_UUID" or "GATT_CLIENT_CHAR_CFG_UUID" cases;
326  // gattserverapp handles this type for reads
327 
328  // simple keys characteristic does not have read permissions, but because it
329  // can be sent as a notification, it must be included here
330  case SK_KEYPRESSED_UUID:
331  *pLen = 1;
332  pValue[0] = *pAttr->pValue;
333  break;
334 
335  default:
336  // Should never get here!
337  *pLen = 0;
338  status = ATT_ERR_ATTR_NOT_FOUND;
339  break;
340  }
341  }
342  else
343  {
344  // 128-bit UUID
345  *pLen = 0;
346  status = ATT_ERR_INVALID_HANDLE;
347  }
348 
349  return ( status );
350 }

352 /*********************************************************************
353  * @fn sk_WriteAttrCB
354  *
355  * @brief Validate attribute data prior to a write operation
356  *
357  * @param connHandle - connection message was received on
358  * @param pAttr - pointer to attribute
359  * @param pValue - pointer to data to be written
360  * @param len - length of data
361  * @param offset - offset of the first octet to be written
362  *
363  * @return Success or Failure
364  */
365 static bStatus_t sk_WriteAttrCB( uint16 connHandle, gattAttribute_t *pAttr,
366  uint8 *pValue, uint8 len, uint16 offset )
367 {
368  bStatus_t status = SUCCESS;
369 
370  if ( pAttr->type.len == ATT_BT_UUID_SIZE )
371  {
372  // 16-bit UUID
373  uint16 uuid = BUILD_UINT16( pAttr->type.uuid[0], pAttr->type.uuid[1]);
374  switch ( uuid )
375  {
376  case GATT_CLIENT_CHAR_CFG_UUID:
377  status = GATTServApp_ProcessCCCWriteReq( connHandle, pAttr, pValue, len,
378  offset, GATT_CLIENT_CFG_NOTIFY );
379  break;
380 
381  default:
382  // Should never get here!
383  status = ATT_ERR_ATTR_NOT_FOUND;
384  break;
385  }
386  }
387  else
388  {
389  // 128-bit UUID
390  status = ATT_ERR_INVALID_HANDLE;
391  }
392 
393  return ( status );
394 }

396 /*********************************************************************
397  * @fn sk_HandleConnStatusCB
398  *
399  * @brief Simple Keys Profile link status change handler function.
400  *
401  * @param connHandle - connection handle
402  * @param changeType - type of change
403  *
404  * @return none
405  */
406 static void sk_HandleConnStatusCB( uint16 connHandle, uint8 changeType )
407 {
408  // Make sure this is not loopback connection
409  if ( connHandle != LOOPBACK_CONNHANDLE )
410  {
411  // Reset Client Char Config if connection has dropped
412  if ( ( changeType == LINKDB_STATUS_UPDATE_REMOVED ) ||
413  ( ( changeType == LINKDB_STATUS_UPDATE_STATEFLAGS ) &&
414  ( !linkDB_Up( connHandle ) ) ) )
415  {
416  GATTServApp_InitCharCfg( connHandle, skConfig );
417  }
418  }
419 }
