251 /******************************************************************************
252  * @fn HalUARTInitISR
253  *
254  * @brief Initialize the UART
255  *
256  * @param none
257  *
258  * @return none
259  *****************************************************************************/
260 static void HalUARTInitISR(void)
261 {
262  // Set P2 priority - USART0 over USART1 if both are defined.
263  P2DIR &= ~P2DIR_PRIPO;
264  P2DIR |= HAL_UART_PRIPO;
265 
266 #if (HAL_UART_ISR == 1)
267  PERCFG &= ~HAL_UART_PERCFG_BIT; // Set UART0 I/O location to P0.
268 #else
269  PERCFG |= HAL_UART_PERCFG_BIT; // Set UART1 I/O location to P1.
270 #endif
271  PxSEL |= HAL_UART_Px_RX_TX; // Enable Tx and Rx on P1.
272  ADCCFG &= ~HAL_UART_Px_RX_TX; // Make sure ADC doesnt use this.
273  UxCSR = CSR_MODE; // Mode is UART Mode.
274  UxUCR = UCR_FLUSH; // Flush it.
275 }

277 /******************************************************************************
278  * @fn HalUARTOpenISR
279  *
280  * @brief Open a port according tp the configuration specified by parameter.
281  *
282  * @param config - contains configuration information
283  *
284  * @return none
285  *****************************************************************************/
286 static void HalUARTOpenISR(halUARTCfg_t *config)
287 {
288  isrCfg.uartCB = config->callBackFunc;
289  // Only supporting subset of baudrate for code size - other is possible.
290  HAL_UART_ASSERT((config->baudRate == HAL_UART_BR_9600) ||
291  (config->baudRate == HAL_UART_BR_19200) ||
292  (config->baudRate == HAL_UART_BR_38400) ||
293  (config->baudRate == HAL_UART_BR_57600) ||
294  (config->baudRate == HAL_UART_BR_115200));
295 
296  if (config->baudRate == HAL_UART_BR_57600 ||
297  config->baudRate == HAL_UART_BR_115200)
298  {
299  UxBAUD = 216;
300  }
301  else
302  {
303  UxBAUD = 59;
304  }
305 
306  switch (config->baudRate)
307  {
308  case HAL_UART_BR_9600:
309  UxGCR = 8;
310  break;
311  case HAL_UART_BR_19200:
312  UxGCR = 9;
313  break;
314  case HAL_UART_BR_38400:
315  case HAL_UART_BR_57600:
316  UxGCR = 10;
317  break;
318  default:
319  // HAL_UART_BR_115200
320  UxGCR = 11;
321  break;
322  }
323 
324  // 8 bits/char; no parity; 1 stop bit; stop bit hi.
325  if (config->flowControl)
326  {
327  UxUCR = UCR_FLOW | UCR_STOP;
328  PxSEL |= HAL_UART_Px_RTS | HAL_UART_Px_CTS;
329  PxOUT &= ~HAL_UART_Px_RTS;
330  PxDIR |= HAL_UART_Px_RTS;
331  }
332  else
333  {
334  UxUCR = UCR_STOP;
335  }
336 
337  URXxIE = 1;
338  UTXxIF = 1; // Prime the ISR pump.
339  UxCSR = (CSR_MODE | CSR_RE);
340 }

342 /*****************************************************************************
343  * @fn HalUARTReadISR
344  *
345  * @brief Read a buffer from the UART
346  *
347  * @param buf - valid data buffer at least 'len' bytes in size
348  * len - max length number of bytes to copy to 'buf'
349  *
350  * @return length of buffer that was read
351  *****************************************************************************/
352 static uint16 HalUARTReadISR(uint8 *buf, uint16 len)
353 {
354  uint16 cnt = 0;
355 
356  while ((isrCfg.rxHead != isrCfg.rxTail) && (cnt < len))
357  {
358  *buf++ = isrCfg.rxBuf[isrCfg.rxHead++];
359  if (isrCfg.rxHead >= HAL_UART_ISR_RX_MAX)
360  {
361  isrCfg.rxHead = 0;
362  }
363  cnt++;
364  }
365 
366  return cnt;
367 }

369 /******************************************************************************
370  * @fn HalUARTWriteISR
371  *
372  * @brief Write a buffer to the UART.
373  *
374  * @param buf - pointer to the buffer that will be written, not freed
375  * len - length of
376  *
377  * @return length of the buffer that was sent
378  *****************************************************************************/
379 static uint16 HalUARTWriteISR(uint8 *buf, uint16 len)
380 {
381  uint16 cnt;
382 
383  // Enforce all or none.
384  if (HAL_UART_ISR_TX_AVAIL() < len)
385  {
386  return 0;
387  }
388 
389  for (cnt = 0; cnt < len; cnt++)
390  {
391  isrCfg.txBuf[isrCfg.txTail] = *buf++;
392  isrCfg.txMT = 0;
393 
394  if (isrCfg.txTail >= HAL_UART_ISR_TX_MAX-1)
395  {
396  isrCfg.txTail = 0;
397  }
398  else
399  {
400  isrCfg.txTail++;
401  }
402 
403  // Keep re-enabling ISR as it might be keeping up with this loop due to other ints.
404  IEN2 |= UTXxIE;
405  }
406 
407  return cnt;
408 }

410 /******************************************************************************
411  * @fn HalUARTPollISR
412  *
413  * @brief Poll a USART module implemented by ISR.
414  *
415  * @param none
416  *
417  * @return none
418  *****************************************************************************/
419 static void HalUARTPollISR(void)
420 {
421  uint16 cnt = HAL_UART_ISR_RX_AVAIL();
422  uint8 evt = 0;
423 
424  if (isrCfg.rxTick)
425  {
426  // Use the LSB of the sleep timer (ST0 must be read first anyway).
427  uint8 decr = ST0 - isrCfg.rxShdw;
428 
429  if (isrCfg.rxTick > decr)
430  {
431  isrCfg.rxTick -= decr;
432  }
433  else
434  {
435  isrCfg.rxTick = 0;
436  }
437  }
438  isrCfg.rxShdw = ST0;
439 
440  if (cnt >= HAL_UART_ISR_RX_MAX-1)
441  {
442  evt = HAL_UART_RX_FULL;
443  }
444  else if (cnt >= HAL_UART_ISR_HIGH)
445  {
446  evt = HAL_UART_RX_ABOUT_FULL;
447  }
448  else if (cnt && !isrCfg.rxTick)
449  {
450  evt = HAL_UART_RX_TIMEOUT;
451  }
452 
453  if (isrCfg.txMT)
454  {
455  isrCfg.txMT = 0;
456  evt |= HAL_UART_TX_EMPTY;
457  }
458 
459  if (evt && (isrCfg.uartCB != NULL))
460  {
461  isrCfg.uartCB(HAL_UART_ISR-1, evt);
462  }
463 }

494 /******************************************************************************
495  * @fn HalUARTSuspendISR
496  *
497  * @brief Suspend UART hardware before entering PM mode 1, 2 or 3.
498  *
499  * @param None
500  *
501  * @return None
502  *****************************************************************************/
503 static void HalUARTSuspendISR( void )
504 {
505 #if defined POWER_SAVING
506  UxCSR = CSR_MODE; // Suspend Rx operations.
507 
508  if (UxUCR & UCR_FLOW)
509  {
510  PxOUT |= HAL_UART_Px_RTS; // Disable Rx flow.
511  PxIFG = (HAL_UART_Px_CTS ^ 0xFF);
512  PxIEN |= HAL_UART_Px_CTS; // Enable the CTS ISR.
513  }
514  else
515  {
516  PxIFG = (HAL_UART_Px_RX ^ 0xFF);
517  PxIEN |= HAL_UART_Px_RX; // Enable the Rx ISR.
518  }
519 
520 #if defined HAL_UART_GPIO_ISR
521  PxIF = 0; // Clear the next level.
522  PICTL |= PICTL_BIT; // Interrupt on falling edge of input.
523  IENx |= IEN_BIT;
524 #endif
525 #endif
526 }

528 /******************************************************************************
529  * @fn HalUARTResumeISR
530  *
531  * @brief Resume UART hardware after exiting PM mode 1, 2 or 3.
532  *
533  * @param None
534  *
535  * @return None
536  *****************************************************************************/
537 static void HalUARTResumeISR( void )
538 {
539 #if defined POWER_SAVING
540 #if defined HAL_UART_GPIO_ISR
541  IENx &= (IEN_BIT ^ 0xFF);
542 #endif
543 
544  if (UxUCR & UCR_FLOW)
545  {
546  PxIEN &= (HAL_UART_Px_CTS ^ 0xFF); // Disable the CTS ISR.
547  PxOUT &= (HAL_UART_Px_RTS ^ 0xFF); // Re-enable Rx flow.
548  }
549  else
550  {
551  PxIEN &= (HAL_UART_Px_RX ^ 0xFF); // Enable the Rx ISR.
552  }
553 
554  UxUCR |= UCR_FLUSH;
555  UxCSR = (CSR_MODE | CSR_RE);
556 #endif
557 }

559 /***************************************************************************************************
560  * @fn halUartRxIsr
561  *
562  * @brief UART Receive Interrupt
563  *
564  * @param None
565  *
566  * @return None
567  ***************************************************************************************************/
568 #if (HAL_UART_ISR == 1)
569 HAL_ISR_FUNCTION( halUart0RxIsr, URX0_VECTOR )
570 #else
571 HAL_ISR_FUNCTION( halUart1RxIsr, URX1_VECTOR )
572 #endif
573 {
574  HAL_ENTER_ISR();
575 
576  uint8 tmp = UxDBUF;
577  isrCfg.rxBuf[isrCfg.rxTail] = tmp;
578 
579  // Re-sync the shadow on any 1st byte received.
580  if (isrCfg.rxHead == isrCfg.rxTail)
581  {
582  isrCfg.rxShdw = ST0;
583  }
584 
585  if (++isrCfg.rxTail >= HAL_UART_ISR_RX_MAX)
586  {
587  isrCfg.rxTail = 0;
588  }
589 
590  isrCfg.rxTick = HAL_UART_ISR_IDLE;
591 
592  HAL_EXIT_ISR();
593 }
