60 /* These attributes define sleep beheaver. The attributes can be changed
61  * for each sleep cycle or when the device characteristic change.
62  */
63 typedef struct
64 {
65  uint16 pwrmgr_task_state;
66  uint16 pwrmgr_next_timeout;
67  uint16 accumulated_sleep_time;
68  uint8 pwrmgr_device;
69 } pwrmgr_attribute_t;
70 
71 /* With PWRMGR_ALWAYS_ON selection, there is no power savings and the
72  * device is most likely on mains power. The PWRMGR_BATTERY selection allows
73  * the HAL sleep manager to enter SLEEP LITE state or SLEEP DEEP state.
74  */
75 #define PWRMGR_ALWAYS_ON 0
76 #define PWRMGR_BATTERY 1
77 
78 /* The PWRMGR_CONSERVE selection turns power savings on, all tasks have to
79  * agree. The PWRMGR_HOLD selection turns power savings off.
80  */
81 #define PWRMGR_CONSERVE 0
82 #define PWRMGR_HOLD 1

97  /*
98  * Initialize the power management system.
99  * This function is called from OSAL.
100  *
101  */
102  extern void osal_pwrmgr_init( void );
103 
104  /*
105  * This function is called by each task to state whether or not this
106  * task wants to conserve power. The task will call this function to
107  * vote whether it wants the OSAL to conserve power or it wants to
108  * hold off on the power savings. By default, when a task is created,
109  * its own power state is set to conserve. If the task always wants
110  * to converse power, it doesn't need to call this function at all.
111  * It is important for the task that changed the power manager task
112  * state to PWRMGR_HOLD to switch back to PWRMGR_CONSERVE when the
113  * hold period ends.
114  */
115  extern uint8 osal_pwrmgr_task_state( uint8 task_id, uint8 state );
116 
117  /*
118  * This function is called on power-up, whenever the device characteristic
119  * change (ex. Battery backed coordinator). This function works with the timer
120  * to set HAL's power manager sleep state when power saving is entered.
121  * This function should be called form HAL initialization. After power up
122  * initialization, it should only be called from NWK or ZDO.
123  */
124  extern void osal_pwrmgr_device( uint8 pwrmgr_device );
125 
126  /*
127  * This function is called from the main OSAL loop when there are
128  * no events scheduled and shouldn't be called from anywhere else.
129  */
130  extern void osal_pwrmgr_powerconserve( void );
