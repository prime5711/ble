48 uint8 hidSendHidInReport(uint8 *pReport, uint8 endPoint, uint8 len)
49 {
50  uint8 result = FALSE;
51 
52  if (endPoint < 6)
53  {
54  uint8 ea = halIntLock();
55 
56  USBFW_SELECT_ENDPOINT(endPoint);
57  if (!(USBCSIL & USBCSIL_INPKT_RDY))
58  {
59  usbfwWriteFifo(((&USBF0) + (endPoint << 1)), len, pReport);
60  USBCSIL |= USBCSIL_INPKT_RDY;
61  result = TRUE;
62  }
63  halIntUnlock(ea);
64  }
