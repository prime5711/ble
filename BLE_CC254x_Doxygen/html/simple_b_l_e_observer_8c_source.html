144 /*********************************************************************
145  * @fn SimpleBLEObserver_Init
146  *
147  * @brief Initialization function for the Simple BLE Observer App Task.
148  * This is called during initialization and should contain
149  * any application specific initialization (ie. hardware
150  * initialization/setup, table initialization, power up
151  * notification).
152  *
153  * @param task_id - the ID assigned by OSAL. This ID should be
154  * used to send messages and set timers.
155  *
156  * @return none
157  */
158 void SimpleBLEObserver_Init( uint8 task_id )
159 {
160  simpleBLETaskId = task_id;
161 
162  // Setup Observer Profile
163  {
164  uint8 scanRes = DEFAULT_MAX_SCAN_RES;
165  GAPObserverRole_SetParameter ( GAPOBSERVERROLE_MAX_SCAN_RES, sizeof( uint8 ), &scanRes );
166  }
167 
168  // Setup GAP
169  GAP_SetParamValue( TGAP_GEN_DISC_SCAN, DEFAULT_SCAN_DURATION );
170  GAP_SetParamValue( TGAP_LIM_DISC_SCAN, DEFAULT_SCAN_DURATION );
171 
172  // Register for all key events - This app will handle all key events
173  RegisterForKeys( simpleBLETaskId );
174 
175  // makes sure LEDs are off
176  HalLedSet( (HAL_LED_1 | HAL_LED_2), HAL_LED_MODE_OFF );
177 
178  // Setup a delayed profile startup
179  osal_set_event( simpleBLETaskId, START_DEVICE_EVT );
180 }

182 /*********************************************************************
183  * @fn SimpleBLEObserver_ProcessEvent
184  *
185  * @brief Simple BLE Observer Application Task event processor. This function
186  * is called to process all events for the task. Events
187  * include timers, messages and any other user defined events.
188  *
189  * @param task_id - The OSAL assigned task ID.
190  * @param events - events to process. This is a bit map and can
191  * contain more than one event.
192  *
193  * @return events not processed
194  */
195 uint16 SimpleBLEObserver_ProcessEvent( uint8 task_id, uint16 events )
196 {
197 
198  VOID task_id; // OSAL required parameter that isn't used in this function
199 
200  if ( events & SYS_EVENT_MSG )
201  {
202  uint8 *pMsg;
203 
204  if ( (pMsg = osal_msg_receive( simpleBLETaskId )) != NULL )
205  {
206  simpleBLEObserver_ProcessOSALMsg( (osal_event_hdr_t *)pMsg );
207 
208  // Release the OSAL message
209  VOID osal_msg_deallocate( pMsg );
210  }
211 
212  // return unprocessed events
213  return (events ^ SYS_EVENT_MSG);
214  }
215 
216  if ( events & START_DEVICE_EVT )
217  {
218  // Start the Device
219  VOID GAPObserverRole_StartDevice( (gapObserverRoleCB_t *) &simpleBLERoleCB );
220 
221  return ( events ^ START_DEVICE_EVT );
222  }
223 
224  // Discard unknown events
225  return 0;
226 }

231  * @brief Process an incoming task message.

235  * @return none

237 static void simpleBLEObserver_ProcessOSALMsg( osal_event_hdr_t *pMsg )
238 {
239  switch ( pMsg->event )
240  {
241  case KEY_CHANGE:
242  simpleBLEObserver_HandleKeys( ((keyChange_t *)pMsg)->state, ((keyChange_t *)pMsg)->keys );
243  break;
244 
245  case GATT_MSG_EVENT:
246  break;
247  }
248 }

250 /*********************************************************************
251  * @fn simpleBLEObserver_HandleKeys
252  *
253  * @brief Handles all key events for this device.
254  *
255  * @param shift - true if in shift/alt.
256  * @param keys - bit field for key events. Valid entries:
257  * HAL_KEY_SW_2
258  * HAL_KEY_SW_1
259  *
260  * @return none
261  */
262 uint8 gStatus;
263 static void simpleBLEObserver_HandleKeys( uint8 shift, uint8 keys )
264 {
265  (void)shift; // Intentionally unreferenced parameter
266 
267  if ( keys & HAL_KEY_UP )
268  {
269  // Start or stop discovery
270  if ( !simpleBLEScanning )
271  {
272  simpleBLEScanning = TRUE;
273  simpleBLEScanRes = 0;
274 
275  LCD_WRITE_STRING( "Discovering...", HAL_LCD_LINE_1 );
276  LCD_WRITE_STRING( "", HAL_LCD_LINE_2 );
277 
278  GAPObserverRole_StartDiscovery( DEFAULT_DISCOVERY_MODE,
279  DEFAULT_DISCOVERY_ACTIVE_SCAN,
280  DEFAULT_DISCOVERY_WHITE_LIST );
281  }
282  else
283  {
284  GAPObserverRole_CancelDiscovery();
285  }
286  }
287 
288  if ( keys & HAL_KEY_LEFT )
289  {
290  // Display discovery results
291  if ( !simpleBLEScanning && simpleBLEScanRes > 0 )
292  {
293  // Increment index of current result (with wraparound)
294  simpleBLEScanIdx++;
295  if ( simpleBLEScanIdx >= simpleBLEScanRes )
296  {
297  simpleBLEScanIdx = 0;
298  }
299 
300  LCD_WRITE_STRING_VALUE( "Device", simpleBLEScanIdx + 1,
301  10, HAL_LCD_LINE_1 );
302  LCD_WRITE_STRING( bdAddr2Str( simpleBLEDevList[simpleBLEScanIdx].addr ),
303  HAL_LCD_LINE_2 );
304  }
305  }
306 
307  if ( keys & HAL_KEY_RIGHT )
308  {
309  }
310 
311  if ( keys & HAL_KEY_CENTER )
312  {
313  }
314 
315  if ( keys & HAL_KEY_DOWN )
316  {
317  }
318 }

320 /*********************************************************************
321  * @fn simpleBLEObserverEventCB
322  *
323  * @brief Observer event callback function.
324  *
325  * @param pEvent - pointer to event structure
326  *
327  * @return none
328  */
329 static void simpleBLEObserverEventCB( gapObserverRoleEvent_t *pEvent )
330 {
331  switch ( pEvent->gap.opcode )
332  {
333  case GAP_DEVICE_INIT_DONE_EVENT:
334  {
335  LCD_WRITE_STRING( "BLE Observer", HAL_LCD_LINE_1 );
336  LCD_WRITE_STRING( bdAddr2Str( pEvent->initDone.devAddr ), HAL_LCD_LINE_2 );
337  }
338  break;
339 
340  case GAP_DEVICE_INFO_EVENT:
341  {
342  simpleBLEAddDeviceInfo( pEvent->deviceInfo.addr, pEvent->deviceInfo.addrType );
343  }
344  break;
345 
346  case GAP_DEVICE_DISCOVERY_EVENT:
347  {
348  // discovery complete
349  simpleBLEScanning = FALSE;
350 
351  // Copy results
352  simpleBLEScanRes = pEvent->discCmpl.numDevs;
353  osal_memcpy( simpleBLEDevList, pEvent->discCmpl.pDevList,
354  (sizeof( gapDevRec_t ) * pEvent->discCmpl.numDevs) );
355 
356  LCD_WRITE_STRING_VALUE( "Devices Found", simpleBLEScanRes,
357  10, HAL_LCD_LINE_1 );
358  if ( simpleBLEScanRes > 0 )
359  {
360  LCD_WRITE_STRING( "<- To Select", HAL_LCD_LINE_2 );
361  }
362 
363  // initialize scan index to last device
364  simpleBLEScanIdx = simpleBLEScanRes;
365  }
366  break;
367 
368  default:
369  break;
370  }
371 }

373 /*********************************************************************
374  * @fn simpleBLEAddDeviceInfo
375  *
376  * @brief Add a device to the device discovery result list
377  *
378  * @return none
379  */
380 static void simpleBLEAddDeviceInfo( uint8 *pAddr, uint8 addrType )
381 {
382  uint8 i;
383 
384  // If result count not at max
385  if ( simpleBLEScanRes < DEFAULT_MAX_SCAN_RES )
386  {
387  // Check if device is already in scan results
388  for ( i = 0; i < simpleBLEScanRes; i++ )
389  {
390  if ( osal_memcmp( pAddr, simpleBLEDevList[i].addr , B_ADDR_LEN ) )
391  {
392  return;
393  }
394  }
395 
396  // Add addr to scan result list
397  osal_memcpy( simpleBLEDevList[simpleBLEScanRes].addr, pAddr, B_ADDR_LEN );
398  simpleBLEDevList[simpleBLEScanRes].addrType = addrType;
399 
400  // Increment scan result count
401  simpleBLEScanRes++;
402  }
403 }

405 /*********************************************************************
406  * @fn bdAddr2Str
407  *
408  * @brief Convert Bluetooth address to string
409  *
410  * @return none
411  */
412 char *bdAddr2Str( uint8 *pAddr )
413 {
414  uint8 i;
415  char hex[] = "0123456789ABCDEF";
416  static char str[B_ADDR_STR_LEN];
417  char *pStr = str;
418 
419  *pStr++ = '0';
420  *pStr++ = 'x';
421 
422  // Start from end of addr
423  pAddr += B_ADDR_LEN;
424 
425  for ( i = B_ADDR_LEN; i > 0; i-- )
426  {
427  *pStr++ = hex[*--pAddr >> 4];
428  *pStr++ = hex[*pAddr & 0x0F];
429  }
430 
431  *pStr = 0;
432 
433  return str;
434 }
