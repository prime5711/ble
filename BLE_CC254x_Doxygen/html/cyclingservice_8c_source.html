301 /*********************************************************************
302  * @fn CyclingService_ProcessEvent
303  *
304  * @brief process incoming event.
305  *
306  * @param task_id - OSAL task id.
307  *
308  * @param events - event bit(s) set for the task(s)
309  *
310  * @return none
311  */
312 uint16 CyclingService_ProcessEvent( uint8 task_id, uint16 events )
313 {
314  VOID task_id;
315 
316  if ( events & SYS_EVENT_MSG )
317  {
318  uint8 *pMsg;
319 
320  if ( (pMsg = osal_msg_receive( cyclingService_TaskID )) != NULL )
321  {
322  cycling_ProcessOSALMsg( (osal_event_hdr_t *)pMsg );
323 
324  // Release the OSAL message
325  VOID osal_msg_deallocate( pMsg );
326  }
327 
328  // return unprocessed events
329  return (events ^ SYS_EVENT_MSG);
330  }
331 
332  if ( events & CSC_CMD_IND_SEND_EVT )
333  {
334  GATT_Indication( connectionHandle, &cscCmdInd, FALSE, cyclingService_TaskID );
335 
336  // Set Control Point Cfg done
337  scOpInProgress = FALSE;
338 
339  return ( events ^ CSC_CMD_IND_SEND_EVT );
340  }
341 
342  return 0;
343 }

561  /*********************************************************************
562  * @fn cycling_ProcessCSCCmd
563  *
564  * @brief process an incoming CSC command.
565  *
566  * @param attrHandle - attribute handle
567  * @param pValue - pointer to data to be written
568  * @param len - length of data
569  *
570  * @return none
571  */
572 static void cycling_ProcessCSCCmd( uint16 attrHandle, uint8 *pValue, uint8 len )
573 {
574  uint8 cscStatus = CSC_SUCCESS;
575 
576  // Set Control Point Cfg in progress
577  scOpInProgress = TRUE;
578 
579  // Set indication info to be sent out
580  cscCmdInd.handle = attrHandle;
581 
582  cscCmdInd.len = 3;
583  cscCmdInd.value[0] = CSC_COMMAND_RSP;
584  cscCmdInd.value[1] = pValue[0];
585 
586  switch ( pValue[0] )
587  {
588  case CSC_SET_CUMM_VAL:
589  // If wheel revolutions is a feature
590  if ( ( len <= 5 ) && ( cyclingFeatures & CSC_WHEEL_REV_SUPP ) )
591  {
592  uint32 cummWheelRevolutions;
593 
594  // full 32 bits were specified.
595  if (( len - 1 ) == 4)
596  {
597  cummWheelRevolutions = BUILD_UINT32( pValue[1], pValue[2], pValue[3], pValue[4]);
598  }
599  else
600  {
601  cummWheelRevolutions = 0;
602 
603  // In case only lower bits were specified and upper bits remain zero.
604  for( int i = 0; i < (len - 1); ++i )
605  {
606  cummWheelRevolutions += pValue[i + 1] << (i*8);
607  }
608  }
609 
610  // Notify app
611  if ( cyclingServiceCB != NULL )
612  {
613  (*cyclingServiceCB)( CSC_CMD_SET_CUMM_VAL, &cummWheelRevolutions );
614  }
615  }
616  else // characteristic not supported.
617  {
618  cscStatus = CSC_INVALID_PARAMETER;
619  }
620  break;
621 
622  case CSC_UPDATE_SENS_LOC:
623  // If multiple sensor locations is supported and that this is a valid location.
624  if ( ( len == 2 ) &&
625  ( cyclingFeatures & CSC_MULTI_SENS_SUPP ) &&
626  ( cycling_SensorLocSupported( pValue[1] ) == TRUE ) )
627  {
628  // Update sensor location
629  cyclingSensLoc = pValue[1];
630 
631  // Notify app
632  if ( cyclingServiceCB != NULL )
633  {
634  (*cyclingServiceCB)( CSC_CMD_UPDATE_SENS_LOC, NULL );
635  }
636  }
637  else // characteristic not supported.
638  {
639  cscStatus = CSC_INVALID_PARAMETER;
640  }
641  break;
642 
643  case CSC_REQ_SUPP_SENS_LOC:
644  // If multiple sensor locations are supported and list requested
645  if ( ( len == 1 ) && ( cyclingFeatures & CSC_MULTI_SENS_SUPP ) )
646  {
647  cscCmdInd.len += supportedSensors;
648  osal_memcpy( &(cscCmdInd.value[3]), supportedSensorLocations, supportedSensors );
649  }
650  else // characteristic not supported.
651  {
652  // Send an indication with the list.
653  cscStatus = CSC_INVALID_PARAMETER;
654  }
655  break;
656 
657  default:
658  // Send an indication with opcode not suported response
659  cscStatus = CSC_OPCODE_NOT_SUPPORTED;
660  break;
661  }
662 
663  // Send indication of operation result
664  cscCmdInd.value[2] = cscStatus;
665 
666  // Ask our task to send out indication
667  osal_set_event( cyclingService_TaskID, CSC_CMD_IND_SEND_EVT );
668 }

696 /*********************************************************************
697  * @fn cycling_ReadAttrCB
698  *
699  * @brief Read an attribute.
700  *
701  * @param connHandle - connection message was received on
702  * @param pAttr - pointer to attribute
703  * @param pValue - pointer to data to be read
704  * @param pLen - length of data to be read
705  * @param offset - offset of the first octet to be read
706  * @param maxLen - maximum length of data to be read
707  *
708  * @return Success or Failure
709  */
710 static uint8 cycling_ReadAttrCB( uint16 connHandle, gattAttribute_t *pAttr,
711  uint8 *pValue, uint8 *pLen, uint16 offset, uint8 maxLen )
712 {
713  // Make sure it's not a blob operation (no attributes in the profile are long)
714  if ( offset > 0 )
715  {
716  return ( ATT_ERR_ATTR_NOT_LONG );
717  }
718 
719  bStatus_t status = SUCCESS;
720  uint16 uuid = BUILD_UINT16( pAttr->type.uuid[0], pAttr->type.uuid[1]);
721 
722  switch ( uuid )
723  {
724  case SENSOR_LOC_UUID:
725  {
726  // Read Sensor Location
727  *pLen = 1;
728  pValue[0] = pAttr->pValue[0];
729  }
730  break;
731 
732  case CSC_FEATURE_UUID:
733  {
734  //Read Cycling Feature
735  *pLen = 2;
736  pValue[0] = LO_UINT16(pAttr->pValue[0]);
737  pValue[1] = HI_UINT16(pAttr->pValue[0]);
738  }
739  break;
740 
741  case GATT_CLIENT_CHAR_CFG_UUID:
742  {
743  // Read Measurement or Command Configuration
744  if ( pAttr->pValue == (uint8*)cyclingMeasClientCharCfg )
745  {
746  *pLen = 1;
747  pValue[0] = GATTServApp_ReadCharCfg(connHandle, cyclingMeasClientCharCfg );
748  }
749  else if ( pAttr->pValue == (uint8*)cyclingCommandClientCharCfg )
750  {
751  *pLen = 1;
752  pValue[0] = GATTServApp_ReadCharCfg(connHandle, cyclingCommandClientCharCfg );
753  }
754  else
755  {
756  status = ATT_ERR_ATTR_NOT_FOUND;
757  }
758  }
759  break;
760 
761  default:
762  status = ATT_ERR_ATTR_NOT_FOUND;
763  break;
764  }
765 
766  // Notify app
767  if ( cyclingServiceCB != NULL )
768  {
769  (*cyclingServiceCB)( CSC_READ_ATTR, NULL );
770  }
771 
772  return ( status );
773 }

776 /*********************************************************************
777  * @fn cycling_WriteAttrCB
778  *
779  * @brief Validate attribute data prior to a write operation
780  *
781  * @param connHandle - connection message was received on
782  * @param pAttr - pointer to attribute
783  * @param pValue - pointer to data to be written
784  * @param len - length of data
785  * @param offset - offset of the first octet to be written
786  *
787  * @return Success or Failure
788  */
789 static bStatus_t cycling_WriteAttrCB( uint16 connHandle, gattAttribute_t *pAttr,
790  uint8 *pValue, uint8 len, uint16 offset )
791 {
792  bStatus_t status = SUCCESS;
793  uint16 uuid = BUILD_UINT16( pAttr->type.uuid[0], pAttr->type.uuid[1]);
794 
795  if ( offset > 0 )
796  {
797  return (ATT_ERR_ATTR_NOT_LONG);
798  }
799 
800  switch ( uuid )
801  {
802  case SC_CTRL_PT_UUID:
803  // Make sure Control Point Cfg is not already in progress
804  if ( scOpInProgress == TRUE )
805  {
806  status = CSC_ERR_PROC_IN_PROGRESS;
807  }
808  // Make sure Control Point Cfg is configured for Indications
809  else if ( (cyclingCommandClientCharCfg[connHandle].value & GATT_CLIENT_CFG_INDICATE) == FALSE )
810  {
811  status = CSC_ERR_CCC_IMPROPER_CFG;
812  }
813  else
814  {
815  // Process CSC command
816  cycling_ProcessCSCCmd( pAttr->handle, pValue, len );
817  connectionHandle = connHandle;
818  }
819  break;
820 
821  // For Measure and Commands CCC
822  case GATT_CLIENT_CHAR_CFG_UUID:
823  if ( pAttr->handle == cyclingAttrTbl[CSC_COMMAND_CFG_POS].handle )
824  {
825  status = GATTServApp_ProcessCCCWriteReq( connHandle, pAttr, pValue, len,
826  offset, GATT_CLIENT_CFG_INDICATE );
827  // Notify app
828  if ( cyclingServiceCB != NULL )
829  {
830  (*cyclingServiceCB)( CSC_WRITE_ATTR, NULL );
831  }
832  }
833  else if ( pAttr->handle == cyclingAttrTbl[CSC_MEAS_CFG_POS].handle )
834  {
835  status = GATTServApp_ProcessCCCWriteReq( connHandle, pAttr, pValue, len,
836  offset, GATT_CLIENT_CFG_NOTIFY );
837  if ( status == SUCCESS )
838  {
839  // Notify app
840  if ( cyclingServiceCB != NULL )
841  {
842  uint16 charCfg = BUILD_UINT16( pValue[0], pValue[1] );
843 
844  (*cyclingServiceCB)( ((charCfg == GATT_CFG_NO_OPERATION) ?
845  CSC_MEAS_NOTI_DISABLED :
846  CSC_MEAS_NOTI_ENABLED ), NULL );
847  }
848  }
849  }
850  break;
851 
852  default:
853  status = ATT_ERR_ATTR_NOT_FOUND;
854  break;
855  }
856 
857  return ( status );
858 }
