104 void HalBuzzerInit( void )
105 {

115 /**************************************************************************************************
116  * @fn HalBuzzerRing
117  *
118  * @brief This function rings the buzzer once.
119  *
120  * input parameters
121  *
122  * @param msec - Number of msec to ring the buzzer
123  * @param tone - Type of tone (low or high)
124  * @param buzzerCback - Callback function to call when ringing of buzzer is finished
125  *
126  * output parameters
127  *
128  * None.
129  *
130  * @return None.
131  */
132 void HalBuzzerRing( uint16 msec,
133  uint8 tone,
134  halBuzzerCBack_t buzzerCback )
135 {
136  /* Register the callback fucntion */
137  pHalBuzzerRingCompleteNotificationFunction = buzzerCback;
138 
139  /* Configure output pin as peripheral since we're using T4 to generate */
140  P1SEL |= (uint8) HAL_BUZZER_P1_GPIO_PINS;
141 
142  if ( tone == HAL_BUZZER_LOW_TONE )
143  {
144  /* Buzzer is "rung" by using T4, channel 0 to generate 2kHz square wave */
145  T4CTL = HAL_T4_TIMER_CTL_DIV64 |
146  HAL_T4_TIMER_CTL_CLEAR |
147  HAL_T4_TIMER_CTL_OPMODE_MODULO;
148  }
149  else // tone == HAL_BUZZER_HIGH_TONE
150  {
151  /* Buzzer is "rung" by using T4, channel 0 to generate 4kHz square wave */
152  T4CTL = HAL_T4_TIMER_CTL_DIV32 |
153  HAL_T4_TIMER_CTL_CLEAR |
154  HAL_T4_TIMER_CTL_OPMODE_MODULO;
155  }
156 
157  T4CCTL0 = HAL_T4_TIMER_CCTL_MODE_COMPARE | HAL_T4_TIMER_CCTL_CMP_TOGGLE;
158  T4CC0 = HAL_T4_CC0_VALUE;
159 
160  /* Start it */
161  T4CTL |= HAL_T4_TIMER_CTL_START;
162 
163  /* Setup timer that will end the buzzing */
164  osal_start_timerEx( Hal_TaskID,
165  HAL_BUZZER_EVENT,
166  msec );
167 
168 }

170 /**************************************************************************************************
171  * @fn HalBuzzerStop
172  *
173  * @brief Halts buzzer
174  *
175  * @param None
176  *
177  * @return None
178  **************************************************************************************************/
179 void HalBuzzerStop( void )
180 {
181  /* Setting T4CTL to 0 disables it and masks the overflow interrupt */
182  T4CTL = 0;
183 
184  /* Return output pin to GPIO */
185  P1SEL &= (uint8) ~HAL_BUZZER_P1_GPIO_PINS;
186 
187  /* Inform application that buzzer is done */
188  if (pHalBuzzerRingCompleteNotificationFunction != NULL)
189  {
190  pHalBuzzerRingCompleteNotificationFunction();
191  }
192 }
