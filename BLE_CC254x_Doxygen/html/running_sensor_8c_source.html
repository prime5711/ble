404 /*********************************************************************
405  * @fn RunningSensor_ProcessEvent
406  *
407  * @brief Running Application Task event processor. This function
408  * is called to process all events for the task. Events
409  * include timers, messages and any other user defined events.
410  *
411  * @param task_id - The OSAL assigned task ID.
412  * @param events - events to process. This is a bit map and can
413  * contain more than one event.
414  *
415  * @return events not processed
416  */
417 uint16 RunningSensor_ProcessEvent( uint8 task_id, uint16 events )
418 {
419 
420  VOID task_id; // OSAL required parameter that isn't used in this function
421 
422  if ( events & SYS_EVENT_MSG )
423  {
424  uint8 *pMsg;
425 
426  if ( (pMsg = osal_msg_receive( sensor_TaskID )) != NULL )
427  {
428  sensor_ProcessOSALMsg( (osal_event_hdr_t *)pMsg );
429 
430  // Release the OSAL message
431  VOID osal_msg_deallocate( pMsg );
432  }
433 
434  // return unprocessed events
435  return (events ^ SYS_EVENT_MSG);
436  }
437 
438  if ( events & START_DEVICE_EVT )
439  {
440  // Start the Device
441  VOID GAPRole_StartDevice( &runningPeripheralCB );
442 
443  // Register with bond manager after starting device
444  GAPBondMgr_Register( (gapBondCBs_t *) &runningBondCB );
445 
446  return ( events ^ START_DEVICE_EVT );
447  }
448 
449  if ( events & RSC_PERIODIC_EVT )
450  {
451  // Perform periodic sensor's periodic task
452  sensorPeriodicTask();
453 
454  return (events ^ RSC_PERIODIC_EVT);
455  }
456 
457  if ( events & RSC_CONN_PARAM_UPDATE_EVT )
458  {
459  // Send param update. If it fails, retry until successful.
460  GAPRole_SendUpdateParam( DEFAULT_DESIRED_MIN_CONN_INTERVAL, DEFAULT_DESIRED_MAX_CONN_INTERVAL,
461  DEFAULT_DESIRED_SLAVE_LATENCY, DEFAULT_DESIRED_CONN_TIMEOUT,
462  GAPROLE_RESEND_PARAM_UPDATE );
463 
464  // Assuming service discovery complete, start neglect timer
465  if ( USING_NEGLECT_TIMEOUT )
466  {
467  osal_start_timerEx( sensor_TaskID, RSC_NEGLECT_TIMEOUT_EVT, NEGLECT_TIMEOUT_DELAY );
468  }
469 
470  return (events ^ RSC_CONN_PARAM_UPDATE_EVT);
471  }
472 
473  if ( events & RSC_NEGLECT_TIMEOUT_EVT )
474  {
475  // No user input, terminate connection.
476  GAPRole_TerminateConnection();
477 
478  return ( events ^ RSC_NEGLECT_TIMEOUT_EVT );
479  }
480 
481  if ( events & RSC_RESET_EVT )
482  {
483  if ( gapProfileState == GAPROLE_CONNECTED )
484  {
485  // Exit the connection
486  GAPRole_TerminateConnection();
487 
488  // There is no callback for manual termination of the link. change state variable here.
489  gapProfileState = GAPROLE_WAITING;
490 
491  // Set timer to give the end advertising event time to finish
492  osal_start_timerEx( sensor_TaskID, RSC_RESET_EVT, 500 );
493  }
494  else if ( gapProfileState == GAPROLE_ADVERTISING )
495  {
496  uint8 value = FALSE;
497 
498  // Turn off advertising
499  GAPRole_SetParameter( GAPROLE_ADVERT_ENABLED, sizeof( uint8 ), &value );
500 
501  // Set timer to give the end advertising event time to finish
502  osal_start_timerEx( sensor_TaskID, RSC_RESET_EVT, 500 );
503  }
504  else if ( USING_WHITE_LIST == TRUE )
505  {
506  //temporary variable
507  uint8 value = GAP_FILTER_POLICY_ALL;
508 
509  // Turn off white list filter policy
510  GAPRole_SetParameter(GAPROLE_ADV_FILTER_POLICY, sizeof( uint8 ), &value);
511 
512  sensorUsingWhiteList = FALSE;
513 
514  // Clear the white list
515  HCI_LE_ClearWhiteListCmd();
516 
517  // Set timer to give the end advertising event time to finish
518  osal_start_timerEx( sensor_TaskID, RSC_RESET_EVT, 500 );
519  }
520  else if ( (gapProfileState == GAPROLE_STARTED) ||
521  (gapProfileState == GAPROLE_WAITING) ||
522  (gapProfileState == GAPROLE_WAITING_AFTER_TIMEOUT) )
523  {
524  uint8 eraseBonds = TRUE;
525 
526  // Erase all bonds
527  GAPBondMgr_SetParameter( GAPBOND_ERASE_ALLBONDS, 0, &eraseBonds );
528 
529  // Turn on GREEN LED for set time
530  HalLedSet( HAL_LED_1, HAL_LED_MODE_BLINK );
531 
532  return (events ^ RSC_RESET_EVT);
533  }
534  }
535 
536  // Discard unknown events
537  return 0;
538 }

559 /*********************************************************************
560  * @fn sensor_HandleKeys
561  *
562  * @brief Handles all key events for this device.
563  *
564  * @param shift - true if in shift/alt.
565  * @param keys - bit field for key events. Valid entries:
566  * HAL_KEY_SW_2
567  * HAL_KEY_SW_1
568  *
569  * @return none
570  */
571 static void sensor_HandleKeys( uint8 shift, uint8 keys )
572 {
573  if ( keys == ( HAL_KEY_SW_1 | HAL_KEY_SW_2 ) )
574  {
575  // Reset in progress has started
576  resetInProgress = TRUE;
577 
578  // Set OSAL timer for reset
579  osal_start_timerEx( sensor_TaskID, RSC_RESET_EVT, RSC_RESET_DELAY );
580  }
581  else if ( keys & HAL_KEY_SW_1 )
582  {
583  if ( resetInProgress == TRUE )
584  {
585  // Cancel the reset
586  resetInProgress = FALSE;
587  osal_stop_timerEx ( sensor_TaskID, RSC_RESET_EVT );
588  }
589 
590  // set simulated measurement flag index
591  if (++sensorFlagsIdx == FLAGS_IDX_MAX)
592  {
593  sensorFlagsIdx = 0;
594  }
595  }
596  else if ( keys & HAL_KEY_SW_2 )
597  {
598  if ( resetInProgress == TRUE )
599  {
600  // Cancel the reset
601  resetInProgress = FALSE;
602  osal_stop_timerEx ( sensor_TaskID, RSC_RESET_EVT );
603  }
604 
605  // if not in a connection, toggle advertising on and off
606  if ( gapProfileState != GAPROLE_CONNECTED )
607  {
608  uint8 status;
609 
610  // Set fast advertising interval for user-initiated connections
611  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MIN, DEFAULT_FAST_ADV_INTERVAL );
612  GAP_SetParamValue( TGAP_GEN_DISC_ADV_INT_MAX, DEFAULT_FAST_ADV_INTERVAL );
613  GAP_SetParamValue( TGAP_GEN_DISC_ADV_MIN, DEFAULT_WHITE_LIST_ADV_DURATION );
614 
615  // toggle GAP advertisement status
616  GAPRole_GetParameter( GAPROLE_ADVERT_ENABLED, &status );
617  status = !status;
618 
619  // If not already using white list, begin to do so.
620  // Only do so if about to begin advertising
621  if ( USING_WHITE_LIST && status == TRUE )
622  {
623  uint8 bondCount = 0;
624 
625  GAPBondMgr_GetParameter( GAPBOND_BOND_COUNT, &bondCount );
626 
627  if ((sensorUsingWhiteList == FALSE) && (bondCount > 0) )
628  {
629  uint8 value = GAP_FILTER_POLICY_WHITE;
630 
631  GAPRole_SetParameter(GAPROLE_ADV_FILTER_POLICY, sizeof( uint8 ), &value);
632 
633  sensorUsingWhiteList = TRUE;
634  }
635  }
636 
637  GAPRole_SetParameter( GAPROLE_ADVERT_ENABLED, sizeof( uint8 ), &status );
638 
639  // Set state variable
640  if (status == FALSE)
641  {
642  sensorAdvCancelled = TRUE;
643  }
644  }
645  else if ( gapProfileState == GAPROLE_CONNECTED )
646  {
647  // If connected, change rate of motion
648  if( (++motionIdx) == MOTION_IDX_MAX )
649  {
650  motionIdx = 0;
651  }
652 
653  motion = motionCycle[motionIdx];
654  }
655  }
656  // end of key press
657  else if ( keys == 0x00 )
658  {
659  if ( resetInProgress == TRUE )
660  {
661  resetInProgress = FALSE;

668 /*********************************************************************
669  * @fn sensorMeasNotify
670  *
671  * @brief Prepare and send a RSC measurement notification
672  *
673  * @return none
674  */
675 static void sensorMeasNotify(void)
676 {
677  static uint16 centimeters = 0;
678  uint8 *p = sensorMeas.value;
679  uint8 flags = sensorFlags[sensorFlagsIdx];
680 
681  switch( motion )
682  {
683  case STANDING_STILL:
684  instSpeed = instCadence = instStrideLength = STANDING_STILL;
685  // 0 for walking bit
686  flags = flags & 0xFB; //0b1111 1011
687  break;
688 
689  case WALKING_MOTION:
690  instStrideLength = STRIDE_LENGTH_WALKING;
691  instCadence = WALKING_CADENCE;
692  instSpeed = WALKING_SPEED;
693  // 0 for walking bit
694  flags = flags & 0xFB;
695  break;
696 
697  case RUNNING_MOTION:
698  instStrideLength = STRIDE_LENGTH_RUNNING;
699  instCadence = RUNNING_CADENCE;
700  instSpeed = RUNNING_SPEED;
701  // set in 1 for walking bit.
702  flags = flags | 0x04;
703  break;
704 
705  default: // Do nothing
706  break;
707  }
708 
709  // Add distance
710  centimeters += (uint16)DISTANCE_TRAVELED( instSpeed );
711 
712  // If traveled at least a meter
713  if ( centimeters >= 100 )
714  {
715  // add distance, truncated to meters
716  totalDistance += (centimeters / 100);
717  // and continue to store the remaining centimeters
718  centimeters %= 100;
719  }
720 
721  // Build RSC measurement structure from simulated values
722  // Flags simulate the isPresent bits.
723  *p++ = flags;
724 
725  //Included regardless of flags.
726  *p++ = LO_UINT16( instSpeed );
727  *p++ = HI_UINT16( instSpeed );
728  *p++ = instCadence;
729 
730  if (flags & RSC_FLAGS_STRIDE)
731  {
732  *p++ = LO_UINT16( instStrideLength );
733  *p++ = HI_UINT16( instStrideLength );
734  }
735 
736  if (flags & RSC_FLAGS_DIST)
737  {
738  *p++ = BREAK_UINT32(totalDistance, 0);
739  *p++ = BREAK_UINT32(totalDistance, 1);
740  *p++ = BREAK_UINT32(totalDistance, 2);
741  *p++ = BREAK_UINT32(totalDistance, 3);
742  }
743 
744  // Get length
745  sensorMeas.len = (uint8) (p - sensorMeas.value);
746 
747  // Send to service to send the notification
748  Running_MeasNotify( gapConnHandle, &sensorMeas );
749 }

851  else if ( ((!USING_WHITE_LIST) || whiteListUsed) ||
852  (GAP_GetParamValue( TGAP_GEN_DISC_ADV_INT_MIN ) == DEFAULT_FAST_ADV_INTERVAL) )
853  {

859  GAPRole_SetParameter( GAPROLE_ADVERT_ENABLED, sizeof( uint8 ), &advState );
860  }
861  }
862  // if started
863  else if (newState == GAPROLE_STARTED)
864  {

876  systemId[3] = 0;
877 
878  DevInfo_SetParameter(DEVINFO_SYSTEM_ID, DEVINFO_SYSTEM_ID_LEN, systemId);
879  }
880 
881  gapProfileState = newState;
882 }

884 /*********************************************************************
885  * @fn SensorCB
886  *
887  * @brief Callback function for RSC service.
888  *
889  * @param event - service event
890  * @param pNewCummVal - pointer to new total distanace data
891  * if specified by event. NULL otherwise.
892  *
893  * @return SUCCESS if operation successful. FAILURE, otherwise.
894  */
895 static bStatus_t SensorCB(uint8 event, uint32 *pNewCummVal)
896 {
897  bStatus_t status = SUCCESS;
898 
899  switch ( event )
900  {
901  case RSC_CMD_SET_CUMM_VAL:

904  {
905  osal_stop_timerEx( sensor_TaskID, RSC_NEGLECT_TIMEOUT_EVT );
906  }
907

910

913  {
914  osal_start_timerEx( sensor_TaskID, RSC_NEGLECT_TIMEOUT_EVT, NEGLECT_TIMEOUT_DELAY );
915  }
916  break;
917 
918  case RSC_CMD_START_SENS_CALIB:
919  if ( sensorFlags[sensorFlagsIdx] != RSC_FLAGS_AT_REST )
920  {
921  // Induce a calibration error for conformance testing
922  status = FAILURE;
923  }
924  break;
925 
926  case RSC_CMD_UPDATE_SENS_LOC:
927  // Cancel neglect timer
928  if ( USING_NEGLECT_TIMEOUT )
929  {
930  osal_stop_timerEx( sensor_TaskID, RSC_NEGLECT_TIMEOUT_EVT );
931  }
932 
933  // Get updated sensor location
934  Running_GetParameter( RSC_SENS_LOC, &sensorLocation );
935 
936  // Start neglect timer
937  if ( USING_NEGLECT_TIMEOUT )
938  {
939  osal_start_timerEx( sensor_TaskID, RSC_NEGLECT_TIMEOUT_EVT, NEGLECT_TIMEOUT_DELAY );
940  }
941  break;
942 
943  case RSC_MEAS_NOTI_ENABLED:
944  // Cancel neglect timer
945  if ( USING_NEGLECT_TIMEOUT )
946  {
947  osal_stop_timerEx( sensor_TaskID, RSC_NEGLECT_TIMEOUT_EVT );
948  }
949 
950  // if connected start periodic measurement
951  if (gapProfileState == GAPROLE_CONNECTED)
952  {
953  osal_start_timerEx( sensor_TaskID, RSC_PERIODIC_EVT, DEFAULT_RSC_PERIOD );
954  }
955  break;
956 
957  case RSC_MEAS_NOTI_DISABLED:
958  // stop periodic measurement
959  osal_stop_timerEx( sensor_TaskID, RSC_PERIODIC_EVT );
960 
961  // Start neglect timer
962  if ( USING_NEGLECT_TIMEOUT )
963  {
964  osal_start_timerEx( sensor_TaskID, RSC_NEGLECT_TIMEOUT_EVT, NEGLECT_TIMEOUT_DELAY );
965  }
966  break;
967 
968  case RSC_READ_ATTR:
969  case RSC_WRITE_ATTR:
970  if ( USING_NEGLECT_TIMEOUT )
971  {
972  // Cancel neglect timer
973  osal_stop_timerEx( sensor_TaskID, RSC_NEGLECT_TIMEOUT_EVT );
974 
975  // Start neglect timer
976  osal_start_timerEx( sensor_TaskID, RSC_NEGLECT_TIMEOUT_EVT, NEGLECT_TIMEOUT_DELAY );
977  }
978  break;
979 
980  default:
981  break;
982  }
983 
984  return ( status );
985 }
