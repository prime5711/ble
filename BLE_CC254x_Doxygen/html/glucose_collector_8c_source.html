333 /*********************************************************************
334  * @fn GlucColl_ProcessEvent
335  *
336  * @brief Glucose Collector Application Task event processor. This function
337  * is called to process all events for the task. Events
338  * include timers, messages and any other user defined events.
339  *
340  * @param task_id - The OSAL assigned task ID.
341  * @param events - events to process. This is a bit map and can
342  * contain more than one event.
343  *
344  * @return events not processed
345  */
346 uint16 GlucColl_ProcessEvent( uint8 task_id, uint16 events )
347 {
348 
349  VOID task_id; // OSAL required parameter that isn't used in this function
350 
351  if ( events & SYS_EVENT_MSG )
352  {
353  uint8 *pMsg;
354 
355  if ( (pMsg = osal_msg_receive( glucCollTaskId )) != NULL )
356  {
357  glucCollCentral_ProcessOSALMsg( (osal_event_hdr_t *)pMsg );
358 
359  // Release the OSAL message
360  VOID osal_msg_deallocate( pMsg );
361  }
362 
363  // return unprocessed events
364  return (events ^ SYS_EVENT_MSG);
365  }
366 
367  if ( events & PROCEDURE_TIMEOUT_EVT )
368  {
369  if ( glucCollState == BLE_STATE_CONNECTED )
370  {
371  // disconnect
372  glucCollState = BLE_STATE_DISCONNECTING;
373 
374  GAPCentralRole_TerminateLink( glucCollConnHandle );
375 
376  LCD_WRITE_STRING( "Timeout", HAL_LCD_LINE_1 );
377  LCD_WRITE_STRING( "", HAL_LCD_LINE_2 );
378  LCD_WRITE_STRING( "", HAL_LCD_LINE_3 );
379  }
380 
381  // return unprocessed events
382  return ( events ^ PROCEDURE_TIMEOUT_EVT );
383  }
384 
385  if ( events & START_DEVICE_EVT )
386  {
387  // Start the Device
388  VOID GAPCentralRole_StartDevice( (gapCentralRoleCB_t *) &glucCollRoleCB );
389 
390  // Register with bond manager after starting device
391  GAPBondMgr_Register( (gapBondCBs_t *) &glucCollBondCB );
392 
393  return ( events ^ START_DEVICE_EVT );
394  }
395 
396  if ( events & START_DISCOVERY_EVT )
397  {
398  if ( glucCollPairingStarted )
399  {
400  // Postpone discovery until pairing completes
401  glucCollDiscPostponed = TRUE;
402  }
403  else
404  {
405  glucCollCentralStartDiscovery( );
406  }
407  return ( events ^ START_DISCOVERY_EVT );
408  }
409 
410  // Discard unknown events
411  return 0;
412 }

437 /*********************************************************************
438  * @fn glucColl_HandleKeys
439  *
440  * @brief Handles all key events for this device.
441  *
442  * @param shift - true if in shift/alt.
443  * @param keys - bit field for key events. Valid entries:
444  * HAL_KEY_SW_2
445  * HAL_KEY_SW_1
446  *
447  * @return none
448  */
449 uint8 gStatus;
450 static void glucColl_HandleKeys( uint8 shift, uint8 keys )
451 {
452  (void)shift; // Intentionally unreferenced parameter
453 
454  if ( keys & HAL_KEY_UP )
455  {
456  // Start or stop discovery
457  if ( glucCollState != BLE_STATE_CONNECTED )
458  {
459  if ( !glucCollScanning )
460  {
461  glucCollScanning = TRUE;
462  glucCollScanRes = 0;
463 
464  LCD_WRITE_STRING( "Discovering...", HAL_LCD_LINE_1 );
465  LCD_WRITE_STRING( "", HAL_LCD_LINE_2 );
466  LCD_WRITE_STRING( "", HAL_LCD_LINE_3 );
467 
468  GAPCentralRole_StartDiscovery( DEFAULT_DISCOVERY_MODE,
469  DEFAULT_DISCOVERY_ACTIVE_SCAN,
470  DEFAULT_DISCOVERY_WHITE_LIST );
471  }
472  else
473  {
474  GAPCentralRole_CancelDiscovery();
475  }
476  }
477  else if ( glucCollState == BLE_STATE_CONNECTED &&
478  glucCollCharHdls == true &&
479  glucCollWritePending == false)
480  {
481  uint8 status;
482 #if GLUCOSE_FILTER_ENABLED
483  // Request number of records
484  status = glucoseCtlPntWriteFilter(CTL_PNT_OP_GET_NUM, CTL_PNT_OPER_RANGE,
485  DEFAULT_FILTER_TYPE, pFilter1, pFilter2);
486 #else
487  // Request number of records
488  status = glucoseCtlPntWrite(CTL_PNT_OP_GET_NUM, CTL_PNT_OPER_ALL);
489 #endif
490 
491  if(status == 0)
492  glucCollWritePending = true;
493  }
494  }
495 
496  if ( keys & HAL_KEY_LEFT )
497  {
498  // Display discovery results
499  if ( glucCollState != BLE_STATE_CONNECTED && !glucCollScanning && glucCollScanRes > 0 )
500  {
501  // Increment index of current result (with wraparound)
502  glucCollScanIdx++;
503  if ( glucCollScanIdx >= glucCollScanRes )
504  {
505  glucCollScanIdx = 0;
506  }
507 
508  LCD_WRITE_STRING_VALUE( "Device", glucCollScanIdx + 1,
509  10, HAL_LCD_LINE_1 );
510  LCD_WRITE_STRING( bdAddr2Str( glucCollDevList[glucCollScanIdx].addr ),
511  HAL_LCD_LINE_2 );
512  LCD_WRITE_STRING( "", HAL_LCD_LINE_3 );
513  }
514  else if ( glucCollState == BLE_STATE_CONNECTED )
515  {
516  uint8 status;
517 
518  // Abort
519  status = glucoseCtlPntWrite(CTL_PNT_OP_ABORT, CTL_PNT_OPER_NULL);
520 
521  if(status == 0)
522  glucCollWritePending = false;
523  }
524  }
525 
526  if ( keys & HAL_KEY_RIGHT )
527  {
528  // Request all records
529  if ( glucCollState == BLE_STATE_CONNECTED &&
530  glucCollCharHdls == true &&
531  glucCollWritePending == false)
532  {
533  uint8 status;
534 #if GLUCOSE_FILTER_ENABLED
535  status = glucoseCtlPntWriteFilter(CTL_PNT_OP_REQ, CTL_PNT_OPER_RANGE,
536  DEFAULT_FILTER_TYPE, pFilter1, pFilter2);
537 #else
538  status = glucoseCtlPntWrite(CTL_PNT_OP_REQ, CTL_PNT_OPER_ALL);
539 #endif
540  if(status == SUCCESS)
541  glucCollWritePending = true;
542  }
543  }
544 
545  if ( keys & HAL_KEY_CENTER )
546  {
547  uint8 addrType;
548  uint8 *peerAddr;
549 
550  // Connect or disconnect
551  if ( glucCollState == BLE_STATE_IDLE )
552  {
553  // if there is a scan result
554  if ( glucCollScanRes > 0 )
555  {
556  // connect to current device in scan result
557  peerAddr = glucCollDevList[glucCollScanIdx].addr;
558  addrType = glucCollDevList[glucCollScanIdx].addrType;
559 
560  glucCollState = BLE_STATE_CONNECTING;
561 
562  GAPCentralRole_EstablishLink( DEFAULT_LINK_HIGH_DUTY_CYCLE,
563  DEFAULT_LINK_WHITE_LIST,
564  addrType, peerAddr );
565 
566  LCD_WRITE_STRING( "Connecting", HAL_LCD_LINE_1 );
567  LCD_WRITE_STRING( bdAddr2Str( peerAddr ), HAL_LCD_LINE_2 );
568  LCD_WRITE_STRING( "", HAL_LCD_LINE_3 );
569  }
570  }
571  else if ( glucCollState == BLE_STATE_CONNECTING ||
572  glucCollState == BLE_STATE_CONNECTED )
573  {
574  // disconnect
575  glucCollState = BLE_STATE_DISCONNECTING;
576 
577  gStatus = GAPCentralRole_TerminateLink( glucCollConnHandle );
578 
579  LCD_WRITE_STRING( "Disconnecting", HAL_LCD_LINE_1 );
580  LCD_WRITE_STRING( "", HAL_LCD_LINE_2 );
581  LCD_WRITE_STRING( "", HAL_LCD_LINE_3 );
582  }
583  }
584 
585  if ( keys & HAL_KEY_DOWN )
586  {
587  // Clear stored records
588  if ( glucCollState == BLE_STATE_CONNECTED &&
589  glucCollCharHdls == true &&
590  glucCollWritePending == false)
591  {
592  uint8 status;
593 #if GLUCOSE_FILTER_ENABLED
594  status = glucoseCtlPntWriteFilter(CTL_PNT_OP_CLR, CTL_PNT_OPER_RANGE,
595  DEFAULT_FILTER_TYPE, pFilter1, pFilter2);
596 #else
597  status = glucoseCtlPntWrite(CTL_PNT_OP_CLR, CTL_PNT_OPER_ALL);
598 #endif
599  if(status == 0)
600  {
601  glucCollWritePending = true;
602  glucCollClearPending = true;
603  }
604  }
605  else if ( glucCollState != BLE_STATE_CONNECTED )
606  {
607  // erase all bonds
608  GAPBondMgr_SetParameter( GAPBOND_ERASE_ALLBONDS, 0, NULL );
609  LCD_WRITE_STRING( "Erasing bonds", HAL_LCD_LINE_1 );
610  LCD_WRITE_STRING( "", HAL_LCD_LINE_2 );
611  LCD_WRITE_STRING( "", HAL_LCD_LINE_3 );
612 
613  // initiate service discovery again
614  glucCollCharHdls = false;
615  }
616  }
617 }

619 /*********************************************************************
620  * @fn glucCollProcessGATTMsg
621  *
622  * @brief Process GATT messages
623  *
624  * @return none
625  */
626 static void glucCollProcessGATTMsg( gattMsgEvent_t *pMsg )
627 {
628  if ( pMsg->method == ATT_HANDLE_VALUE_NOTI ||
629  pMsg->method == ATT_HANDLE_VALUE_IND )
630  {
631  glucoseIndGattMsg( pMsg );
632  }
633  else if ( pMsg->method == ATT_READ_RSP ||
634  pMsg->method == ATT_WRITE_RSP)
635  {
636  if(glucCollCharHdls == true)
637  {
638  glucoseCtlPntGattMsg(pMsg);
639  }
640  else
641  {
642  glucCollConfigState = glucoseConfigGattMsg ( glucCollConfigState, pMsg );
643  }
644  }
645  else if(glucCollDiscState != DISC_IDLE)
646  {
647  glucCollDiscState = glucoseDiscGattMsg(glucCollDiscState, pMsg);
648  if ( glucCollDiscState == DISC_IDLE )
649  {
650  // Start characteristic configuration
651  glucCollConfigState = glucoseConfigNext( GLUCOSE_CONFIG_START );
652  }
653  }
654  else if (pMsg->method == ATT_ERROR_RSP)
655  {
656  if(glucCollCharHdls == true)
657  {
658  glucoseCtlPntGattMsg(pMsg);
659  }
660  else
661  {
662  glucCollConfigState = glucoseConfigGattMsg ( glucCollConfigState, pMsg );
663  }
664  }
665 }

904 /*********************************************************************
905  * @fn glucCollFindSvcUuid
906  *
907  * @brief Find a given UUID in an advertiser's service UUID list.
908  *
909  * @return TRUE if service UUID found
910  */
911 static bool glucCollFindSvcUuid( uint16 uuid, uint8 *pData, uint8 dataLen )
912 {
913  uint8 adLen;
914  uint8 adType;
915  uint8 *pEnd;
916 
917  pEnd = pData + dataLen - 1;
918 
919  // While end of data not reached
920  while ( pData < pEnd )
921  {
922  // Get length of next AD item
923  adLen = *pData++;
924  if ( adLen > 0 )
925  {
926  adType = *pData;
927 
928  // If AD type is for 16-bit service UUID
929  if ( adType == GAP_ADTYPE_16BIT_MORE || adType == GAP_ADTYPE_16BIT_COMPLETE )
930  {
931  pData++;
932  adLen--;
933 
934  // For each UUID in list
935  while ( adLen >= 2 && pData < pEnd )
936  {
937  // Check for match
938  if ( pData[0] == LO_UINT16(uuid) && pData[1] == HI_UINT16(uuid) )
939  {
940  // Match found
941  return TRUE;
942  }
943 
944  // Go to next
945  pData += 2;
946  adLen -= 2;
947  }
948 
949  // Handle possible erroneous extra byte in UUID list
950  if ( adLen == 1 )
951  {
952  pData++;
953  }
954  }
955  else
956  {
957  // Go to next item
958  pData += adLen;
959  }
960  }
961  }
962 
963  // Match not found
964  return FALSE;
965 }
