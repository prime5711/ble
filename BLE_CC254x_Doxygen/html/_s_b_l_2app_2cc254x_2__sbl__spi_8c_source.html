97 /**************************************************************************************************
98  * @fn sblRun
99  *
100  * @brief Serial Boot run code for the SPI transport.
101  *
102  * input parameters
103  *
104  * None.
105  *
106  * output parameters
107  *
108  * None.
109  *
110  * @return None.
111  */
112 static void sblRun(void)
113 {
114  halUARTCfg_t uartConfig;
115  uartConfig.callBackFunc = sblRxCB;
116 
117  HAL_DMA_SET_ADDR_DESC1234(dmaCh1234);
118 
119  HalUARTInitSPI();
120  HalUARTOpenSPI(&uartConfig);
121 
122  while (1)
123  {
124  HalUARTPollSPI();
125 
126  if (HAL_DMA_CHECK_IRQ(HAL_SPI_CH_TX))
127  {
128  HAL_DMA_CLEAR_IRQ(HAL_SPI_CH_TX);
129  HalUART_DMAIsrSPI();
130 
131  if (sblReset)
132  {
133  HAL_SYSTEM_RESET();
134  }
135  }
136  }
137 }
