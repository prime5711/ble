39 /*********************************************************************
40 NOTE: If polling is used, the hal_driver task schedules the KeyRead()
41 to occur every 100ms. This should be long enough to naturally
42 debounce the keys. The KeyRead() function remembers the key
43 state of the previous poll and will only return a non-zero
44 value if the key state changes.
45 
46 NOTE: If interrupts are used, the KeyRead() function is scheduled
47 25ms after the interrupt occurs by the ISR. This delay is used
48 for key debouncing. The ISR disables any further Key interrupt
49 until KeyRead() is executed. KeyRead() will re-enable Key
50 interrupts after executing. Unlike polling, when interrupts
51 are enabled, the previous key state is not remembered. This
52 means that KeyRead() will return the current state of the keys
53 (not a change in state of the keys).
54 
55 NOTE: If interrupts are used, the KeyRead() fucntion is scheduled by
56 the ISR. Therefore, the joystick movements will only be detected
57 during a pushbutton interrupt caused by S1 or the center joystick
58 pushbutton.
59 
60 NOTE: When a switch like S1 is pushed, the S1 signal goes from a normally
61 high state to a low state. This transition is typically clean. The
62 duration of the low state is around 200ms. When the signal returns
63 to the high state, there is a high likelihood of signal bounce, which
64 causes a unwanted interrupts. Normally, we would set the interrupt
65 edge to falling edge to generate an interrupt when S1 is pushed, but
66 because of the signal bounce, it is better to set the edge to rising
67 edge to generate an interrupt when S1 is released. The debounce logic
68 can then filter out the signal bounce. The result is that we typically
69 get only 1 interrupt per button push. This mechanism is not totally
70 foolproof because occasionally, signal bound occurs during the falling
71 edge as well. A similar mechanism is used to handle the joystick
72 pushbutton on the DB. For the EB, we do not have independent control
73 of the interrupt edge for the S1 and center joystick pushbutton. As
74 a result, only one or the other pushbuttons work reasonably well with
75 interrupts. The default is the make the S1 switch on the EB work more
76 reliably.

130 /**************************************************************************************************
131 * @fn HalKeyInit
132 *
133 * @brief Initilize Key Service
134 *
135 * @param none
136 *
137 * @return None
138 **************************************************************************************************/
139 void HalKeyInit( void )
140 {
141  /* Initialize previous key to 0 */
142  halKeySavedKeys = 0;
143 
144  HAL_KEY_SW_1_SEL &= ~(HAL_KEY_SW_1_BIT); /* Set pin function to GPIO */
145  HAL_KEY_SW_1_DIR &= ~(HAL_KEY_SW_1_BIT); /* Set pin direction to Input */
146 
147  HAL_KEY_SW_2_SEL &= ~(HAL_KEY_SW_2_BIT); /* Set pin function to GPIO */
148  HAL_KEY_SW_2_DIR &= ~(HAL_KEY_SW_2_BIT); /* Set pin direction to Input */
149 
150  HAL_KEY_SW_3_SEL &= ~(HAL_KEY_SW_3_BIT); /* Set pin function to GPIO */
151  HAL_KEY_SW_3_DIR &= ~(HAL_KEY_SW_3_BIT); /* Set pin direction to Input */
152 
153 
154  /* Initialize callback function */
155  pHalKeyProcessFunction = NULL;
156 
157  /* Start with key is not configured */
158  HalKeyConfigured = FALSE;
159 }

162 /**************************************************************************************************
163 * @fn HalKeyConfig
164 *
165 * @brief Configure the Key serivce
166 *
167 * @param interruptEnable - TRUE/FALSE, enable/disable interrupt
168 * cback - pointer to the CallBack function
169 *
170 * @return None
171 **************************************************************************************************/
172 void HalKeyConfig (bool interruptEnable, halKeyCBack_t cback)
173 {
174  /* Enable/Disable Interrupt or */
175  Hal_KeyIntEnable = interruptEnable;
176 
177  /* Register the callback fucntion */
178  pHalKeyProcessFunction = cback;
179 
180  /* Determine if interrupt is enable or not */
181  if (Hal_KeyIntEnable)
182  {
183  /* Rising/Falling edge configuratinn */
184  PICTL |= (HAL_KEY_SW_1_EDGEBIT | HAL_KEY_SW_2_3_EDGEBIT); /* Set the edge bit to set falling edge to give int */
185 
186  HAL_KEY_SW_1_ICTL |= HAL_KEY_SW_1_ICTLBIT; /* enable interrupt generation at port */
187  HAL_KEY_SW_1_IEN |= HAL_KEY_SW_1_IENBIT; /* enable CPU interrupt */
188  HAL_KEY_SW_1_PXIFG = ~(HAL_KEY_SW_1_BIT); /* Clear any pending interrupt */
189 
190  HAL_KEY_SW_2_ICTL |= HAL_KEY_SW_2_ICTLBIT; /* enable interrupt generation at port */
191  HAL_KEY_SW_2_IEN |= HAL_KEY_SW_2_IENBIT; /* enable CPU interrupt */
192  HAL_KEY_SW_2_PXIFG = ~(HAL_KEY_SW_2_BIT); /* Clear any pending interrupt */
193 
194  HAL_KEY_SW_3_ICTL |= HAL_KEY_SW_3_ICTLBIT; /* enable interrupt generation at port */
195  HAL_KEY_SW_3_IEN |= HAL_KEY_SW_3_IENBIT; /* enable CPU interrupt */
196  HAL_KEY_SW_3_PXIFG = (uint8)(~(HAL_KEY_SW_3_BIT)); /* Clear any pending interrupt */
197 
198  /* Do this only after the hal_key is configured - to work with sleep stuff */
199  if (HalKeyConfigured == TRUE)
200  {
201  osal_stop_timerEx(Hal_TaskID, HAL_KEY_EVENT); /* Cancel polling if active */
202  }
203  }
204  else /* Interrupts NOT enabled */
205  {
206  HAL_KEY_SW_1_ICTL &= ~(HAL_KEY_SW_1_ICTLBIT); /* don't generate interrupt */
207  HAL_KEY_SW_1_IEN &= ~(HAL_KEY_SW_1_IENBIT); /* Clear interrupt enable bit */
208 
209  HAL_KEY_SW_2_ICTL &= ~(HAL_KEY_SW_2_ICTLBIT); /* don't generate interrupt */
210  HAL_KEY_SW_2_IEN &= ~(HAL_KEY_SW_2_IENBIT); /* Clear interrupt enable bit */
211 
212  HAL_KEY_SW_3_ICTL &= ~(HAL_KEY_SW_3_ICTLBIT); /* don't generate interrupt */
213  HAL_KEY_SW_3_IEN &= ~(HAL_KEY_SW_3_IENBIT); /* Clear interrupt enable bit */
214 
215  osal_set_event(Hal_TaskID, HAL_KEY_EVENT);
216  }
217 
218  /* Key now is configured */
219  HalKeyConfigured = TRUE;
220 }

223 /**************************************************************************************************
224 * @fn HalKeyRead
225 *
226 * @brief Read the current value of a key
227 *
228 * @param None
229 *
230 * @return keys - current keys status
231 **************************************************************************************************/
232 uint8 HalKeyRead ( void )
233 {
234  uint8 keys = 0;
235 
236  if (!(HAL_KEY_SW_1_PORT & HAL_KEY_SW_1_BIT)) /* Key is active low */
237  {
238  keys |= HAL_KEY_SW_1;
239  }
240  if (!(HAL_KEY_SW_2_PORT & HAL_KEY_SW_2_BIT)) /* Key is active low */
241  {
242  keys |= HAL_KEY_SW_2;
243  }
244  if (!(HAL_KEY_SW_3_PORT & HAL_KEY_SW_3_BIT)) /* Key is active low */
245  {
246  keys |= HAL_KEY_SW_3;
247  }
248 
249  return keys;
250 }

253 /**************************************************************************************************
254 * @fn HalKeyPoll
255 *
256 * @brief Called by hal_driver to poll the keys
257 *
258 * @param None
259 *
260 * @return None
261 **************************************************************************************************/
262 void HalKeyPoll (void)
263 {
264  uint8 keys = 0;
265  uint8 notify = 0;
266 
267  if (!(HAL_KEY_SW_1_PORT & HAL_KEY_SW_1_BIT)) /* Key is active low */
268  {
269  keys |= HAL_KEY_SW_1;
270  }
271  if (!(HAL_KEY_SW_2_PORT & HAL_KEY_SW_2_BIT)) /* Key is active low */
272  {
273  keys |= HAL_KEY_SW_2;
274  }
275  if (!(HAL_KEY_SW_3_PORT & HAL_KEY_SW_3_BIT)) /* Key is active low */
276  {
277  keys |= HAL_KEY_SW_3;
278  }
279 
280  (void) keys;
281 
282  /* If interrupts are not enabled, previous key status and current key status
283  * are compared to find out if a key has changed status.
284  */
285  if (!Hal_KeyIntEnable)
286  {
287  if (keys == halKeySavedKeys)
288  {
289  /* Exit - since no keys have changed */
290  return;
291  }
292  else
293  {
294  notify = 1;
295  }
296  }
297  else
298  {
299  /* Key interrupt handled here */
300  if (keys)
301  {
302  notify = 1;
303  }
304  }
305 
306  /* Store the current keys for comparation next time */
307  halKeySavedKeys = keys;
308 
309  /* Invoke Callback if new keys were depressed */
310  if (notify && (pHalKeyProcessFunction))
311  {
312  (pHalKeyProcessFunction) (keys, HAL_KEY_STATE_NORMAL);
313 
314  }
315 }

317 /**************************************************************************************************
318 * @fn halProcessKeyInterrupt
319 *
320 * @brief Checks to see if it's a valid key interrupt, saves interrupt driven key states for
321 * processing by HalKeyRead(), and debounces keys by scheduling HalKeyRead() 25ms later.
322 *
323 * @param
324 *
325 * @return
326 **************************************************************************************************/
327 void halProcessKeyInterrupt (void)
328 {
329  bool valid=FALSE;
330 
331  if( HAL_KEY_SW_1_PXIFG & HAL_KEY_SW_1_BIT) /* Interrupt Flag has been set by SW1 */
332  {
333  HAL_KEY_SW_1_PXIFG = ~(HAL_KEY_SW_1_BIT); /* Clear Interrupt Flag */
334  valid = TRUE;
335  }
336 
337  if (HAL_KEY_SW_2_PXIFG & HAL_KEY_SW_2_BIT) /* Interrupt Flag has been set by SW2 */
338  {
339  HAL_KEY_SW_2_PXIFG = ~(HAL_KEY_SW_2_BIT); /* Clear Interrupt Flag */
340  valid = TRUE;
341  }
342  if (HAL_KEY_SW_3_PXIFG & HAL_KEY_SW_3_BIT) /* Interrupt Flag has been set by SW3 */
343  {
344  HAL_KEY_SW_3_PXIFG = (uint8)(~(HAL_KEY_SW_3_BIT)); /* Clear Interrupt Flag */
345  valid = TRUE;
346  }
347 
348  if (valid)
349  {
350  osal_start_timerEx (Hal_TaskID, HAL_KEY_EVENT, HAL_KEY_DEBOUNCE_VALUE);
351  }
352 }
