122 /**************************************************************************************************
123  * @fn HalKeyConfig
124  *
125  * @brief This function is called by HalDriverInit to initialize the H/W keys.
126  *
127  * input parameters
128  *
129  * @param interruptEnable - TRUE/FALSE to enable the key interrupt.
130  * @param cback - The callback function for the key change event.
131  *
132  * output parameters
133  *
134  * None.
135  *
136  * @return None.
137  **************************************************************************************************
138  */
139 void HalKeyConfig(bool interruptEnable, halKeyCBack_t cback)
140 {
141  if ((Hal_KeyIntEnable = interruptEnable))
142  {
143  HAL_KEY_CLR_INT(); // Clear spurious ints.
144  PICTL |= 0x01; // P1ICONL: Falling edge ints on pins 0-3.
145  P1IEN |= PUSH1_BV | PUSH2_BV; // Enable specific P1 bits for ints by bit mask.
146  IEN2 |= 0x10; // Enable general P1 interrupts.
147  }
148  else
149  {
150  (void)osal_set_event(Hal_TaskID, HAL_KEY_EVENT);
151  }
152 
153  pHalKeyProcessFunction = cback;
154 }

156 /**************************************************************************************************
157  * @fn HalKeyPoll
158  *
159  * @brief This function is called by Hal_ProcessEvent() on a HAL_KEY_EVENT.
160  *
161  * input parameters
162  *
163  * None.
164  *
165  * output parameters
166  *
167  * None.
168  *
169  * @return None.
170  **************************************************************************************************
171  */
172 void HalKeyPoll(void)
173 {
174  uint8 newKeys;
175 
176  if (Hal_KeyIntEnable)
177  {
178  halIntState_t intState;
179  HAL_ENTER_CRITICAL_SECTION(intState);
180  newKeys = isrKeys;
181  isrKeys = 0;
182  HAL_EXIT_CRITICAL_SECTION(intState);
183  }
184  else
185  {
186  uint8 keys = HalKeyRead();
187  newKeys = (halKeys ^ keys) & keys;
188  halKeys = keys;
189  }
190 
191  if (newKeys && pHalKeyProcessFunction)
192  {
193  (pHalKeyProcessFunction)(newKeys, HAL_KEY_STATE_NORMAL);
194  }
195 }

213 uint8 HalKeyRead(void)

220  }
221

224  keys |= HAL_KEY_SW_2;
225  }
226 
227  return keys;
