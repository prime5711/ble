292 /*********************************************************************
293  * @fn KeyFobApp_Init
294  *
295  * @brief Initialization function for the Key Fob App Task.
296  * This is called during initialization and should contain
297  * any application specific initialization (ie. hardware
298  * initialization/setup, table initialization, power up
299  * notificaiton ... ).
300  *
301  * @param task_id - the ID assigned by OSAL. This ID should be
302  * used to send messages and set timers.
303  *
304  * @return none
305  */
306 void KeyFobApp_Init( uint8 task_id )
307 {
308  keyfobapp_TaskID = task_id;
309 
310  // Setup the GAP
311  VOID GAP_SetParamValue( TGAP_CONN_PAUSE_PERIPHERAL, DEFAULT_CONN_PAUSE_PERIPHERAL );
312 
313  // Setup the GAP Peripheral Role Profile
314  {
315  // For the CC2540DK-MINI keyfob, device doesn't start advertising until button is pressed
316  uint8 initial_advertising_enable = FALSE;
317 
318  // By setting this to zero, the device will go into the waiting state after
319  // being discoverable for 30.72 second, and will not being advertising again
320  // until the enabler is set back to TRUE
321  uint16 gapRole_AdvertOffTime = 0;
322 
323  uint8 enable_update_request = DEFAULT_ENABLE_UPDATE_REQUEST;
324  uint16 desired_min_interval = DEFAULT_DESIRED_MIN_CONN_INTERVAL;
325  uint16 desired_max_interval = DEFAULT_DESIRED_MAX_CONN_INTERVAL;
326  uint16 desired_slave_latency = DEFAULT_DESIRED_SLAVE_LATENCY;
327  uint16 desired_conn_timeout = DEFAULT_DESIRED_CONN_TIMEOUT;
328 
329  // Set the GAP Role Parameters
330  GAPRole_SetParameter( GAPROLE_ADVERT_ENABLED, sizeof( uint8 ), &initial_advertising_enable );
331  GAPRole_SetParameter( GAPROLE_ADVERT_OFF_TIME, sizeof( uint16 ), &gapRole_AdvertOffTime );
332 
333  GAPRole_SetParameter( GAPROLE_SCAN_RSP_DATA, sizeof ( deviceName ), deviceName );
334  GAPRole_SetParameter( GAPROLE_ADVERT_DATA, sizeof( advertData ), advertData );
335 
336  GAPRole_SetParameter( GAPROLE_PARAM_UPDATE_ENABLE, sizeof( uint8 ), &enable_update_request );
337  GAPRole_SetParameter( GAPROLE_MIN_CONN_INTERVAL, sizeof( uint16 ), &desired_min_interval );
338  GAPRole_SetParameter( GAPROLE_MAX_CONN_INTERVAL, sizeof( uint16 ), &desired_max_interval );
339  GAPRole_SetParameter( GAPROLE_SLAVE_LATENCY, sizeof( uint16 ), &desired_slave_latency );
340  GAPRole_SetParameter( GAPROLE_TIMEOUT_MULTIPLIER, sizeof( uint16 ), &desired_conn_timeout );
341  }
342 
343  // Set the GAP Attributes
344  GGS_SetParameter( GGS_DEVICE_NAME_ATT, GAP_DEVICE_NAME_LEN, attDeviceName );
345 
346  // Setup the GAP Bond Manager
347  {
348  uint8 pairMode = GAPBOND_PAIRING_MODE_WAIT_FOR_REQ;
349  uint8 mitm = TRUE;
350  uint8 ioCap = GAPBOND_IO_CAP_DISPLAY_ONLY;
351  uint8 bonding = TRUE;
352 
353  GAPBondMgr_SetParameter( GAPBOND_PAIRING_MODE, sizeof ( uint8 ), &pairMode );
354  GAPBondMgr_SetParameter( GAPBOND_MITM_PROTECTION, sizeof ( uint8 ), &mitm );
355  GAPBondMgr_SetParameter( GAPBOND_IO_CAPABILITIES, sizeof ( uint8 ), &ioCap );
356  GAPBondMgr_SetParameter( GAPBOND_BONDING_ENABLED, sizeof ( uint8 ), &bonding );
357  }
358 
359  // Initialize GATT attributes
360  GGS_AddService( GATT_ALL_SERVICES ); // GAP
361  GATTServApp_AddService( GATT_ALL_SERVICES ); // GATT attributes
362  DevInfo_AddService(); // Device Information Service
363  ProxReporter_AddService( GATT_ALL_SERVICES ); // Proximity Reporter Profile
364  Batt_AddService( ); // Battery Service
365  Accel_AddService( GATT_ALL_SERVICES ); // Accelerometer Profile
366  SK_AddService( GATT_ALL_SERVICES ); // Simple Keys Profile
367 
368  keyfobProximityState = KEYFOB_PROXSTATE_INITIALIZED;
369 
370  // Initialize Tx Power Level characteristic in Proximity Reporter
371  {
372  int8 initialTxPowerLevel = 0;
373 
374  ProxReporter_SetParameter( PP_TX_POWER_LEVEL, sizeof ( int8 ), &initialTxPowerLevel );
375  }
376 
377  keyfobAlertState = ALERT_STATE_OFF;
378 
379  // make sure buzzer is off
380  buzzerStop();
381 
382  // makes sure LEDs are off
383  HalLedSet( (HAL_LED_1 | HAL_LED_2), HAL_LED_MODE_OFF );
384 
385  // For keyfob board set GPIO pins into a power-optimized state
386  // Note that there is still some leakage current from the buzzer,
387  // accelerometer, LEDs, and buttons on the PCB.
388 
389  P0SEL = 0; // Configure Port 0 as GPIO
390  P1SEL = 0x40; // Configure Port 1 as GPIO, except P1.6 for peripheral function for buzzer
391  P2SEL = 0; // Configure Port 2 as GPIO
392 
393  P0DIR = 0xFC; // Port 0 pins P0.0 and P0.1 as input (buttons),
394  // all others (P0.2-P0.7) as output
395  P1DIR = 0xFF; // All port 1 pins (P1.0-P1.7) as output
396  P2DIR = 0x1F; // All port 1 pins (P2.0-P2.4) as output
397 
398  P0 = 0x03; // All pins on port 0 to low except for P0.0 and P0.1 (buttons)
399  P1 = 0; // All pins on port 1 to low
400  P2 = 0; // All pins on port 2 to low
401 
402 
403  // initialize the ADC for battery reads
404  HalAdcInit();
405 
406  // Register for all key events - This app will handle all key events
407  RegisterForKeys( keyfobapp_TaskID );
408 
409 #if defined ( DC_DC_P0_7 )
410 
411  // Enable stack to toggle bypass control on TPS62730 (DC/DC converter)
412  HCI_EXT_MapPmIoPortCmd( HCI_EXT_PM_IO_PORT_P0, HCI_EXT_PM_IO_PORT_PIN7 );
413 
414 #endif // defined ( DC_DC_P0_7 )
415 
416  // Setup a delayed profile startup
417  osal_start_timerEx( keyfobapp_TaskID, KFD_START_DEVICE_EVT, STARTDELAY );
418 }

420 /*********************************************************************
421  * @fn KeyFobApp_ProcessEvent
422  *
423  * @brief Key Fob Application Task event processor. This function
424  * is called to process all events for the task. Events
425  * include timers, messages and any other user defined events.
426  *
427  * @param task_id - The OSAL assigned task ID.
428  * @param events - events to process. This is a bit map and can
429  * contain more than one event.
430  *
431  * @return none
432  */
433 uint16 KeyFobApp_ProcessEvent( uint8 task_id, uint16 events )
434 {
435  if ( events & SYS_EVENT_MSG )
436  {
437  uint8 *pMsg;
438 
439  if ( (pMsg = osal_msg_receive( keyfobapp_TaskID )) != NULL )
440  {
441  keyfobapp_ProcessOSALMsg( (osal_event_hdr_t *)pMsg );
442 
443  // Release the OSAL message
444  VOID osal_msg_deallocate( pMsg );
445  }
446 
447  // return unprocessed events
448  return (events ^ SYS_EVENT_MSG);
449  }
450 
451  if ( events & KFD_START_DEVICE_EVT )
452  {
453  // Start the Device
454  VOID GAPRole_StartDevice( &keyFob_PeripheralCBs );
455 
456  // Start Bond Manager
457  VOID GAPBondMgr_Register( &keyFob_BondMgrCBs );
458 
459  // Start the Proximity Profile
460  VOID ProxReporter_RegisterAppCBs( &keyFob_ProximityCBs );
461 
462  // Set timer for first battery read event
463  osal_start_timerEx( keyfobapp_TaskID, KFD_BATTERY_CHECK_EVT, BATTERY_CHECK_PERIOD );
464 
465  // Start the Accelerometer Profile
466  VOID Accel_RegisterAppCBs( &keyFob_AccelCBs );
467 
468  //Set the proximity attribute values to default
469  ProxReporter_SetParameter( PP_LINK_LOSS_ALERT_LEVEL, sizeof ( uint8 ), &keyfobProxLLAlertLevel );
470  ProxReporter_SetParameter( PP_IM_ALERT_LEVEL, sizeof ( uint8 ), &keyfobProxIMAlertLevel );
471  ProxReporter_SetParameter( PP_TX_POWER_LEVEL, sizeof ( int8 ), &keyfobProxTxPwrLevel );
472 
473  // Set LED1 on to give feedback that the power is on, and a timer to turn off
474  HalLedSet( HAL_LED_1, HAL_LED_MODE_ON );
475  osal_pwrmgr_device( PWRMGR_ALWAYS_ON ); // To keep the LED on continuously.
476  osal_start_timerEx( keyfobapp_TaskID, KFD_POWERON_LED_TIMEOUT_EVT, 1000 );
477 
478  return ( events ^ KFD_START_DEVICE_EVT );
479  }
480 
481  if ( events & KFD_POWERON_LED_TIMEOUT_EVT )
482  {
483  osal_pwrmgr_device( PWRMGR_BATTERY ); // Revert to battery mode after LED off
484  HalLedSet( HAL_LED_1, HAL_LED_MODE_OFF );
485  return ( events ^ KFD_POWERON_LED_TIMEOUT_EVT );
486  }
487 
488  if ( events & KFD_ACCEL_READ_EVT )
489  {
490  bStatus_t status = Accel_GetParameter( ACCEL_ENABLER, &accelEnabler );
491 
492  if (status == SUCCESS)
493  {
494  if ( accelEnabler )
495  {
496  // Restart timer
497  if ( ACCEL_READ_PERIOD )
498  {
499  osal_start_timerEx( keyfobapp_TaskID, KFD_ACCEL_READ_EVT, ACCEL_READ_PERIOD );
500  }
501 
502  // Read accelerometer data
503  accelRead();
504  }
505  else
506  {
507  // Stop the acceleromter
508  osal_stop_timerEx( keyfobapp_TaskID, KFD_ACCEL_READ_EVT);
509  }
510  }
511  else
512  {
513  //??
514  }
515  return (events ^ KFD_ACCEL_READ_EVT);
516  }
517 
518  if ( events & KFD_BATTERY_CHECK_EVT )
519  {
520  // Restart timer
521  if ( BATTERY_CHECK_PERIOD )
522  {
523  osal_start_timerEx( keyfobapp_TaskID, KFD_BATTERY_CHECK_EVT, BATTERY_CHECK_PERIOD );
524  }
525 
526  // perform battery level check
527  Batt_MeasLevel( );
528 
529  return (events ^ KFD_BATTERY_CHECK_EVT);
530  }
531 
532  if ( events & KFD_TOGGLE_BUZZER_EVT )
533  {
534  // if this event was triggered while buzzer is on, turn it off, increment beep_count,
535  // check whether max has been reached, and if not set the OSAL timer for next event to
536  // turn buzzer back on.
537  if ( buzzer_state == BUZZER_ON )
538  {
539  buzzerStop();
540  buzzer_state = BUZZER_OFF;
541  buzzer_beep_count++;
542  #if defined ( POWER_SAVING )
543  osal_pwrmgr_device( PWRMGR_BATTERY );
544  #endif
545 
546  // check to see if buzzer has beeped maximum number of times
547  // if it has, then don't turn it back on
548  if ( ( buzzer_beep_count < BUZZER_MAX_BEEPS ) &&
549  ( ( keyfobProximityState == KEYFOB_PROXSTATE_LINK_LOSS ) ||
550  ( keyfobProximityState == KEYFOB_PROXSTATE_PATH_LOSS ) ) )
551  {
552  osal_start_timerEx( keyfobapp_TaskID, KFD_TOGGLE_BUZZER_EVT, 800 );
553  }
554  }
555  else if ( keyfobAlertState != ALERT_STATE_OFF )
556  {
557  // if this event was triggered while the buzzer is off then turn it on if appropriate
558  keyfobapp_PerformAlert();
559  }
560 
561  return (events ^ KFD_TOGGLE_BUZZER_EVT);
562  }
563 
564 
565 #if defined ( PLUS_BROADCASTER )
566  if ( events & KFD_ADV_IN_CONNECTION_EVT )
567  {
568  uint8 turnOnAdv = TRUE;
569  // Turn on advertising while in a connection
570  GAPRole_SetParameter( GAPROLE_ADVERT_ENABLED, sizeof( uint8 ), &turnOnAdv );
571  }
572 #endif
573 
574  // Discard unknown events
575  return 0;
576 }

597 /*********************************************************************
598  * @fn keyfobapp_HandleKeys
599  *
600  * @brief Handles all key events for this device.
601  *
602  * @param shift - true if in shift/alt.
603  * @param keys - bit field for key events. Valid entries:
604  * HAL_KEY_SW_2
605  * HAL_KEY_SW_1
606  *
607  * @return none
608  */
609 static void keyfobapp_HandleKeys( uint8 shift, uint8 keys )
610 {
611  uint8 SK_Keys = 0;
612 
613  (void)shift; // Intentionally unreferenced parameter
614 
615  if ( keys & HAL_KEY_SW_1 )
616  {
617  SK_Keys |= SK_KEY_LEFT;
618 
619  // if is active, pressing the left key should toggle
620  // stop the alert

824 /*********************************************************************
825  * @fn peripheralStateNotificationCB
826  *
827  * @brief Notification from the profile of a state change.
828  *
829  * @param newState - new state
830  *
831  * @return none
832  */
833 static void peripheralStateNotificationCB( gaprole_States_t newState )
834 {
835  uint16 connHandle = INVALID_CONNHANDLE;
836  uint8 valFalse = FALSE;
837 
838  if ( gapProfileState != newState )
839  {
840  switch( newState )
841  {
842  case GAPROLE_STARTED:
843  {
844  // Set the system ID from the bd addr
845  uint8 systemId[DEVINFO_SYSTEM_ID_LEN];
846  GAPRole_GetParameter(GAPROLE_BD_ADDR, systemId);
847 
848  // shift three bytes up
849  systemId[7] = systemId[5];
850  systemId[6] = systemId[4];
851  systemId[5] = systemId[3];
852 
853  // set middle bytes to zero
854  systemId[4] = 0;
855  systemId[3] = 0;
856 
857  DevInfo_SetParameter(DEVINFO_SYSTEM_ID, DEVINFO_SYSTEM_ID_LEN, systemId);
858  }
859  break;
860 
861  //if the state changed to connected, initially assume that keyfob is in range
862  case GAPROLE_ADVERTISING:
863  {
864  // Visual feedback that we are advertising.
865  HalLedSet( HAL_LED_2, HAL_LED_MODE_ON );
866  }
867  break;
868 
869  //if the state changed to connected, initially assume that keyfob is in range
870  case GAPROLE_CONNECTED:
871  {
872  // set the proximity state to either path loss alert or in range depending
873  // on the value of keyfobProxIMAlertLevel (which was set by proximity monitor)
874  if( keyfobProxIMAlertLevel != PP_ALERT_LEVEL_NO )
875  {
876  keyfobProximityState = KEYFOB_PROXSTATE_PATH_LOSS;
877  // perform alert
878  keyfobapp_PerformAlert();
879  buzzer_beep_count = 0;
880  }
881  else // if keyfobProxIMAlertLevel == PP_ALERT_LEVEL_NO
882  {
883  keyfobProximityState = KEYFOB_PROXSTATE_CONNECTED_IN_RANGE;
884  keyfobapp_StopAlert();

888 
889  #if defined ( PLUS_BROADCASTER )
