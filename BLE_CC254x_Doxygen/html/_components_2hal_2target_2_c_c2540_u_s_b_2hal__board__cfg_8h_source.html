181 /* Setting Clocks */
182 
183 // switch to the 16MHz HSOSC and wait until it is stable
184 #define SET_OSC_TO_HSOSC() \
185 { \
186  CLKCONCMD = (CLKCONCMD & 0x80) | CLKCONCMD_16MHZ; \
187  while ( (CLKCONSTA & ~0x80) != CLKCONCMD_16MHZ ); \
188 }
189 
190 // switch to the 32MHz XOSC and wait until it is stable
191 #define SET_OSC_TO_XOSC() \
192 { \
193  CLKCONCMD = (CLKCONCMD & 0x80) | CLKCONCMD_32MHZ; \
194  while ( (CLKCONSTA & ~0x80) != CLKCONCMD_32MHZ ); \
195 }
196 
197 // set 32kHz OSC and wait until it is stable
198 #define SET_32KHZ_OSC() \
199 { \
200  CLKCONCMD = (CLKCONCMD & ~0x80) | OSC_32KHZ; \
201  while ( (CLKCONSTA & 0x80) != OSC_32KHZ ); \
202 }
203 
204 #define START_HSOSC_XOSC() \
205 { \
206  SLEEPCMD &= ~OSC_PD; /* start 16MHz RCOSC & 32MHz XOSC */ \
207  while (!(SLEEPSTA & XOSC_STB)); /* wait for stable 32MHz XOSC */ \
208 }
209 
210 #define STOP_HSOSC() \
211 { \
212  SLEEPCMD |= OSC_PD; /* stop 16MHz RCOSC */ \
213 }

215 /* Board Initialization */
216 #if defined HAL_USB_BOOT_CODE
217 #define HAL_BOARD_INIT() \
218 { \
219  /* Set to 16Mhz to set 32kHz OSC, then back to 32MHz */ \
220  START_HSOSC_XOSC(); \
221  SET_OSC_TO_HSOSC(); \
222  SET_32KHZ_OSC(); \
223  SET_OSC_TO_XOSC(); \
224  STOP_HSOSC(); \
225  \
226  /* Disconnect D+ signal to host. */ \
227  HAL_USB_PULLUP_DISABLE(); \
