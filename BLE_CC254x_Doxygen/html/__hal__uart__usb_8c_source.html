221 /***********************************************************************************
222 * @fn halUartPollEvt
223 *
224 * @brief Poll for USB events which are not directly related to the UART.
225 *
226 * @param none
227 *
228 * @return none
229 */
230 static void halUartPollEvt(void)
231 {
232  // Handle reset signaling on the bus
233  if (USBIRQ_GET_EVENT_MASK() & USBIRQ_EVENT_RESET)
234  {
235  USBIRQ_CLEAR_EVENTS(USBIRQ_EVENT_RESET);
236  usbfwResetHandler();
237  }
238 
239  // Handle packets on EP0
240  if (USBIRQ_GET_EVENT_MASK() & USBIRQ_EVENT_SETUP)
241  {
242  USBIRQ_CLEAR_EVENTS(USBIRQ_EVENT_SETUP);
243  usbfwSetupHandler();
244  }
245 
246  // Handle USB suspend
247  if (USBIRQ_GET_EVENT_MASK() & USBIRQ_EVENT_SUSPEND)
248  {
249  // Clear USB suspend interrupt
250  USBIRQ_CLEAR_EVENTS(USBIRQ_EVENT_SUSPEND);
251 
252 #if HAL_UART_USB_SUSPEND
253  // Take the chip into PM1 until a USB resume is deteceted.
254  usbsuspEnter();
255 #endif
256 
257  // Running again; first clear RESUME interrupt
258  USBIRQ_CLEAR_EVENTS(USBIRQ_EVENT_RESUME);
259  }
260 }

262 /***********************************************************************************
263 * @fn halUartPollRx
264 *
265 * @brief Poll for data from USB.
266 *
267 * @param none
268 *
269 * @return none
270 */
271 static void halUartPollRx(void)
272 {
273  uint8 cnt;
274  uint8 ep = USBFW_GET_SELECTED_ENDPOINT();
275  USBFW_SELECT_ENDPOINT(4);
276 
277  // If the OUT endpoint has received a complete packet.
278  if (USBFW_OUT_ENDPOINT_DISARMED())
279  {
280  halIntState_t intState;
281 
282  HAL_ENTER_CRITICAL_SECTION(intState);
283  // Get length of USB packet, this operation must not be interrupted.
284  cnt = USBFW_GET_OUT_ENDPOINT_COUNT_LOW();
285  cnt += USBFW_GET_OUT_ENDPOINT_COUNT_HIGH() >> 8;
286  HAL_EXIT_CRITICAL_SECTION(intState);
287 
288  while (cnt--)
289  {
290  halUartRxQ[halUartRxT++] = USBF4;
291  }
292  USBFW_ARM_OUT_ENDPOINT();
293 
294 #if !defined HAL_SB_BOOT_CODE
295  // If the USB has transferred in more Rx bytes, reset the Rx idle timer.
296 
297  // Re-sync the shadow on any 1st byte(s) received.
298  if (rxTick == 0)
299  {
300  rxShdw = ST0;
301  }
302  rxTick = HAL_UART_USB_IDLE;
303 #endif
304  }
305 #if !defined HAL_SB_BOOT_CODE
306  else if (rxTick)
307  {
308  // Use the LSB of the sleep timer (ST0 must be read first anyway).
309  uint8 decr = ST0 - rxShdw;
310 
311  if (rxTick > decr)
312  {
313  rxTick -= decr;
314  rxShdw = ST0;
315  }
316  else
317  {
318  rxTick = 0;
319  }
320  }
321 
322  {
323  uint8 evt = 0;
324  cnt = halUartRxT - halUartRxH;
325 
326  if (cnt >= HAL_UART_USB_HIGH)
327  {
328  evt = HAL_UART_RX_ABOUT_FULL;
329  }
330  else if (cnt && !rxTick)
331  {
332  evt = HAL_UART_RX_TIMEOUT;
333  }
334 
335  if (evt && (NULL != usbCB))
336  {
337  usbCB(0, evt);
338  }
339  }
340 #endif
341 
342  USBFW_SELECT_ENDPOINT(ep);
343 }

345 /***********************************************************************************
346 * @fn halUartPollTx
347 *
348 * @brief Poll for data to USB.
349 *
350 * @param none
351 *
352 * @return none
353 */
354 static void halUartPollTx(void)
355 {
356  uint8 ep = USBFW_GET_SELECTED_ENDPOINT();
357  USBFW_SELECT_ENDPOINT(4);
358 
359  // If the IN endpoint is ready to accept data.
360  if (USBFW_IN_ENDPOINT_DISARMED())
361  {
362  if (halUartTxT == halUartTxH)
363  {
364 #if !defined HAL_SB_BOOT_CODE
365  if (!usbTxMT && usbCB)
366  {
367  usbTxMT = TRUE;
368  usbCB(0, HAL_UART_TX_EMPTY);
369  }
370 #endif
371  }
372  else
373  {
374  uint8 max = HAL_UART_USB_TX_MAX;
375 
376  do
377  {
378  USBF4 = halUartTxQ[halUartTxH++];
379  } while ((halUartTxH != halUartTxT) && (0 != --max));
380 
381  USBFW_ARM_IN_ENDPOINT();
382  }
383  }
384 
385  USBFW_SELECT_ENDPOINT(ep);
386 }
