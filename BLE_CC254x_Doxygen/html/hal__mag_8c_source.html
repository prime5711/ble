135 /**************************************************************************************************
136  * @fn HalMagStatus
137  *
138  * @brief Read the magnetometer status
139  *
140  * @return none
141  **************************************************************************************************/
142 Magnetometer_States_t HalMagStatus( void )
143 {
144  if( sensorState == MAG3110_IDLE)
145  {
146  uint8 status = 0;
147 
148  HalMagSelect();
149  HalSensorReadReg( MAG_REG_ADDR_DR_STATUS, &status, 1);
150  if(status!=0)
151  {
152  sensorState = MAG3110_DATA_READY;
153  }
154  }
155 
156  return sensorState;
157 }

192 /**************************************************************************************************
193 * @fn HalMagRead
194 *
195 * @brief Read data from the magnetometer
196 *
197 * @param pBuf - buffer to hold the data
198 *
199 * @return TRUE if valid data
200 */
201 bool HalMagRead(uint8 *pBuf)
202 {
203  uint8 tmp[MAG_REG_READ_ALL_LEN];
204  bool f;
205 
206  HalMagSelect();
207 
208  f = HalSensorReadReg(MAG_REG_ADDR_READ_START,tmp,MAG_REG_READ_ALL_LEN);
209  if (f)
210  {
211  // Swap bytes in each value-pair
212  pBuf[0] = tmp[1];
213  pBuf[1] = tmp[0];
214  pBuf[2] = tmp[3];
215  pBuf[3] = tmp[2];
216  pBuf[4] = tmp[5];
217  pBuf[5] = tmp[4];
218  }
219  sensorState = MAG3110_IDLE;
220 
221  return f;
222 }

225 /**************************************************************************************************
226  * @fn HalMagTest
227  *
228  * @brief Run a sensor self-test
229  *
230  * @return TRUE if passed, FALSE if failed
231  **************************************************************************************************/
232 bool HalMagTest(void)
233 {
234  uint8 val;
235 
236  // Select this sensor on the I2C bus
237  HalMagSelect();
238 
239  // Check who-am-i register
240  ST_ASSERT(HalSensorReadReg(MAG_REG_ADDR_WHO_AM_I, &val, sizeof(val)));
241  ST_ASSERT(val==WHO_AM_I_VALUE);
242 
243  // Check CTRL_REG1 in standby mode
244  ST_ASSERT(HalSensorReadReg(MAG_REG_ADDR_CTRL_1, &val, sizeof(val)));
245  ST_ASSERT(val==MAG_REG_CTRL_OFF);
246 
247  // Check that CTRL_REG1 can be written
248  val = CTRL1_TEST_VALUE;
249  ST_ASSERT(HalSensorWriteReg(MAG_REG_ADDR_CTRL_1, &val, sizeof(val)));
250  ST_ASSERT(HalSensorReadReg(MAG_REG_ADDR_CTRL_1, &val, sizeof(val)));
251  ST_ASSERT(val==CTRL1_TEST_VALUE);
252 
253  // Restore default value
254  val = MAG_REG_CTRL_OFF;
255  ST_ASSERT(HalSensorWriteReg(MAG_REG_ADDR_CTRL_1, &val, sizeof(val)));
256 
257  // Check that the sensor is in standby mode
258  ST_ASSERT(HalSensorReadReg(MAG_REG_ADDR_SYSMOD, &val, sizeof(val)));
259  ST_ASSERT(val==0);
260 
261  return TRUE;
262 }
