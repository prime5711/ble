58 // Minimum supported information payload for the Basic information frame (B-frame)
59 #define L2CAP_MTU_SIZE 23
60 
61 // Minimum supported information payload for the Control frame (C-frame)
62 #define L2CAP_SIG_MTU_SIZE 23
63 
64 // Basic L2CAP header: Length (2 bytes) + Channel ID (2 bytes)
65 #define L2CAP_HDR_SIZE 4
66 
67 // Minimum size of PDU received from lower layer protocol (incoming
68 // packet), or delivered to lower layer protocol (outgoing packet).
69 #define L2CAP_PDU_SIZE ( L2CAP_HDR_SIZE + L2CAP_MTU_SIZE )

71 // L2CAP Channel Identifiers. Identifiers from 0x0001 to 0x003F are
72 // reserved for specific L2CAP functions. Identifiers 0x0001-0x0003
73 // are reserved by BR/EDR.
74 #define L2CAP_CID_NULL 0x0000 // Illegal Identifier
75 #define L2CAP_CID_ATT 0x0004 // Attribute Protocol
76 #define L2CAP_CID_SIG 0x0005 // L2CAP Signaling
77 #define L2CAP_CID_SMP 0x0006 // Security Management Protocol
78 #define L2CAP_CID_GENERIC 0x0007 // Generic Fixed Channel
79 
80 // L2CAP Dynamic Channel Identifiers
81 #define L2CAP_BASE_DYNAMIC_CID 0x0040
82 #define L2CAP_LAST_DYNAMIC_CID ( BASE_DYNAMIC_CID + L2CAP_NUM_CHANNELS - 1 )
83 
84 // Number of Fixed channels: one for each of ATT, Signaling, SMP channels and one Generic Channel
85 #define L2CAP_NUM_FIXED_CHANNELS 4
86 
87 // Number of Protocols supported -- for future use
88 #define L2CAP_NUM_PROTOCOLS 0
89 
90 // Number of Auxiliary channels: one for each of Echo Request, Information
91 // Request and Connection Parameter Update Request
92 #define L2CAP_NUM_AUX_CHANNELS 3
93 
94 // Number of Dynamic channels: one per each protocol supported on each physical connection
95 #define L2CAP_NUM_DYNAMIC_CHANNELS ( L2CAP_NUM_PROTOCOLS * MAX_NUM_LL_CONN )
96 
97 // Total number of L2CAP channels: Dynamic channels plus Auxiliary channels
98 #define L2CAP_NUM_CHANNELS ( L2CAP_NUM_DYNAMIC_CHANNELS + L2CAP_NUM_AUX_CHANNELS )
99 
100 // L2CAP Response Timeout expired (RTX) value for Signaling commands (in seconds).
101 // The RTX timer is used for response timeout or to terminate a dynamic channel
102 // when the remote device is unresponsive to signaling requests. Its value may
103 // range from 1 to 60 seconds.
104 #define L2CAP_RTX_TIMEOUT 30
105 
106 // L2CAP Signaling Codes (type of commands)
107 #define L2CAP_CMD_REJECT 0x01
108 #define L2CAP_ECHO_REQ 0x08 // No longer supported
109 #define L2CAP_ECHO_RSP 0x09 // No longer supported
110 #define L2CAP_INFO_REQ 0x0a // No longer supported
111 #define L2CAP_INFO_RSP 0x0b // No longer supported
112 #define L2CAP_PARAM_UPDATE_REQ 0x12
113 #define L2CAP_PARAM_UPDATE_RSP 0x13
