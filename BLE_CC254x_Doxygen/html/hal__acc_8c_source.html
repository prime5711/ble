173 /**************************************************************************************************
174 * @fn HalAccSetRange
175 *
176 * @brief Set the range of the accelerometer
177 *
178 * @param range: HAL_ACC_RANGE_2G, HAL_ACC_RANGE_4G, HAL_ACC_RANGE_8G
179 *
180 * @return None
181 */
182 void HalAccSetRange(uint8 range)
183 {
184  accRange = range;
185 
186  switch (accRange)
187  {
188  case HAL_ACC_RANGE_2G:
189  accSensorConfig = ACC_REG_CTRL_ON_2G;
190  accSensorOff = ACC_REG_CTRL_OFF_2G;
191  break;
192  case HAL_ACC_RANGE_4G:
193  accSensorConfig = ACC_REG_CTRL_ON_4G;
194  accSensorOff = ACC_REG_CTRL_OFF_4G;
195  break;
196  case HAL_ACC_RANGE_8G:
197  accSensorConfig = ACC_REG_CTRL_ON_8G;
198  accSensorOff = ACC_REG_CTRL_OFF_8G;
199  break;
200  default:
201  // Should not get here
202  break;
203  }
204 }

206 /**************************************************************************************************
207 * @fn HalAccRead
208 *
209 * @brief Read data from the accelerometer - X, Y, Z - 3 bytes
210 *
211 * @return TRUE if valid data, FALSE if not
212 */
213 bool HalAccRead(uint8 *pBuf )
214 {
215  uint8 x;
216  uint8 y;
217  uint8 z;
218  bool success;
219 
220  // Select this sensor
221  HalAccSelect();
222 
223  // Turn on sensor
224  HalSensorWriteReg(ACC_REG_ADDR_CTRL_REG1, &accSensorConfig, sizeof(accSensorConfig));
225 
226  // Wait for measurement ready (appx. 1.45 ms)
227  ST_HAL_DELAY(180);
228 
229  // Read the three registers
230  success = HalSensorReadReg( ACC_REG_ADDR_XOUT_H, &x, sizeof(x));
231  if (success)
232  {
233  success = HalSensorReadReg( ACC_REG_ADDR_YOUT_H, &y, sizeof(y));
234  if (success)
235  {
236  success = HalSensorReadReg( ACC_REG_ADDR_ZOUT_H, &z, sizeof(z));
237  }
238  }
239 
240  if (success)
241  {
242  // Valid data
243  pBuf[0] = x;
244  pBuf[1] = y;
245  pBuf[2] = z;
246  }
247 
248  // Turn off sensor
249  HalSensorWriteReg(ACC_REG_ADDR_CTRL_REG1, &accSensorOff, sizeof(accSensorOff));
250 
251  return success;
252 }

255 /**************************************************************************************************
256  * @fn HalAccTest
257  *
258  * @brief Run a sensor self-test
259  *
260  * @return TRUE if passed, FALSE if failed
261  */
262 bool HalAccTest(void)
263 {
264  uint8 val;
265 
266  // Select this sensor on the I2C bus
267  HalAccSelect();
268 
269  // Check the DCST_RESP (pattern 0x55)
270  ST_ASSERT(HalSensorReadReg(ACC_REG_ADDR_DCST_RESP, &val, 1));
271  ST_ASSERT(val==0x55);
272 
273  // Check the DCST_RESP (pattern 0xAA)
274  val = 0x10; // Sets the DCST bit
275  ST_ASSERT(HalSensorWriteReg(ACC_REG_ADDR_CTRL_REG3, &val, 1));
276  ST_ASSERT(HalSensorReadReg(ACC_REG_ADDR_DCST_RESP, &val, 1));
277  ST_ASSERT(val==0xAA);
278 
279  // Check the WHO AM I register
280  ST_ASSERT(HalSensorReadReg(ACC_REG_ADDR_WHO_AM_I, &val, 1));
281  ST_ASSERT(val==REG_VAL_WHO_AM_I);
282 
283  return TRUE;
284 }
