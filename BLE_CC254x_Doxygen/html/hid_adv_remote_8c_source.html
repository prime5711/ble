430  /*********************************************************************
431  * @fn HidAdvRemote_Init
432  *
433  * @brief Initialization function for the Advance Remote App Task.
434  * This is called during initialization and should contain
435  * any application specific initialization (ie. hardware
436  * initialization/setup, table initialization, power up
437  * notificaiton ... ).
438  *
439  * @param task_id - the ID assigned by OSAL. This ID should be
440  * used to send messages and set timers.
441  *
442  * @return none
443  */
444 void HidAdvRemote_Init( uint8 task_id )
445 {
446  hidAdvRemote_TaskId = task_id;
447 
448  // Setup the GAP
449  VOID GAP_SetParamValue( TGAP_CONN_PAUSE_PERIPHERAL, DEFAULT_CONN_PAUSE_PERIPHERAL );
450 
451  // Setup the GAP Peripheral Role Profile
452  {
453  uint8 initial_advertising_enable = FALSE;
454 
455  // By setting this to zero, the device will go into the waiting state after
456  // being discoverable for 30.72 second, and will not being advertising again
457  // until the enabler is set back to TRUE
458  uint16 gapRole_AdvertOffTime = 0;
459 
460  uint8 enable_update_request = DEFAULT_ENABLE_UPDATE_REQUEST;
461  uint16 desired_min_interval = DEFAULT_DESIRED_MIN_CONN_INTERVAL;
462  uint16 desired_max_interval = DEFAULT_DESIRED_MAX_CONN_INTERVAL;
463  uint16 desired_slave_latency = DEFAULT_DESIRED_SLAVE_LATENCY;
464  uint16 desired_conn_timeout = DEFAULT_DESIRED_CONN_TIMEOUT;
465 
466  // Set the GAP Role Parameters
467  GAPRole_SetParameter( GAPROLE_ADVERT_ENABLED, sizeof( uint8 ), &initial_advertising_enable );
468  GAPRole_SetParameter( GAPROLE_ADVERT_OFF_TIME, sizeof( uint16 ), &gapRole_AdvertOffTime );
469 
470  GAPRole_SetParameter( GAPROLE_SCAN_RSP_DATA, sizeof ( scanRspData ), scanRspData );
471  GAPRole_SetParameter( GAPROLE_ADVERT_DATA, sizeof( advertData ), advertData );
472 
473  GAPRole_SetParameter( GAPROLE_PARAM_UPDATE_ENABLE, sizeof( uint8 ), &enable_update_request );
474  GAPRole_SetParameter( GAPROLE_MIN_CONN_INTERVAL, sizeof( uint16 ), &desired_min_interval );
475  GAPRole_SetParameter( GAPROLE_MAX_CONN_INTERVAL, sizeof( uint16 ), &desired_max_interval );
476  GAPRole_SetParameter( GAPROLE_SLAVE_LATENCY, sizeof( uint16 ), &desired_slave_latency );
477  GAPRole_SetParameter( GAPROLE_TIMEOUT_MULTIPLIER, sizeof( uint16 ), &desired_conn_timeout );
478  }
479 
480  // Set the GAP Characteristics
481  {
482  gapPeriConnectParams_t connParams =
483  {
484  DEFAULT_DESIRED_MIN_CONN_INTERVAL,
485  DEFAULT_DESIRED_MAX_CONN_INTERVAL,
486  DEFAULT_DESIRED_SLAVE_LATENCY,
487  DEFAULT_DESIRED_CONN_TIMEOUT
488  };
489 
490  GGS_SetParameter( GGS_PERI_CONN_PARAM_ATT, sizeof( gapPeriConnectParams_t ), (void *)&connParams );
491  GGS_SetParameter( GGS_DEVICE_NAME_ATT, sizeof ( attDeviceName ) , attDeviceName );
492  GGS_SetParameter( GGS_APPEARANCE_ATT, sizeof( uint16 ), (void *)&appearance );
493  }
494 
495  // Setup the GAP Bond Manager
496  {
497  uint32 passkey = DEFAULT_PASSCODE;
498  uint8 pairMode = DEFAULT_PAIRING_MODE;
499  uint8 mitm = DEFAULT_MITM_MODE;
500  uint8 ioCap = DEFAULT_IO_CAPABILITIES;
501  uint8 bonding = DEFAULT_BONDING_MODE;
502 
503  GAPBondMgr_SetParameter( GAPBOND_DEFAULT_PASSCODE, sizeof( uint32 ), &passkey );
504  GAPBondMgr_SetParameter( GAPBOND_PAIRING_MODE, sizeof( uint8 ), &pairMode );
505  GAPBondMgr_SetParameter( GAPBOND_MITM_PROTECTION, sizeof( uint8 ), &mitm );
506  GAPBondMgr_SetParameter( GAPBOND_IO_CAPABILITIES, sizeof( uint8 ), &ioCap );
507  GAPBondMgr_SetParameter( GAPBOND_BONDING_ENABLED, sizeof( uint8 ), &bonding );
508  }
509 
510  // Setup Battery Characteristic Values
511  {
512  uint8 critical = DEFAULT_BATT_CRITICAL_LEVEL;
513 
514  Batt_SetParameter( BATT_PARAM_CRITICAL_LEVEL, sizeof (uint8), &critical );
515 
516  Batt_Setup( HAL_ADC_CHN_AIN6, BATT_ADC_LEVEL_2_0V, BATT_ADC_LEVEL_4_5V,
517  hidAdvRemoteBattMeasSetupCB, hidAdvRemoteBattMeasTeardownCB, NULL );
518 
519  /* Control of transistor in the battery measurement voltage divider */
520  P1DIR |= (1<<5);
521  P1SEL &= ~(1<<5);
522  P1_5 = 0; // Stop feeding current through voltage divider
523 
524  /* Analog measuring pin setup. This will be retained during ADC read */
525  P0DIR |= (1<<6);
526  P0SEL &= ~(1<<6);
527  P0_6 = 0; // LOW output to stop leakage current due to default pull-up
528  }
529

544 }

630 /*********************************************************************
631  *
632  * @fn hidAdvRemoteKeyCback
633  *
634  * @brief Callback service for keys
635  *
636  * @param keys - key that was pressed (i.e. the scanned row/col index)
637  * state - shifted
638  *
639  * @return void
640  */
641 static void hidAdvRemoteKeyCback(uint8 keys, uint8 state)
642 {
643  static uint8 lastKey = HAL_KEY_CODE_NOKEY;
644  static uint8 lastCmd = HID_KEYBOARD_RESERVED;
645  bool sendMouseReport = FALSE;
646 
647  // Get the key code based on the row/col index
648  uint8 cmd = keycodeMap[keys].keycode;
649 
650  (void) state; // unused argument
651 
652  if ( keys == HAL_KEY_CODE_NOKEY )
653  {
654  // Overwrite the invalid cmd lookup on key release.
655  cmd = HID_KEYBOARD_RESERVED;
656  keyRepeated = 0;
657 
658  if ( mouseButtonStates != 0 )
659  {
660  /* While mouse buttons are sensed as buttons, they are reported in
661  * HID Mouse Reports, which are sent from the HAR_MotionSensorCback
662  * function. The global "mouseButtonStates" keeps track of the state
663  * of the mouse buttons, and this callback function simply sets/clears
664  * the state of these buttons.
665  */
666  lastKey = HAL_KEY_CODE_NOKEY;
667  mouseButtonStates = 0;
668  sendMouseReport = TRUE;
669  }
670  else if ( lastCmd == ACT_AIR_MOUSE_CURSOR_CTL )
671  {
672  hidAdvRemoteAirMouseCursorCtlKeyRelease();
673  lastCmd = HID_KEYBOARD_RESERVED;
674  }
675  }
676  else if ( lastKey == keys )
677  {
678  // same key is still pressed
679  if ( keyRepeated < 255 )
680  {
681  keyRepeated++;
682  }
683  }
684  else
685  {
686  // remember last row/col index
687  lastKey = keys;
688  lastCmd = cmd;
689  keyRepeated = 0;
690  }
691 
692  /* Mouse button clicks are simply recorded in the global "mouseButtonStates"
693  * and reported by the motion sensor callback function.
694  */
695  if ( cmd == HID_MOUSE_BUTTON_LEFT )
696  {
697  mouseButtonStates |= 0x01;
698  sendMouseReport = TRUE;
699  }
700  else if ( cmd == HID_MOUSE_BUTTON_RIGHT )
701  {
702  mouseButtonStates |= 0x02;
703  sendMouseReport = TRUE;
704  }
705  else if ( cmd >= ACT_START )
706  {
707  // action table command
708  harSpecialKeyActions[cmd - ACT_START]();
709  }
710  else if ( sendMouseReport == FALSE ) // we have a valid key command
711  {
712  uint8 key;
713  uint8 usagePage;
714 
715  if ( keys == HAL_KEY_CODE_NOKEY )
716  {
717  // indicate no key pressed anymore
718  key = 0;
719  keyPressed = FALSE;
720 
721  // Get the last key code based on the row/col index.
722  cmd = keycodeMap[lastKey].keycode;
723  usagePage = keycodeMap[lastKey].usagePage;
724  lastKey = HAL_KEY_CODE_NOKEY;
725  }
726  else // key pressed
727  {
728  // insert key that was pressed
729  key = cmd;
730  keyPressed = TRUE;
731 
732  usagePage = keycodeMap[keys].usagePage;
733  }
734 
735  // Only send out functional keys
736  if ( usagePage == USAGE_PAGE_GD )
737  {
738  // Keyboard key
739  hidAdvRemoteHandleKeys( key, keyPressed, keyRepeated );
740  }
741  else if ( usagePage == USAGE_PAGE_CC )
742  {
743  // Consumer control key
744  hidCCSendReport( cmd, keyPressed, keyRepeated );
745  }
746  else
747  {
748  // Control button
749  }
750  }
751 
752  if ( ( sendMouseReport == TRUE ) &&
753  ( arcMotionDetectorState == MOTION_DETECTOR_DISABLED ) &&
754  ( mouseButtonStates != savedMouseButtonStates ) )
755  {
756  hidMouseSendReport( mouseButtonStates, 0, 0 );
757 
758  savedMouseButtonStates = mouseButtonStates;
759  }
760 }
