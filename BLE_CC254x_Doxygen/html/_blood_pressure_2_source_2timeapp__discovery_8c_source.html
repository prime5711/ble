110 uint8 timeAppDiscStart( void )
111 {
112  // Clear handle cache
113  osal_memset( timeAppHdlCache, 0, sizeof(timeAppHdlCache) );
114 
115  // Start discovery with first service
116  return timeAppDiscGattMsg( DISC_CURR_TIME_START, NULL );

119 /*********************************************************************
120  * @fn timeAppDiscGattMsg()
121  *
122  * @brief Handle GATT messages for characteristic discovery.
123  *
124  * @param state - Discovery state.
125  * @param pMsg - GATT message.
126  *
127  * @return New discovery state.
128  */
129 uint8 timeAppDiscGattMsg( uint8 state, gattMsgEvent_t *pMsg )
130 {
131  // Execute discovery function for service
132  do
133  {
134  switch ( state & 0xF0 )
135  {
136  // Current time service
137  case DISC_CURR_TIME_START:
138  state = TimeAppDiscCurrTime( state, pMsg );
139  if ( state == DISC_FAILED )
140  {
141  state = DISC_FAILED;
142  }
143  else if ( state == DISC_IDLE )
144  {
145  state = DISC_IDLE;
146  }
147  break;
148 
149  default:
150  break;
151  }
152 
153  } while ( (state != 0) && ((state & 0x0F) == 0) );
154 
155  return state;
156 }

158 /*********************************************************************
159  * @fn TimeAppDiscCurrTime()
160  *
161  * @brief Current time service and characteristic discovery.
162  *
163  * @param state - Discovery state.
164  * @param pMsg - GATT message.
165  *
166  * @return New discovery state.
167  */
168 static uint8 TimeAppDiscCurrTime( uint8 state, gattMsgEvent_t *pMsg )
169 {
170  uint8 newState = state;
171 
172  switch ( state )
173  {
174  case DISC_CURR_TIME_START:
175  {
176  uint8 uuid[ATT_BT_UUID_SIZE] = { LO_UINT16(CURRENT_TIME_SERV_UUID),
177  HI_UINT16(CURRENT_TIME_SERV_UUID) };
178 
179  // Initialize service discovery variables
180  timeAppSvcStartHdl = timeAppSvcEndHdl = 0;
181  timeAppEndHdlIdx = 0;
182 
183  // Discover service by UUID
184  GATT_DiscPrimaryServiceByUUID( gapConnHandle, uuid,
185  ATT_BT_UUID_SIZE, bloodPressureTaskId );
186 
187  newState = DISC_CURR_TIME_SVC;
188  }
189  break;
190 
191  case DISC_CURR_TIME_SVC:
192  // Service found, store handles
193  if ( pMsg->method == ATT_FIND_BY_TYPE_VALUE_RSP &&
194  pMsg->msg.findByTypeValueRsp.numInfo > 0 )
195  {
196  timeAppSvcStartHdl = pMsg->msg.findByTypeValueRsp.handlesInfo[0].handle;
197  timeAppSvcEndHdl = pMsg->msg.findByTypeValueRsp.handlesInfo[0].grpEndHandle;
198  }
199 
200  // If procedure complete
201  if ( ( pMsg->method == ATT_FIND_BY_TYPE_VALUE_RSP &&
202  pMsg->hdr.status == bleProcedureComplete ) ||
203  ( pMsg->method == ATT_ERROR_RSP ) )
204  {
205  // If service found
206  if ( timeAppSvcStartHdl != 0 )
207  {
208  // Discover all characteristics
209  GATT_DiscAllChars( gapConnHandle, timeAppSvcStartHdl,
210  timeAppSvcEndHdl, bloodPressureTaskId );
211 
212  newState = DISC_CURR_TIME_CHAR;
213  }
214  else
215  {
216  // Service not found
217  newState = DISC_FAILED;
218  }
219  }
220  break;
221 
222  case DISC_CURR_TIME_CHAR:
223  {
224  // Characteristics found
225  if ( pMsg->method == ATT_READ_BY_TYPE_RSP &&
226  pMsg->msg.readByTypeRsp.numPairs > 0 &&
227  pMsg->msg.readByTypeRsp.len == CHAR_DESC_HDL_UUID16_LEN)
228  {
229  uint8 i;
230  uint8 *p;
231  uint16 handle;
232  uint16 uuid;
233  // For each characteristic declaration
234  p = pMsg->msg.readByTypeRsp.dataList;
235  for ( i = pMsg->msg.readByTypeRsp.numPairs; i > 0; i-- )
236  {
237  // Parse characteristic declaration
238  handle = BUILD_UINT16(p[3], p[4]);
239  uuid = BUILD_UINT16(p[5], p[6]);
240 
241  // If looking for end handle
242  if ( timeAppEndHdlIdx != 0 )
243  {
244  // End handle is one less than handle of characteristic declaration
245  timeAppHdlCache[timeAppEndHdlIdx] = BUILD_UINT16(p[0], p[1]) - 1;
246 
247  timeAppEndHdlIdx = 0;
248  }
249 
250  // If UUID is of interest, store handle
251  switch ( uuid )
252  {
253  case CT_TIME_UUID:
254  timeAppHdlCache[HDL_CURR_TIME_CT_TIME_START] = handle;
255  timeAppEndHdlIdx = HDL_CURR_TIME_CT_TIME_END;
256  break;
257 
258  default:
259  break;
260  }
261 
262  p += CHAR_DESC_HDL_UUID16_LEN;
263  }
264 
265  }
266 
267  // If procedure complete
268  if ( ( pMsg->method == ATT_READ_BY_TYPE_RSP &&
269  pMsg->hdr.status == bleProcedureComplete ) ||
270  ( pMsg->method == ATT_ERROR_RSP ) )
271  {
272  // Special case of end handle at end of service
273  if ( timeAppEndHdlIdx != 0 )
274  {
275  timeAppHdlCache[timeAppEndHdlIdx] = timeAppSvcEndHdl;
276  timeAppEndHdlIdx = 0;
277  }
278 
279  // If didn't find time characteristic
280  if ( timeAppHdlCache[HDL_CURR_TIME_CT_TIME_START] == 0 )
281  {
282  newState = DISC_FAILED;
283  }
284 
285  else if ( timeAppHdlCache[HDL_CURR_TIME_CT_TIME_START] <timeAppHdlCache[HDL_CURR_TIME_CT_TIME_END] )
286  {
287  // Discover characteristic descriptors
288  GATT_DiscAllCharDescs( gapConnHandle,
289  timeAppHdlCache[HDL_CURR_TIME_CT_TIME_START] + 1,
290  timeAppHdlCache[HDL_CURR_TIME_CT_TIME_END],
291  bloodPressureTaskId );
292 
293  newState = DISC_CURR_TIME_CT_TIME_CCCD;
294  }
295  else
296  {
297  newState = DISC_IDLE;
298  }
299  }
300  }
301  break;
302 
303  case DISC_CURR_TIME_CT_TIME_CCCD:
304  {
305  // Characteristic descriptors found
306  if ( pMsg->method == ATT_FIND_INFO_RSP &&
307  pMsg->msg.findInfoRsp.numInfo > 0 &&
308  pMsg->msg.findInfoRsp.format == ATT_HANDLE_BT_UUID_TYPE )
309  {
310  uint8 i;
311  // For each handle/uuid pair
312  for ( i = 0; i < pMsg->msg.findInfoRsp.numInfo; i++ )
313  {
314  // Look for CCCD
315  if ( (pMsg->msg.findInfoRsp.info.btPair[i].uuid[0] ==
316  LO_UINT16(GATT_CLIENT_CHAR_CFG_UUID)) &&
317  (pMsg->msg.findInfoRsp.info.btPair[i].uuid[1] ==
318  HI_UINT16(GATT_CLIENT_CHAR_CFG_UUID)) )
319  {
320  // CCCD found
321  timeAppHdlCache[HDL_CURR_TIME_CT_TIME_CCCD] =
322  pMsg->msg.findInfoRsp.info.btPair[i].handle;
323 
324  break;
325  }
326  }
327  }
328 
329  // If procedure complete
330  if ( ( pMsg->method == ATT_FIND_INFO_RSP &&
331  pMsg->hdr.status == bleProcedureComplete ) ||
332  ( pMsg->method == ATT_ERROR_RSP ) )
333  {
334  newState = DISC_IDLE;
335  }
336  }
337  break;
338 
339  default:
340  break;
341  }
342 
343  return newState;
344 }
