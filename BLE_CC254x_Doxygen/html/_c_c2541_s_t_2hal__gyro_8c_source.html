181 /**************************************************************************************************
182  * @fn HalGyroTurnOn
183  *
184  * @brief Turn the sensor on
185  *
186  * @return none
187  **************************************************************************************************/
188 void HalGyroTurnOn(void)
189 {
190  bool success;
191  uint8 clk_select;
192 
193  HalDcDcControl(ST_GYRO,true);
194 
195  HalGyroSelect();
196 
197  // Default to internal oscillator if no axis is enabled (should not happen)
198  clk_select = HAL_GYRO_PWR_MGM_CLOCK_INT_OSC;
199  if(!(mDisabledAxes & HAL_GYRO_PWR_MGM_STBY_XG))
200  {
201  clk_select = HAL_GYRO_PWR_MGM_CLOCK_PLL_X;
202  }
203  else if(!(mDisabledAxes & HAL_GYRO_PWR_MGM_STBY_YG))
204  {
205  clk_select = HAL_GYRO_PWR_MGM_CLOCK_PLL_Y;
206  }
207  else if(!(mDisabledAxes & HAL_GYRO_PWR_MGM_STBY_ZG))
208  {
209  clk_select = HAL_GYRO_PWR_MGM_CLOCK_PLL_Z;
210  }
211 
212  // Wake up from sleep, disable standby for all gyros, select reference PLL according to selected axes
213  cfgOn = mDisabledAxes & (~HAL_GYRO_PWR_MGM_SLEEP | clk_select);
214  success = HalSensorWriteReg(HAL_GYRO_REG_PWR_MGM,&cfgOn,1);
215  if(success)
216  {
217  mStatus = HAL_GYRO_DATA_READY;
218  }
219  else
220  {
221  mStatus = HAL_GYRO_STATE_ERROR;
222  }
223 }

225 /**************************************************************************************************
226  * @fn HalGyroTurnOff
227  *
228  * @brief Turn the sensor off
229  *
230  * @return none
231  **************************************************************************************************/
232 void HalGyroTurnOff(void)
233 {
234  bool success;
235  uint8 val;
236 
237  HalGyroSelect();
238 
239  // Standby, 5 uA current consumption
240  val = HAL_GYRO_PWR_MGM_SLEEP;
241  success = HalSensorWriteReg(HAL_GYRO_REG_PWR_MGM,&val,1);
242 
243  if(success)
244  {
245  uint8 rv;
246 
247  success = HalSensorReadReg(HAL_GYRO_REG_PWR_MGM,&rv,1);
248  if (success && rv == val)
249  {
250  mStatus = HAL_GYRO_STOPPED;
251  mDisabledAxes = HAL_GYRO_PWR_MGM_STBY_ALL;
252  }
253  }
254 
255  if (!success)
256  {
257  mStatus = HAL_GYRO_STATE_ERROR;
258  }
259 
260  HalDcDcControl(ST_GYRO,false);
261 }

264 /**************************************************************************************************
265  * @fn HalGyroRead
266  *
267  * @brief Read gyro data
268  *
269  * @param Voltage and temperature in raw format [X_LOW, X_HIGH, Y_LOW, Y_HIGH, Z_LOW, Z_HIGH]
270  *
271  * @return TRUE if valid data
272  **************************************************************************************************/
273 bool HalGyroRead(uint8 *pBuf)
274 {
275  bool success;
276  uint8 buf[HAL_GYRO_DATA_SIZE];
277  uint8 val;
278 
279  HalGyroSelect();
280 
281  // Read sensor
282  success = HalSensorReadReg(HAL_GYRO_REG_GYRO_XOUT_H,buf,HAL_GYRO_DATA_SIZE);
283 
284  if (success)
285  {
286  // Result in LE
287  pBuf[0] = buf[1];
288  pBuf[1] = buf[0];
289  pBuf[2] = buf[3];
290  pBuf[3] = buf[2];
291  pBuf[4] = buf[5];
292  pBuf[5] = buf[4];
293  }
294 
295  // Put gyro back to sleep
296  val = HAL_GYRO_PWR_MGM_SLEEP;
297  success = HalSensorWriteReg(HAL_GYRO_REG_PWR_MGM,&val,1);
298  mStatus = HAL_GYRO_SLEEP;
299 
300  return success;
301 }

334 /**************************************************************************************************
335  * @fn HalGyroSelectAxes
336  *
337  * @brief Select axes for gyroscope
338  *
339  * @param Bitmask representing enabled axes (1 enable, 0 disable)
340  * Bit 0: X axis
341  * Bit 1: Y axis
342  * Bit 2: Z axis
343  * Bit 3-7: reserved
344  * @return none
345  **************************************************************************************************/
346 void HalGyroSelectAxes(uint8 axes)
347 {
348  mDisabledAxes = HAL_GYRO_PWR_MGM_STBY_ALL;
349 
350  if(axes & 0x01)
351  {
352  mDisabledAxes ^= HAL_GYRO_PWR_MGM_STBY_XG;
353  }
354 
355  if(axes & 0x02)
356  {
357  mDisabledAxes ^= HAL_GYRO_PWR_MGM_STBY_YG;
358  }
359 
360  if(axes & 0x04)
361  {
362  mDisabledAxes ^= HAL_GYRO_PWR_MGM_STBY_ZG;
363  }
364 
365  if(HalGyroStatus() != HAL_GYRO_STOPPED)
366  {
367  HalGyroTurnOn();
368  }
369 }
