150 /**************************************************************************************************
151  * @fn HalIRTempRead
152  *
153  * @brief Read the sensor voltage and sensor temperature registers
154  *
155  * @param Voltage and temperature in raw format (2 + 2 bytes)
156  *
157  * @return TRUE if valid data
158  **************************************************************************************************/
159 bool HalIRTempRead(uint8 *pBuf)
160 {
161  uint16 v;
162  uint16 t;
163  bool success;
164 
165  if (irtSensorState != TMP006_DATA_READY)
166  {
167  return FALSE;
168  }
169 
170  HalIRTempSelect();
171 
172  // Read the sensor registers
173  success = HalSensorReadReg(TMP006_REG_ADDR_VOLTAGE, (uint8 *)&v,IRTEMP_REG_LEN );
174  if (success)
175  {
176  success = HalSensorReadReg(TMP006_REG_ADDR_TEMPERATURE, (uint8 *)&t,IRTEMP_REG_LEN );
177  }
178 
179  if (success)
180  {
181  // Store values
182  pBuf[0] = HI_UINT16( v );
183  pBuf[1] = LO_UINT16( v );
184  pBuf[2] = HI_UINT16( t );
185  pBuf[3] = LO_UINT16( t );
186  }
187 
188  // Turn off sensor
189  if (HalSensorWriteReg(TMP006_REG_ADDR_CONFIG, configSensorOff, IRTEMP_REG_LEN))
190  {
191  irtSensorState = TMP006_OFF;
192  }
193  HalDcDcControl(ST_IRTEMP,false);
194 
195  return success;
196 }

199 /**************************************************************************************************
200  * @fn HalIRTempStatus
201  *
202  * @brief Read the state of the sensor
203  *
204  * @return none
205  **************************************************************************************************/
206 IRTemperature_States_t HalIRTempStatus(void)
207 {
208  if (irtSensorState != TMP006_OFF)
209  {
210  bool success;
211  uint16 v;
212 
213  // Select this sensor on the I2C bus
214  HalIRTempSelect();
215 
216  // Read the data ready bit
217  success = HalSensorReadReg(TMP006_REG_ADDR_CONFIG, (uint8 *)&v,IRTEMP_REG_LEN );
218  if ((v & DATA_RDY_BIT) && success)
219  {
220  irtSensorState = TMP006_DATA_READY;
221  }
222  }
223 
224  return irtSensorState;
225 }

228 /**************************************************************************************************
229  * @fn HalIRTempTest
230  *
231  * @brief Run a sensor self-test
232  *
233  * @return TRUE if passed, FALSE if failed
234  **************************************************************************************************/
235 bool HalIRTempTest(void)
236 {
237  uint16 val;
238 
239  // Select this sensor on the I2C bus
240  HalIRTempSelect();
241 
242  // Check manufacturer ID
243  ST_ASSERT(HalSensorReadReg(TMP006_REG_MANF_ID, (uint8 *)&val, IRTEMP_REG_LEN));
244  val = (LO_UINT16(val) << 8) | HI_UINT16(val);
245  ST_ASSERT(val == TMP006_VAL_MANF_ID);
246 
247  // Reset sensor
248  ST_ASSERT(HalSensorWriteReg(TMP006_REG_ADDR_CONFIG, configSensorReset, IRTEMP_REG_LEN));
249 
250  // Check config register (reset)
251  ST_ASSERT(HalSensorReadReg(TMP006_REG_ADDR_CONFIG, (uint8 *)&val, IRTEMP_REG_LEN));
252  val = ((LO_UINT16(val) << 8) | HI_UINT16(val));
253  ST_ASSERT(val == TMP006_VAL_CONFIG_RESET);
254 
255  // Turn sensor off
256  ST_ASSERT(HalSensorWriteReg(TMP006_REG_ADDR_CONFIG, configSensorOff,IRTEMP_REG_LEN));
257 
258  // Check config register (off)
259  ST_ASSERT(HalSensorReadReg(TMP006_REG_ADDR_CONFIG, (uint8 *)&val, IRTEMP_REG_LEN));
260  val = ((LO_UINT16(val) << 8) | HI_UINT16(val));
261  ST_ASSERT(val == TMP006_VAL_CONFIG_OFF);
262 
263  // Turn sensor on
264  ST_ASSERT(HalSensorWriteReg(TMP006_REG_ADDR_CONFIG, configSensorOn, IRTEMP_REG_LEN));
265 
266  // Check config register (on)
267  ST_ASSERT(HalSensorReadReg(TMP006_REG_ADDR_CONFIG, (uint8 *)&val, IRTEMP_REG_LEN));
268  val = ((LO_UINT16(val) << 8) | HI_UINT16(val));
269  ST_ASSERT(val == TMP006_VAL_CONFIG_ON);
270 
271  // Turn sensor off
272  ST_ASSERT(HalSensorWriteReg(TMP006_REG_ADDR_CONFIG, configSensorOff, IRTEMP_REG_LEN));
273 
274  return TRUE;
275 }
