156 /**************************************************************************************************
157  * @fn HalGyroInit
158  *
159  * @brief Initilize Gyro hardware
160  *
161  * @param none
162  *
163  * @return None
164  **************************************************************************************************/
165 void HalGyroInit( void )
166 {
167  /* Initialize outputs */
168  HAL_GYRO_POWER_PIN = 1;
169  HAL_GYRO_CLKIN_PIN = 0;
170  HAL_GYRO_INT_PIN = 0;
171 
172  /* Configure pin function as GPIO for pins related to gyro */
173  P0SEL &= (uint8) ~HAL_GYRO_P0_GPIO_PINS;
174  P1SEL &= (uint8) ~HAL_GYRO_P1_GPIO_PINS;
175 
176  /* Configure direction of pins related to gyro */
177  P0DIR |= (uint8) HAL_GYRO_P0_OUTPUT_PINS;
178  P1DIR |= (uint8) HAL_GYRO_P1_OUTPUT_PINS;
179 
180  /* Indicate no callback */
181  pHalGyroEnableProcessFunction = NULL;
182 
183  /* Start timer to wait for outputs to become stable */
184  osal_start_timerEx( Hal_TaskID,
185  HAL_GYRO_REGISTER_ACCESS_READY_EVENT,
186  100 );
187 }

209  100 );

212 /**************************************************************************************************
213  * @fn HalGyroHandleGyroRegisterAccessReadyEvent
214  *
215  * @brief Callback to handle expiry of gyro powerup wait timer.
216  *
217  * @param None.
218  *
219  * @return None.
220  **************************************************************************************************/
221 void HalGyroHandleGyroRegisterAccessReadyEvent( void )
222 {
223  uint8 i;
224 
225  if (pHalGyroEnableProcessFunction != NULL)
226  {
227  /* Gyro is ready. Configure it. */
228  for (i = 0; i < HAL_GYRO_CONFIG_TABLE_SIZE; i++)
229  {
230  HalMotionI2cWrite( HAL_MOTION_DEVICE_GYRO,
231  HalGyroConfigTable[i].addr,
232  &HalGyroConfigTable[i].data,
233  1 );
234  }
235 
236  /* config is complete, so notify caller */
237  pHalGyroEnableProcessFunction();
238  }
239  else
240  {
241  /* No callback, so just put gyro in sleep mode */
242  gyroSleep();
243  }
244 }

261 /**************************************************************************************************
262  * @fn HalGyroRead
263  *
264  * @brief Reads channels connected to X and Z output on Gyroscope.
265  *
266  * @param None
267  *
268  * @return X and Z channel readings
269  **************************************************************************************************/
270 void HalGyroRead( int16 *x, int16 *y, int16 *z )
271 {
272  uint8 temp[6];
273 
274  /* Get all 3 axes of gyro samples as the gyro seems to hang the I2C bus if we only read 2 of them. */
275  HalMotionI2cRead( HAL_MOTION_DEVICE_GYRO,
276  IMU3000_REG_ADDR_GYRO_XOUT_H,
277  6,
278  (uint8 *)temp );
279 
280  /* Extract X and Z axis info, accounting for endian difference */
281  *x = (temp[2] << 8) | temp[3];
282  *y = (temp[0] << 8) | temp[1];
283  *z = (temp[4] << 8) | temp[5];
284 }

286 /**************************************************************************************************
287  * @fn HalGyroReadAccelData
288  *
289  * @brief Reads X, Y and Z data out registers from Accelerometer.
290  *
291  * input parameters
292  *
293  * None.
294  *
295  * output parameters
296  *
297  * @param *x - X register value.
298  * @param *y - Y register value.
299  * @param *z - Z register value.
300  *
301  * @return None.
302  */
303 void HalGyroReadAccelData( int8 *x, int8 *y, int8 *z )
304 {
305  uint8 temp[6];
306 
307  /* Get all 3 axes of gyro samples as the gyro seems to hang the I2C bus if we only read 2 of them. */
308  HalMotionI2cRead( HAL_MOTION_DEVICE_GYRO,
309  IMU3000_REG_ADDR_AUX_XOUT_H,
310  6,
311  (uint8 *)temp );
312 
313  /* Extract Y and Z axis info. Data is laid out as follows:
314  *
315  * ----------
316  * | XOUT_L | B3 B2 B1 B0 x x x x
317  * |--------|
318  * | XOUT_H | B11 B10 B9 B8 B7 B6 B5 B4
319  * |--------|
320  * | YOUT_L |
321  * |--------|
322  * | YOUT_H |
323  * |--------|
324  * | ZOUT_L |
325  * |--------|
326  * | ZOUT_H |
327  * ----------
328  *
329  * Since the data we're interested in lies in the 8 most significant bits, all
330  * we need to do is read the XOUT_H, YOUT_H, and ZOUT_H registers.
331  */
332  *x = temp[1];
333  *y = temp[3];
334  *z = temp[5];
335 }

398 /**************************************************************************************************
399  * @fn gyroSleep
400  *
401  * @brief This function puts the gyro in sleep mode.
402  *
403  * @param None.
404  *
405  * @return None.
406  **************************************************************************************************/
407 static void gyroSleep( void )
408 {
409  /* All we need to do is put gyro to sleep */
410  uint8 reg = PWR_MGM_SLEEP | PWR_MGM_STBY_XG | PWR_MGM_STBY_YG | PWR_MGM_STBY_ZG;
411  HalMotionI2cWrite( HAL_MOTION_DEVICE_GYRO,
412  IMU3000_REG_ADDR_PWR_MGM,
413  &reg,
414  1 );
415 }

433  &reg,
434  1 );
