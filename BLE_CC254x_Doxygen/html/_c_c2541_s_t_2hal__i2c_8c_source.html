102 // Must clear SI before setting STA and then STA must be manually cleared.
103 #define I2C_STRT() st ( \
104  I2CCFG &= ~I2C_SI; \
105  I2CCFG |= I2C_STA; \
106  while ((I2CCFG & I2C_SI) == 0); \
107  I2CCFG &= ~I2C_STA; \
108 )
109 
110 // Must set STO before clearing SI.
111 #define I2C_STOP() st ( \
112  I2CCFG |= I2C_STO; \
113  I2CCFG &= ~I2C_SI; \
114  while ((I2CCFG & I2C_STO) != 0); \
115 )
116 
117 // Stop clock-stretching and then read when it arrives.
118 #define I2C_READ(_X_) st ( \
119  I2CCFG &= ~I2C_SI; \
120  while ((I2CCFG & I2C_SI) == 0); \
121  (_X_) = I2CDATA; \
122 )
123 
124 // First write new data and then stop clock-stretching.
125 #define I2C_WRITE(_X_) st ( \
126  I2CDATA = (_X_); \
127  I2CCFG &= ~I2C_SI; \

138 /**************************************************************************************************
139  * @fn i2cMstStrt
140  *
141  * @brief Attempt to send an I2C bus START and Slave Address as an I2C bus Master.
142  *
143  * input parameters
144  *
145  * @param RD_WRn - The LSB of the Slave Address as Read/~Write.
146  *
147  * @return The I2C status of the START request or of the Slave Address Ack.
148  */
149 static uint8 i2cMstStrt(uint8 RD_WRn)
150 {
151  I2C_STRT();
152 
153  if (I2CSTAT == mstStarted) /* A start condition has been transmitted */
154  {
155  I2C_WRITE(i2cAddr | RD_WRn);
156  }
157 
158  return I2CSTAT;
159 }

177 void HalI2CInit(uint8 address, i2cClock_t clockRate)

187 /**************************************************************************************************
188  * @fn HalI2CRead
189  *
190  * @brief Read from the I2C bus as a Master.
191  *
192  * input parameters
193  *
194  * @param len - Number of bytes to read.
195  * @param pBuf - Pointer to the data buffer to put read bytes.
196  *
197  * output parameters
198  *
199  * None.
200  *
201  * @return The number of bytes successfully read.
202  */
203 uint8 HalI2CRead(uint8 len, uint8 *pBuf)
204 {
205  uint8 cnt = 0;
206 
207  if (i2cMstStrt(I2C_MST_RD_BIT) != mstAddrAckR)
208  {
209  len = 0;
210  }
211 
212  // All bytes are ACK'd except for the last one which is NACK'd. If only
213  // 1 byte is being read, a single NACK will be sent. Thus, we only want
214  // to enable ACK if more than 1 byte is going to be read.
215  if (len > 1)
216  {
217  I2C_SET_ACK();
218  }
219 
220  while (len > 0)
221  {
222  // slave devices require NACK to be sent after reading last byte
223  if (len == 1)
224  {
225  I2C_SET_NACK();
226  }
227 
228  // read a byte from the I2C interface
229  I2C_READ(*pBuf++);
230  cnt++;
231  len--;
232 
233  if (I2CSTAT != mstDataAckR)
234  {
235  if (I2CSTAT != mstDataNackR)
236  {
237  // something went wrong, so don't count last byte
238  cnt--;
239  }
240  break;
241  }
242  }
243  I2C_STOP();
244 
245  return cnt;
246 }

248 /**************************************************************************************************
249  * @fn HalI2CWrite
250  *
251  * @brief Write to the I2C bus as a Master.
252  *
253  * input parameters
254  *
255  * @param len - Number of bytes to write.
256  * @param pBuf - Pointer to the data buffer to write.
257  *
258  * output parameters
259  *
260  * None.
261  *
262  * @return The number of bytes successfully written.
263  */
264 uint8 HalI2CWrite(uint8 len, uint8 *pBuf)
265 {
266  if (i2cMstStrt(0) != mstAddrAckW)
267  {
268  len = 0;
269  }
270 
271  for (uint8 cnt = 0; cnt < len; cnt++)
272  {
273  I2C_WRITE(*pBuf++);
274 
275  if (I2CSTAT != mstDataAckW)
276  {
277  if (I2CSTAT == mstDataNackW)
278  {
279  len = cnt + 1;
280  }
281  else
282  {
283  len = cnt;
284  }
285  break;
286  }
287  }
288 
289  I2C_STOP();
290 
291  return len;
292 }
