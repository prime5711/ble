254 /**************************************************************************************************
255  * @fn Hal_ProcessEvent
256  *
257  * @brief Hal Process Event
258  *
259  * @param task_id - Hal TaskId
260  * events - events
261  *
262  * @return None
263  **************************************************************************************************/
264 uint16 Hal_ProcessEvent( uint8 task_id, uint16 events )
265 {
266  uint8 *msgPtr;
267 
268  (void)task_id; // Intentionally unreferenced parameter
269 
270  if ( events & SYS_EVENT_MSG )
271  {
272  msgPtr = osal_msg_receive(Hal_TaskID);
273 
274  while (msgPtr)
275  {
276  /* Do something here - for now, just deallocate the msg and move on */
277 
278  /* De-allocate */
279  osal_msg_deallocate( msgPtr );
280  /* Next */
281  msgPtr = osal_msg_receive( Hal_TaskID );
282  }
283  return events ^ SYS_EVENT_MSG;
284  }
285 
286  if ( events & HAL_LED_BLINK_EVENT )
287  {
288 #if (defined (BLINK_LEDS)) && (HAL_LED == TRUE)
289  HalLedUpdate();
290 #endif /* BLINK_LEDS && HAL_LED */
291  return events ^ HAL_LED_BLINK_EVENT;
292  }
293 
294  if (events & HAL_KEY_EVENT)
295  {
296 
297 #if (defined HAL_KEY) && (HAL_KEY == TRUE)
298  /* Check for keys */
299  HalKeyPoll();
300 
301  /* if interrupt disabled, do next polling */
302  if (!Hal_KeyIntEnable)
303  {
304  osal_start_timerEx( Hal_TaskID, HAL_KEY_EVENT, 100);
305  }
306 #endif // HAL_KEY
307 
308  return events ^ HAL_KEY_EVENT;
309  }
310 
311 #if (defined HAL_MOTION) && (HAL_MOTION == TRUE) && (!defined NO_POINTING_LIB)
312  if (events & HAL_MOTION_EVENT)
313  {
314  uint16 eventHandled = HalMotionHandleOSEvent( events );
315  return events ^ eventHandled;
316  }
317 
318 #if (defined HAL_I2C) && (HAL_I2C == TRUE)
319  if (events & HAL_GYRO_REGISTER_ACCESS_READY_EVENT)
320  {
321  HalGyroHandleGyroRegisterAccessReadyEvent();
322  return events ^ HAL_GYRO_REGISTER_ACCESS_READY_EVENT;
323  }
324 #endif // #if (defined HAL_I2C) && (HAL_I2C == TRUE)
325 #endif // #if (defined HAL_MOTION) && (HAL_MOTION == TRUE)
326 
327 #if (defined HAL_BUZZER) && (HAL_BUZZER == TRUE)
328  if (events & HAL_BUZZER_EVENT)
329  {
330  HalBuzzerStop();
331  return events ^ HAL_BUZZER_EVENT;
332  }
333 #endif
334 
335  /* Nothing interested, discard the message */
336  return 0;
337 
338 }
