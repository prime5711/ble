360 /*********************************************************************
361  * @fn Accel_AddService
362  *
363  * @brief Initializes the Accelerometer service by
364  * registering GATT attributes with the GATT server. Only
365  * call this function once.
366  *
367  * @param services - services to add. This is a bit map and can
368  * contain more than one service.
369  *
370  * @return Success or Failure
371  */
372 bStatus_t Accel_AddService( uint32 services )
373 {
374  uint8 status = SUCCESS;
375 
376  // Initialize Client Characteristic Configuration attributes
377  GATTServApp_InitCharCfg( INVALID_CONNHANDLE, accelXConfigCoordinates );
378  GATTServApp_InitCharCfg( INVALID_CONNHANDLE, accelYConfigCoordinates );
379  GATTServApp_InitCharCfg( INVALID_CONNHANDLE, accelZConfigCoordinates );
380 
381  // Register with Link DB to receive link status change callback
382  VOID linkDB_Register( accel_HandleConnStatusCB );
383 
384  if ( services & ACCEL_SERVICE )
385  {
386  // Register GATT attribute list and CBs with GATT Server App
387  status = GATTServApp_RegisterService( accelAttrTbl,
388  GATT_NUM_ATTRS( accelAttrTbl ),
389  &accelCBs );
390  }
391 
392  return ( status );
393 }

420 /*********************************************************************
421  * @fn Accel_SetParameter
422  *
423  * @brief Set an Accelerometer Profile parameter.
424  *
425  * @param param - Profile parameter ID
426  * @param len - length of data to right
427  * @param value - pointer to data to write. This is dependent on
428  * the parameter ID and WILL be cast to the appropriate
429  * data type (example: data type of uint16 will be cast to
430  * uint16 pointer).
431  *
432  * @return bStatus_t
433  */
434 bStatus_t Accel_SetParameter( uint8 param, uint8 len, void *value )
435 {
436  bStatus_t ret = SUCCESS;
437 
438  switch ( param )
439  {
440  case ACCEL_ENABLER:
441  if ( len == sizeof ( uint8 ) )
442  {
443  accelEnabled = *((uint8*)value);
444  }
445  else
446  {
447  ret = bleInvalidRange;
448  }
449  break;
450 
451  case ACCEL_RANGE:
452  if ( (len == sizeof ( uint16 )) && ((*((uint8*)value)) <= ACCEL_RANGE_8G) )
453  {
454  accelRange = *((uint16*)value);
455  }
456  else
457  {
458  ret = bleInvalidRange;
459  }
460  break;
461 
462  case ACCEL_X_ATTR:
463  if ( len == sizeof ( int8 ) )
464  {
465  accelXCoordinates = *((int8*)value);
466 
467  // See if Notification has been enabled
468  GATTServApp_ProcessCharCfg( accelXConfigCoordinates, (uint8 *)&accelXCoordinates,
469  FALSE, accelAttrTbl, GATT_NUM_ATTRS( accelAttrTbl ),
470  INVALID_TASK_ID );
471  }
472  else
473  {
474  ret = bleInvalidRange;
475  }
476  break;
477 
478  case ACCEL_Y_ATTR:
479  if ( len == sizeof ( int8 ) )
480  {
481  accelYCoordinates = *((int8*)value);

483  // See if Notification has been enabled

485  FALSE, accelAttrTbl, GATT_NUM_ATTRS( accelAttrTbl ),

564 /*********************************************************************
565  * @fn accel_ReadAttr
566  *
567  * @brief Read an attribute.
568  *
569  * @param pAttr - pointer to attribute
570  * @param pLen - length of data to be read
571  * @param pValue - pointer to data to be read
572  * @param signature - whether to include Authentication Signature
573  *
574  * @return Success or Failure
575  */
576 static uint8 accel_ReadAttrCB( uint16 connHandle, gattAttribute_t *pAttr,
577  uint8 *pValue, uint8 *pLen, uint16 offset, uint8 maxLen )
578 {
579  uint16 uuid;
580  bStatus_t status = SUCCESS;
581 
582  // Make sure it's not a blob operation
583  if ( offset > 0 )
584  {
585  return ( ATT_ERR_ATTR_NOT_LONG );
586  }
587 
588  if ( pAttr->type.len == ATT_BT_UUID_SIZE )
589  {
590  // 16-bit UUID
591  uuid = BUILD_UINT16( pAttr->type.uuid[0], pAttr->type.uuid[1]);
592  switch ( uuid )
593  {
594  // No need for "GATT_SERVICE_UUID" or "GATT_CLIENT_CHAR_CFG_UUID" cases;
595  // gattserverapp handles those types for reads
596  case ACCEL_RANGE_UUID:
597  *pLen = 2;
598  pValue[0] = LO_UINT16( *((uint16 *)pAttr->pValue) );
599  pValue[1] = HI_UINT16( *((uint16 *)pAttr->pValue) );
600  break;
601 
602  case ACCEL_ENABLER_UUID:
603  case ACCEL_X_UUID:
604  case ACCEL_Y_UUID:
605  case ACCEL_Z_UUID:
606  *pLen = 1;
607  pValue[0] = *pAttr->pValue;
608  break;
609 
610  default:
611  // Should never get here!
612  *pLen = 0;
613  status = ATT_ERR_ATTR_NOT_FOUND;
614  break;
615  }
616  }
617  else
618  {
619  // 128-bit UUID
620  *pLen = 0;
621  status = ATT_ERR_INVALID_HANDLE;
622  }
623 
624 
625  return ( status );
626 }

628 /*********************************************************************
629  * @fn accel_WriteAttrCB
630  *
631  * @brief Validate attribute data prior to a write operation
632  *
633  * @param connHandle – connection message was received on
634  * @param pReq - pointer to request
635  *
636  * @return Success or Failure
637  */
638 static bStatus_t accel_WriteAttrCB( uint16 connHandle, gattAttribute_t *pAttr,
639  uint8 *pValue, uint8 len, uint16 offset )
640 {
641  bStatus_t status = SUCCESS;
642  uint8 notify = 0xFF;
643 
644  if ( pAttr->type.len == ATT_BT_UUID_SIZE )
645  {
646 
647  uint16 uuid = BUILD_UINT16( pAttr->type.uuid[0], pAttr->type.uuid[1]);
648  switch ( uuid )
649  {
650  case ACCEL_ENABLER_UUID:
651  //Validate the value
652  // Make sure it's not a blob oper
653  if ( offset == 0 )
654  {
655  if ( len > 1 )
656  status = ATT_ERR_INVALID_VALUE_SIZE;
657  else if ( pValue[0] != FALSE && pValue[0] != TRUE )
658  status = ATT_ERR_INVALID_VALUE;
659  }
660  else
661  {
662  status = ATT_ERR_ATTR_NOT_LONG;
663  }
664 
665  //Write the value
666  if ( status == SUCCESS )
667  {
668  uint8 *pCurValue = (uint8 *)pAttr->pValue;
669 
670  *pCurValue = pValue[0];
671  notify = ACCEL_ENABLER;
672  }
673 
674  break;
675 
676  case GATT_CLIENT_CHAR_CFG_UUID:
677  status = GATTServApp_ProcessCCCWriteReq( connHandle, pAttr, pValue, len,
678  offset, GATT_CLIENT_CFG_NOTIFY );
679  break;
680 
681  default:
682  // Should never get here!
683  status = ATT_ERR_ATTR_NOT_FOUND;
684  }
685  }
686  else
687  {
688  // 128-bit UUID
689  status = ATT_ERR_INVALID_HANDLE;
690  }
691 
692  // If an attribute changed then callback function to notify application of change
693  if ( (notify != 0xFF) && accel_AppCBs && accel_AppCBs->pfnAccelEnabler )
694  accel_AppCBs->pfnAccelEnabler();
695 
696  return ( status );
697 }

699 /*********************************************************************
700  * @fn accel_HandleConnStatusCB
701  *
702  * @brief Accelerometer Service link status change handler function.
703  *
704  * @param connHandle - connection handle
705  * @param changeType - type of change
706  *
707  * @return none
708  */
709 static void accel_HandleConnStatusCB( uint16 connHandle, uint8 changeType )
710 {
711  // Make sure this is not loopback connection
712  if ( connHandle != LOOPBACK_CONNHANDLE )
713  {
714  // Reset Client Char Config if connection has dropped
715  if ( ( changeType == LINKDB_STATUS_UPDATE_REMOVED ) ||
716  ( ( changeType == LINKDB_STATUS_UPDATE_STATEFLAGS ) &&
717  ( !linkDB_Up( connHandle ) ) ) )
718  {
719  GATTServApp_InitCharCfg( connHandle, accelXConfigCoordinates );
720  GATTServApp_InitCharCfg( connHandle, accelYConfigCoordinates );
721  GATTServApp_InitCharCfg( connHandle, accelZConfigCoordinates );
722  }
723  }
