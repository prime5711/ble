138 void __code* usbdpFindNext(uint8 wantedType, uint8 haltAtType)
139 {
140  void __code *pResult;
141  pResult = NULL;
142 
143  // As long as we haven't reached the end...
144  while (usbdpData.pDesc != (void __code *) usbDescriptorMarker.pUsbDescEnd) {
145 
146  // If we have a match on wantedType...
147  if (usbdpData.pDesc[DESC_TYPE_IDX] == wantedType) {
148  pResult = (void __code*) usbdpData.pDesc;
149  usbdpData.pDesc += usbdpData.pDesc[DESC_LENGTH_IDX];
150  break;
151 
152  // If we have a match on haltAtType...
153  } else if (usbdpData.pDesc[DESC_TYPE_IDX] == haltAtType) {
154  if (haltAtType) break;
155  }
156 
157  // Move on to the next descriptor
158  usbdpData.pDesc += usbdpData.pDesc[DESC_LENGTH_IDX];
159  }
160 
161  return pResult;
162 } // usbdpFindNext
